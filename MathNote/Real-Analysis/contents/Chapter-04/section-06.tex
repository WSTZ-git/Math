\section{习题}

\begin{example}
设$f\in L(E)$，$n$是自然数。证明：
\begin{align*}
\lim\limits_{n\rightarrow \infty}\int_{\left\{ x\in E:\left| f\left( x \right) \right|<\frac{1}{n} \right\}}\left| f\left( x \right) \right|\mathrm{d}x=0.
\end{align*}
\end{example}
\begin{proof}

记$A_n\triangleq \left\{ x\in E:\left| f\left( x \right) \right|<\frac{1}{n} \right\}\left( \forall n\in \mathbb{N} \right)$，则对$\forall n\in \mathbb{N}$，有
\begin{align*}
\left| f\left( x \right) \right|\cdot \chi _{A_n}\left( x \right) \leqslant \left| f\left( x \right) \right|,\quad \forall x\in E.
\end{align*}
又由$f\in L\left( E \right)$和\refthe{theorem:f与f绝对值的可积性等价}知$\left| f \right|\in L\left( E \right)$.于是由\hyperref[theorem:控制收敛定理]{控制收敛定理}知
\begin{align*}
\lim\limits_{n\rightarrow \infty}\int_{A_n}{\left| f\left( x \right) \right|\mathrm{d}x}=\lim\limits_{n\rightarrow \infty}\int_E{\left| f\left( x \right) \right|\cdot \chi _{A_n}\left( x \right) \mathrm{d}x}=\int_E{\left| f\left( x \right) \right|\lim\limits_{n\rightarrow \infty}\chi _{A_n}\left( x \right) \mathrm{d}x}.
\end{align*}
注意到$\left\{ A_n \right\} _{n=1}^{\infty}$是递减集列，故
\begin{align*}
\lim\limits_{n\rightarrow \infty}A_n=\bigcap\limits_{n=1}^{\infty}{A_n}=\bigcap\limits_{n=1}^{\infty}{\left\{ x\in E:\left| f\left( x \right) \right|<\frac{1}{n} \right\}}=\left\{ x\in E\mid f\left( x \right) =0 \right\}.
\end{align*}
从而由\hyperref[proposition:特征函数的基本性质-8]{特征函数的基本性质\ref{proposition:特征函数的基本性质-8}}知
\begin{align*}
\left| f\left( x \right) \right|\lim\limits_{n\rightarrow \infty}\chi _{A_n}\left( x \right) =\left| f\left( x \right) \right|\chi _{\lim\limits_{n\rightarrow \infty}A_n}\left( x \right) =\left| f\left( x \right) \right|\chi _{\left\{ x\in E\mid f\left( x \right) =0 \right\}}\left( x \right) =0,\quad \forall x\in E.
\end{align*}
因此
\begin{align*}
\lim\limits_{n\rightarrow \infty}\int_{A_n}{\left| f\left( x \right) \right|\mathrm{d}x}=\int_E{\left| f\left( x \right) \right|\lim\limits_{n\rightarrow \infty}\chi _{A_n}\left( x \right) \mathrm{d}x}=0.
\end{align*}
\end{proof}
