\section{测地坐标系和法坐标系}

\begin{definition}
设$S$为正则参数曲面，$D\subseteq S$。若存在依赖单参数的曲线族$\Sigma$，使得对$\forall\,p\in D$，有且仅有一条$\Sigma$中的测地线经过点$p$，则称$\Sigma$为覆盖区域$D$的\textbf{测地线族}。
\end{definition}

\begin{proposition}
设$\Sigma$为覆盖区域$D$的测地线族，$\Sigma_1$为$D$内与$\Sigma$中曲线正交的轨线族，则在$D$内任一点的邻域中存在正交参数系$(u,v)$，使得$u$-曲线族为$\Sigma$，$v$-曲线族为$\Sigma_1$，曲面第一基本形式为
\begin{align*}
\mathrm{I}=E(u,v)(\mathrm{d}u)^2+G(u,v)(\mathrm{d}v)^2.
\end{align*}
且可通过参数变换化为
\begin{align}
\mathrm{I}=\mathrm{d}\tilde{u}^2+\tilde{G}(\tilde{u},\tilde{v})(\mathrm{d}\tilde{v})^2.\label{eq:geo-baseform}
\end{align}
\end{proposition}
\begin{proof}
由$\S6.1$的Liouville公式，$u$-曲线为测地线，其测地曲率$\kappa_{g1}=-\frac{1}{2\sqrt{G}}\frac{\partial\ln E}{\partial v}=0$，故$\frac{\partial E}{\partial v}=0$，即$E=E(u)$。
作参数变换
\begin{align*}
\tilde{u}=\int_{u_0}^u\sqrt{E(u)}\mathrm{d}u,\quad \tilde{v}=v,
\end{align*}
代入第一基本形式即得\eqref{eq:geo-baseform}。

\end{proof}

\begin{theorem}
设$\Sigma$是覆盖区域$D$的测地线族，$\Sigma_1$为$D$内与$\Sigma$正交的轨线族，则$\Sigma_1$中任意两条曲线，在$\Sigma$的每条测地线上截得的线段长度相等。
\end{theorem}
\begin{proof}
取参数系$(\tilde{u},\tilde{v})$使第一基本形式为\eqref{eq:geo-baseform}。取$\Sigma_1$中两条曲线$C_1:\tilde{u}=u_1$，$C_2:\tilde{u}=u_2$，$\Sigma$中任意测地线$C:\tilde{v}=v_0$，则截得线段的长度为
\begin{align*}
\int_{u_1}^{u_2}\sqrt{\mathrm{I}}\bigg|_{\tilde{v}=v_0}=\int_{u_1}^{u_2}\mathrm{d}\tilde{u}=u_2-u_1,
\end{align*}
该长度与$v_0$无关。

\end{proof}
\begin{remark}
$\Sigma_1$中任意两条正交轨线之间距离处处相等，称此类正交轨线为\textbf{测地平行}的。
\end{remark}

\begin{theorem}
设$C$是曲面$S$上连接$p,q$两点的测地线，若$C$可嵌入覆盖区域$D$的测地线族$\Sigma$，且$p,q\in D$，则$C$是$D$内连接$p,q$的最短曲线。
\end{theorem}
\begin{proof}
取参数系$(\tilde{u},\tilde{v})$满足\eqref{eq:geo-baseform}，设$C:\tilde{v}=0$，$p(0,0)$，$q(l,0)$。
任取$D$内连接$p,q$的曲线$\tilde{C}:u=u(t),\,v=v(t)\ (0\leqslant t\leqslant l)$，满足$u(0)=0,\,u(l)=l,\,v(0)=v(l)=0$，其长度为
\begin{align*}
L(\tilde{C})&=\int_0^l\sqrt{\left(\frac{\mathrm{d}u(t)}{\mathrm{d}t}\right)^2+\tilde{G}(u(t),v(t))\left(\frac{\mathrm{d}v(t)}{\mathrm{d}t}\right)^2}\mathrm{d}t\\
&\geqslant\int_0^l\left|\frac{\mathrm{d}u(t)}{\mathrm{d}t}\right|\mathrm{d}t\geqslant\int_0^l\mathrm{d}u(t)=l=L(C).
\end{align*}
故$L(\tilde{C})\geqslant L(C)$。

\end{proof}

\begin{definition}
在曲面$S$上取定一条测地线$C$，过$C$上每一点作与$C$正交的测地线，构成覆盖$C$邻域$U$的测地线族$\Sigma$。
取参数系$(u,v)$，使$\Sigma$为$u$-曲线族，$C$对应$u=0$，且第一基本形式为
\begin{align*}
\mathrm{I}=(\mathrm{d}u)^2+G(u,v)(\mathrm{d}v)^2,
\end{align*}
满足$G(0,v)=1$，$\frac{\partial G}{\partial u}(0,v)=0$，则称$(u,v)$为$S$在$C$邻域内的\textbf{测地平行坐标系}。
\end{definition}

\begin{theorem}
曲面$S$上任意一点$p$的充分小邻域$U$内，必存在测地平行坐标系$(u,v)$，使得$p$对应$u=0,\,v=0$，第一基本形式为
\begin{align}
\mathrm{I}=(\mathrm{d}u)^2+G(u,v)(\mathrm{d}v)^2,\label{eq:geo-parallel-1}
\end{align}
且
\begin{align}
G(0,v)=1,\quad \frac{\partial G}{\partial u}(0,v)=0.\label{eq:geo-parallel-2}
\end{align}
\end{theorem}
\begin{proof}
取过$p$的测地线$C$，作其正交测地线族得到正交参数系。
$C:u=0$为测地线，由Liouville公式，其测地曲率$\kappa_{g2}\big|_{u=0}=\frac{1}{2}\frac{\partial\ln G}{\partial u}\big|_{u=0}=0$，故$\frac{\partial G}{\partial u}(0,v)=0$。
再通过参数变换使$G(0,v)=1$，即得\eqref{eq:geo-parallel-1}与\eqref{eq:geo-parallel-2}。

\end{proof}
\begin{remark}
平面上的测地平行坐标系即为平面笛卡尔直角坐标系。
\end{remark}

\begin{definition}
设$p\in S$，$T_pS$为$S$在$p$点的切空间。对任意切向量$\boldsymbol{v}\in T_pS$，存在唯一过$p$、以$\boldsymbol{v}$为初始切向量的测地线$\gamma(s;\boldsymbol{v})$，满足$\gamma(0)=p,\,\gamma'(0)=\boldsymbol{v}$。
取$T_pS$中原点的充分小邻域$U$，定义映射
\begin{align*}
\exp_p:U\to S,\quad \exp_p(\boldsymbol{v})=\gamma(1;\boldsymbol{v}),
\end{align*}
称$\exp_p$为$S$在$p$点的\textbf{指数映射}。
\end{definition}

\begin{proposition}
指数映射$\exp_p:U\to S$为连续可微映射，且对任意常数$\lambda>0$，有$\gamma(\lambda t;\boldsymbol{v})=\gamma(t;\lambda\boldsymbol{v})$。
\end{proposition}
\begin{proof}
对测地线作参数变换$\tilde{\gamma}(t)=\gamma(\lambda t;\boldsymbol{v})$，代入测地线微分方程可知$\tilde{\gamma}(t)$仍为测地线，且$\tilde{\gamma}'(0)=\lambda\boldsymbol{v}$，故$\tilde{\gamma}(t)=\gamma(t;\lambda\boldsymbol{v})$。
由常微分方程初值解的连续可微依赖性，$\exp_p$连续可微。

\end{proof}

\begin{definition}
在切空间$T_pS$上取单位正交标架$\{\boldsymbol{r}_1|_p,\boldsymbol{r}_2|_p\}$，对应坐标$(v^1,v^2)$。通过指数映射$\exp_p$将切空间坐标拉回至曲面$S$，得到$S$在$p$点附近的参数系$(v^1,v^2)$，该参数系称为$S$在$p$点的\textbf{法坐标系}。
\end{definition}

\begin{theorem}
曲面$S$在任意点$p$的附近必存在法坐标系$(v^1,v^2)$，满足：
\begin{enumerate}[(1)]
\item 从$p$出发、以$(v_0^1,v_0^2)$为初始切向量的测地线参数方程为$v^1(t)=tv_0^1,\,v^2(t)=tv_0^2$；
\item 第一类基本量$\tilde{g}_{\alpha\beta}$满足
\begin{align*}
\tilde{g}_{11}(p)=\tilde{g}_{22}(p)=1,\quad \tilde{g}_{12}(p)=\tilde{g}_{21}(p)=0,\quad \tilde{\Gamma}_{\alpha\beta}^\gamma(p)=0,
\end{align*}
其中$\tilde{\Gamma}_{\alpha\beta}^\gamma$为对应Christoffel记号；
\item $\displaystyle\frac{\partial\tilde{g}_{\alpha\beta}}{\partial v^\gamma}(p)=0,\ \alpha,\beta,\gamma=1,2$.
\end{enumerate}
\end{theorem}
\begin{proof}
由指数映射性质，测地线$\gamma(t;\boldsymbol{v})$在法坐标系下为线性形式$v^\alpha(t)=tv^\alpha$。
将其代入测地线微分方程$\frac{\mathrm{d}^2v^\gamma}{\mathrm{d}t^2}+\tilde{\Gamma}_{\alpha\beta}^\gamma\frac{\mathrm{d}v^\alpha}{\mathrm{d}t}\frac{\mathrm{d}v^\beta}{\mathrm{d}t}=0$，令$t=0$得$\tilde{\Gamma}_{\alpha\beta}^\gamma(p)v^\alpha v^\beta=0$，对任意$\boldsymbol{v}$成立，故$\tilde{\Gamma}_{\alpha\beta}^\gamma(p)=0$。
结合Christoffel记号与第一基本量的关系，可得$\frac{\partial\tilde{g}_{\alpha\beta}}{\partial v^\gamma}(p)=0$。

\end{proof}
\begin{remark}
法坐标系仅在$p$点处满足$\tilde{g}_{12}=0$，在$p$点外未必为正交参数系。
\end{remark}

\begin{definition}
在切空间$T_pS$的法坐标系$(v^1,v^2)$上取极坐标$(s,\theta)$，满足$v^1=s\cos\theta,\,v^2=s\sin\theta$。
通过指数映射$\exp_p$将极坐标拉回至曲面$S$，得到$S$上除去一条从$p$出发的测地线后的参数系$(s,\theta)$，该参数系称为$S$在$p$点的\textbf{测地极坐标系}。
固定$s=s_0$、$\theta$变化得到的曲线，称为以$p$为中心、$s_0$为半径的\textbf{测地圆}。
\end{definition}

\begin{lemma}
从点$p$出发的测地线与以点$p$为中心的测地圆彼此正交，即曲线族$\Sigma_1$中的每一条曲线是测地线族$\Sigma$的正交轨线。
\end{lemma}

\begin{proof}
取点$p$附近的正交参数系$(u^1,u^2)$，其中点$p$对应$u^1=0,\,u^2=0$，曲面$S$的第一基本形式为
$$\mathrm{I}=g_{\alpha\beta}\mathrm{d}u^\alpha\mathrm{d}u^\beta.$$
设$\theta$为点$p$处切向量与$u^1$-曲线的夹角，过点$p$、与$u^1$-曲线夹角为$\theta$的测地线记为$C_\theta$，参数方程为$u^\alpha=u^\alpha(s,\theta)\ (\alpha=1,2)$，$s$为测地线的弧长参数，且$u^\alpha(s,\theta)$关于$s,\theta$连续可微。
测地线族$\Sigma$为$\theta=\text{常数}$的曲线，测地圆族$\Sigma_1$为$s=\text{常数}$的曲线。取测地线$C:u^\alpha=u^\alpha(s,\theta_0)\ (0\leqslant s\leqslant s_0)$，则$\Sigma$是$C$的一个变分，对应的变分向量场为
$$w^\alpha(s)=\left.\frac{\partial u^\alpha(s,\theta)}{\partial\theta}\right|_{\theta=\theta_0}.$$
由于所有变分曲线均过点$p$，故$u^\alpha(0,\theta)=0$，因此$w^\alpha(0)=0\ (\alpha=1,2)$。
$w^\alpha(s_0)\boldsymbol{r}_\alpha$为测地圆$u^\alpha=u^\alpha(s_0,\theta)$在$\theta=\theta_0$处的切向量。由弧长第一变分公式
\begin{align}
\left.\frac{\mathrm{d}}{\mathrm{d}\theta}\right|_{\theta=\theta_0}L(C_\theta)&=g_{\alpha\beta}w^\alpha(s)\left.\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\right|_{s=0}-\int_{0}^{s_0}g_{\alpha\beta}w^\alpha\left(\frac{\mathrm{d}^2u^\beta}{\mathrm{d}s^2}+\Gamma^\beta_{\gamma\delta}\frac{\mathrm{d}u^\gamma}{\mathrm{d}s}\frac{\mathrm{d}u^\delta}{\mathrm{d}s}\right)\mathrm{d}s\nonumber\\
&=g_{\alpha\beta}w^\alpha(s_0)\left.\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\right|_{s=s_0}.
\label{eq:3.28}
\end{align}
每条测地线$C_\theta$的长度均为$s_0$，即$L(C_\theta)=s_0$，故式\eqref{eq:3.28}左端为零，因此
$$g_{\alpha\beta}w^\alpha(s_0)\left.\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\right|_{s=s_0}=0,$$
即测地线$C$与半径为$s_0$的测地圆正交。

\end{proof}

由上述引理，$(s,\theta)$可作为曲面局部参数系，此时曲面$S$的第一基本形式可写为
\begin{align}
\mathrm{I}=\mathrm{d}s^2+G(s,\theta)\mathrm{d}\theta^2.
\label{eq:3.29}
\end{align}

设曲面$S$在点$p$的法坐标系为$(v^1,v^2)$，法坐标系与测地极坐标系$(s,\theta)$满足坐标变换
$$(v^1)^2+(v^2)^2=s^2,\quad \frac{v^2}{v^1}=\tan\theta,$$
微分关系为
$$\mathrm{d}v^1=\cos\theta\mathrm{d}s-s\sin\theta\mathrm{d}\theta,\quad \mathrm{d}v^2=\sin\theta\mathrm{d}s+s\cos\theta\mathrm{d}\theta,$$
解得
$$\mathrm{d}s=\frac{v^1\mathrm{d}v^1+v^2\mathrm{d}v^2}{s},\quad \mathrm{d}\theta=\frac{v^1\mathrm{d}v^2-v^2\mathrm{d}v^1}{s^2}.$$
将其代入式\eqref{eq:3.29}得
\begin{align*}
\mathrm{I}&=\left(\frac{v^1\mathrm{d}v^1+v^2\mathrm{d}v^2}{s}\right)^2+G(s,\theta)\left(\frac{v^1\mathrm{d}v^2-v^2\mathrm{d}v^1}{s^2}\right)^2\\
&=\left(\frac{(v^1)^2}{s^2}+G\frac{(v^2)^2}{s^4}\right)(\mathrm{d}v^1)^2+\frac{2v^1v^2}{s^2}\left(1-\frac{G}{s^2}\right)\mathrm{d}v^1\mathrm{d}v^2+\left(\frac{(v^2)^2}{s^2}+G\frac{(v^1)^2}{s^4}\right)(\mathrm{d}v^2)^2\\
&=\left(1+\frac{(v^2)^2}{s^2}\left(\frac{G}{s^2}-1\right)\right)(\mathrm{d}v^1)^2+\frac{2v^1v^2}{s^2}\left(1-\frac{G}{s^2}\right)\mathrm{d}v^1\mathrm{d}v^2+\left(1+\frac{(v^1)^2}{s^2}\left(\frac{G}{s^2}-1\right)\right)(\mathrm{d}v^2)^2.
\end{align*}
对比法坐标系下第一基本形式的性质，可得
\begin{align}
\lim_{s\to0}\frac{G(s,\theta)}{s^2}=1.
\label{eq:3.31}
\end{align}
由此推得
$$G(s,\theta)=s^2+o(s^2),\quad \sqrt{G(s,\theta)}=s+o(s),$$
进而有
$$\lim_{s\to0}\sqrt{G(s,\theta)}=0,\quad \lim_{s\to0}\frac{\partial}{\partial s}\sqrt{G(s,\theta)}=1.$$

\begin{theorem}
在曲面$S$的每一点$p$的邻域内，除去从点$p$出发的一条测地线外，必存在测地极坐标系$(s,\theta)$，使得曲面$S$的第一基本形式为
$$\mathrm{I}=\mathrm{d}s^2+G(s,\theta)\mathrm{d}\theta^2,$$
其中函数$G(s,\theta)$满足
\begin{align}
\lim_{s\to0}\sqrt{G(s,\theta)}=0,\quad \lim_{s\to0}\frac{\partial}{\partial s}\sqrt{G(s,\theta)}=1.
\label{eq:3.32}
\end{align}
\end{theorem}
\begin{remark}
平面上的极坐标系即为测地极坐标系。
\end{remark}

\begin{example}
设曲面$S$的第一基本形式是$\mathrm{I}=(\mathrm{d}u)^2+G(u,v)(\mathrm{d}v)^2$，求$\Gamma_{\alpha\beta}^{\gamma}$和Gauss曲率$K$。
\end{example}

\begin{example}
设曲面$S$的第一基本形式是$\mathrm{I}=(\mathrm{d}u)^2+G(u,v)(\mathrm{d}v)^2$，并且函数$G(u,v)$满足条件$G(0,v)=1$，$G_u(0,v)=0$。证明：
\begin{align*}
G(u,v)=1-u^2K(0,v)+o(u^2).
\end{align*}
\end{example}

\begin{example}
设曲面$S$上以点$p$为中心、以$r$为半径的测地圆的周长是$L_r$，其所围的面积是$A_r$。证明：曲面$S$在点$p$处的Gauss曲率是
\begin{align*}
K(p)=\lim\limits_{r\to0}\frac{3}{\pi}\cdot\frac{2\pi r-L_r}{r^3}=\lim\limits_{r\to0}\frac{12}{\pi}\cdot\frac{\pi r^2-A_r}{r^4}.
\end{align*}
\end{example}
