\section{抽象曲面的内蕴几何}

\section{抽象曲面的概念}
\begin{definition}
设$M$是 Hausdorff 拓扑空间，若对于任意一点$p\in M$，都存在$p$的开邻域$U\subset M$，使得$U$与平面$\mathbb{R}^2$中的一个开区域$D$同胚，则称$M$为**二维拓扑流形**，简称**曲面**。
\end{definition}

\begin{definition}
设$M$是二维拓扑流形，若在$M$上给定一个容许坐标映射族$\Sigma=\{(U_\alpha,\varphi_\alpha)\}$，满足：
\begin{enumerate}[(1)]
\item $\bigcup\limits_{\alpha}U_\alpha=M$；
\item 对任意$\alpha$，$\varphi_\alpha:U_\alpha\to D_\alpha\subset\mathbb{R}^2$是同胚；
\item 若$U_\alpha\cap U_\beta\neq\emptyset$，则坐标变换$\varphi_\beta\circ\varphi_\alpha^{-1}:\varphi_\alpha(U_\alpha\cap U_\beta)\to\varphi_\beta(U_\alpha\cap U_\beta)$是$C^\infty$类的，
\end{enumerate}
则称$\Sigma$为$M$上的一个**光滑结构**，赋予光滑结构的二维拓扑流形称为**光滑曲面**。
\end{definition}

\begin{definition}
若光滑曲面$M$上存在坐标覆盖$\{(U_\alpha,\varphi_\alpha)\}$，使得所有坐标变换的Jacobi行列式都大于零，则称$M$是**可定向曲面**。
\end{definition}

\section{切向量与切空间}
\begin{definition}
设$M$是光滑曲面，$p\in M$，$C^\infty(p)$表示在$p$的邻域内有定义的$C^\infty$函数的全体。若映射$v:C^\infty(p)\to\mathbb{R}$满足：
\begin{enumerate}[(1)]
\item $v(f+g)=v(f)+v(g)$，$\forall f,g\in C^\infty(p)$；
\item $v(\lambda f)=\lambda v(f)$，$\forall\lambda\in\mathbb{R}$，$f\in C^\infty(p)$；
\item $v(fg)=f(p)v(g)+g(p)v(f)$，
\end{enumerate}
则称$v$为$M$在点$p$处的**切向量**。
\end{definition}

\begin{definition}
曲面$M$在点$p$处的全体切向量构成的二维线性空间，称为$M$在点$p$处的**切空间**，记作$T_pM$。
\end{definition}

\begin{theorem}
设$(U;u,v)$是点$p\in M$的局部坐标系，则$\left\{\dfrac{\partial}{\partial u}\bigg|_p,\dfrac{\partial}{\partial v}\bigg|_p\right\}$是$T_pM$的一组基，任意切向量$v\in T_pM$可表示为
\[v=v^1\dfrac{\partial}{\partial u}\bigg|_p+v^2\dfrac{\partial}{\partial v}\bigg|_p,\]
其中$v^1,v^2\in\mathbb{R}$称为切向量$v$的分量。
\end{theorem}

\section{黎曼度量}
\begin{definition}
光滑曲面$M$上的一个**黎曼度量**$g$是指对每一点$p\in M$，在切空间$T_pM$上指定一个正定对称双线性函数$g_p:T_pM\times T_pM\to\mathbb{R}$，且$g$关于局部坐标是光滑的。
\end{definition}

\begin{definition}
在局部坐标系$(U;u,v)$下，黎曼度量$g$的系数为
\[E=g\left(\dfrac{\partial}{\partial u},\dfrac{\partial}{\partial u}\right),\quad F=g\left(\dfrac{\partial}{\partial u},\dfrac{\partial}{\partial v}\right),\quad G=g\left(\dfrac{\partial}{\partial v},\dfrac{\partial}{\partial v}\right),\]
且$E>0$，$EG-F^2>0$，黎曼度量的局部表达式为
\[g=E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2.\]
\end{definition}

\begin{remark}
赋予黎曼度量的光滑曲面称为**黎曼曲面**，黎曼度量是曲面的内蕴结构，决定了曲面上的长度、夹角、面积等内蕴几何量。
\end{remark}

\section{协变微分}
\begin{definition}
设$M$是黎曼曲面，$X,Y$是$M$上的光滑切向量场，$X=X^1\dfrac{\partial}{\partial u}+X^2\dfrac{\partial}{\partial v}$，$Y=Y^1\dfrac{\partial}{\partial u}+Y^2\dfrac{\partial}{\partial v}$，则$Y$关于$X$的**协变导数**为
\begin{align*}
\nabla_XY&=\left(X(Y^1)+X^1Y^1\Gamma^1_{11}+X^1Y^2\Gamma^1_{12}+X^2Y^1\Gamma^1_{12}+X^2Y^2\Gamma^1_{22}\right)\dfrac{\partial}{\partial u}\\
&\quad+\left(X(Y^2)+X^1Y^1\Gamma^2_{11}+X^1Y^2\Gamma^2_{12}+X^2Y^1\Gamma^2_{12}+X^2Y^2\Gamma^2_{22}\right)\dfrac{\partial}{\partial v},
\end{align*}
其中$\Gamma^i_{jk}$是由黎曼度量决定的Christoffel记号。
\end{definition}

\begin{proposition}
协变导数满足下列性质：
\begin{enumerate}[(1)]
\item $\nabla_{fX+gY}Z=f\nabla_XZ+g\nabla_YZ$；
\item $\nabla_X(Y+Z)=\nabla_XY+\nabla_XZ$；
\item $\nabla_X(fY)=X(f)Y+f\nabla_XY$；
\item $X(g(Y,Z))=g(\nabla_XY,Z)+g(Y,\nabla_XZ)$。
\end{enumerate}
\end{proposition}

\section{测地线}
\begin{definition}
设$\gamma:I\to M$是黎曼曲面上的光滑曲线，若其切向量场$\gamma'(t)$沿自身的协变导数为零，即
\[\nabla_{\gamma'(t)}\gamma'(t)=0,\]
则称$\gamma$为$M$上的**测地线**。
\end{definition}

\begin{theorem}
在局部坐标系$(u,v)$下，测地线的微分方程为
\begin{align}
&\dfrac{\mathrm{d}^2u}{\mathrm{d}t^2}+\Gamma^1_{11}\left(\dfrac{\mathrm{d}u}{\mathrm{d}t}\right)^2+2\Gamma^1_{12}\dfrac{\mathrm{d}u}{\mathrm{d}t}\dfrac{\mathrm{d}v}{\mathrm{d}t}+\Gamma^1_{22}\left(\dfrac{\mathrm{d}v}{\mathrm{d}t}\right)^2=0,\label{eq:geodesic1}\\
&\dfrac{\mathrm{d}^2v}{\mathrm{d}t^2}+\Gamma^2_{11}\left(\dfrac{\mathrm{d}u}{\mathrm{d}t}\right)^2+2\Gamma^2_{12}\dfrac{\mathrm{d}u}{\mathrm{d}t}\dfrac{\mathrm{d}v}{\mathrm{d}t}+\Gamma^2_{22}\left(\dfrac{\mathrm{d}v}{\mathrm{d}t}\right)^2=0.\label{eq:geodesic2}
\end{align}
\end{theorem}

\begin{proof}
由测地线的定义$\nabla_{\gamma'(t)}\gamma'(t)=0$，将切向量场代入协变导数的表达式，直接可得测地线微分方程。

\end{proof}

\begin{remark}
测地线是欧氏空间中直线的推广，是曲面上的最短曲线（局部意义下）。
\end{remark}

\section{高斯-博内定理}
\begin{definition}
黎曼曲面上的**高斯曲率**$K$是内蕴几何量，在局部坐标系下可表示为
\[K=\dfrac{-1}{(EG-F^2)^2}\left(\begin{vmatrix}
-\dfrac{1}{2}E_{vv}+F_{uv}-\dfrac{1}{2}G_{uu} & \dfrac{1}{2}E_u & F_u-\dfrac{1}{2}E_v\\
F_v-\dfrac{1}{2}G_u & E & F\\
\dfrac{1}{2}G_v & F & G
\end{vmatrix}\right).\]
\end{definition}

\begin{theorem}[高斯-博内定理]
设$M$是紧致可定向的光滑黎曼曲面，则
\[\iint\limits_M K\mathrm{d}\sigma=2\pi\chi(M),\]
其中$\mathrm{d}\sigma=\sqrt{EG-F^2}\mathrm{d}u\mathrm{d}v$是曲面的面积元素，$\chi(M)$是曲面的欧拉示性数。
\end{theorem}

\begin{proof}
将曲面分解为有限个测地三角形，对每个测地三角形应用局部高斯-博内公式，再求和即可得到整体公式。

\end{proof}

\begin{corollary}
紧致可定向黎曼曲面的高斯曲率的积分是拓扑不变量，与黎曼度量的选取无关。
\end{corollary}

\section{常曲率曲面}
\begin{definition}
若黎曼曲面$M$的高斯曲率$K$为常数，则称$M$为**常曲率曲面**。
\end{definition}

\begin{proposition}
常曲率曲面的黎曼度量可化为标准形式：
\begin{enumerate}[(1)]
\item 当$K=0$时，$g=(\mathrm{d}u)^2+(\mathrm{d}v)^2$（欧氏平面）；
\item 当$K>0$时，$g=(\mathrm{d}u)^2+\dfrac{1}{K}\sin^2(\sqrt{K}u)(\mathrm{d}v)^2$（球面）；
\item 当$K<0$时，$g=(\mathrm{d}u)^2+\dfrac{1}{-K}\sinh^2(\sqrt{-K}u)(\mathrm{d}v)^2$（双曲平面）。
\end{enumerate}
\end{proposition}

\begin{example}
单位球面$S^2$的高斯曲率$K\equiv1$，伪球面的高斯曲率$K\equiv-1$，欧氏平面的高斯曲率$K\equiv0$。
\end{example}

\section{指数映射与法坐标系}
\begin{definition}
设$p\in M$，$v\in T_pM$，$\gamma_v(t)$是满足$\gamma_v(0)=p$，$\gamma_v'(0)=v$的测地线，定义**指数映射**$\exp_p:T_pM\to M$为
\[\exp_p(v)=\gamma_v(1).\]
\end{definition}

\begin{theorem}
指数映射$\exp_p$在$T_pM$的原点邻域内是微分同胚，由此诱导的局部坐标系称为**法坐标系**。
\end{theorem}

\begin{remark}
在法坐标系下，点$p$处的Christoffel记号$\Gamma^i_{jk}(p)=0$，黎曼度量的一阶偏导数在$p$处为零。
\end{remark}

\section{洛伦兹空间中的伪球面}
\begin{definition}
三维洛伦兹空间$\mathbb{R}^3_1$中的向量内积为
\[v\cdot w=v^1w^1+v^2w^2-v^3w^3,\]
满足$v\cdot v=-1$的点的轨迹称为**伪球面**。
\end{definition}

\begin{example}
伪球面的参数方程为
\[\bm{r}(u,v)=\left(\dfrac{\sin u}{\cosh v},\dfrac{\sinh v}{\cosh v},\cos u\right),\]
其高斯曲率$K\equiv-1$，是典型的负常曲率曲面模型。
\end{example}
