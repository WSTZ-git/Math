\section{测地曲率和测地挠率}

\begin{definition}
设正则参数曲面$S$的方程为$\boldsymbol{r}=\boldsymbol{r}(u^1,u^2)$，$C$为$S$上以弧长$s$为参数的曲线$\boldsymbol{r}=\boldsymbol{r}(u^1(s),u^2(s))$。沿$C$定义\textbf{正交标架场}$\{\boldsymbol{r};\boldsymbol{e}_1,\boldsymbol{e}_2,\boldsymbol{e}_3\}$：
\begin{align}
\boldsymbol{e}_1&=\frac{\mathrm{d}\boldsymbol{r}(s)}{\mathrm{d}s}=\boldsymbol{\alpha}(s),\label{eq:e1}\\
\boldsymbol{e}_3&=\boldsymbol{n}(s),\label{eq:e3}\\
\boldsymbol{e}_2&=\boldsymbol{e}_3\times\boldsymbol{e}_1=\boldsymbol{n}(s)\times\boldsymbol{\alpha}(s),\label{eq:e2}
\end{align}
其中$\boldsymbol{\alpha}(s)$为$C$的单位切向量，$\boldsymbol{n}(s)$为曲面$S$的单位法向量。

该标架场的运动公式为：
\begin{align}
\begin{cases}
\displaystyle\frac{\mathrm{d}\boldsymbol{r}(s)}{\mathrm{d}s}=\boldsymbol{e}_1,\\
\displaystyle\frac{\mathrm{d}\boldsymbol{e}_1(s)}{\mathrm{d}s}=\kappa_g\boldsymbol{e}_2+\kappa_n\boldsymbol{e}_3,\\
\displaystyle\frac{\mathrm{d}\boldsymbol{e}_2(s)}{\mathrm{d}s}=-\kappa_g\boldsymbol{e}_1+\tau_g\boldsymbol{e}_3,\\
\displaystyle\frac{\mathrm{d}\boldsymbol{e}_3(s)}{\mathrm{d}s}=-\kappa_n\boldsymbol{e}_1-\tau_g\boldsymbol{e}_2,
\end{cases}\label{eq:moveformula}
\end{align}
其中：
\begin{enumerate}[(1)]
\item $\kappa_n=\frac{\mathrm{d}\boldsymbol{e}_1(s)}{\mathrm{d}s}\cdot\boldsymbol{e}_3=\boldsymbol{r}''(s)\cdot\boldsymbol{n}$，称为$C$沿切方向的\textbf{法曲率}；
\item $\kappa_g=\frac{\mathrm{d}^2\boldsymbol{r}(s)}{\mathrm{d}s^2}\cdot\boldsymbol{e}_2=(\boldsymbol{n}(s),\boldsymbol{r}'(s),\boldsymbol{r}''(s))$，称为$C$的\textbf{测地曲率}；
\item $\tau_g=\frac{\mathrm{d}\boldsymbol{e}_2}{\mathrm{d}s}\cdot\boldsymbol{n}=(\boldsymbol{n}(s),\boldsymbol{n}'(s),\boldsymbol{r}'(s))$，称为$C$的\textbf{测地挠率}。
\end{enumerate}
\end{definition}

\begin{proposition}
设$\kappa$为空间曲线$C$的曲率，$\kappa_g,\kappa_n$分别为其测地曲率、法曲率，则
\begin{align*}
\kappa^2=\kappa_g^2+\kappa_n^2.
\end{align*}
\end{proposition}
\begin{proof}
由空间曲线Frenet公式，$\kappa\boldsymbol{\beta}(s)=\frac{\mathrm{d}\boldsymbol{e}_1}{\mathrm{d}s}=\kappa_g\boldsymbol{e}_2+\kappa_n\boldsymbol{n}$，又$\boldsymbol{e}_2\perp\boldsymbol{n}$，对等式两边作模长平方即得。

\end{proof}

\begin{theorem}
设$C$是曲面$S$上的正则曲线，则$C$在点$p$的测地曲率，等于$C$投影到$S$在$p$的切平面上所得曲线$\tilde{C}$在该点的\textbf{相对曲率}，切平面正向由$S$在$p$的法向量$\boldsymbol{n}$给出。
\end{theorem}
\begin{proof}
设$S$在$p$的切平面为$\Pi$，过$C$各点作$\Pi$的垂线构成柱面$\tilde{S}$，则$C=S\cap\tilde{S}$，$\boldsymbol{e}_2=\boldsymbol{n}\times\boldsymbol{e}_1$为$\tilde{S}$的法向量。

投影曲线$\tilde{C}=\tilde{S}\cap\Pi$，$\Pi$是$\tilde{S}$的法截面，$\tilde{C}$是$\tilde{S}$上与$C$相切的法截线。于是
\begin{align*}
\kappa_g&=\frac{\mathrm{d}^2\boldsymbol{r}(s)}{\mathrm{d}s^2}\cdot\boldsymbol{e}_2\\
&=\text{$C$作为$\tilde{S}$上曲线的法曲率}\\
&=\text{$\tilde{C}$作为平面$\Pi$上曲线的相对曲率}.
\end{align*}

\end{proof}

\begin{theorem}
曲面$S$上任意曲线$C$的测地曲率$\kappa_g$在$S$的\textbf{保长对应}下保持不变，即测地曲率是曲面的\textbf{内蕴几何量}。
\end{theorem}
\begin{proof}
设$C:u^\alpha=u^\alpha(s)\ (\alpha=1,2)$，$s$为弧长参数，有
\begin{align*}
\boldsymbol{e}_1=\boldsymbol{r}_\alpha\frac{\mathrm{d}u^\alpha}{\mathrm{d}s},
\end{align*}
\begin{align}
\frac{\mathrm{d}\boldsymbol{e}_1}{\mathrm{d}s}=\left(\frac{\mathrm{d}^2u^\gamma}{\mathrm{d}s^2}+\Gamma_{\alpha\beta}^\gamma\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\right)\boldsymbol{r}_\gamma+b_{\alpha\beta}\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\boldsymbol{n},\label{eq:de1ds}
\end{align}
其中$\Gamma_{\alpha\beta}^\gamma$为克里斯托费尔符号，$b_{\alpha\beta}$为第二基本形式系数。

代入测地曲率定义：
\begin{align*}
\kappa_g=\frac{\mathrm{d}\boldsymbol{e}_1}{\mathrm{d}s}\cdot\boldsymbol{e}_2=\left(\frac{\mathrm{d}^2u^\gamma}{\mathrm{d}s^2}+\Gamma_{\alpha\beta}^\gamma\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\right)\boldsymbol{r}_\gamma\cdot\boldsymbol{e}_2.
\end{align*}
计算得
\begin{align}
\kappa_g=\sqrt{g_{11}g_{22}-(g_{12})^2}
\begin{vmatrix}
\frac{\mathrm{d}u^1}{\mathrm{d}s}&\frac{\mathrm{d}^2u^1}{\mathrm{d}s^2}+\Gamma_{\alpha\beta}^1\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\\
\frac{\mathrm{d}u^2}{\mathrm{d}s}&\frac{\mathrm{d}^2u^2}{\mathrm{d}s^2}+\Gamma_{\alpha\beta}^2\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}
\end{vmatrix},\label{eq:kgintrinsic}
\end{align}
该式仅依赖曲面\textbf{第一基本量}、克里斯托费尔符号及曲线参数方程，与第二基本形式无关，故保长变形下不变。

\end{proof}

\begin{theorem}
设$(u,v)$为曲面$S$的\textbf{正交参数系}，第一基本形式为$\mathrm{I}=E(\mathrm{d}u)^2+G(\mathrm{d}v)^2$，$C:u=u(s),v=v(s)$为$S$上以弧长$s$为参数的曲线，$\theta$为$C$与$u$-曲线的夹角，则$C$的测地曲率满足
\begin{align}
\kappa_g=\frac{\mathrm{d}\theta}{\mathrm{d}s}-\frac{1}{2\sqrt{G}}\frac{\partial\ln E}{\partial v}\cos\theta+\frac{1}{2\sqrt{E}}\frac{\partial\ln G}{\partial u}\sin\theta.\label{eq:liouville}
\end{align}
\end{theorem}
\begin{proof}
记$u$-曲线、$v$-曲线的单位切向量为
\begin{align*}
\boldsymbol{\alpha}_1=\frac{1}{\sqrt{E}}\boldsymbol{r}_u,\quad \boldsymbol{\alpha}_2=\frac{1}{\sqrt{G}}\boldsymbol{r}_v,
\end{align*}
则$C$的单位切向量$\boldsymbol{e}_1=\cos\theta\boldsymbol{\alpha}_1+\sin\theta\boldsymbol{\alpha}_2$，且
\begin{align*}
\cos\theta=\sqrt{E}\frac{\mathrm{d}u}{\mathrm{d}s},\quad \sin\theta=\sqrt{G}\frac{\mathrm{d}v}{\mathrm{d}s}.
\end{align*}
由$\boldsymbol{e}_2=\boldsymbol{n}\times\boldsymbol{e}_1$，得$\boldsymbol{e}_2=-\sin\theta\boldsymbol{\alpha}_1+\cos\theta\boldsymbol{\alpha}_2$。

对$\boldsymbol{e}_1$求导：
\begin{align*}
\frac{\mathrm{d}^2\boldsymbol{r}}{\mathrm{d}s^2}=\frac{\mathrm{d}\boldsymbol{e}_1}{\mathrm{d}s}=\frac{\mathrm{d}\theta}{\mathrm{d}s}(-\sin\theta\boldsymbol{\alpha}_1+\cos\theta\boldsymbol{\alpha}_2)+\cos\theta\frac{\mathrm{d}\boldsymbol{\alpha}_1}{\mathrm{d}s}+\sin\theta\frac{\mathrm{d}\boldsymbol{\alpha}_2}{\mathrm{d}s}.
\end{align*}
代入测地曲率定义$\kappa_g=\frac{\mathrm{d}^2\boldsymbol{r}}{\mathrm{d}s^2}\cdot\boldsymbol{e}_2$，化简得
\begin{align*}
\kappa_g=\frac{\mathrm{d}\theta}{\mathrm{d}s}+\frac{\mathrm{d}\boldsymbol{\alpha}_1}{\mathrm{d}s}\cdot\boldsymbol{\alpha}_2.
\end{align*}
计算$\frac{\mathrm{d}\boldsymbol{\alpha}_1}{\mathrm{d}s}\cdot\boldsymbol{\alpha}_2$：
\begin{align*}
\frac{\mathrm{d}\boldsymbol{\alpha}_1}{\mathrm{d}s}\cdot\boldsymbol{\alpha}_2=-\frac{1}{2\sqrt{G}}\frac{\partial\ln E}{\partial v}\cos\theta+\frac{1}{2\sqrt{E}}\frac{\partial\ln G}{\partial u}\sin\theta,
\end{align*}
代入即得\eqref{eq:liouville}。

\end{proof}

\begin{remark}
由Liouville公式可得正交参数系下坐标曲线的测地曲率：
\begin{enumerate}[(1)]
\item $u$-曲线（$\theta=0$）：$\displaystyle\kappa_{g1}=-\frac{1}{2\sqrt{G}}\frac{\partial\ln E}{\partial v}$；
\item $v$-曲线（$\theta=\frac{\pi}{2}$）：$\displaystyle\kappa_{g2}=\frac{1}{2\sqrt{E}}\frac{\partial\ln G}{\partial u}$。
\end{enumerate}
此时Liouville公式可改写为
\begin{align*}
\kappa_g=\frac{\mathrm{d}\theta}{\mathrm{d}s}+\kappa_{g1}\cos\theta+\kappa_{g2}\sin\theta.
\end{align*}
\end{remark}

\begin{proposition}
曲面$S$上曲线$C$的测地挠率可表示为行列式形式
\begin{align}
\tau_g=\frac{1}{\sqrt{g}}
\begin{vmatrix}
\left(\frac{\mathrm{d}u^2}{\mathrm{d}s}\right)^2&-\frac{\mathrm{d}u^1}{\mathrm{d}s}\frac{\mathrm{d}u^2}{\mathrm{d}s}&\left(\frac{\mathrm{d}u^1}{\mathrm{d}s}\right)^2\\
g_{11}&g_{12}&g_{22}\\
b_{11}&b_{12}&b_{22}
\end{vmatrix},\label{eq:taugdet}
\end{align}
其中$g=\det(g_{\alpha\beta})$，$g_{\alpha\beta},b_{\alpha\beta}$分别为第一、第二基本形式系数。
\end{proposition}
\begin{proof}
由$\frac{\mathrm{d}\boldsymbol{n}}{\mathrm{d}s}=\frac{\partial\boldsymbol{n}}{\partial u^\alpha}\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}=-b_\alpha^\beta\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\boldsymbol{r}_\beta$，代入测地挠率定义$\tau_g=(\boldsymbol{n},\boldsymbol{n}',\boldsymbol{r}')$，结合第一、第二基本形式系数关系化简即得。

\end{proof}

\begin{remark}
\begin{enumerate}[(1)]
\item 测地挠率仅依赖曲面切方向，\textbf{不是内蕴几何量}；
\item 曲面的\textbf{主方向}等价于测地挠率为$0$的切方向，故\textbf{曲率线}是沿切方向测地挠率为$0$的曲线；
\item 有关系式$\displaystyle\tau_g=\frac{1}{2}\frac{\mathrm{d}\kappa_n(\theta)}{\mathrm{d}\theta}$，即法曲率取极值的方向为主方向。
\end{enumerate}
\end{remark}

\begin{theorem}
曲面$S$上\textbf{非直线的渐近曲线}$C$的挠率，等于$S$沿$C$切方向的测地挠率。
\end{theorem}
\begin{proof}
$C$为非直线渐近曲线，故空间曲率$\kappa\neq0$，法曲率$\kappa_n\equiv0$。代入标架运动公式\eqref{eq:moveformula}得
\begin{align*}
\begin{cases}
\displaystyle\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}s}=\boldsymbol{e}_1,\\
\displaystyle\frac{\mathrm{d}\boldsymbol{e}_1}{\mathrm{d}s}=|\kappa_g|(\varepsilon\boldsymbol{e}_2),\quad \varepsilon=\mathrm{sign}\,\kappa_g,\\
\displaystyle\frac{\mathrm{d}(\varepsilon\boldsymbol{e}_2)}{\mathrm{d}s}=-|\kappa_g|\boldsymbol{e}_1+\tau_g(\varepsilon\boldsymbol{e}_3),\\
\displaystyle\frac{\mathrm{d}(\varepsilon\boldsymbol{e}_3)}{\mathrm{d}s}=-\tau_g(\varepsilon\boldsymbol{e}_2).
\end{cases}
\end{align*}
又$\kappa^2=\kappa_g^2+\kappa_n^2=\kappa_g^2\neq0$，故$\{\boldsymbol{r};\boldsymbol{e}_1,\varepsilon\boldsymbol{e}_2,\varepsilon\boldsymbol{e}_3\}$为$C$的Frenet标架，因此$C$的挠率为$\tau_g$。

\end{proof}

\begin{example}
证明: 旋转曲面上的纬线的测地曲率是常数, 该常数的倒数恰好等于, 过纬线上一点的经线的切线从切点到该切线与旋转轴的交点之间的长度.
\end{example}

\begin{example}
证明: 在球面$S$
\begin{align*}
\bm{r}=(a\cos u\cos v,a\cos u\sin v,a\sin u),\quad -\frac{\pi}{2}<u<\frac{\pi}{2},\ 0<v<2\pi
\end{align*}
上, 曲线$C$的测地曲率可以表示成
\begin{align*}
\kappa_g=\frac{\mathrm{d}\theta(s)}{\mathrm{d}s}-\sin(u(s))\frac{\mathrm{d}v(s)}{\mathrm{d}s},
\end{align*}
其中$(u(s),v(s))$是球面$S$上的曲线$C$的参数方程, $s$是曲线$C$的弧长参数, $\theta(s)$是曲线$C$与球面上的经线 (即$u$-曲线) 之间的夹角.
\end{example}

\begin{example}
证明: 在曲面$S$的任意的参数系$(u,v)$下, 曲线$C:u=u(s),v=v(s)$的测地曲率是
\begin{align*}
\kappa_g=\sqrt{g}\big(Bu'(s)-Av'(s)+u'(s)v''(s)-v'(s)u''(s)\big),
\end{align*}
其中$s$是曲线$C$的弧长参数, $g=EG-F^2$, 并且
\begin{align*}
A&=\Gamma_{11}^1(u'(s))^2+2\Gamma_{12}^1u'(s)v'(s)+\Gamma_{22}^1(v'(s))^2,\\
B&=\Gamma_{11}^2(u'(s))^2+2\Gamma_{12}^2u'(s)v'(s)+\Gamma_{22}^2(v'(s))^2.
\end{align*}
特别是, 参数曲线的测地曲率是
\begin{align*}
\kappa_{g1}=\sqrt{g}\,\Gamma_{11}^2(u'(s))^3,\quad \kappa_{g2}=-\sqrt{g}\,\Gamma_{22}^1(v'(s))^3.
\end{align*}
\end{example}

\begin{example}
假定$\Phi$是曲面$S$上由保长对应构成的一个变换群, 并且它的每一个成员在曲面上的作用, 是把曲面$S$上一条给定的曲线$C$变到它自身. 证明: 如果变换群$\Phi$限制在曲线$C$上的作用是传递的, 即对于曲线$C$上的任意两点$p,q$, 必有$\Phi$中的一个成员$\sigma$, 它限制在$C$上的作用是把点$p$映到点$q$, 则曲线$C$的测地曲率是常数.
\end{example}

\begin{example}
设$\bm{e}_1,\bm{e}_2$是曲面$S$在点$p$的两个彼此正交的主方向, 其对应的主曲率分别是$\kappa_1,\kappa_2$. 证明: 曲面$S$在点$p$与$\bm{e}_1$成$\theta$角的切方向的测地挠率是
\begin{align*}
\tau_g=\frac{1}{2}(\kappa_2-\kappa_1)\sin2\theta=\frac{1}{2}\frac{\mathrm{d}\kappa_n(\theta)}{\mathrm{d}\theta}.
\end{align*}
\end{example}

\begin{example}
假定曲面$S$上经过一个双曲点$p$的两条渐近曲线在该点的曲率不为零. 证明: 这两条曲线在该点的挠率的绝对值相等、符号相反, 并且这两个挠率之积等于曲面$S$在点$p$的Gauss曲率$K$.
\end{example}

\begin{example}
证明: 曲面$S$上的曲线$C$的法曲率$\kappa_n$和测地挠率$\tau_g$满足关系式
\begin{align*}
\kappa_n^2+\tau_g^2-2H\kappa_n+K=0,
\end{align*}
其中$H$和$K$分别是曲面$S$的平均曲率和Gauss曲率.
\end{example}
