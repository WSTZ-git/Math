\section{法曲率}

\begin{theorem}
设 $S$ 为正则参数曲面，$P$ 为 $S$ 上一点，$\boldsymbol{n}$ 为 $S$ 在 $P$ 处的单位法向量。
给定 $P$ 处一个单位切向量 $\boldsymbol{v}$，记 $\kappa_n = \mathrm{I\!I}(\boldsymbol{v},\boldsymbol{v})$ 为 $S$ 沿 $\boldsymbol{v}$ 方向的法曲率，且假设 $\kappa_n \neq 0$。
考虑所有经过 $P$ 且以 $\boldsymbol{v}$ 为切向量的正则曲线（弧长参数化使得 $\boldsymbol{\alpha}(P)=\boldsymbol{v}$）。
对于每条这样的曲线 $C$，设 $\kappa$ 为其在 $P$ 处的曲率（$\kappa>0$），$\boldsymbol{\beta}$ 为主法向量（指向曲率中心），则曲率中心为
\[
\boldsymbol{c} = P + \frac{1}{\kappa}\boldsymbol{\beta}.
\]
则所有这样的曲率中心 $\boldsymbol{c}$（包括极限点 $P$）位于过 $P$ 且垂直于 $\boldsymbol{v}$ 的平面 $\Pi = \{\boldsymbol{w}\in\mathbb{R}^3: \boldsymbol{w}\cdot\boldsymbol{v}=0\}$ 上，且满足方程
\[
\left|\boldsymbol{c} - \left(P + \frac{1}{2\kappa_n}\boldsymbol{n}\right)\right| = \frac{1}{2|\kappa_n|},
\]
即它们构成该平面内以 $P + \frac{1}{2\kappa_n}\boldsymbol{n}$ 为圆心、$\frac{1}{2|\kappa_n|}$ 为半径的圆。该圆经过 $P$ 点，直径为 $\frac{1}{|\kappa_n|}$。
\end{theorem}
\begin{proof}
由于 $\boldsymbol{v}$ 是切向量，$\boldsymbol{n}\perp\boldsymbol{v}$，故 $\boldsymbol{n}\in\Pi$。
取 $\boldsymbol{b} = \boldsymbol{v}\times\boldsymbol{n}$，则 $\{\boldsymbol{n},\boldsymbol{b}\}$ 构成 $\Pi$ 的一组标准正交基。
对于任意以 $\boldsymbol{v}$ 为切向量的曲线 $C$，其主法向量 $\boldsymbol{\beta}$ 满足 $\boldsymbol{\beta}\perp\boldsymbol{v}$，因此 $\boldsymbol{\beta}\in\Pi$，可表示为
\[
\boldsymbol{\beta} = \cos\theta\,\boldsymbol{n} + \sin\theta\,\boldsymbol{b},
\]
其中 $\theta = \angle(\boldsymbol{\beta},\boldsymbol{n})\in[0,\pi]$。
由法曲率公式 $\kappa_n = \kappa\cos\theta$（$\kappa$ 为曲线曲率）得 $\cos\theta = \kappa_n/\kappa$，因此 $\cos\theta$ 与 $\kappa_n$ 同号且 $|\cos\theta|\leqslant 1$。
曲率中心为
\[
\boldsymbol{c} = P + \frac{1}{\kappa}\boldsymbol{\beta}
= P + \frac{1}{\kappa_n}\cos\theta\,\boldsymbol{\beta}
= P + \frac{1}{\kappa_n}\left(\cos^2\theta\,\boldsymbol{n} + \cos\theta\sin\theta\,\boldsymbol{b}\right).
\]
记 $\boldsymbol{c}-P$ 在基 $\{\boldsymbol{n},\boldsymbol{b}\}$ 下的坐标为 $(x,y)$，则
\[
x = \frac{\cos^2\theta}{\kappa_n},\qquad y = \frac{\cos\theta\sin\theta}{\kappa_n}.
\]
计算得
\[
x^2+y^2 = \frac{\cos^4\theta+\cos^2\theta\sin^2\theta}{\kappa_n^2} = \frac{\cos^2\theta}{\kappa_n^2},
\]
且 $x = \kappa_n(x^2+y^2)$，即
\[
\kappa_n(x^2+y^2) - x = 0.
\]
配方得
\[
\left(x - \frac{1}{2\kappa_n}\right)^2 + y^2 = \left(\frac{1}{2\kappa_n}\right)^2.
\]
因此
\[
\left|\boldsymbol{c} - \left(P + \frac{1}{2\kappa_n}\boldsymbol{n}\right)\right| = \frac{1}{2|\kappa_n|}.
\]
当 $\theta$ 变化时，$\cos\theta$ 可取遍 $(0,1]$（若 $\kappa_n>0$）或 $[-1,0)$（若 $\kappa_n<0$），对应的 $(x,y)$ 覆盖整个圆（$\theta\to\pi/2$ 时得到原点 $P$）。
故所有曲率中心（连同极限点 $P$）位于该圆上。
\end{proof}

\begin{definition}
设正则参数曲面$S:\boldsymbol{r}=\boldsymbol{r}(u,v)$，$\mathrm{I},\mathrm{II}$分别为其第一基本形式、第二基本形式，则
\begin{align}
\kappa_n=\frac{\mathrm{II}}{\mathrm{I}}=\frac{L(\mathrm{d}u)^2+2M\mathrm{d}u\mathrm{d}v+N(\mathrm{d}v)^2}{E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2}\label{eq:normal_curvature_def}
\end{align}
称为曲面$S$在点$(u,v)$处沿切方向$(\mathrm{d}u,\mathrm{d}v)$的\textbf{法曲率}。
\end{definition}

\begin{theorem}
曲面$S$在点$(u,v)$处沿切方向$(\mathrm{d}u,\mathrm{d}v)$的法曲率$\kappa_n$，等于以该切方向确定的法截线作为对应有向法截面上平面曲线的相对曲率$\kappa_r$。
\end{theorem}
\begin{proof}
法截线以曲面法向量$\boldsymbol{n}$为法向量，即其主法向量$\boldsymbol{\beta}=\pm\boldsymbol{n}$，故$\theta=0$或$\pi$。代入\eqref{eq:normal_curvature_angle}得$\kappa=|\kappa_n|$，即法曲率等于法截线的相对曲率。

\end{proof}

\begin{example}
求平面、圆柱面和球面的法曲率。
\end{example}
\begin{solution}
平面的第二基本形式满足$\mathrm{II}=0$，因此平面上任意点沿任意切方向的法曲率$\kappa_n=0$，其法截线为直线。

圆柱面$\boldsymbol{r}=\left(a\cos\frac{u}{a},a\sin\frac{u}{a},v\right)$的第一、第二基本形式为
\begin{align*}
\mathrm{I}=(\mathrm{d}u)^2+(\mathrm{d}v)^2,\quad \mathrm{II}=-\frac{1}{a}(\mathrm{d}u)^2,
\end{align*}
代入\eqref{eq:normal_curvature_def}得
\begin{align*}
\kappa_n=-\frac{(\mathrm{d}u)^2}{a\left((\mathrm{d}u)^2+(\mathrm{d}v)^2\right)}.
\end{align*}
设切方向$(\mathrm{d}u,\mathrm{d}v)$与$u$-曲线切方向的夹角为$\theta$，则$\cos\theta=\frac{\mathrm{d}u}{\sqrt{(\mathrm{d}u)^2+(\mathrm{d}v)^2}}$，因此
\begin{align}
\kappa_n=-\frac{1}{a}\cos^2\theta.\label{eq:cylinder_normal_curvature}
\end{align}
$\theta=0$时$\kappa_n$取最小值$-\frac{1}{a}$，$\theta=\frac{\pi}{2}$时$\kappa_n$取最大值$0$。负号表示法截线向$-\boldsymbol{n}$方向弯曲。

球面的法曲率为常数$\kappa_n=-\frac{1}{R}$，球面法截线为半径为$R$的大圆，故球面上曲线的曲率恒不为零。

\end{solution}

\begin{theorem}
正则参数曲面在任意固定点，其法曲率必在两个彼此正交的切方向上分别取最大值和最小值。
\end{theorem}
\begin{proof}
取曲面的正交参数系，即$F=0$，此时$\mathrm{I}=E(\mathrm{d}u)^2+G(\mathrm{d}v)^2$，$\mathrm{II}=L(\mathrm{d}u)^2+2M\mathrm{d}u\mathrm{d}v+N(\mathrm{d}v)^2$。设切方向$(\mathrm{d}u,\mathrm{d}v)$与$u$-曲线切方向的夹角为$\theta$，则
$$\cos\theta=\frac{\sqrt{E}\mathrm{d}u}{\sqrt{E(\mathrm{d}u)^2+G(\mathrm{d}v)^2}},\quad \sin\theta=\frac{\sqrt{G}\mathrm{d}v}{\sqrt{E(\mathrm{d}u)^2+G(\mathrm{d}v)^2}},$$
代入\eqref{eq:normal_curvature_def}得
\begin{align*}
\kappa_n(\theta)=\frac{L}{E}\cos^2\theta+\frac{2M}{\sqrt{EG}}\cos\theta\sin\theta+\frac{N}{G}\sin^2\theta.
\end{align*}
利用三角恒等式化简得
\begin{align*}
\kappa_n(\theta)=\frac{1}{2}\left(\frac{L}{E}+\frac{N}{G}\right)+\frac{1}{2}\left(\frac{L}{E}-\frac{N}{G}\right)\cos2\theta+\frac{M}{\sqrt{EG}}\sin2\theta.
\end{align*}
令
\begin{align}
A=\sqrt{\left(\frac{1}{2}\left(\frac{L}{E}-\frac{N}{G}\right)\right)^2+\left(\frac{M}{\sqrt{EG}}\right)^2},\label{eq:A_def}
\end{align}
若$A\neq0$，引入角$\theta_0$满足
\begin{align}
\cos2\theta_0=\frac{1}{2A}\left(\frac{L}{E}-\frac{N}{G}\right),\quad \sin2\theta_0=\frac{M}{A\sqrt{EG}},\label{eq:theta0_def}
\end{align}
则
\begin{align*}
\kappa_n(\theta)=\frac{1}{2}\left(\frac{L}{E}+\frac{N}{G}\right)+A\cos2(\theta-\theta_0).
\end{align*}

法曲率的最大值、最小值分别为
\begin{align}
\kappa_1&=\frac{1}{2}\left(\frac{L}{E}+\frac{N}{G}\right)+\sqrt{\left(\frac{1}{2}\left(\frac{L}{E}-\frac{N}{G}\right)\right)^2+\left(\frac{M}{\sqrt{EG}}\right)^2},\label{eq:max_principal_curvature}\\
\kappa_2&=\frac{1}{2}\left(\frac{L}{E}+\frac{N}{G}\right)-\sqrt{\left(\frac{1}{2}\left(\frac{L}{E}-\frac{N}{G}\right)\right)^2+\left(\frac{M}{\sqrt{EG}}\right)^2}.\label{eq:min_principal_curvature}
\end{align}
若$A=0$，则$\kappa_n(\theta)=\frac{1}{2}\left(\frac{L}{E}+\frac{N}{G}\right)$，法曲率与切方向无关。
\end{proof}

\begin{definition}
正则参数曲面在任意固定点，法曲率取最大值、最小值的方向称为曲面在该点的\textbf{主方向}，对应的法曲率称为曲面在该点的\textbf{主曲率}。
\end{definition}

\begin{proposition}
正交参数曲线网下，主方向与$u$-曲线的夹角为$\theta_0$和$\theta_0+\frac{\pi}{2}$，其中$\theta_0$由\eqref{eq:theta0_def}给出；主曲率由\eqref{eq:max_principal_curvature}、\eqref{eq:min_principal_curvature}给出。沿方向角为$\theta$的切方向的法曲率满足\textbf{Euler公式}:
\begin{align}
\kappa_n(\theta)=\kappa_1\cos^2(\theta-\theta_0)+\kappa_2\sin^2(\theta-\theta_0).\label{eq:euler_formula112121}
\end{align}
\end{proposition}

\begin{definition}
曲面上一点处法曲率为零的切方向称为曲面在该点的\textbf{渐近方向}；曲面上每一点的切方向均为渐近方向的曲线称为曲面的\textbf{渐近曲线}。
\end{definition}

固定点$(u,v)$处，渐近方向$(\mathrm{d}u,\mathrm{d}v)$满足方程
\begin{align}
L(\mathrm{d}u)^2+2M\mathrm{d}u\mathrm{d}v+N(\mathrm{d}v)^2=0.\label{eq:asymptotic_dir_eq}
\end{align}
曲面在该点存在实渐近方向的充分必要条件为$LN-M^2\leqslant0$。
\begin{enumerate}[(1)]
\item 当$LN-M^2<0$时，存在两个不同的实渐近方向：
$$\frac{\mathrm{d}u}{\mathrm{d}v}=\frac{-M\pm\sqrt{M^2-LN}}{L}=\frac{N}{-M\mp\sqrt{M^2-LN}}.$$
\item 当$LN-M^2=0$时，存在唯一实渐近方向：
$$\frac{\mathrm{d}u}{\mathrm{d}v}=-\frac{M}{L}=-\frac{N}{M}.$$
\end{enumerate}

\begin{theorem}
曲面上的参数曲线网是渐近曲线网的充分必要条件是$L=N=0$。
\end{theorem}
\begin{proof}
{\heiti 必要性：}若$u$-曲线、$v$-曲线均为渐近曲线，则$(\mathrm{d}u,\mathrm{d}v)=(1,0),(0,1)$均为渐近方向，代入\eqref{eq:asymptotic_dir_eq}得$L=N=0$。

{\heiti 充分性：}若$L=N=0$，则\eqref{eq:asymptotic_dir_eq}化为$2M\mathrm{d}u\mathrm{d}v=0$，其解为$\mathrm{d}v=0$和$\mathrm{d}u=0$，即$u$-曲线、$v$-曲线为渐近曲线。

\end{proof}

\begin{theorem}
曲面上的一条曲线是渐近曲线，当且仅当该曲线为直线，或其密切平面恰好是曲面的切平面。
\end{theorem}
\begin{proof}
由\eqref{eq:euler_formula112121}，$\kappa_n=0$等价于$\kappa=0$或$\cos\theta=0$。
若$\kappa=0$，则曲线为直线；若$\cos\theta=0$，则$\boldsymbol{\beta}\perp\boldsymbol{n}$，即曲线的密切平面与曲面相切。

\end{proof}
