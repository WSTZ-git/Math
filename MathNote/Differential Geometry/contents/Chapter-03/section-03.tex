\section{Weingarten映射和主曲率}

\begin{definition}
设正则参数曲面$S:\boldsymbol{r}=\boldsymbol{r}(u,v)$，$\boldsymbol{n}(u,v)$为其单位法向量。定义可微映射$g:S\to\Sigma$（$\Sigma$为$\mathbb{E}^3$中单位球面）满足
\begin{align}
g(\boldsymbol{r}(u,v))=\boldsymbol{n}(u,v),\label{eq:gauss_map}
\end{align}
称$g$为\textbf{Gauss映射}。
\end{definition}

\begin{figure}[H]
\centering
\includegraphics[scale=0.3]{Gauss映射.png}
\caption{Gauss映射}
\label{figure:gauss_map}
\end{figure}

\begin{definition}
Gauss映射$g:S\to\Sigma$诱导切映射$g_*:T_pS\to T_{g(p)}\Sigma$。由于曲面$S$与球面$\Sigma$在对应点的切平面平行，$T_pS$与$T_{g(p)}\Sigma$可自然等同。令
\begin{align}
W=-g_*:T_pS\to T_pS,\label{eq:weingarten_map}
\end{align}
称$W$为曲面$S$在点$p$的\textbf{Weingarten映射}，且满足$g_*(\boldsymbol{r}_u)=\boldsymbol{n}_u,\ g_*(\boldsymbol{r}_v)=\boldsymbol{n}_v$。
\end{definition}

\begin{theorem}
曲面$S$的第二基本形式$\mathrm{II}$可由Weingarten映射表示为
\begin{align}
\mathrm{II}=W(\mathrm{d}\boldsymbol{r})\cdot\mathrm{d}\boldsymbol{r}.\label{eq:ii_weingarten}
\end{align}
\end{theorem}
\begin{proof}
曲面任意切向量可表示为$\mathrm{d}\boldsymbol{r}=\boldsymbol{r}_u\mathrm{d}u+\boldsymbol{r}_v\mathrm{d}v$，由Weingarten映射定义
\begin{align*}
W(\mathrm{d}\boldsymbol{r})&=-g_*(\boldsymbol{r}_u\mathrm{d}u+\boldsymbol{r}_v\mathrm{d}v)\\
&=-(\boldsymbol{n}_u\mathrm{d}u+\boldsymbol{n}_v\mathrm{d}v)=-\mathrm{d}\boldsymbol{n}.
\end{align*}
因此$\mathrm{II}=-\mathrm{d}\boldsymbol{n}\cdot\mathrm{d}\boldsymbol{r}=W(\mathrm{d}\boldsymbol{r})\cdot\mathrm{d}\boldsymbol{r}.$

\end{proof}

\begin{theorem}
Weingarten映射$W:T_pS\to T_pS$为自共轭映射，即对曲面$S$在点$(u,v)$的任意两个切方向$\mathrm{d}\boldsymbol{r},\delta\boldsymbol{r}$，有
\begin{align}
W(\mathrm{d}\boldsymbol{r})\cdot\delta\boldsymbol{r}=\mathrm{d}\boldsymbol{r}\cdot W(\delta\boldsymbol{r}).\label{eq:self_conjugate}
\end{align}
\end{theorem}
\begin{proof}
设$\mathrm{d}\boldsymbol{r}=\boldsymbol{r}_u\mathrm{d}u+\boldsymbol{r}_v\mathrm{d}v,\ \delta\boldsymbol{r}=\boldsymbol{r}_u\delta u+\boldsymbol{r}_v\delta v$，则
\begin{align*}
W(\mathrm{d}\boldsymbol{r})&=-(\boldsymbol{n}_u\mathrm{d}u+\boldsymbol{n}_v\mathrm{d}v),\quad W(\delta\boldsymbol{r})=-(\boldsymbol{n}_u\delta u+\boldsymbol{n}_v\delta v).
\end{align*}
代入计算得
\begin{align*}
W(\mathrm{d}\boldsymbol{r})\cdot\delta\boldsymbol{r}&=-(\boldsymbol{n}_u\mathrm{d}u+\boldsymbol{n}_v\mathrm{d}v)\cdot(\boldsymbol{r}_u\delta u+\boldsymbol{r}_v\delta v)\\
&=L\mathrm{d}u\delta u+M(\mathrm{d}u\delta v+\mathrm{d}v\delta u)+N\mathrm{d}v\delta v\\
&=-(\boldsymbol{r}_u\mathrm{d}u+\boldsymbol{r}_v\mathrm{d}v)\cdot(\boldsymbol{n}_u\delta u+\boldsymbol{n}_v\delta v)\\
&=\mathrm{d}\boldsymbol{r}\cdot W(\delta\boldsymbol{r}).
\end{align*}

\end{proof}

\begin{definition}
若非零切向量$\mathrm{d}\boldsymbol{r}$与实数$\lambda$满足
\begin{align}
W(\mathrm{d}\boldsymbol{r})=\lambda\mathrm{d}\boldsymbol{r},\label{eq:eigen_weingarten}
\end{align}
则称$\lambda$为$W$的\textbf{特征值}，$\mathrm{d}\boldsymbol{r}$为对应\textbf{特征向量}。此时曲面沿$\mathrm{d}\boldsymbol{r}$的法曲率满足
\begin{align}
\kappa_n=\frac{\mathrm{II}}{\mathrm{I}}=\frac{W(\mathrm{d}\boldsymbol{r})\cdot\mathrm{d}\boldsymbol{r}}{\mathrm{d}\boldsymbol{r}\cdot\mathrm{d}\boldsymbol{r}}=\lambda.\label{eq:kn_eigenvalue}
\end{align}
\end{definition}

\begin{theorem}
正则参数曲面在每一点的Weingarten映射的两个特征值恰好是该点的\textbf{主曲率}，对应的特征方向是曲面的\textbf{主方向}。
\end{theorem}
\begin{proof}
取切空间$T_pS$的单位正交特征基底$\{\boldsymbol{e}_1,\boldsymbol{e}_2\}$，满足$W(\boldsymbol{e}_1)=\lambda_1\boldsymbol{e}_1,\ W(\boldsymbol{e}_2)=\lambda_2\boldsymbol{e}_2$，其中$\lambda_1\geqslant\lambda_2$。
对任意切向量$\boldsymbol{e}=\cos\theta\boldsymbol{e}_1+\sin\theta\boldsymbol{e}_2$，有
\begin{align*}
W(\boldsymbol{e})=\lambda_1\cos\theta\boldsymbol{e}_1+\lambda_2\sin\theta\boldsymbol{e}_2,
\end{align*}
则法曲率
\begin{align*}
\kappa_n(\theta)&=\frac{W(\boldsymbol{e})\cdot\boldsymbol{e}}{\boldsymbol{e}\cdot\boldsymbol{e}}\\
&=\lambda_1\cos^2\theta+\lambda_2\sin^2\theta.
\end{align*}
当$\theta=0$时$\kappa_n$取最大值$\lambda_1$，$\theta=\frac{\pi}{2}$时取最小值$\lambda_2$，故$\lambda_1,\lambda_2$为主曲率，对应方向为主方向。

\end{proof}

\begin{theorem}[Euler公式]
设$\boldsymbol{e}_1,\boldsymbol{e}_2$为曲面$S$在点$p$处两个彼此正交的主方向单位向量，对应的主曲率为$\kappa_1,\kappa_2$，则曲面沿任意切向量$\boldsymbol{e}=\cos\theta\boldsymbol{e}_1+\sin\theta\boldsymbol{e}_2$的法曲率为
\begin{align}
\kappa_n(\theta)=\kappa_1\cos^2\theta+\kappa_2\sin^2\theta.\label{eq:euler_formula}
\end{align}
\end{theorem}

\begin{definition}
若曲面某点满足$\kappa_1=\kappa_2$，则该点称为\textbf{脐点}。脐点处法曲率与切方向无关，即
\begin{align}
\frac{L}{E}=\frac{M}{F}=\frac{N}{G}.\label{eq:umbilical_point}
\end{align}
其中，比值为零时称脐点为\textbf{平点}，比值非零时称\textbf{圆点}。
\end{definition}

\begin{proposition}
曲面$S$是平面，当且仅当$S$上的点都是平点；曲面$S$是球面，当且仅当$S$上的点都是圆点。
\end{proposition}

\begin{definition}
设$C$是正则曲面$S$上的一条曲线，若$C$在每一点的切向量都是$S$在该点的主方向，则称$C$为$S$上的一条\textbf{曲率线}。
\end{definition}

\begin{theorem}[Rodrigues定理]
曲面$S:\boldsymbol{r}=\boldsymbol{r}(u,v)$上的曲线$C:u=u(t),v=v(t)$是曲率线的充要条件是，曲面沿$C$的法向量场的导数与$C$相切，即
\begin{align}
\frac{\mathrm{d}\boldsymbol{n}(u(t),v(t))}{\mathrm{d}t}\parallel\frac{\mathrm{d}\boldsymbol{r}(u(t),v(t))}{\mathrm{d}t}.\label{eq:rodrigues}
\end{align}
\end{theorem}
\begin{proof}
$C$为曲率线$\iff W\left(\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}\right)=\lambda\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}$，结合$W\left(\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}\right)=-\frac{\mathrm{d}\boldsymbol{n}}{\mathrm{d}t}$，得$-\frac{\mathrm{d}\boldsymbol{n}}{\mathrm{d}t}=\lambda\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}$，即$\frac{\mathrm{d}\boldsymbol{n}}{\mathrm{d}t}\parallel\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}.$

\end{proof}

\begin{theorem}
曲面$S$上的曲线$C$是曲率线的充要条件是，$S$沿$C$的法线构成一个可展曲面。
\end{theorem}
\begin{proof}
设$C$的弧长参数表示为$\boldsymbol{r}=\boldsymbol{r}(s)$，沿$C$的法线构成的直纹面为$\boldsymbol{r}=\boldsymbol{r}(s)+t\boldsymbol{n}(s)$。
直纹面可展的充要条件为$(\boldsymbol{r}'(s),\boldsymbol{n}(s),\boldsymbol{n}'(s))=0$。
由$\boldsymbol{r}'(s)\cdot\boldsymbol{n}(s)=0,\ \boldsymbol{n}'(s)\cdot\boldsymbol{n}(s)=0$，得$\boldsymbol{n}(s)\parallel\boldsymbol{r}'(s)\times\boldsymbol{n}'(s)$，设$\boldsymbol{r}'(s)\times\boldsymbol{n}'(s)=\lambda\boldsymbol{n}(s)$，代入可展条件得$\lambda=0$，即$\boldsymbol{r}'(s)\parallel\boldsymbol{n}'(s)$。
由Rodrigues定理，该条件等价于$C$为曲率线。

\end{proof}

\begin{example}
求旋转面上的曲率线。
\end{example}
\begin{solution}
旋转面的经线所在平面包含旋转轴，沿经线的法线均落在该平面内，其构成的直纹面为可展曲面；旋转面的纬线（平行圆）的法线均经过旋转轴上定点，构成锥面，仍为可展曲面。
由曲率线的充要条件，旋转面上的经线和纬线均为曲率线，二者构成正交曲线网。

\end{solution}

\begin{example}
求可展曲面上的曲率线。
\end{example}
\begin{solution}
可展曲面为直纹面，其切平面沿直母线保持不变，即单位法向量沿直母线为常向量场，故法向量沿直母线的导数为零。
由Rodrigues定理，可展曲面的直母线必为曲率线；直母线的正交轨线为另一主方向场的积分曲线，是另一族曲率线。

\end{solution}

\end{example}

\begin{example}
设曲面$S$上的一条曲率线不是渐近曲线,并且它的密切平面与曲面$S$的切平面相交成定角,则该曲线必定是一条平面曲线.
\end{example}

\begin{example}
证明:曲面$S$上任意一点$p$的某个邻域内都有正交参数系$(u,v)$,使得参数曲线在点$p$处的切方向是曲面$S$在该点的两个彼此正交的主方向.
\end{example}

\begin{example}
设在曲面$S$的一个固定点$p$的切方向与一个主方向的夹角为$\theta$,该切方向所对应的法曲率记为$\kappa_n(\theta)$,则
\begin{align*}
\frac{1}{2\pi}\int_{0}^{2\pi}\kappa_n(\theta)\mathrm{d}\theta = H,
\end{align*}
其中$H=\frac{\kappa_1+\kappa_2}{2}$.
\end{example}

\begin{example}
设点$p$是曲面$S$的一个非脐点.如果曲面$S$在点$p$的任意两个夹角为$\theta_0$的切方向上的法曲率之和为常数,则该夹角$\theta_0$必等于$\frac{\pi}{2}$.
\end{example}
