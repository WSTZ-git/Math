\section{外形式}

\begin{definition}
设$V$为$n$维实向量空间，$\{e_1,\dots,e_n\}$为$V$的一组基，则$\forall\boldsymbol{x}\in V$可唯一表示为
\begin{align*}
\boldsymbol{x}=x^1e_1+\cdots+x^ne_n\equiv x^ie_i,
\end{align*}
其中采用\textbf{Einstein求和约定}：指标$i,j,k,l$取值范围为$1\sim n$，重复指标表示求和；$x^1,\dots,x^n$称为$\boldsymbol{x}$在基$\{e_i\}$下的分量。
\end{definition}

\begin{definition}
设函数$f:V\to\mathbb{R}$，若$\forall\boldsymbol{x},\boldsymbol{y}\in V,\alpha\in\mathbb{R}$，满足
\begin{align*}
f(\boldsymbol{x}+\boldsymbol{y})&=f(\boldsymbol{x})+f(\boldsymbol{y}),\\
f(\alpha\boldsymbol{x})&=\alpha f(\boldsymbol{x}),
\end{align*}
则称$f$为$V$上的\textbf{线性函数（1-形式）}。
\end{definition}

\begin{definition}
对基$\{e_i\}$，定义函数$e^j:V\to\mathbb{R}$满足$e^j(\boldsymbol{x})=x^j$（$\boldsymbol{x}=x^ie_i$），称$\{e^1,\dots,e^n\}$为$\{e_i\}$的\textbf{对偶基}。
\end{definition}

\begin{definition}
Kronecker $\delta$记号定义为
\begin{align}
\delta_i^j=
\begin{cases}
1, & i=j,\\
0, & i\neq j,
\end{cases}
\label{eq:kronecker_delta}
\end{align}
满足$e^j(e_i)=\delta_i^j$。
\end{definition}

\begin{proposition}
$V$上全体线性函数构成$n$维向量空间，记为$V^*$，称为$V$的\textbf{对偶空间}；对偶基$\{e^1,\dots,e^n\}$是$V^*$的一组基。
\end{proposition}
\begin{proof}
1. 线性函数的加法、数乘运算封闭，故$V^*$为向量空间。
2. 对任意$f\in V^*$，记$f_i=f(e_i)$，则$\forall\boldsymbol{x}=x^ie_i\in V$，
\begin{align*}
f(\boldsymbol{x})=f(x^ie_i)=x^if(e_i)=f_ix^i=f_ie^i(\boldsymbol{x}),
\end{align*}
即$f=f_ie^i$，任意线性函数可由$\{e^i\}$线性表示。
3. 若$\alpha_ie^i=0$，则$\forall k$，$0=\alpha_ie^i(e_k)=\alpha_i\delta_k^i=\alpha_k$，故$\{e^i\}$线性无关。

综上$\dim V^*=n$，$\{e^i\}$为$V^*$的基。

\end{proof}

\begin{definition}
设$r\in\mathbb{N}^*$，函数$f:\underbrace{V\times\cdots\times V}_{r\text{个}}\to\mathbb{R}$，若对每个自变量均为线性函数，则称$f$为$V$上的\textbf{$r$重线性函数}；全体$r$重线性函数构成向量空间，记为$\bigotimes^r V^*$。
\end{definition}

\begin{definition}
设$f$为$r$重线性函数，$g$为$s$重线性函数，二者的\textbf{张量积}$f\otimes g$定义为
\begin{align}
f\otimes g(\boldsymbol{x}_1,\dots,\boldsymbol{x}_{r+s})=f(\boldsymbol{x}_1,\dots,\boldsymbol{x}_r)g(\boldsymbol{x}_{r+1},\dots,\boldsymbol{x}_{r+s}),
\label{eq:tensor_product}
\end{align}
其中$\boldsymbol{x}_1,\dots,\boldsymbol{x}_{r+s}\in V$。
\end{definition}

\begin{proposition}
张量积满足：
\begin{enumerate}[(1)]
\item 分配律：$(f+g)\otimes h=f\otimes h+g\otimes h,\quad h\otimes(f+g)=h\otimes f+h\otimes g$；
\item 结合律：$(f\otimes g)\otimes h=f\otimes(g\otimes h)=f\otimes g\otimes h$。
\end{enumerate}
\end{proposition}
\begin{proof}
直接代入张量积定义验证即可。

\end{proof}

\begin{proposition}
$\{e^{i_1}\otimes\cdots\otimes e^{i_r}\mid 1\leq i_1,\dots,i_r\leq n\}$是$\bigotimes^r V^*$的一组基，且$\dim\bigotimes^r V^*=n^r$。
\end{proposition}
\begin{proof}
对任意$f\in\bigotimes^r V^*$，$\forall\boldsymbol{x}_1=x_1^{i_1}e_{i_1},\dots,\boldsymbol{x}_r=x_r^{i_r}e_{i_r}\in V$，
\begin{align*}
f(\boldsymbol{x}_1,\dots,\boldsymbol{x}_r)&=f(x_1^{i_1}e_{i_1},\dots,x_r^{i_r}e_{i_r})\\
&=x_1^{i_1}\cdots x_r^{i_r}f(e_{i_1},\dots,e_{i_r})\\
&=f_{i_1\cdots i_r}e^{i_1}\otimes\cdots\otimes e^{i_r}(\boldsymbol{x}_1,\dots,\boldsymbol{x}_r),
\end{align*}
其中$f_{i_1\cdots i_r}=f(e_{i_1},\dots,e_{i_r})$，即$f=f_{i_1\cdots i_r}e^{i_1}\otimes\cdots\otimes e^{i_r}$。
该组函数共$n^r$个且线性无关，故为$\bigotimes^r V^*$的基，$\dim\bigotimes^r V^*=n^r$。

\end{proof}

\begin{definition}
设$f\in\bigotimes^r V^*$，若对任意$1\leq s<t\leq r$及$\boldsymbol{x}_1,\dots,\boldsymbol{x}_r\in V$，满足
\begin{align*}
f(\boldsymbol{x}_1,\dots,\boldsymbol{x}_s,\dots,\boldsymbol{x}_t,\dots,\boldsymbol{x}_r)=-f(\boldsymbol{x}_1,\dots,\boldsymbol{x}_t,\dots,\boldsymbol{x}_s,\dots,\boldsymbol{x}_r),
\end{align*}
则称$f$为\textbf{反对称$r$重线性函数}，或\textbf{$r$次外形式（$r$-形式）}；全体$r$-形式构成向量空间，记为$\bigwedge^r V^*$。
\end{definition}

\begin{proposition}
设$f$为$r$-形式，$\sigma\in\mathfrak{S}_r$（$r$元置换群），则
\begin{align}
f(\boldsymbol{x}_{\sigma(1)},\dots,\boldsymbol{x}_{\sigma(r)})=\mathrm{sign}(\sigma)f(\boldsymbol{x}_1,\dots,\boldsymbol{x}_r),
\label{eq:alternate_permutation}
\end{align}
其中置换符号定义为
\begin{align}
\mathrm{sign}(\sigma)=
\begin{cases}
1, & \sigma\text{为偶置换},\\
-1, & \sigma\text{为奇置换}.
\end{cases}
\label{eq:perm_sign}
\end{align}
\end{proposition}
\begin{proof}
置换可分解为对换的乘积，每次对换使函数变号，代入即得结论。

\end{proof}

\begin{definition}
取$1\leq j_1<\dots<j_r\leq n$，定义
\begin{align}
D^{j_1\cdots j_r}(\boldsymbol{x}_1,\dots,\boldsymbol{x}_r)=
\begin{vmatrix}
x_1^{j_1} & \cdots & x_1^{j_r}\\
\vdots & & \vdots\\
x_r^{j_1} & \cdots & x_r^{j_r}
\end{vmatrix},
\label{eq:det_form}
\end{align}
则$D^{j_1\cdots j_r}$为$r$-形式。
\end{definition}

\begin{definition}
对$f\in\bigotimes^r V^*$，其\textbf{反对称化}$[f]$定义为
\begin{align}
[f](\boldsymbol{x}_1,\dots,\boldsymbol{x}_r)=\frac{1}{r!}\sum\limits_{\sigma\in\mathfrak{S}_r}\mathrm{sign}(\sigma)f(\boldsymbol{x}_{\sigma(1)},\dots,\boldsymbol{x}_{\sigma(r)}).
\label{eq:antisymmetrization}
\end{align}
\end{definition}

\begin{proposition}
$[f]$为$r$-形式；若$f$本身为$r$-形式，则$[f]=f$。
\end{proposition}
\begin{proof}
直接验证反对称性；$r$-形式代入置换性质可证$[f]=f$。

\end{proof}

\begin{definition}
设$f\in\bigwedge^r V^*$，$g\in\bigwedge^s V^*$，二者的\textbf{外积}$f\wedge g$为$(r+s)$-形式，定义为
\begin{align}
f\wedge g=\frac{(r+s)!}{r!s!}[f\otimes g].
\label{eq:wedge_product}
\end{align}
\end{definition}

\begin{example}
设$f,g$为$V$上的1-形式，求$f\wedge g$的求值公式。
\end{example}
\begin{solution}
由外积定义，$f\wedge g=\frac{2!}{1!1!}[f\otimes g]=2[f\otimes g]$。
$\forall\boldsymbol{x},\boldsymbol{y}\in V$，
\begin{align*}
f\wedge g(\boldsymbol{x},\boldsymbol{y})&=2[f\otimes g](\boldsymbol{x},\boldsymbol{y})=f\otimes g(\boldsymbol{x},\boldsymbol{y})-f\otimes g(\boldsymbol{y},\boldsymbol{x})\\
&=f(\boldsymbol{x})g(\boldsymbol{y})-f(\boldsymbol{y})g(\boldsymbol{x})=
\begin{vmatrix}
f(\boldsymbol{x}) & f(\boldsymbol{y})\\
g(\boldsymbol{x}) & g(\boldsymbol{y})
\end{vmatrix}.
\end{align*}

\end{solution}

\begin{example}
设$f,g,h$为$V$上的1-形式，求$f\wedge(g\wedge h)$的求值公式。
\end{example}
\begin{solution}
由外积定义，
\begin{align*}
f\wedge(g\wedge h)=\frac{3!}{1!2!}[f\otimes(g\wedge h)]=6[f\otimes g\otimes h].
\end{align*}
$\forall\boldsymbol{x},\boldsymbol{y},\boldsymbol{z}\in V$，
\begin{align*}
f\wedge(g\wedge h)(\boldsymbol{x},\boldsymbol{y},\boldsymbol{z})&=6[f\otimes g\otimes h](\boldsymbol{x},\boldsymbol{y},\boldsymbol{z})\\
&=
\begin{vmatrix}
f(\boldsymbol{x}) & f(\boldsymbol{y}) & f(\boldsymbol{z})\\
g(\boldsymbol{x}) & g(\boldsymbol{y}) & g(\boldsymbol{z})\\
h(\boldsymbol{x}) & h(\boldsymbol{y}) & h(\boldsymbol{z})
\end{vmatrix}.
\end{align*}

\end{solution}

\begin{theorem}
外积运算满足：
\begin{enumerate}[(1)]
\item 分配律：$(f_1+f_2)\wedge g=f_1\wedge g+f_2\wedge g$；
\item 反交换律：若$f\in\bigwedge^r V^*$，$g\in\bigwedge^s V^*$，则$f\wedge g=(-1)^{rs}g\wedge f$；
\item 结合律：$f\wedge(g\wedge h)=(f\wedge g)\wedge h$。
\end{enumerate}
\end{theorem}
\begin{proof}
(1) 由张量积与反对称化的线性性直接可得。

(2) 对任意$\boldsymbol{x}_1,\dots,\boldsymbol{x}_{r+s}\in V$，
\begin{align*}
f\wedge g(\boldsymbol{x}_1,\dots,\boldsymbol{x}_{r+s})&=\frac{(r+s)!}{r!s!}[f\otimes g](\boldsymbol{x}_1,\dots,\boldsymbol{x}_{r+s})\\
&=\frac{1}{r!s!}\sum\limits_{\sigma\in\mathfrak{S}_{r+s}}\mathrm{sign}(\sigma)f(\boldsymbol{x}_{\sigma(1)},\dots,\boldsymbol{x}_{\sigma(r)})g(\boldsymbol{x}_{\sigma(r+1)},\dots,\boldsymbol{x}_{\sigma(r+s)})\\
&=\frac{1}{r!s!}\mathrm{sign}\begin{pmatrix}s+1 & \cdots & s+r & 1 & \cdots & s\\1 & \cdots & r & r+1 & \cdots & r+s\end{pmatrix}\sum\limits_{\sigma\in\mathfrak{S}_{r+s}}\mathrm{sign}(\sigma)g(\boldsymbol{x}_{\sigma(1)},\dots,\boldsymbol{x}_{\sigma(s)})f(\boldsymbol{x}_{\sigma(s+1)},\dots,\boldsymbol{x}_{\sigma(s+r)})\\
&=(-1)^{rs}g\wedge f(\boldsymbol{x}_1,\dots,\boldsymbol{x}_{r+s}).
\end{align*}

(3) 对$f\in\bigwedge^r V^*,g\in\bigwedge^s V^*,h\in\bigwedge^t V^*$，可验证
\begin{align*}
f\wedge(g\wedge h)=\frac{(r+s+t)!}{r!s!t!}[f\otimes g\otimes h]=(f\wedge g)\wedge h.
\end{align*}

\end{proof}

\begin{corollary}
设$f,g$为1-形式，则$f\wedge g=-g\wedge f$；特别地$f\wedge f=0$。若$f,g$中至少一个为偶次形式，则$f\wedge g=g\wedge f$。
\end{corollary}
\begin{proof}
反交换律中取$r=s=1$，得$f\wedge g=-g\wedge f$；令$g=f$，则$f\wedge f=-f\wedge f$，即$f\wedge f=0$。偶次形式满足$rs$为偶数，$(-1)^{rs}=1$，故交换成立。

\end{proof}

\begin{proposition}
设$f^1,\dots,f^r$为1-形式，则
\begin{align}
f^1\wedge\cdots\wedge f^r=r![f^1\otimes\cdots\otimes f^r],
\label{eq:r_wedge_1form}
\end{align}
且
\begin{align*}
f^1\wedge\cdots\wedge f^r(\boldsymbol{x}_1,\dots,\boldsymbol{x}_r)=
\begin{vmatrix}
f^1(\boldsymbol{x}_1) & \cdots & f^1(\boldsymbol{x}_r)\\
\vdots & & \vdots\\
f^r(\boldsymbol{x}_1) & \cdots & f^r(\boldsymbol{x}_r)
\end{vmatrix}.
\end{align*}
\end{proposition}
\begin{proof}
数学归纳法结合外积结合律可证。

\end{proof}

\begin{proposition}
对偶基的外积满足
\begin{align}
e^{j_1}\wedge\cdots\wedge e^{j_r}(\boldsymbol{x}_1,\dots,\boldsymbol{x}_r)=
\begin{vmatrix}
x_1^{j_1} & \cdots & x_1^{j_r}\\
\vdots & & \vdots\\
x_r^{j_1} & \cdots & x_r^{j_r}
\end{vmatrix}=D^{j_1\cdots j_r}(\boldsymbol{x}_1,\dots,\boldsymbol{x}_r),
\label{eq:basis_wedge_det}
\end{align}
即$D^{j_1\cdots j_r}=e^{j_1}\wedge\cdots\wedge e^{j_r}$。
\end{proposition}
\begin{proof}
直接代入对偶基定义与行列式定义验证。

\end{proof}

\begin{definition}
广义Kronecker $\delta$记号定义为
\begin{align}
\delta_{i_1\cdots i_r}^{j_1\cdots j_r}=e^{j_1}\wedge\cdots\wedge e^{j_r}(e_{i_1},\dots,e_{i_r})=
\begin{vmatrix}
\delta_{i_1}^{j_1} & \cdots & \delta_{i_r}^{j_1}\\
\vdots & & \vdots\\
\delta_{i_1}^{j_r} & \cdots & \delta_{i_r}^{j_r}
\end{vmatrix},
\label{eq:general_kronecker}
\end{align}
即
\begin{align*}
\delta_{i_1\cdots i_r}^{j_1\cdots j_r}=
\begin{cases}
1, & i_1,\dots,i_r\text{互异且}j_1,\dots,j_r\text{为其偶排列},\\
-1, & i_1,\dots,i_r\text{互异且}j_1,\dots,j_r\text{为其奇排列},\\
0, & \text{其他情形}.
\end{cases}
\end{align*}
\end{definition}

\begin{proposition}
对任意$\sigma\in\mathfrak{S}_r$，
\begin{align}
f^{\sigma(1)}\wedge\cdots\wedge f^{\sigma(r)}=\mathrm{sign}(\sigma)f^1\wedge\cdots\wedge f^r=\delta_{1\cdots r}^{\sigma(1)\cdots\sigma(r)}f^1\wedge\cdots\wedge f^r.
\label{eq:wedge_perm}
\end{align}
\end{proposition}
\begin{proof}
由反交换律，交换因子依次变号，置换符号为总符号，结合广义Kronecker $\delta$定义可得。

\end{proof}

\begin{theorem}
设$\{e^1,\dots,e^n\}$是对偶向量空间$V^*$的一个基底，则$\{e^{j_1}\land\cdots\land e^{j_r},\,1\leqslant j_1<\cdots<j_r\leqslant n\}$是$r$次外形式空间$\bigwedge^r V^*$的基底。特别地，$\dim\bigwedge^r V^* = \mathrm{C}_n^r$；当$r>n$时，$\bigwedge^r V^* = \{0\}$。
\end{theorem}

\begin{proof}
任取$f\in\bigwedge^r V^*$，$f$作为$r$重反对称线性函数，可表示为
\begin{align*}
f = f_{j_1\cdots j_r}e^{j_1}\otimes\cdots\otimes e^{j_r},
\end{align*}
其中$f_{j_1\cdots j_r}=f(e_{j_1},\dots,e_{j_r})$，$\{e_i\}$是$V$中与$\{e^i\}$对偶的基底。
由$f$的反对称性，系数满足
\begin{align*}
f_{j_{\sigma(1)}\cdots j_{\sigma(r)}} = \mathrm{sign}(\sigma)f_{j_1\cdots j_r},\quad \forall\sigma\in\mathfrak{S}_r.
\end{align*}
结合外积定义可得
\begin{align}
f &= \frac{1}{r!}f_{j_1\cdots j_r}e^{j_1}\land\cdots\land e^{j_r} \nonumber \\
&= \sum\limits_{1\leqslant j_1<\cdots<j_r\leqslant n} f_{j_1\cdots j_r}e^{j_1}\land\cdots\land e^{j_r}. \label{eq:r-form-basis-expand}
\end{align}
因此任意$r$次外形式均可表示为$\{e^{j_1}\land\cdots\land e^{j_r}\mid 1\leqslant j_1<\cdots<j_r\leqslant n\}$的线性组合。

下证该组外形式线性无关。设存在实数$f_{j_1\cdots j_r}$使得
\begin{align*}
\sum\limits_{1\leqslant j_1<\cdots<j_r\leqslant n} f_{j_1\cdots j_r}e^{j_1}\land\cdots\land e^{j_r}=0,
\end{align*}
将等式两侧作用于向量组$e_{i_1},\dots,e_{i_r}\,(1\leqslant i_1<\cdots<i_r\leqslant n)$，由外积求值性质得
\begin{align*}
0 = \sum\limits_{1\leqslant j_1<\cdots<j_r\leqslant n} f_{j_1\cdots j_r}\delta_{i_1\cdots i_r}^{j_1\cdots j_r}=f_{i_1\cdots i_r},
\end{align*}
即所有系数均为零，故该组外形式线性无关。

综上，$\{e^{j_1}\land\cdots\land e^{j_r},\,1\leqslant j_1<\cdots<j_r\leqslant n\}$为$\bigwedge^r V^*$的基底，维数$\dim\bigwedge^r V^*=\mathrm{C}_n^r$；当$r>n$时无满足条件的严格递增指标，故$\bigwedge^r V^*=\{0\}$。

\end{proof}

\begin{definition}
设$W$为$n$维实向量空间，基底为$\{w_1,\dots,w_n\}$，定义\textbf{外积运算}$\land$满足反交换律、结合律与分配律：
\begin{enumerate}[(1)]
\item 反交换律：$w_i\land w_j = -w_j\land w_i,\ \forall\,1\leqslant i,j\leqslant n$；
\item 结合律：$(w_i\land w_j)\land w_k = w_i\land(w_j\land w_k)$；
\item 分配律：$w_i\land(w_j+w_k)=w_i\land w_j+w_i\land w_k$，$(w_j+w_k)\land w_i=w_j\land w_i+w_k\land w_i$。
\end{enumerate}
由反交换律可得$w_i\land w_i=0$，因此外积中同一向量不可重复出现。

对任意$k\in\mathbb{N}$，定义\textbf{$k$次外多项式空间}$\bigwedge^k W$为$\{w_{i_1}\land\cdots\land w_{i_k}\mid 1\leqslant i_1<\cdots<i_k\leqslant n\}$的实线性组合构成的向量空间；零次外多项式空间$\bigwedge^0 W=\mathbb{R}$。

向量空间$W$上的\textbf{外代数}定义为分次直和
\begin{align*}
\bigwedge(W) = \sum\limits_{k=0}^n \bigwedge^k W,
\end{align*}
其元素为各次外多项式的形式和，外积运算可自然延拓至$\bigwedge(W)$上。
\end{definition}

\begin{remark}
\begin{enumerate}[(1)]
\item 向量空间$V$上的外形式空间$\bigwedge^r V^*$可视为对偶空间$V^*$上的$r$次外多项式空间，即$\bigwedge^r V^*=\bigwedge^r(V^*)$；本节通过反对称重线性函数构造外形式，给出了外代数的具体实现。
\item $\mathbb{R}^3$关于向量积（叉积）满足反交换律，但不满足结合律，仅满足Jacobi恒等式，因此$\mathbb{R}^3$关于向量积是李代数，而非外代数。
\end{enumerate}
\end{remark}

\begin{theorem}
设$\omega^1,\dots,\omega^r,\theta_1,\dots,\theta_r$是$n$维向量空间$V$上的$2r$个一次形式，且$\omega^1,\dots,\omega^r$线性无关。若
\begin{align}
\sum\limits_{\alpha=1}^r \omega^\alpha\land\theta_\alpha=0, \label{eq:cartan-identity}
\end{align}
则对任意$\alpha$，存在对称实系数$a_{\alpha\beta}=a_{\beta\alpha}$使得
\begin{align*}
\theta_\alpha = \sum\limits_{\beta=1}^r a_{\alpha\beta}\omega^\beta.
\end{align*}
\end{theorem}

\begin{proof}
将线性无关的$\{\omega^1,\dots,\omega^r\}$扩充为$V^*$的基底$\{\omega^1,\dots,\omega^r,\omega^{r+1},\dots,\omega^n\}$，则每个一次形式$\theta_\alpha$可表示为
\begin{align}
\theta_\alpha = \sum\limits_{i=1}^n a_{\alpha i}\omega^i. \label{eq:theta-expand}
\end{align}
将\eqref{eq:theta-expand}代入\eqref{eq:cartan-identity}，结合$\bigwedge^2 V^*$的基底$\{\omega^i\land\omega^j\mid 1\leqslant i<j\leqslant n\}$展开得
\begin{align*}
0 &= \sum\limits_{\alpha=1}^r \omega^\alpha\land\left(\sum\limits_{i=1}^n a_{\alpha i}\omega^i\right) \\
&= \sum\limits_{\alpha=1}^r\sum\limits_{\beta=1}^r a_{\alpha\beta}\omega^\alpha\land\omega^\beta + \sum\limits_{\alpha=1}^r\sum\limits_{\xi=r+1}^n a_{\alpha\xi}\omega^\alpha\land\omega^\xi \\
&= \sum\limits_{1\leqslant\alpha<\beta\leqslant r} (a_{\alpha\beta}-a_{\beta\alpha})\omega^\alpha\land\omega^\beta + \sum\limits_{\alpha=1}^r\sum\limits_{\xi=r+1}^n a_{\alpha\xi}\omega^\alpha\land\omega^\xi.
\end{align*}
由$\bigwedge^2 V^*$的基底线性无关，各项系数均为零，即
\begin{align*}
\begin{cases}
a_{\alpha\beta}-a_{\beta\alpha}=0,\quad \forall\,1\leqslant\alpha<\beta\leqslant r, \\
a_{\alpha\xi}=0,\quad \forall\,1\leqslant\alpha\leqslant r,\,r<\xi\leqslant n.
\end{cases}
\end{align*}

\end{proof}
