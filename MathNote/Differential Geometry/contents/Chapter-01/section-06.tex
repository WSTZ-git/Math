\section{曲线参数方程在一点的标准展开}

设$\bm{r}(s)$为弧长参数曲线，在$s=0$处作Taylor展开：
\[
\bm{r}(s) = \bm{r}(0) + s\bm{r}'(0) + \dfrac{s^2}{2!}\bm{r}''(0) + \dfrac{s^3}{3!}\bm{r}'''(0) + o(s^3).
\]
代入Frenet公式$\bm{r}'=\bm{\alpha},\ \bm{r}''=\kappa\bm{\beta},\ \bm{r}'''=-\kappa^2\bm{\alpha}+\kappa'\bm{\beta}+\kappa\tau\bm{\gamma}$，得**标准展开式**：
\[
\begin{cases}
x(s) = s - \dfrac{\kappa_0^2}{6}s^3 + o(s^3),\\
y(s) = \dfrac{\kappa_0}{2}s^2 + \dfrac{\kappa_0'}{6}s^3 + o(s^3),\\
z(s) = \dfrac{\kappa_0\tau_0}{6}s^3 + o(s^3),
\end{cases}
\]
其中$\kappa_0=\kappa(0),\kappa_0'=\kappa'(0),\tau_0=\tau(0)$。


\begin{definition}[切触阶]
两条曲线在交点处，若弧长$\Delta s\to0$时，距离为$(\Delta s)^n$的高阶无穷小，则称它们有**$n$阶切触**。
\end{definition}

\begin{theorem}
两条弧长参数曲线在$s=0$处有$n$阶切触的充要条件是
\[
\bm{r}_1^{(k)}(0)=\bm{r}_2^{(k)}(0),\ k=1,2,\dots,n,\quad \bm{r}_1^{(n+1)}(0)\neq\bm{r}_2^{(n+1)}(0).
\]
\end{theorem}


\begin{definition}[密切球面]
与曲线在一点有**3阶切触**的球面称为**密切球面**，其球心为
\[
\bm{r}(s) + \dfrac{1}{\kappa(s)}\bm{\beta}(s) + \dfrac{1}{\tau(s)}\left(\dfrac{1}{\kappa(s)}\right)'\bm{\gamma}(s),
\]
半径为
\[
\sqrt{\left(\dfrac{1}{\kappa(s)}\right)^2 + \left(\dfrac{1}{\tau(s)}\left(\dfrac{1}{\kappa(s)}\right)'\right)^2}.
\]
\end{definition}

过曲率中心、垂直于密切平面的直线称为**曲率轴**。
