\section{存在对应关系的曲线偶}

\begin{definition}[Bertrand曲线偶]
如果两条互不重合的曲线$C_1$和$C_2$之间存在一个对应，使得它们在每一对对应点有公共的主法线，则称这两条曲线为\textbf{Bertrand曲线偶}，其中每一条曲线称为另外一条曲线的\textbf{侣线}（或\textbf{共轭曲线}）．
\end{definition}
\begin{remark}
任意一条平面曲线都有侣线,构成Bertrand曲线偶．事实上，设平面曲线$r=r(s)$以弧长$s$为参数，则
\begin{align*}
r''(s)=\kappa(s)n(s),
\end{align*}
其中$n(s)$是沿曲线定义的单位法向量场.令$$r_1(s)=r(s)+\lambda n(s),$$
其中$\lambda$是任意给定的一个非零常数.又由$n(s)$是单位法向量场知
$$r_1'(s)\cdot n(s)=0,\quad n^2(s)=1\Longrightarrow n'(s)\cdot n(s)=0,$$
故
\begin{align*}
r_1'(s)\cdot n(s)= r'(s)\cdot n(s)+\lambda n'(s)\cdot n(s)=0,
\end{align*}
即$n(s)$也是$r_1(s)$的法向量，故$r(s)$与$r_1(s)$在对应点有相同的法线（即主法线）.因此寻求Bertrand曲线偶应在空间挠曲线（挠率非零的曲线）中去找．
\end{remark}

\begin{theorem}\label{theorem:微分几何-定理7.1}
设曲线$C_1$和$C_2$是Bertrand曲线偶，则$C_1$和$C_2$的对应点之间的距离是常数，并且它们在对应点的切线成定角．
\end{theorem}
\begin{proof}
设$C_1$和$C_2$的参数方程分别为$r_1(s)$和$r_2(s)$，其中$s$是$C_1$的弧长参数．记$C_1$的Frenet标架为$\{r_1(s);\alpha_1(s),\beta_1(s),\gamma_1(s)\}$，$C_2$的Frenet标架为$\{r_2(s);\alpha_2(s),\beta_2(s),\gamma_2(s)\}$，$C_2$的弧长参数记为$\tilde{s}$.
曲线$C_1,C_2$的曲率和挠率分别记为$\kappa_1(s),\kappa_2(s)$和$\tau_1(s),\tau_2(s)$.
由Bertrand曲线定义，对应点有相同的主法线，故存在函数$\lambda(s)$,使得
\begin{align}\label{eq::20rj90j30923jr3-1}
r_2(s)=r_1(s)+\lambda(s)\beta_1(s),\qquad \beta_1(s)=\pm\beta_2(s).
\end{align}
对\eqref{eq::20rj90j30923jr3-1}式关于$s$求导，利用\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
\begin{gather}\label{eq::20rj90j30923jr3-2}
\begin{aligned}
\alpha _2\left( s \right) \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}&=\frac{\mathrm{d}r_2\left( s \right)}{\mathrm{d}\tilde{s}}\frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\frac{\mathrm{d}r_2\left( s \right)}{\mathrm{d}s}=\alpha _1\left( s \right) +\lambda ' \left( s \right) \beta _1\left( s \right) +\lambda \left( s \right) (-\kappa _1\left( s \right) \alpha _1\left( s \right) +\tau _1\left( s \right) \gamma _1\left( s \right) )
\\
&=(1-\lambda \left( s \right) \kappa _1\left( s \right) )\alpha _1\left( s \right) +\lambda ' \left( s \right) \beta _1\left( s \right) +\lambda \left( s \right) \tau _1\left( s \right) \gamma _1\left( s \right) .
\end{aligned}
\end{gather}
因为$\beta_1\left( s \right)=\pm\beta_2\left( s \right)$，所以\eqref{eq::20rj90j30923jr3-2}式两边与$\beta_2\left( s \right)$点乘得
\begin{align*}
0=\alpha _2\left( s \right) \cdot \beta _2\left( s \right) \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\pm \lambda ' \beta _{1}^{2}\left( s \right) =\pm \lambda ' \left( s \right) ,
\end{align*}
所以$\lambda'\left( s \right)=0$，$\lambda\left( s \right)$为常数，即$|r_2(s)-r_1(s)|=|\lambda\left( s \right)|$为常数．再对$\alpha_1\left( s \right)\cdot\alpha_2\left( s \right)$求导得
\begin{align*}
\frac{\mathrm{d}}{\mathrm{d}s}(\alpha _1\left( s \right) \cdot \alpha _2\left( s \right) )=\kappa _1\left( s \right) \beta _1\left( s \right) \cdot \alpha _2\left( s \right) +\alpha _1\left( s \right) \cdot \frac{\mathrm{d}\alpha _2}{\mathrm{d}\tilde{s}}\frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\kappa _1\left( s \right) \beta _1\left( s \right) \cdot \alpha _2\left( s \right) +\kappa _2\left( s \right) \alpha _1\left( s \right) \cdot \beta _2\left( s \right) \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=0,
\end{align*}
因此$\alpha_1\left( s \right)\cdot\alpha_2\left( s \right)$为常数，即$C_1$与$C_2$在对应点的切线成定角．

\end{proof}

\begin{theorem}\label{theorem:微分几何-定理7.2}
设正则参数曲线$C$的曲率$\kappa(s)$和挠率$\tau(s)$均非零，则存在另一条正则参数曲线$C_1$使得$C$与$C_1$构成Bertrand曲线偶的充分必要条件是:存在常数$\lambda\neq0$和$\mu$，使得
\begin{align*}
\lambda\kappa(s)+\mu\tau(s)=1.
\end{align*}
\end{theorem}
\begin{proof}
{\heiti 必要性:}设$C$有侣线$C_1$，参数方程分别为$r(s)$和$r_1(s)$，其中$s$是$C$的弧长参数．由\refthe{theorem:微分几何-定理7.1}知，存在常数$\lambda\neq0$使得
\begin{align*}
r_1(s)=r(s)+\lambda\beta(s).
\end{align*}
记$C_1$的弧长参数为$\tilde{s}$，对上式求导得
\begin{align}\label{eq:fj2349j0iJIOJJIOfjioewm}
\alpha _1\left( s \right) \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\frac{\mathrm{d}r_1}{\mathrm{d}\tilde{s}}\frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\frac{\mathrm{d}r_1}{\mathrm{d}s}=\left( 1-\lambda \kappa \left( s \right) \right) \alpha \left( s \right) +\lambda \tau \left( s \right) \gamma \left( s \right) .
\end{align}
于是对上式两边平方得
\begin{align*}
\left| \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s} \right|^2=\left( 1-\lambda \kappa \left( s \right) \right) ^2+\left( \lambda \tau \left( s \right) \right) ^2.
\end{align*}
又由\refthe{theorem:微分几何-定理7.1}知,可设$\alpha(s)\cdot\alpha_1(s)=C$为常数，故\eqref{eq:fj2349j0iJIOJJIOfjioewm}式两边与$\alpha(s)$点乘得
\begin{align*}
C\frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\alpha \left( s \right) \cdot \alpha _1\left( s \right) \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=1-\lambda \kappa \left( s \right) ,
\end{align*}
所以
\begin{align*}
\frac{\left( \frac{1-\lambda \kappa \left( s \right)}{\tau \left( s \right)} \right) ^2}{\left( \frac{1-\lambda \kappa \left( s \right)}{\tau \left( s \right)} \right) ^2+\lambda ^2}=\frac{\left( 1-\lambda \kappa \left( s \right) \right) ^2}{\left( 1-\lambda \kappa \left( s \right) \right) ^2+\left( \lambda \tau \left( s \right) \right) ^2}=\frac{C\left| \frac{\mathrm{d}\widetilde{s}}{\mathrm{d}s} \right|^2}{\left| \frac{\mathrm{d}\widetilde{s}}{\mathrm{d}s} \right|^2}=C.
\end{align*}
从而$\frac{1-\lambda\kappa}{\tau}$为常数，记作$\mu$，即$\lambda\kappa+\mu\tau=1$．

{\heiti 充分性:} 设$\lambda\neq0$和$\mu$是常数，使得$\lambda\kappa(s)+\mu\tau(s)=1$．构造曲线
\begin{align}
r_1(s)=r(s)+\lambda\beta(s).
\end{align}
求导得
\begin{align*}
r_1' \left( s \right) =(1-\lambda \kappa \left( s \right) )\alpha \left( s \right) +\lambda \tau \left( s \right) \gamma \left( s \right) =\mu \tau \left( s \right) \alpha \left( s \right) +\lambda \tau \left( s \right) \gamma \left( s \right) =\tau \left( s \right) (\mu \alpha \left( s \right) +\lambda \gamma \left( s \right) ).
\end{align*}
因此由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}可得
\begin{align*}
\alpha _1(s)=\frac{{r_1}^{\prime}\left( s \right)}{\left| r_{2}^{\prime}\left( s \right) \right|}=\frac{\tau \left( s \right) \left( \mu \alpha \left( s \right) +\lambda \gamma \left( s \right) \right)}{\tau \left( s \right) \sqrt{\lambda ^2+\mu ^2}}=\frac{\mu}{\sqrt{\lambda ^2+\mu ^2}}\alpha (s)+\frac{\lambda}{\sqrt{\lambda ^2+\mu ^2}}\gamma (s).
\end{align*}
再对上式求导并利用\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}，得
\begin{align*}
\frac{\mathrm{d}\alpha _1}{\mathrm{d}s}=\frac{\mu}{\sqrt{\lambda ^2+\mu ^2}}(\kappa \left( s \right) \beta \left( s \right) )+\frac{\lambda}{\sqrt{\lambda ^2+\mu ^2}}(-\tau \left( s \right) \beta \left( s \right) )=\frac{\mu \kappa \left( s \right) -\lambda \tau \left( s \right)}{\sqrt{\lambda ^2+\mu ^2}}\beta \left( s \right) .
\end{align*}
记$\tilde{s}$为$C_1$的弧长参数，则$\frac{\mathrm{d}\alpha _1}{\mathrm{d}\tilde{s}}=\kappa _1\left( s \right) \beta _1\left( s \right) $,进而
\begin{align*}
\frac{\mathrm{d}\alpha _1}{\mathrm{d}\tilde{s}}\frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\kappa _1\left( s \right) \beta _1\left( s \right) \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\frac{\mu \kappa \left( s \right) -\lambda \tau \left( s \right)}{\sqrt{\lambda ^2+\mu ^2}}\beta \left( s \right) .
\end{align*}
因此$\beta_1(s)$与$\beta(s)$平行，即$\beta_1(s)=\pm\beta(s)$，故$C$与$C_1$在对应点有相同的主法线，它们构成Bertrand曲线偶．

\end{proof}

\begin{definition}
如果在曲线$C_1$和$C_2$之间存在一个对应，使得$C_1$在任意一点的切线恰好是$C_2$在对应点的(主)法线，则称$C_2$是$C_1$的\textbf{渐伸线}，同时称$C_1$是$C_2$的\textbf{渐缩线}.
\end{definition}

\begin{theorem}[曲线渐伸线的参数方程]\label{theorem:微分几何-定理7.3}
设正则参数曲线$C$的参数方程为$r(s)$，$s$是弧长参数，则$C$的渐伸线的参数方程为
\begin{align}\label{eq::3902jr0209rj20JJIJ}
r_1(s)=r(s)+(c-s)\alpha(s),
\end{align}
其中$c$为任意常数．
\end{theorem}
\begin{remark}
曲线的渐伸线可以看作该曲线的切线族的正交轨线．\eqref{eq::3902jr0209rj20JJIJ}式的几何解释：将一条软线沿曲线放置，一端固定于$s=c$处，另一端慢慢离开原曲线并始终保持伸直的部分为原曲线的切线，则这另一端描出的曲线就是原曲线的渐伸线．
\end{remark}
\begin{proof}
设渐伸线为$r_1(s)=r(s)+\lambda(s)\alpha(s)$，则$\alpha(s)$应是$r_1(s)$的法向量，故由渐伸线定义知$r_1'(s)\cdot\alpha(s)=0$,即
\begin{align*}
r_1'(s)=\alpha(s)+\lambda'(s)\alpha(s)+\lambda(s)\kappa(s)\beta(s)=(1+\lambda'(s))\alpha(s)+\lambda(s)\kappa(s)\beta(s),
\end{align*}
于是
\begin{align*}
r_1'(s)\cdot\alpha(s)=1+\lambda'(s)=0,
\end{align*}
解得$\lambda(s)=c-s$，其中$c$为常数.代入即得\eqref{eq::3902jr0209rj20JJIJ}式.

\end{proof}

\begin{theorem}[曲线渐缩线的参数方程]\label{theorem:曲线渐缩线的参数方程}
设正则参数曲线$C$的参数方程为$r(s)$，$s$是弧长参数，$\kappa(s),\tau(s)$分别是曲线$C$的曲率和挠率.则$C$的渐缩线的参数方程为
\begin{align}\label{eq::ri239jrj23jj2IU}
r_1(s)=r(s)+\frac{1}{\kappa(s)}\beta(s)-\frac{1}{\kappa(s)}\tan\left(\int\tau(s)\mathrm{d}s\right)\gamma(s).
\end{align}
\end{theorem}
\begin{proof}
由渐缩线的定义知$r_1(s)-r(s)\in \left( \mathrm{span}\left\{ \alpha \left( s \right) \right\} \right) ^{\bot}=\mathrm{span}\left\{ \beta \left( s \right) ,\gamma \left( s \right) \right\} $,故可设渐缩线为
\begin{align}\label{eq::AAFJIOWJjioijo}
r_1(s)=r(s)+\lambda(s)\beta(s)+\mu(s)\gamma(s),
\end{align}
则向量$\lambda(s)\beta(s)+\mu(s)\gamma(s)$应是$r_1(s)$的切向量,即$\lambda(s)\beta(s)+\mu(s)\gamma(s)$与$r_1'$(s)平行.对上式求导得
\begin{align*}
r_1' \left( s \right) &=\alpha \left( s \right) +\lambda ' \left( s \right) \beta \left( s \right) +\lambda \left( s \right) (-\kappa \left( s \right) \alpha \left( s \right) +\tau \left( s \right) \gamma \left( s \right) )+\mu ' \left( s \right) \gamma \left( s \right) +\mu \left( s \right) (-\tau \left( s \right) \beta \left( s \right) )
\\
&=(1-\lambda \left( s \right) \kappa \left( s \right) )\alpha \left( s \right) +(\lambda ' \left( s \right) -\mu \left( s \right) \tau \left( s \right) )\beta \left( s \right) +(\mu ' \left( s \right) +\lambda \left( s \right) \tau \left( s \right) )\gamma \left( s \right) .
\end{align*}
由于$\lambda(s)\beta(s)+\mu(s)\gamma(s)$与$r_1'$(s)平行，故$\alpha(s)$方向的系数必须为零，即
\begin{align*}
1-\lambda\kappa(s)=0\Longrightarrow \lambda=\frac{1}{\kappa(s)}.
\end{align*}
同时$(\lambda'-\mu\tau(s),\ \mu'+\lambda\tau(s))$与$(\lambda,\mu)$成比例，故
\begin{align*}
\frac{\lambda'-\mu\tau(s)}{\lambda}=\frac{\mu'+\lambda\tau(s)}{\mu}
\Longrightarrow
\lambda'\mu-\mu'\lambda=(\lambda^2+\mu^2)\tau.
\end{align*}
注意到$\frac{\mathrm{d}}{\mathrm{d}s}\left(\frac{\mu}{\lambda}\right)=\frac{\lambda\mu'-\mu\lambda'}{\lambda^2}$，故
\begin{align*}
\frac{\mathrm{d}}{\mathrm{d}s}\left(\frac{\mu}{\lambda}\right)=-\frac{(\lambda^2+\mu^2)\tau}{\lambda^2}=-\left(1+\left(\frac{\mu}{\lambda}\right)^2\right)\tau.
\end{align*}
令$\theta=\arctan\frac{\mu}{\lambda}$，则$\frac{\mathrm{d}\theta}{\mathrm{d}s}=-\tau$，积分得$\theta=-\int\tau(s)\mathrm{d}s$，即
\begin{align*}
\frac{\mu}{\lambda}=\tan\left(-\int\tau(s)\mathrm{d}s\right)=-\tan\int\tau(s)\mathrm{d}s.
\end{align*}
再由$\lambda=\frac{1}{\kappa(s)}$得$\mu=-\frac{1}{\kappa(s)}\tan\int\tau(s)\mathrm{d}s$．代入\eqref{eq::AAFJIOWJjioijo}式即得\eqref{eq::ri239jrj23jj2IU}式．

\end{proof}


\begin{example}
设$C$是挠率为非零常数的正则参数曲线，$\tilde{C}$是沿曲线$C$的次法线到曲线$C$的距离为常数$c$的点的轨迹，求曲线$C$和$\tilde{C}$在对应点的次法线之间的夹角.
\end{example}
\begin{solution}

设曲线$C$的参数方程为$\bm{r}=\bm{r}(s)$，其中$s$为曲线$C$的弧长参数。其在$s$处的Frenet标架为$\left\{ \bm{r}(s); \bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s) \right\}$。记曲线$C$的曲率和挠率分别为$\kappa(s)$和$\tau(s)$。再设曲线$\tilde{C}$的参数方程为$\bm{r}=\bm{r}_1(s)$，其在$s$处的Frenet标架为$\left\{ \bm{r}_1(s); \bm{\alpha}_1(s), \bm{\beta}_1(s), \bm{\gamma}_1(s) \right\}$，记曲线$\tilde{C}$的曲率和挠率分别为$\kappa_1(s)$和$\tau_1(s)$。由条件可知，曲线$\tilde{C}$的参数方程满足
\begin{align*}
\bm{r}_1(s) = \bm{r}(s) + c\bm{\gamma}(s),\quad \forall s.
\end{align*}
对上式关于$s$求导并利用\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
\begin{align*}
\bm{r}_{1}'(s) = \bm{r}'(s) + c\bm{\gamma}'(s) = \bm{\alpha}(s) - c\tau(s)\bm{\beta}(s).
\end{align*}
再由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}知
\begin{align}
\bm{\alpha}_1(s) = \frac{\bm{r}_{1}'(s)}{\left| \bm{r}_{1}'(s) \right|} = \frac{\bm{\alpha}(s) - c\tau(s)\bm{\beta}(s)}{\sqrt{1+c^2\tau^2(s)}}.
\label{eq:--29fj3290fKLIJI31.1}
\end{align}
并且曲线$\tilde{C}$的弧长元素$\mathrm{d}\tilde{s}=\left| \bm{r}_{1}'(s) \right|\mathrm{d}s=\sqrt{1+c^2\tau^2(s)}\mathrm{d}s$，进而$\frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\sqrt{1+c^2\tau^2(s)}$。再对\eqref{eq:--29fj3290fKLIJI31.1}式关于$s$求导并利用\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
\begin{align*}
\frac{\mathrm{d}\bm{\alpha}_1(s)}{\mathrm{d}\tilde{s}}\sqrt{1+c^2\tau^2(s)}=\frac{\mathrm{d}\bm{\alpha}_1(s)}{\mathrm{d}\tilde{s}}\cdot \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}=\frac{\mathrm{d}\bm{\alpha}_1(s)}{\mathrm{d}s}=\frac{\bm{\alpha}'(s) - c\tau(s)\bm{\beta}'(s)}{\sqrt{1+c^2\tau^2(s)}}=\frac{c\tau(s)\kappa(s)\bm{\alpha}(s) + \kappa(s)\bm{\beta}(s) - c\tau^2(s)\bm{\gamma}(s)}{\sqrt{1+c^2\tau^2(s)}}.
\end{align*}
因此
\begin{align*}
\frac{\mathrm{d}\bm{\alpha}_1(s)}{\mathrm{d}\tilde{s}}=\frac{c\tau(s)\kappa(s)\bm{\alpha}(s) + \kappa(s)\bm{\beta}(s) - c\tau^2(s)\bm{\gamma}(s)}{1+c^2\tau^2(s)}.
\end{align*}
而$\frac{\mathrm{d}\bm{\alpha}_1(s)}{\mathrm{d}\tilde{s}}$与曲线$\tilde{C}$在$s$处的主法线同向，故
\begin{align*}
\bm{\beta}_1(s) = \frac{\frac{\mathrm{d}\bm{\alpha}_1(s)}{\mathrm{d}\tilde{s}}}{\left| \frac{\mathrm{d}\bm{\alpha}_1(s)}{\mathrm{d}\tilde{s}} \right|}=\frac{\frac{c\tau(s)\kappa(s)\bm{\alpha}(s) + \kappa(s)\bm{\beta}(s) - c\tau^2(s)\bm{\gamma}(s)}{1+c^2\tau^2(s)}}{\frac{\sqrt{c^2\tau^2(s)\kappa^2(s) + \kappa^2(s) + c^2\tau^4(s)}}{1+c^2\tau^2(s)}}=\frac{c\tau(s)\kappa(s)\bm{\alpha}(s) + \kappa(s)\bm{\beta}(s) - c\tau^2(s)\bm{\gamma}(s)}{\sqrt{c^2\tau^2(s)\kappa^2(s) + \kappa^2(s) + c^2\tau^4(s)}}.
\end{align*}
因此
\begin{align*}
\bm{\gamma}_1(s) &= \bm{\alpha}_1(s) \times \bm{\beta}_1(s) \\
&=\frac{c^2\tau^3(s)}{\sqrt{\left( 1+c^2\tau^2(s) \right) \left( c^2\tau^2(s)\kappa^2(s) + \kappa^2(s) + c^2\tau^4(s) \right)}}\bm{\alpha}(s) \\
&\quad+\frac{c\tau^2(s)}{\sqrt{\left( 1+c^2\tau^2(s) \right) \left( c^2\tau^2(s)\kappa^2(s) + \kappa^2(s) + c^2\tau^4(s) \right)}}\bm{\beta}(s) \\
&\quad+\frac{\kappa(s) \sqrt{1+c^2\tau^2(s)}}{\sqrt{c^2\tau^2(s)\kappa^2(s) + \kappa^2(s) + c^2\tau^4(s)}}\bm{\gamma}(s).
\end{align*}
对$\forall s$，记$\bm{\gamma}(s)$与$\bm{\gamma}_1(s)$的夹角为$\theta$，则
\begin{align*}
\cos\theta = \bm{\gamma}(s) \cdot \bm{\gamma}_1(s) = \frac{\kappa(s)\sqrt{1+c^2\tau^2(s)}}{\sqrt{c^2\tau^2(s)\kappa^2(s) + \kappa^2(s) + c^2\tau^4(s)}} \Longrightarrow \theta = \arccos\left( \frac{\kappa(s)\sqrt{1+c^2\tau^2(s)}}{\sqrt{c^2\tau^2(s)\kappa^2(s) + \kappa^2(s) + c^2\tau^4(s)}} \right).
\end{align*}

\end{align*}
故由\hyperref[theorem:微分几何-曲线挠率的计算公式]{曲线挠率的计算公式}知渐伸线$\bm{\rho}(s)$的挠率为零，因此由\refthe{theorem:挠率为0等价于是平面曲线}知它是平面曲线。

显然圆螺旋线$\bm{r}(s)$的轴线方向为$\bm{n}=(0,0,1)$，注意到
\begin{align*}
\bm{\rho}(s)\cdot\bm{n}= \bigl(\bm{r}(s)+(c-s)\bm{\alpha}(s)\bigr)\cdot\bm{n}=bks+(c-s)bk=bkc\in \mathbb{R},
\end{align*}
故$\bm{\rho}(s)$落在平面$z=bkc$上，该平面与轴线垂直.

最后证明$\bm{\rho}(s)$也是圆柱面$x^2+y^2=a^2$与平面$z=bkc$交线的渐伸线. 交线为圆$\bm{\varphi}(\theta)=(a\cos\theta,a\sin\theta,bkc)$，其参数为$\theta\in [0,2\pi]$.注意到交线的弧长参数为$\tilde{s}=a\theta$,故交线以弧长$\tilde{s}$为参数的参数方程为
\begin{align*}
\varphi(\tilde{s})=(a\cos\frac{\tilde{s}}{a},a\sin\frac{\tilde{s}}{a},bkc).
\end{align*}
从而交线的单位切向量场为
\begin{align*}
\tilde{\bm{\alpha}}(\tilde{s})=\frac{\mathrm{d}\boldsymbol{\varphi }}{\mathrm{d}\tilde{s}}=(-\sin\frac{\tilde{s}}{a},\cos\frac{\tilde{s}}{a},0).
\end{align*}
再由\hyperref[theorem:微分几何-定理7.3]{曲线渐伸线的参数方程}可得交线的渐伸线方程为
\begin{align*}
\tilde{\bm{\rho}}(\tilde{s}) = \bm{\varphi}(\tilde{s}) + (c-\tilde{s})\tilde{\bm{\alpha}}(\tilde{s})
= \left(a\cos\frac{\tilde{s}}{a}-(c-\tilde{s})\sin\frac{\tilde{s}}{a},\; a\sin\frac{\tilde{s}}{a}+(c-\tilde{s})\cos\frac{\tilde{s}}{a},\; bkc\right),
\end{align*}
其中$c$为任意常数.
这正是以$\theta$为参数的圆的渐伸线方程\eqref{eq::2jru3ju233jKJIJI}. 因此$\bm{\rho}(s)$即为该交线圆的渐伸线.

\end{proof}
