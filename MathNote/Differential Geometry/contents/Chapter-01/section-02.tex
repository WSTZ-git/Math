\section{曲线的弧长}

\begin{definition}[曲线的弧长]
设$E^3$中的一条正则曲线$C$的参数方程是$\bm{r}(t)$，$t\in[a,b]$，其\textbf{弧长}定义为
\[
s = \int_a^b \big|\bm{r}'(t)\big|\mathrm{d}t.
\]
其中$\big|\bm{r}'(t)\big|=\sqrt{x'(t)^2+y'(t)^2+z'(t)^2}$为切向量的长度。
\end{definition}

\begin{theorem}
设$E^3$中的一条正则曲线$C$的参数方程是$\bm{r}(t)$，$t\in[a,b]$，对$\forall t\in \left[ a,b \right]$,令
\begin{align*}
s\left( t \right) =\int_a^t{\left| \bm{r}' \left( t \right) \right|\mathrm{d}t},
\end{align*}
是曲线$C$从$a$到$t$的弧长,则存在反函数
\begin{align}\label{eq::0293jr2jsjf9hj3fp48afp4fp4fp4afp4afp4afp4afp4afp4afp4afp4afp4afp4afp4afp4afp4afp4}
t=t\left( s \right),\quad s\in \left[ 0,s\left( b \right) \right].
\end{align}
使得曲线$C$的参数方程也可以表示为
\begin{align*}
\bm{r}\left( t \right) =\bm{r}\left( t\left( s \right) \right),\quad s\in \left[ 0,s\left( b \right) \right].
\end{align*}
此时称参数$s$为曲线$C$的\textbf{弧长参数}.由\eqref{eq::0293jr2jsjf9hj3fp48afp4fp4fp4afp4afp4afp4afp4afp4afp4afp4afp4afp4afp4afp4afp4afp4}式还可得
\begin{align}\label{eq:9j03jj3f0j9090u9i9i0iu3th47gh}
\mathrm{d}s=\left| \boldsymbol{r}'\left( t \right) \right|\mathrm{d}t,
\end{align}
将$\mathrm{d}s$称为曲线$C$的\textbf{弧长元素}.
\end{theorem}
\begin{proof}
显然$s\left( t \right)$严格递增,又由$s\left( t \right) \in C^k\left[ a,b \right] \left( k\geqslant 3 \right)$,且
\begin{align*}
\frac{\mathrm{d}s\left( t \right)}{\mathrm{d}t}=\left| \bm{r}' \left( t \right) \right|>0,
\end{align*}
知$s\left( t \right)$是曲线$C$的保持定向的参数变换.进而存在反函数
\begin{align*}
t=t\left( s \right),\quad s\in \left[ 0,s\left( b \right) \right].
\end{align*}
代入原参数方程即得.

\end{proof}

\begin{theorem}
设$E^3$中的一条正则曲线$C$的参数方程是$\bm{r}(t)$，$t\in[a,b]$,则曲线$C$的弧长$s$,弧长参数$s(t)$,弧长元素$\mathrm{d}s$都是是曲线$C$的一个不变量，即它与空间$E^3$中的笛卡尔直角坐标系的选取无关，也与曲线$\bm{r}(t)$的保持定向的参数变换无关.
\end{theorem}
\begin{proof}
前者是因为在笛卡儿直角坐标变换下, 切向量的长度$|\bm{r}'(t)|$是不变的, 故$s$不变. 关于后者, 不妨设参数变换是
\begin{align}
t = t(u),\quad t'(u) > 0,\quad \alpha \leqslant u \leqslant \beta,
\label{eq:::-39jt349gu9ejg8eu4gpdrsg34uyrgxfg2}
\end{align}
并且
\begin{align*}
t(\alpha) = a,\quad t(\beta) = b,
\end{align*}
因此
\begin{align}
\left| \frac{\mathrm{d}\bm{r}(t(u))}{\mathrm{d}u} \right| = \left| \frac{\mathrm{d}\bm{r}}{\mathrm{d}t}(t(u)) \right| \cdot \frac{\mathrm{d}t}{\mathrm{d}u}.
\label{eq:::-39jt349gu9ejg8eu4gpdrsg34uyrgxfg3}
\end{align}
根据积分的变量替换公式有
\begin{align*}
\int_a^b \left| \frac{\mathrm{d}\bm{r}(t)}{\mathrm{d}t} \right| \mathrm{d}t = \int_\alpha^\beta \left| \frac{\mathrm{d}\bm{r}}{\mathrm{d}t}(t(u)) \right| \cdot \frac{\mathrm{d}t}{\mathrm{d}u} \mathrm{d}u 
= \int_\alpha^\beta \left| \frac{\mathrm{d}\bm{r}(t(u))}{\mathrm{d}u} \right| \mathrm{d}u,
\end{align*}
所以$s$与参数变换\eqref{eq:::-39jt349gu9ejg8eu4gpdrsg34uyrgxfg2}是无关的.弧长参数与弧长参数的不变性同理可证.

\end{proof}

\begin{theorem}\label{theorem:微分几何-定理2.1}
设$\bm{r} = \bm{r}(t)(a \leqslant t \leqslant b)$是$E^3$中的一条正则曲线，则$t$是它的弧长参数的充分必要条件是$|\bm{r}'(t)| \equiv 1$.
\end{theorem}
\begin{remark}
上述定理的几何意义是: \textbf{曲线以弧长为参数的充分必要条件是它的切向量场是单位切向量场.}
\end{remark}
\begin{proof}
因为$\frac{\mathrm{d}s}{\mathrm{d}t} = |\bm{r}'(t)|$, 当$t$是它的弧长参数$s$时, $\mathrm{d}s = \mathrm{d}t$, 所以由\eqref{eq:9j03jj3f0j9090u9i9i0iu3th47gh}式得知$|\bm{r}'(t)| \equiv 1$. 反之亦然.

\end{proof}

\begin{example}
求下列曲线在指定范围内的弧长:
\begin{enumerate}[(1)]
\item $\bm{r}(t) = (\cosh t, \sinh t, t),\ -1 \leqslant t \leqslant 1$;
\item $\bm{r}(t) = (\sin^2 t, \sin t \cos t, \ln \cos t),\ t\in[0, t_0]$;
\item $\bm{r}(t) = \left(t + \frac{a^2}{t}, t - \frac{a^2}{t}, 2a \ln \frac{t}{a}\right)$, 在曲线与$x$轴的交点和参数为$t_0$的点之间;
\item 曳物线$\bm{r}(t) = (\cos t, \ln(\sec t + \tan t) - \sin t),\ [0, t_0]$;
\item 曲线$y = \frac{x^2}{2a},\ z = \frac{x^3}{6a^2}$在原点$O(0,0,0)$和点$p(x_0, y_0, z_0)$之间.
\end{enumerate}
\end{example}
\begin{solution}
\begin{enumerate}[(1)]
\item 由题设可得
\begin{align*}
\bm{r}'(t)=(\sinh t,\cosh t,1),\quad |\bm{r}'(t)|=\sqrt{\sinh^2 t+\cosh^2 t+1}=\sqrt{2}\cosh t.
\end{align*}
从而所求曲线的弧长为
\begin{align*}
s=\int_{-1}^1 |\bm{r}'(t)|\mathrm{d}t=\sqrt{2}\int_{-1}^1 \cosh t\mathrm{d}t=\sqrt{2}\left( \sinh 1-\sinh (-1) \right)=\sqrt{2}\left( \mathrm{e}-\mathrm{e}^{-1} \right).
\end{align*}

\item 由题设可得
\begin{align*}
\bm{r}'(t)=(2\sin t\cos t,\cos^2 t-\sin^2 t,-\tan t),\quad |\bm{r}'(t)|=\sqrt{1+\tan^2 t}=|\sec t|.
\end{align*}
因为$\bm{r}(t)$在$[0,t_0]$上有定义,即$\ln\cos t$在$[0,t_0]$上有定义,所以
\begin{align*}
\cos t>0,\forall t\in [0,t_0] \Longrightarrow t_0\in \left[0,\frac{\pi}{2}\right).
\end{align*}
从而所求曲线的弧长为
\begin{align*}
s=\int_0^{t_0} |\bm{r}'(t)|\mathrm{d}t=\int_0^{t_0} |\sec t|\mathrm{d}t=\int_0^{t_0} \sec t\mathrm{d}t=\ln \left( \tan t_0+\sec t_0 \right).
\end{align*}

\item 由题设可得
\begin{align*}
\bm{r}'(t)=\left(1-\frac{a^2}{t^2},1+\frac{a^2}{t^2},\frac{2a}{t}\right),\quad |\bm{r}'(t)|=\sqrt{2+\frac{2a^4}{t^4}+\frac{4a^2}{t^2}}=\sqrt{2}\left(1+\frac{a^2}{t^2}\right).
\end{align*}
令$t-\frac{a^2}{t}=2a\ln \frac{t}{a}=0$得$t=a$,故曲线$\bm{r}(t)$与$x$轴交点为$\bm{r}(a)=(2a,0,0)$.从而所求曲线的弧长为
\begin{align*}
s=\int_a^{t_0} |\bm{r}'(t)|\mathrm{d}t=\sqrt{2}\int_a^{t_0} \left(1+\frac{a^2}{t^2}\right)\mathrm{d}t=\sqrt{2}\left( t_0-\frac{a^2}{t_0} \right).
\end{align*}

\item 由题设可得
\begin{align*}
\bm{r}'(t)=(-\sin t,\sec t-\cos t),\quad |\bm{r}'(t)|=\sqrt{\sec^2 t-1}=|\tan t|.
\end{align*}
因为$\bm{r}(t)$在$[0,t_0]$上有定义,即$\ln(\sec t+\tan t)$在$[0,t_0]$上有定义,所以
\begin{align*}
\sec t+\tan t>0,\forall t\in [0,t_0] \Longrightarrow \cos t>0,\forall t\in [0,t_0] \Longrightarrow t_0\in \left[0,\frac{\pi}{2}\right).
\end{align*}
从而所求曲线的弧长为
\begin{align*}
s=\int_0^{t_0} |\bm{r}'(t)|\mathrm{d}t=\int_0^{t_0} |\tan t|\mathrm{d}t=-\ln \left( \cos t_0 \right).
\end{align*}

\item 由题设可知,曲线的参数方程为
\begin{align*}
\bm{r}(t)=\left(x,\frac{x^2}{2a},\frac{x^3}{6a^2}\right),
\end{align*}
进而
\begin{align*}
\bm{r}'(t)=\left(1,\frac{x}{a},\frac{x^2}{2a^2}\right),\quad |\bm{r}'(t)|=\sqrt{1+\frac{x^2}{a^2}+\frac{x^4}{4a^4}}=1+\frac{x^2}{2a^2}.
\end{align*}
从而所求曲线的弧长为
\begin{align*}
s=\int_0^{x_0} |\bm{r}'(t)|\mathrm{d}t=\int_0^{x_0} \left(1+\frac{x^2}{2a^2}\right)\mathrm{d}t=x_0+\frac{x_0^3}{6a^2}.
\end{align*}

\end{enumerate}

\end{solution}

\begin{example}
求下列曲线的单位切向量场:
\begin{enumerate}[(1)]
\item $\bm{r}(t) = (\cos^3 t, \sin^3 t, \cos 2t)$;
\item $\bm{r}(t) = (3t, 3t^2, 2t^3)$.
\end{enumerate}
\end{example}
\begin{solution}
\begin{enumerate}[(1)]
\item 由题设可知
\begin{align*}
\bm{r}'(t)=\left(-3\sin t\cos^2 t,3\sin^2 t\cos t,-2\sin 2t\right),\quad |\bm{r}'(t)|=5\sin t\cos t.
\end{align*}
故所求曲线的单位切向量场为
\begin{align*}
\alpha=\frac{\bm{r}'(t)}{|\bm{r}'(t)|}=\left(-\frac{3}{5}\cos t,\frac{3}{5}\sin t,-\frac{4}{5}\right).
\end{align*}
\item 由题设可知
\begin{align*}
\bm{r}'(t)=\left(3,6t,6t^2\right),\quad |\bm{r}'(t)|=3\left(1+2t^2\right).
\end{align*}
故所求曲线的单位切向量场为
\begin{align*}
\alpha=\frac{\bm{r}'(t)}{|\bm{r}'(t)|}=\left(\frac{1}{1+2t^2},\frac{2t}{1+2t^2},\frac{2t^2}{1+2t^2}\right).
\end{align*}

\end{enumerate}

\end{solution}

\begin{example}
设曲线$C$是下面两个柱面的交线:
\[
\frac{x^2}{a^2} - \frac{y^2}{b^2} = 1,\quad x = a \cosh \frac{z}{a},
\]
$a,b>0$是常数. 求曲线$C$从点$(a,0,0)$到点$(x_0, y_0, z_0)$的弧长.
\end{example}
\begin{solution}
由题设可知曲线$C$满足
\begin{align*}
\cosh^2\frac{z}{a}-\frac{y^2}{b^2}=1\Longrightarrow y=\sqrt{b^2\left( \cosh^2\frac{z}{a}-1 \right)}=\sqrt{b^2\sinh^2\frac{z}{a}}=b\left| \sinh\frac{z}{a} \right|.
\end{align*}
因此曲线$C$从点$(a,0,0)$到点$(x_0,y_0,z_0)$的参数方程为
\begin{align*}
\bm{r}(z)=\left( a\cosh\frac{z}{a},b\left| \sinh\frac{z}{a} \right|,z \right)=\left( a\cosh\frac{z}{a},b\sinh\frac{z}{a},z \right),\quad z\in [0,z_0].
\end{align*}
从而
\begin{align*}
\bm{r}'(z)=\left( \sinh\frac{z}{a},\frac{b}{a}\cosh\frac{z}{a},1 \right),\quad |\bm{r}'(z)|=\frac{\sqrt{1+b^2}}{a}\cosh\frac{z}{a},\quad z\in [0,z_0].
\end{align*}
于是所求弧长为
\begin{align*}
s=\int_0^{z_0} |\bm{r}'(z)|\mathrm{d}z=\frac{\sqrt{1+b^2}}{a}\int_0^{z_0} \cosh\frac{z}{a}\mathrm{d}z=\sqrt{1+b^2}\sinh\frac{z_0}{a}.
\end{align*}

\end{solution}

\begin{example}
证明: 曲线$x^2 = 3y,\ 2xy = 9z$的切向量和某个固定的方向成定角. 确定这个固定的方向和该角的大小.
\end{example}
\begin{proof}
由题设可知曲线的参数方程为
\begin{align*}
\bm{r}(x)=\left(x,\frac{x^2}{3},\frac{2}{27}x^3\right).
\end{align*}
从而
\begin{align*}
\bm{r}'(x)=\left(1,\frac{2}{3}x,\frac{2}{9}x^2\right),\quad |\bm{r}'(x)|=\sqrt{1+\frac{4}{9}x^2+\frac{4}{81}x^4}=1+\frac{2}{9}x^2.
\end{align*}
取单位向量$\bm{a}=\left(\frac{1}{\sqrt{2}},0,\frac{1}{\sqrt{2}}\right)$,则对$\forall x\in \mathbb{R}$,记$\bm{r}'(x)$与$\bm{a}$的夹角为$\theta_x$,都有
\begin{align*}
\cos \theta _x=\frac{\boldsymbol{a}\cdot \boldsymbol{r}' (x)}{|\boldsymbol{r}' (x)|}=\frac{1}{\sqrt{2}}\cdot \frac{1+\frac{2}{9}x^2}{1+\frac{2}{9}x^2}=\frac{1}{\sqrt{2}}\Longrightarrow \theta _x=\arccos \frac{1}{\sqrt{2}}=\frac{\pi}{4}.
\end{align*}
故曲线的切向量与固定的单位向量$\bm{a}$成定角$\frac{\pi}{4}$.

\end{proof}

\begin{example}
求曲线$\bm{r}(t) = \left(\frac{t^3}{3}, \frac{t^2}{2}, t\right)$上其切线平行于平面$x + 3y + 2z = 0$的点.
\end{example}
\begin{solution}
由题设知
\begin{align*}
\bm{r}'(t)=\left(t^2,t,1\right),\quad \forall t\in \mathbb{R}.
\end{align*}
记平面$\pi:x+3y+2z=0$的法向量为$\bm{n}=(1,3,2)$,则
\begin{align*}
\bm{r}'(t)\cdot \bm{n}=0\Longleftrightarrow t^2+3t+2=0\Longleftrightarrow t=-1\text{或}t=-2.
\end{align*}
因此所求点为$\boldsymbol{r}\left( -1 \right) =\left( -\frac{1}{3},\frac{1}{2},-1 \right)$ 和$\boldsymbol{r}\left( -2 \right) =\left( -\frac{8}{3},2,-2 \right) $.

\end{solution}

\begin{example}

\end{example}
\begin{solution}

\begin{align*}
\bm{r}\left( t \right) =\bm{r}\left( 0 \right) +\int_0^t \bm{r}'\left( t \right) \mathrm{d}t=\left( 1,0,-5 \right) +\left( \frac{t^3}{3},\frac{t^2}{2},\mathrm{e}^t \right) -\left( 0,0,1 \right) 
=\left( 1+\frac{t^3}{3},\frac{t^2}{2},\mathrm{e}^t-6 \right)
\end{align*}

\end{solution}
