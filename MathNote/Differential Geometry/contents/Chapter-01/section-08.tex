\section{平面曲线}

\begin{definition}
设平面正则曲线$C$在平面$E^2$的右手笛卡儿直角坐标系下的参数方程为
\begin{align*}
\bm{r}(s)=(x(s),y(s)),
\end{align*}
其中$s$为弧长参数，则其\textbf{单位切向量}为
\begin{align*}
\bm{\alpha}(s)=(x'(s),y'(s)),\quad(x'(s))^2+(y'(s))^2=1.
\end{align*}
将$\bm{\alpha}(s)$沿正向旋转$90^\circ$，得到\textbf{单位法向量}
\begin{align*}
\bm{\beta}(s)=(-y'(s),x'(s)),
\end{align*}
则右手单位正交标架$\{\bm{r}(s);\bm{\alpha}(s),\bm{\beta}(s)\}$称为平面曲线$C$的$\mathbf{Frenet}$\textbf{标架}。

由于$\bm{\alpha}(s)$为单位向量场，故$\bm{\alpha}(s)\perp\bm{\alpha}'(s)$，因此存在函数$\kappa_r(s)$使得
\begin{align*}
\bm{\alpha}'(s)=\kappa_r(s)\bm{\beta}(s),
\end{align*}
称$\kappa_r(s)$为平面曲线$C$的\textbf{相对曲率}。
\end{definition}
\begin{remark}
$\bm{\beta}(s)$是曲线$C$的法向量与$\bm{\alpha}'(s)$共线,它与曲线$C$的主法向量可能只差一个正负号.
\end{remark}

\begin{theorem}\label{theorem:相对曲率公式}
平面正则曲线$C: \bm{r}(s)=(x(s),y(s))$的相对曲率$\kappa_r(s)$满足
\begin{align*}
\kappa_r(s)=x'(s)y''(s)-x''(s)y'(s)=\begin{vmatrix}
x'(s) & y'(s) \\
x''(s) & y''(s)
\end{vmatrix}.
\end{align*}
如果曲线$C$的曲率为$\kappa(s)$,则有$$\kappa_{r}(s)=\pm \kappa(s).$$
其中“+”号表示平面曲线$C$朝其单位法向量$\bm{\beta}(s)$所指的方向弯曲，$\bm{\beta}(s)$恰好是曲线$C$的主法向量$\bm{n}(s)$，而“$-$”号表示曲线朝$\bm{\beta}(s)$所指的相反方向弯曲，曲线$C$的主法向量$\bm{n}(s)$是$-\bm{\beta}(s)$（见\reffig{figure:平面曲线}）.此即
\begin{gather*}
\kappa_r(s)=\kappa(s)\iff \bm{\beta}(s)=\bm{n}(s),
\\
\kappa_r(s)=-\kappa(s)\iff \bm{\beta}(s)=-\bm{n}(s).
\end{gather*}
\end{theorem}
\begin{proof}
由定义，$\kappa_r(s)=\bm{\alpha}'(s)\cdot\bm{\beta}(s)$。计算得$\bm{\alpha}'(s)=(x''(s),y''(s))$，因此
\begin{align*}
\kappa_r(s)&=(x''(s),y''(s))\cdot(-y'(s),x'(s)) 
=-x''(s)y'(s)+y''(s)x'(s) \\
&=x'(s)y''(s)-x''(s)y'(s) 
=\begin{vmatrix}
x'(s) & y'(s) \\
x''(s) & y''(s)
\end{vmatrix}.
\end{align*}
根据曲率和相对曲率的定义可立得
$$\kappa_{\tau}(s)=\pm \kappa(s).$$

\label{figure:平面曲线}
\end{figure}

\end{proof}

\begin{theorem}\label{theorem:平面曲线的Frenet公式和渐缩线公式}
设平面正则曲线$C:\bm{r}(s)=(x(s),y(s))$,其中$s$是弧长参数.记其单位切向量、单位法向量、相对曲率分别为$\bm{\alpha}(s)$、$\bm{\beta}(s)$、$\kappa_r(s)$,则平面曲线$C$的Frenet标架满足运动公式
\begin{align}\label{eq::20jrJIOUJFEIOJOJIUJN}
\begin{cases}
\bm{r}'(s)=\bm{\alpha}(s), \\
\bm{\alpha}'(s)=\kappa_r(s)\bm{\beta}(s), \\
\bm{\beta}'(s)=-\kappa_r(s)\bm{\alpha}(s).
\end{cases}
\end{align}
并且平面曲线 $C$ 在$s$处的曲率中心为$\boldsymbol{r}\left( s \right) +\frac{\boldsymbol{\beta }\left( s \right)}{\kappa _r\left( s \right)}$,进而曲线$C$的曲率中心的轨迹也是曲线$C$的渐缩线,参数方程为
\begin{align}
\bm{r}_1(s)=\boldsymbol{r}\left( s \right) +\frac{\boldsymbol{\beta }\left( s \right)}{\kappa _r\left( s \right)}. \label{eq:evolute}
\end{align}
若曲线$C$的参数方程为$\bm{r}=\bm{r}(t)$,其中$t$为任意参数.由于$\bm{\beta}$和$\kappa_r$与曲线的参数变换无关，故其渐缩线参数方程为
\begin{align*}
\bm{r}_1(t)=\boldsymbol{r}\left( t \right) +\frac{\boldsymbol{\beta }\left( t \right)}{\kappa _r\left( t \right)}. 
\end{align*}
\end{theorem}
\begin{proof}
\eqref{eq::20jrJIOUJFEIOJOJIUJN}式中第一式由弧长参数定义直接成立，\eqref{eq::20jrJIOUJFEIOJOJIUJN}式中第二式为相对曲率的定义。对$\bm{\beta}(s)=(-y'(s),x'(s))$求导，得
\begin{align*}
\bm{\beta}'(s)=(-y''(s),x''(s)).
\end{align*}
由\refthe{theorem:相对曲率公式}知$\kappa_r(s)=x'(s)y''(s)-x''(s)y'(s)$，结合$\bm{\alpha}(s)=(x'(s),y'(s))$得
\begin{align*}
-\kappa_r(s)\bm{\alpha}(s)&=-(x'(s)y''(s)-x''(s)y'(s))(x'(s),y'(s)) \\
&=(-x'(s)(x'(s)y''(s)-x''(s)y'(s)), -y'(s)(x'(s)y''(s)-x''(s)y'(s))) \\
&=(-(x'(s))^2y''(s)+x'(s)x''(s)y'(s), -x'(s)y'(s)y''(s)+x''(s)(y'(s))^2).
\end{align*}
另一方面，对$(x'(s))^2+(y'(s))^2=1$求导得
\begin{align*}
2x'(s)x''(s)+2y'(s)y''(s)=0\Longrightarrow x'(s)x''(s)=-y'(s)y''(s).
\end{align*}
因此
\begin{gather*}
-(x'(s))^2y''(s)+x'(s)x''(s)y'(s)=-(x'(s))^2y''(s)+(-y'(s)y''(s))y'(s) 
=-y''(s)((x'(s))^2+(y'(s))^2) 
=-y''(s), \\
-x'(s)y'(s)y''(s)+x''(s)(y'(s))^2=x''(s)(y'(s))^2 + x'(s)(x'(s)x''(s)) 
=x''(s)((y'(s))^2+(x'(s))^2) 
=x''(s).
\end{gather*}
故$\bm{\beta}'(s)=(-y''(s),x''(s))=-\kappa_r(s)\bm{\alpha}(s)$，\eqref{eq::20jrJIOUJFEIOJOJIUJN}式中第三式得证。

设曲线$C$的曲率为$\kappa(s)$，主法向量为$\bm{n}(s)$，则由\refthe{theorem:相对曲率公式}和曲率中心的定义知，曲线$C$在$s$处的曲率中心为
\begin{align*}
\bm{r}_1(s)=\bm{r}(s)+\frac{\bm{n}(s)}{\kappa(s)}=\bm{r}(s)+\frac{\pm\bm{\beta}(s)}{\pm\kappa_r(s)}=\bm{r}(s)+\frac{\bm{\beta}(s)}{\kappa_r(s)}.
\end{align*}
再由\hyperref[theorem:曲线渐缩线的参数方程]{曲线渐缩线的参数方程}和\refthe{theorem:挠率为0等价于是平面曲线}知曲线$C$的渐缩线方程为
\begin{align*}
\bm{r}_2(s)=\bm{r}(s)+\frac{\bm{n}(s)}{\kappa(s)}=\bm{r}(s)+\frac{\bm{\beta}(s)}{\kappa_r(s)}.
\end{align*}
故$\bm{r}_1(s)=\bm{r}_2(s)$，即曲线$C$的曲率中心轨迹就是曲线$C$的渐缩线.

\end{proof}

\begin{theorem}\label{theorem:平面曲线一般参数公式}
设平面正则曲线 \(C\) 的参数方程为$\boldsymbol{r}(t) = (x(t),\, y(t)), \quad t \in I$.则其弧长元素为
\[
\mathrm{d}s=\left| \boldsymbol{r}' \left( t \right) \right|\mathrm{d}t=\sqrt{x'^2\left( t \right) +y'^2\left( t \right)}\mathrm{d}t,
\]
单位切向量为
\[
\boldsymbol{\alpha }\left( t \right) =\frac{\boldsymbol{r}^{\prime}\left( t \right)}{\left| \boldsymbol{r}^{\prime}\left( t \right) \right|}=\left( \frac{x^{\prime}\left( t \right)}{\sqrt{{x^{\prime}}^2\left( t \right) +{y^{\prime}}^2\left( t \right)}},\;\frac{y^{\prime}\left( t \right)}{\sqrt{{x^{\prime}}^2\left( t \right) +{y^{\prime}}^2\left( t \right)}} \right) ,
\]
单位法向量为
\[
\boldsymbol{\beta }\left( t \right) =\left( -\frac{y^{\prime}\left( t \right)}{\sqrt{{x^{\prime}}^2\left( t \right) +{y^{\prime}}^2\left( t \right)}},\;\frac{x^{\prime}\left( t \right)}{\sqrt{{x^{\prime}}^2\left( t \right) +{y^{\prime}}^2\left( t \right)}} \right) ,
\]
相对曲率为
\[
\kappa_r(t) = \frac{x'(t)y''(t) - x''(t)y'(t)}{\bigl(x'^2(t) + y'^2(t)\bigr)^{\frac{3}{2}}}.
\]
\end{theorem}
\begin{proof}
由弧长元素定义知
\begin{align*}
\mathrm{d}s=|\bm{r}'(t)|\mathrm{d}t=\sqrt{x'^2(t)+y'^2(t)}\mathrm{d}t,
\end{align*}
进而$\frac{\mathrm{d}t}{\mathrm{d}s}=\frac{1}{\sqrt{x'^2(t)+y'^2(t)}}.$由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}可得
\begin{align*}
\bm{\alpha}(t)=\frac{\bm{r}'(t)}{|\bm{r}'(t)|}=\left( \frac{x'(t)}{\sqrt{x'^2(t)+y'^2(t)}},\quad\frac{y'(t)}{\sqrt{x'^2(t)+y'^2(t)}} \right).
\end{align*}
由单位法向量的定义即得
\begin{align*}
\bm{\beta}(t)=\left( -\frac{y'(t)}{\sqrt{x'^2(t)+y'^2(t)}},\quad\frac{x'(t)}{\sqrt{x'^2(t)+y'^2(t)}} \right).
\end{align*}
再由求导的链式法则和相对曲率的定义可得
\begin{align*}
\kappa_r(s)=\bm{\alpha}'(s)\cdot\bm{\beta}(s)=\frac{\mathrm{d}\bm{\alpha}(t)}{\mathrm{d}t}\cdot\frac{\mathrm{d}t}{\mathrm{d}s}\cdot\bm{\beta}(s)=\frac{x'(t)y''(t)-x''(t)y'(t)}{\bigl( x'^2(t)+y'^2(t) \bigr)^{\frac{3}{2}}}.
\end{align*}

\end{proof}

\begin{definition}
设平面正则曲线$C:\bm{r}=\bm{r}(s)$,其中$s$是弧长参数.记其单位切向量为$\bm{\alpha}(s)$.再设$\theta(s)$为单位切向量$\bm{\alpha}(s)$与$x$轴正向的夹角，称为$\bm{\alpha}(s)$的\textbf{方向角}。方向角是一个多值函数，但在$s$的邻域内，总可取出函数$\theta(s)$的一个连续分支，称为曲线$C$的\textbf{转角}。
\end{definition}

\begin{theorem}\label{theorem:平面曲线转角和相对曲率参数方程的联系}
设平面正则曲线$C:\bm{r}(s)=(x(s),y(s))$,其中$s$是弧长参数.记其单位切向量、单位法向量分别为$\bm{\alpha}(s)$、$\bm{\beta}(s)$,则平面正则曲线$C$的相对曲率$\kappa_r(s)$等于其转角$\theta(s)$对弧长参数$s$的导数，即
\begin{align*}
\kappa_r(s)=\theta'(s).
\end{align*}
进而有
\begin{align*}
\theta \left( s \right) &= \theta \left( s_0 \right) +\int_{s_0}^s{\kappa _r\left( s \right) \mathrm{d}s},
\\
x\left( s \right) &= x\left( s_0 \right) +\int_{s_0}^s{\cos \theta \left( s \right) \mathrm{d}s},
\\
y\left( s \right) &= y\left( s_0 \right) +\int_{s_0}^s{\sin \theta \left( s \right) \mathrm{d}s}.
\end{align*}
\end{theorem}
\begin{remark}
这个结论清晰地说明了平面曲线相对曲率的几何意义.
\end{remark}
\begin{proof}
由方向角的定义，单位切向量可表示为
\begin{align*}
\bm{\alpha}(s)=(\cos\theta(s),\sin\theta(s)).
\end{align*}
对$s$求导，得
\begin{align*}
\bm{\alpha}'(s)=(-\sin\theta(s)\cdot\theta'(s),\cos\theta(s)\cdot\theta'(s))=\theta'(s)(-\sin\theta(s),\cos\theta(s)).
\end{align*}
另一方面，由相对曲率的定义$\bm{\alpha}'(s)=\kappa_r(s)\bm{\beta}(s)$，且$\bm{\beta}(s)=(-y'(s),x'(s))=(-\sin\theta(s),\cos\theta(s))$，因此
\begin{align*}
\kappa_r(s)\bm{\beta}(s)=\theta'(s)\bm{\beta}(s).
\end{align*}
由于$\bm{\beta}(s)$为单位向量，非零，故$\kappa_r(s)=\theta'(s)$。再分别对$\kappa_r(s)=\theta'(s)$和$\bm{r}'(s)=\alpha(s)$积分即得
\begin{align*}
\theta \left( s \right) = \theta \left( s_0 \right) +\int_{s_0}^s{\kappa _r\left( s \right) \mathrm{d}s},
\quad
x\left( s \right) = x\left( s_0 \right) +\int_{s_0}^s{\cos \theta \left( s \right) \mathrm{d}s},
\quad
y\left( s \right) = y\left( s_0 \right) +\int_{s_0}^s{\sin \theta \left( s \right) \mathrm{d}s}.
\end{align*}

\end{proof}

\begin{definition}[光滑闭曲线]
设 $\bm{r}=\bm{r}(s), a\leqslant s\leqslant b$ 是 $E^2$ 上的一条光滑曲线，并且满足
\[
\bm{r}(b)=\bm{r}(a),\quad \bm{r}'(b)=\bm{r}'(a),\quad \bm{r}''(b)=\bm{r}''(a),\quad \cdots
\]
则称它为\textbf{光滑闭曲线}.
\end{definition}

\begin{definition}[分段光滑闭曲线]
如果曲线 $\bm{r}=\bm{r}(s), a\leqslant s\leqslant b$ 是若干段光滑曲线首尾相接而成的，并且 $\bm{r}(b)=\bm{r}(a)$，则称它是\textbf{分段光滑的闭曲线}.
\end{definition}

\begin{definition}[简单闭曲线]
设 $\bm{r}=\bm{r}(s), a\leqslant s\leqslant b$ 是 $E^2$ 上的一条闭曲线，并且对于任意的 $a\leqslant s_1<s_2<b$，都有 $\bm{r}(s_1)\neq\bm{r}(s_2)$，则称该曲线是\textbf{简单闭曲线}或\textbf{简单的}. 简单闭曲线就是没有自交点的闭曲线.
\end{definition}

\begin{theorem}
对于连续可微曲线 $C:\bm{r}=\bm{r}(s),a\leqslant s\leqslant b$，可以在整个区间上选取单位切向量 $\bm{\alpha}(s)$ 的方向角 $\theta(s)$ 的连续分支. 则对任意两个这样的连续分支,都存在$k\in \mathbb{Z}$,使得这两个连续分支相差 $2k\pi$.因此方向角的总变差 $\theta(b)-\theta(a)$ 与连续分支的选取无关，且满足
\begin{align}
\theta(b)-\theta(a)=\int_a^b \kappa_r(s)\,\mathrm{d}s. \label{eq:total_variation}
\end{align}
\end{theorem}
\begin{proof}
在每一点处单位切向量 $\bm{\alpha}(s)$ 的方向角确定到相差 $2\pi$ 的整数倍. 将区间划分为充分小的子区间
\[
a=s_0<s_1<\cdots<s_n=b,
\]
使得在每一段小区间 $[s_i,s_{i+1}]$ 上方向角的一个连续分支 $\theta(s)$ 的变差不超过 $\pi$. 从 $[s_0,s_1]$ 的一个连续分支出发，依次唯一地确定各个区间上的连续分支，从而得到定义在整条曲线上的方向角连续分支. 若 $\theta(s)$ 与 $\tilde{\theta}(s)$ 是两个这样的连续分支，则对$\forall s$,都存在整数$k(s)$,使得 $\tilde{\theta}(s)-\theta(s)=2k(s)\pi$.又左边是 $s$ 的连续函数，故 $k(s)$ 为常数,记为$k$. 因此总变差与分支无关. 根据相对曲率的定义 $\kappa_r(s)=\frac{\mathrm{d}\theta}{\mathrm{d}s}$，积分即得 \eqref{eq:total_variation}.

\end{proof}

\begin{definition}[旋转指标]
设 $C:\bm{r}=\bm{r}(s),a\leqslant s\leqslant b$ 是一条连续可微的闭曲线，其方向角连续分支的总变差为 $\theta(b)-\theta(a)$，则称
\begin{align}
i(C)=\frac{1}{2\pi}\bigl(\theta(b)-\theta(a)\bigr) \label{eq:rotation_index}
\end{align}
为闭曲线 $C$ 的\textbf{旋转指标}.
\end{definition}

\begin{proposition}[分段光滑简单闭曲线的旋转指标]
设 $C$ 是分段光滑的简单闭曲线，角点处的参数为 $s_i$，外角 $\theta_i=\angle\bigl(\bm{\alpha}(s_i-0),\bm{\alpha}(s_i+0)\bigr)$ 满足 $-\pi<\theta_i<\pi$，则方向角的总变差为
\begin{align}
\theta(b)-\theta(a)=\int_a^b \kappa_r(s)\,\mathrm{d}s+\sum_i \theta_i. \label{eq:piecewise_total_variation}
\end{align}
此时旋转指标仍满足 $i(C)=\pm 1$.
\end{proposition}
\begin{proof}
在每个光滑段上应用 \eqref{eq:total_variation}，并将各段的方向角变化相加；在角点处方向角的跳跃恰好等于外角 $\theta_i$，因此总变差为各段曲率积分与外角和之和. 对于分段光滑的简单闭曲线，\hyperref[theorem:旋转指标定理]{旋转指标定理}依然成立,可通过光滑逼近或直接利用连续可微情形的结论得到.

\end{proof}

\item 双曲线: $r=(a\cosh t,b\sinh t)$, $-\infty<t<\infty$;
\item 抛物线: $r=(t,at^2)$, $-\infty<t<\infty$;
\item 摆线: $r=(a(t-\sin t),a(1-\cos t))$, $0\leqslant t\leqslant 2\pi$;
\item 悬链线: $r=(t,a\cosh t)$, $-\infty<t<\infty$;
\item 曳物线: $r=(a\cos t,a\ln(\sec t+\tan t)-a\sin t)$, $0\leqslant t<\frac{\pi}{2}$.
\end{enumerate}
\end{example}

\begin{enumerate}[(1)]

r'(t)=(-a\sin t,b\cos t),\quad r''(t)=(-a\cos t,-b\sin t).

\end{gather*}

r_1(t)=r(t)+\frac{\beta(t)}{\kappa_r(t)}=\left(\frac{a^2-b^2}{a}\cos^3 t,\frac{b^2-a^2}{b}\sin^3 t\right).
\end{align*}

r'(t)=(a\sinh t,b\cosh t),\quad r''(t)=(a\cosh t,b\sinh t).
\end{align*}

\begin{gather*}
\kappa_r(t)=-\frac{ab}{(a^2\sinh^2 t+b^2\cosh^2 t)^{\frac{3}{2}}},\\
\alpha(t)=\frac{r'(t)}{|r'(t)|}=\left(-\frac{a\sinh t}{\sqrt{a^2\sinh^2 t+b^2\cosh^2 t}},\frac{b\cosh t}{\sqrt{a^2\sinh^2 t+b^2\cosh^2 t}}\right),\\
\beta(t)=\left(-\frac{b\cosh t}{\sqrt{a^2\sinh^2 t+b^2\cosh^2 t}},\frac{a\sinh t}{\sqrt{a^2\sinh^2 t+b^2\cosh^2 t}}\right).
\end{gather*}
再利用\refthe{theorem:平面曲线的Frenet公式和渐缩线公式}
可得所求曲线的曲率中心轨迹方程为
\begin{align*}
r_1(t)=r(t)+\frac{\beta(t)}{\kappa_r(t)}=\left(\frac{a^2+b^2}{a}\cosh^3 t,-\frac{a^2+b^2}{b}\sinh^3 t\right).
\end{align*}

\item 由题设可得

r'(t)=(1,2at),\quad r''(t)=(0,2a).
\]
利用\refthe{theorem:平面曲线一般参数公式}可得所求曲线相对曲率、单位切向量、单位法向量分别为

\]

r'(t)=(a(1-\cos t),a\sin t),\quad r''(t)=(a\sin t,a\cos t).

r_1(t)=r(t)+\frac{\beta(t)}{\kappa_r(t)}=\left(t-\frac{\sinh t(1+a^2\sinh^2t)}{a\cosh t},\;a\cosh t+\frac{1+a^2\sinh^2t}{a\cosh t}\right).

\quad
\alpha(t)=(-\cos t,\sin t),

\[

\]
\end{enumerate}

\end{solution}

\end{solution}

r(x) = \left( x,f(x) \right) ,\quad x\in \left[ a,b \right] .

\end{solution}

\begin{proposition}\label{proposition:满足微分方程的曲线的相对曲率}
已知平面正则曲线 $C$ 满足微分方程
\begin{align*}
P(x,y)\mathrm{d}x+Q(x,y)\mathrm{d}y=0,
\end{align*}
其中$P,Q\in C^1(\mathbb{R}^2)$.则它的相对曲率的表达式为
\begin{align*}
\kappa_r=\frac{P_xQ^2-PQ(P_y+Q_x)+P^2Q_y}{(P^2+Q^2)^{\frac{3}{2}}}.
\end{align*}
\end{proposition}
\begin{solution}
由曲线$C$的正则性知$P(x,y),Q(x,y)\neq0,\forall(x,y)\in\mathbb{R}^2.$设曲线$C$的参数方程为$r(\tilde{t})=(x(\tilde{t}),y(\tilde{t}))$，其中$\tilde{t}$为任意参数。则由条件知
\begin{align*}
P(x(\tilde{t}),y(\tilde{t}))\frac{\mathrm{d}x(\tilde{t})}{\mathrm{d}\tilde{t}}+Q(x(\tilde{t}),y(\tilde{t}))\frac{\mathrm{d}y(\tilde{t})}{\mathrm{d}\tilde{t}}=0.
\end{align*}
即曲线$C$在$\tilde{t}$处的切向量$(\frac{\mathrm{d}x(\tilde{t})}{\mathrm{d}\tilde{t}},\frac{\mathrm{d}y(\tilde{t})}{\mathrm{d}\tilde{t}})$与$(P(x(\tilde{t}),y(\tilde{t})),Q(x(\tilde{t}),y(\tilde{t})))$正交。
又注意到$(-Q(x(\tilde{t}),y(\tilde{t})),P(x(\tilde{t}),y(\tilde{t})))$也与$(P(x(\tilde{t}),y(\tilde{t})),Q(x(\tilde{t}),y(\tilde{t})))$正交，故$(\frac{\mathrm{d}x(\tilde{t})}{\mathrm{d}\tilde{t}},\frac{\mathrm{d}y(\tilde{t})}{\mathrm{d}\tilde{t}})$与$(-Q(x(\tilde{t}),y(\tilde{t})),P(x(\tilde{t}),y(\tilde{t})))$共线。于是可设
\begin{align*}
\left(\frac{\mathrm{d}x(\tilde{t})}{\mathrm{d}\tilde{t}},\frac{\mathrm{d}y(\tilde{t})}{\mathrm{d}\tilde{t}}\right)=\lambda(\tilde{t})\left(-Q(x(\tilde{t}),y(\tilde{t})),P(x(\tilde{t}),y(\tilde{t}))\right),
\end{align*}
其中$\lambda(\tilde{t})\neq 0,\forall \tilde{t}$.对上式两边同时与$(-Q,P)$点乘得
\begin{align*}
\lambda \left( \tilde{t} \right) \left( -Q,P \right) ^2=\left( \frac{\mathrm{d}x}{\mathrm{d}\tilde{t}},\frac{\mathrm{d}y}{\mathrm{d}\tilde{t}} \right) \cdot \left( -Q,P \right) \Longrightarrow \lambda \left( \tilde{t} \right) =\frac{\left( \frac{\mathrm{d}x}{\mathrm{d}\tilde{t}},\frac{\mathrm{d}y}{\mathrm{d}\tilde{t}} \right) \cdot \left( -Q,P \right)}{\left( -Q,P \right) ^2}=\frac{-Q\frac{\mathrm{d}x}{\mathrm{d}\tilde{t}}+P\frac{\mathrm{d}y}{\mathrm{d}\tilde{t}}}{P^2+Q^2}.
\end{align*}
故$\lambda(\tilde{t})$是连续可微函数.由介值定理知,$\lambda(\tilde{t})$恒不变号.令$t(\tilde{t})=\int_{\tilde{t}_0}^{\tilde{t}}\frac{1}{\lambda(u)}\mathrm{d}u$，则由$\lambda(u)$连续可微且不变号知，$t=t(\tilde{t})$是参数变换，且$\frac{\mathrm{d}t}{\mathrm{d}\tilde{t}}=\frac{1}{\lambda(\tilde{t})}$。记$t=t(\tilde{t})$的反函数为$\tilde{t}=\tilde{t}(t)$，再记
\begin{align*}
r(t)=r(\tilde{t})=r(\tilde{t}(t))=(x(\tilde{t}(t)),y(\tilde{t}(t)))=(x(t),y(t)).
\end{align*}
于是
\begin{align*}
\left(\frac{\mathrm{d}x(t)}{\mathrm{d}t},\frac{\mathrm{d}y(t)}{\mathrm{d}t}\right)=\frac{\mathrm{d}r(t)}{\mathrm{d}t}=\frac{\mathrm{d}r(t)}{\mathrm{d}\tilde{t}}\cdot \frac{\mathrm{d}\tilde{t}}{\mathrm{d}t}=\lambda(\tilde{t})(-Q,P)\cdot \frac{1}{\lambda(\tilde{t})}=(-Q,P).
\end{align*}
对上式关于$t$求导得
\begin{align*}
\frac{\mathrm{d}^2x(t)}{\mathrm{d}t^2}&=-\frac{\mathrm{d}}{\mathrm{d}t}Q(x(t),y(t))=Q_xQ-Q_yP,\\
\frac{\mathrm{d}^2y(t)}{\mathrm{d}t^2}&=\frac{\mathrm{d}}{\mathrm{d}t}P(x(t),y(t))=P_yP-P_xQ.
\end{align*}
再利用\refthe{theorem:平面曲线一般参数公式}可得曲线$C$在$t$处的相对曲率为
\begin{align*}
\kappa_r=\frac{\begin{vmatrix}
\frac{\mathrm{d}x(t)}{\mathrm{d}t}&\frac{\mathrm{d}y(t)}{\mathrm{d}t}\\
\frac{\mathrm{d}^2x(t)}{\mathrm{d}t^2}&\frac{\mathrm{d}^2y(t)}{\mathrm{d}t^2}
\end{vmatrix}}{\left(\left(\frac{\mathrm{d}x(t)}{\mathrm{d}t}\right)^2+\left(\frac{\mathrm{d}y(t)}{\mathrm{d}t}\right)^2\right)^{\frac{3}{2}}}=\frac{P_xQ^2-PQ(P_y+Q_x)+P^2Q_y}{(P^2+Q^2)^{\frac{3}{2}}}.
\end{align*}

\end{solution}

\begin{corollary}
已知平面正则曲线 $C$ 的隐式方程是 $\Phi(x,y)=0$, 其中$\Phi\in C^2(\mathbb{R}^2)$.则它的相对曲率的表达式为
\begin{align*}
\kappa _r=\frac{\Phi _{xx}{\Phi _y}^2-2\Phi _x\Phi _y\Phi _{xy}+{\Phi _x}^2\Phi _{yy}}{\left( \Phi _{x}^{2}+\Phi _{y}^{2} \right) ^{\frac{3}{2}}}.
\end{align*}
\end{corollary}
\begin{solution}
对$\Phi(x,y)=0$微分得
\begin{align*}
\Phi _x\left( x,y \right) \mathrm{d}x+\Phi _y\left( x,y \right) \mathrm{d}y=0.
\end{align*}
于是由\refpro{proposition:满足微分方程的曲线的相对曲率}可知曲线$C$的相对曲率为
\begin{align*}
\kappa _r=\frac{\Phi _{xx}{\Phi _y}^2-2\Phi _x\Phi _y\Phi _{xy}+{\Phi _x}^2\Phi _{yy}}{\left( \Phi _{x}^{2}+\Phi _{y}^{2} \right) ^{\frac{3}{2}}}.
\end{align*}

\end{solution}

\begin{example}
已知曲线的相对曲率为
\begin{enumerate}[(1)]
\item $\kappa_r(s)=\frac{1}{1+s^2}$;
\item $\kappa_r(s)=\frac{1}{1+s}$;
\item $\kappa_r(s)=\frac{1}{\sqrt{1-s^2}}$,
\end{enumerate}
其中 $s$ 是弧长参数, 求各条平面曲线的参数方程.
\end{example}
\begin{solution}
\begin{enumerate}[(1)]
\item 由\refthe{theorem:平面曲线转角和相对曲率参数方程的联系}知曲线在$s$处的转角为
\begin{align*}
\theta(s)=\theta(0)+\int_0^s\kappa_r(s)\mathrm{d}s=\int_0^s\frac{1}{1+s^2}\mathrm{d}s=\arctan s.
\end{align*}
再利用\refthe{theorem:平面曲线转角和相对曲率参数方程的联系}可得
\begin{align*}
&x(s)=x(0)+\int_0^s\cos\theta(s)\mathrm{d}s=\int_0^s\cos(\arctan s)\mathrm{d}s\xlongequal{t=\arctan s}\int_0^{\arctan s}\sec t\mathrm{d}t=\ln\left(\sec(\arctan s)+s\right),\\
&y(s)=y(0)+\int_0^s{\sin \theta (s)\mathrm{d}s}=\int_0^s{\mathrm{sin(}\arctan s)\mathrm{d}s}\xlongequal{t=\arctan s}\int_0^{\arctan s}{\frac{\sin t}{\cos ^2t}\mathrm{d}t}=\sec \left( \arctan s \right) -1.
\end{align*}
故所求平面的参数方程为
\begin{align*}
r(s)=\left(x(s),y(s)\right)=\left(\ln\left(\sec(\arctan s)+s\right),\sec \left( \arctan s \right) -1\right).
\end{align*}

\item 由\refthe{theorem:平面曲线转角和相对曲率参数方程的联系}知曲线在 \(s\) 处的转角为
\[
\theta(s)=\theta(0)+\int_0^s\kappa_r(s)\,\mathrm{d}s=\int_0^s\frac{1}{1+s}\,\mathrm{d}s=\ln(1+s).
\]
再利用\refthe{theorem:平面曲线转角和相对曲率参数方程的联系}可得
\[
x(s)=x(0)+\int_0^s\cos\theta(s)\,\mathrm{d}s=\int_0^s\cos(\ln(1+s))\,\mathrm{d}s,
\]
\[
y(s)=y(0)+\int_0^s\sin\theta(s)\,\mathrm{d}s=\int_0^s\sin(\ln(1+s))\,\mathrm{d}s.
\]
令 \(t=\ln(1+s)\)，则 \(s=\mathrm{e}^t-1\)，\(\mathrm{d}s=\mathrm{e}^t\mathrm{d}t\)，积分限 \(t\) 从 \(0\) 到 \(\ln(1+s)\)。于是
\[
x(s)=\int_0^{\ln(1+s)}\cos t\cdot\mathrm{e}^t\,\mathrm{d}t
=\left.\frac{\mathrm{e}^t}{2}(\cos t+\sin t)\right|_0^{\ln(1+s)}
=\frac{1}{2}\bigl[(1+s)(\cos(\ln(1+s))+\sin(\ln(1+s)))-1\bigr],
\]
\[
y(s)=\int_0^{\ln(1+s)}\sin t\cdot\mathrm{e}^t\,\mathrm{d}t
=\left.\frac{\mathrm{e}^t}{2}(\sin t-\cos t)\right|_0^{\ln(1+s)}
=\frac{1}{2}\bigl[(1+s)(\sin(\ln(1+s))-\cos(\ln(1+s)))+1\bigr].
\]
故所求平面的参数方程为
\[
r(s)=\bigl(x(s),y(s)\bigr)=\Bigl(\frac{1}{2}\bigl[(1+s)(\cos(\ln(1+s))+\sin(\ln(1+s)))-1\bigr],\;
\frac{1}{2}\bigl[(1+s)(\sin(\ln(1+s))-\cos(\ln(1+s)))+1\bigr]\Bigr).
\]

\item 由\refthe{theorem:平面曲线转角和相对曲率参数方程的联系}知曲线在 \(s\) 处的转角为
\[
\theta(s)=\theta(0)+\int_0^s\kappa_r(s)\,\mathrm{d}s=\int_0^s\frac{1}{\sqrt{1-s^2}}\,\mathrm{d}s=\arcsin s.
\]
再利用\refthe{theorem:平面曲线转角和相对曲率参数方程的联系}可得
\[
x(s)=x(0)+\int_0^s\cos\theta(s)\,\mathrm{d}s=\int_0^s\cos(\arcsin s)\,\mathrm{d}s=\int_0^s\sqrt{1-s^2}\,\mathrm{d}s=\frac{1}{2}\bigl(s\sqrt{1-s^2}+\arcsin s\bigr),
\]
\[
y\left( s \right) =y\left( 0 \right) +\int_0^s{\sin \theta \left( s \right) \mathrm{d}s}=\int_0^s{\sin \left( \arcsin s \right) \mathrm{d}s}=\int_0^s{s\mathrm{d}s}=\frac{s^2}{2}.
\]
故所求平面的参数方程为
\[
r(s)=\bigl(x(s),y(s)\bigr)=\Bigl(\frac{1}{2}\bigl(s\sqrt{1-s^2}+\arcsin s\bigr),\;\frac{s^2}{2}\Bigr).
\]
\end{enumerate}

\end{solution}

\begin{example}
求下列曲线的渐伸线:
\begin{enumerate}[(1)]
\item 圆: $x^2+y^2=a^2$;
\item 摆线: $r=(a(t-\sin t),a(1-\cos t))$, $0\leqslant t\leqslant 2\pi$;
\item 悬链线: $y=a\cosh x$.
\end{enumerate}
\end{example}
\begin{solution}
\begin{enumerate}[(1)]
\item 取弧长参数化 \(\boldsymbol{r}(s)=(a\cos\frac{s}{a},\,a\sin\frac{s}{a})\)，则单位切向量 \(\boldsymbol{\alpha}(s)=(-\sin\frac{s}{a},\,\cos\frac{s}{a})\)。  
利用\hyperref[theorem:微分几何-定理7.3]{曲线渐伸线的参数方程}可得所求渐伸线方程为
\[
\boldsymbol{r}_1(s)=\bigl(a\cos\frac{s}{a}-(c-s)\sin\frac{s}{a},\; a\sin\frac{s}{a}+(c-s)\cos\frac{s}{a}\bigr).
\]
其中$c$为任意常数.

\item 计算弧长微分\(\dfrac{\mathrm{d}s}{\mathrm{d}t}=2a\sin\frac{t}{2}\)，故弧长\(s(t)=4a\bigl(1-\cos\frac{t}{2}\bigr)\),单位切向量为
\begin{align*}
\boldsymbol{\alpha }(t)=\frac{\boldsymbol{r}^{\prime}\left( t \right)}{\left| \boldsymbol{r}^{\prime}\left( t \right) \right|}=\frac{\left( a-a\cos t,a+a\sin t \right)}{4a\left( 1-\cos \frac{t}{2} \right)}=\bigl( \sin \frac{t}{2},\,\cos \frac{t}{2} \bigr) .
\end{align*} 
又曲线的单位切向量与参数变换无关,故利用\hyperref[theorem:微分几何-定理7.3]{曲线渐伸线的参数方程}可得所求渐伸线方程为
\begin{align*}
\boldsymbol{r}_1(t)&=\boldsymbol{r}(t)+\bigl( c-s(t) \bigr) \boldsymbol{\alpha }(t)
\\
&=\left( a(t-\sin t)+\bigl( c-4a(1-\cos \frac{t}{2}) \bigr) \sin \frac{t}{2},a(1-\cos t)+\bigl( c-4a(1-\cos \frac{t}{2}) \bigr) \cos \frac{t}{2} \right) ,
\end{align*}
其中$c$为任意常数.

\item 由题可知曲线参数方程为\(\boldsymbol{r}(t)=\bigl(t,\,a\cosh\frac{t}{a}\bigr)\)，其中$t$为任意参数.则 \(\dfrac{\mathrm{d}s}{\mathrm{d}t}=\cosh\frac{t}{a}\)，弧长 \(s(t)=a\sinh\frac{t}{a}\)。  
单位切向量为
\begin{align*}
\boldsymbol{\alpha }(t)=\frac{\boldsymbol{r}^{\prime}\left( t \right)}{\left| \boldsymbol{r}^{\prime}\left( t \right) \right|}=\frac{\left( 1,\sinh \frac{t}{a} \right)}{a\sinh \frac{t}{a}}=\bigl( \frac{1}{\cosh \frac{t}{a}},\,\tanh \frac{t}{a} \bigr) .
\end{align*}  
又曲线的单位切向量与参数变换无关,故利用\hyperref[theorem:微分几何-定理7.3]{曲线渐伸线的参数方程}可得所求渐伸线方程为
\begin{align*}
\boldsymbol{r}_1(t)=\boldsymbol{r}(t)+\bigl( c-s(t) \bigr) \boldsymbol{\alpha }(t)=\left( t+\frac{1}{\cosh \frac{t}{a}}-a\tanh \frac{t}{a},a\frac{1}{\cosh \frac{t}{a}}+c\tanh \frac{t}{a} \right) ,
\end{align*}
其中$c$为任意常数.
\end{enumerate}


\end{solution}
