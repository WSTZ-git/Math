\section{正则参数曲线}

\begin{definition}[参数曲线]
设区间$[a,b]\subset \mathbb{R}$，将连续映射
\[
\bm{r}: [a,b] \to \mathbb{R}^3,\quad t\mapsto \bm{r}(t)=\big(x(t),y(t),z(t)\big)
\]
称为\textbf{参数曲线}，,称$t$为\textbf{参数},还把参数$t$增大的方向称为该参数曲线的\textbf{正向}。

若$\bm{r}(t)$还满足$x(t),y(t),z(t)$均为连续可微函数，则称$\bm{r}(t)$为$\mathbb{R}^3$中的\textbf{连续可微参数曲线}.
\end{definition}

\begin{definition}[切向量与正则点]
设$\bm{r}(t)$是连续可微参数曲线，根据导数的定义，将
\[
\bm{r}'(t_0) =\lim_{\Delta t\rightarrow 0} \frac{r\left( t_0+\Delta t \right) -r\left( t \right)}{\Delta t} = \left(x'(t_0),y'(t_0),z'(t_0)\right).
\]
称为参数曲线$\bm{r}(t)$在$t_0$处的\textbf{切向量}。

若$\bm{r}'(t_0)\neq \bm{0}$,则称$t_0$为\textbf{正则点}。若$\bm{r}'(t)=\bm{0}$,则称$t_0$为\textbf{非正则点}。

\end{definition}

\begin{proposition}
设$\bm{r}(t)$是连续可微参数曲线，$t_0$是$\bm{r}(t)$的一个正则点，则
\begin{enumerate}[(1)]
\item $\bm{r}'(t_0)$的方向正好指向曲线$\bm{r}(t)$的正向.

\item 曲线在正则点$t_0$的切线$\boldsymbol{X}\left( u \right)$的方程是
\begin{align*}
\boldsymbol{X}\left( u \right) =\bm{r}\left( t_0 \right) +\bm{r}'\left( t_0 \right)(u-t_0) ,\quad \forall u\in \mathbb{R} .
\end{align*}
\end{enumerate}
\end{proposition}
\begin{proof}
定义自明.

\end{proof}

\begin{definition}[正则参数曲线和参数变换]
若连续可微参数曲线$\bm{r}(t)$满足
\begin{enumerate}[(1)]
\item $\bm{r}(t)$至少是自变量$t$的三次以上连续可微的向量函数；

\item 处处是正则点，即$\bm{r}'(t)\neq \bm{0}$，$\forall t\in[a,b]$.
\end{enumerate}
则称曲线$\bm{r}(t)$为\textbf{正则参数曲线}.此时还称$\bm{r}'(t)$为曲线$\bm{r}(t)$的\textbf{切向量场},$\frac{\bm{r}'(t)}{|\bm{r}'(t)|}$为曲线$\bm{r}(t)$的\textbf{单位切向量场}.

设$t=t(u)$是$\mathbb{R}$上的变换，且满足
\begin{enumerate}[(1)]
\item $t=t(u)$是$\mathbb{R}$上的双射;

\item $t(u)$是$u$的三次以上连续可微函数;

\item $t'(u)$处处不为零.
\end{enumerate}
则称$t=t(u)$是\textbf{容许的参数变换}或\textbf{参数变换}.
\end{definition}
\begin{remark}
要求$\bm{r}(t)$至少三次以上连续可微因为：曲线的几何不变量涉及$\bm{r}(t)$的三次导数。
\end{remark}

\begin{theorem}
设$\bm{r_1}(u),\bm{r_2}(u)$是正则参数曲线，记关系$\sim_1,\,\sim_2$分别为
\begin{gather*}
\boldsymbol{r}_1(u)\sim_1 \boldsymbol{r}_2(u)\Longleftrightarrow \text{存在参数变换}t=t\left( u \right) ,\text{使}\boldsymbol{r}_1\left( u \right) =\boldsymbol{r}_2\left( t\left( u \right) \right),
\\
\boldsymbol{r}_1(u)\sim_2 \boldsymbol{r}_2(u)\Longleftrightarrow \text{存在参数变换}t=t\left( u \right) ,\text{满足}t\prime \left( u \right) >0\left( \forall u\in \mathbb{R} \right) ,\text{使}\boldsymbol{r}_1\left( u \right) =\boldsymbol{r}_2\left( t\left( u \right) \right) .
\end{gather*}
则关系$\sim_1,\,\sim_2$都是等价关系.由关系$\sim_1$确定的一个等价类称为\textbf{一条正则曲线}.
由关系$\sim_2$确定的一个等价类称为\textbf{一条有向的正则曲线}.
\end{theorem}
\begin{note}
关系$\sim_1$中的参数变换保持曲线的定向不变.
\end{note}
\begin{proof}
只需注意到
\begin{align*}
r^{\prime}\left( u \right) =\frac{\mathrm{d}}{\mathrm{d}u}r\left( t\left( u \right) \right) =r^{\prime}\left( t\left( u \right) \right) \cdot t^{\prime}\left( u \right) ,
\end{align*}
再结合定义立得.

\end{proof}

\begin{theorem}[正则参数曲线的显式形式]
设$r(t)=(x(t),y(t),z(t))$是一条正则参数曲线，即$r'(t)\neq \bm{0}$. 若在点$t_0$处有$x'(t_0)\neq \bm{0}$，则存在$\varepsilon>0$使得在区间$(t_0-\varepsilon,t_0+\varepsilon)$上曲线可以表示为
\begin{align}\label{eq::-2jrj03j0wjo8f0jwfdsjge389hfgjp4e}
r(x)=(x,y(x),z(x)),
\end{align}
其中$y(x),z(x)$是$x$的函数.
\end{theorem}
\begin{proof}
由$x'(t_0)\neq 0$及\hyperref[Basis of Analytics-theorem:反函数存在定理]{反函数存在定理}，存在局部反函数$t=t(x)$.显然$t=t(x)$是参数变换,代入原参数方程即得$r(t(x))=(x,y(t(x)),z(t(x)))$，记$y(x)=y(t(x))$，$z(x)=z(t(x))$，即为所求.

\end{proof}

\begin{theorem}[正则参数曲线的隐式形式]
设$F,G\in C^k(\mathbb{R}^3)(k\geqslant 3)$,曲线$C$由方程组
\begin{gather}\label{eq:asfdsteyergrh}
\begin{cases}
f(x,y,z)=0,\\
g(x,y,z)=0
\end{cases}
\end{gather}
确定，且点$p=(x_0,y_0,z_0)$满足方程组$\eqref{eq:asfdsteyergrh}$. 若矩阵
\begin{align*}
\begin{pmatrix}
\frac{\partial f}{\partial x} & \frac{\partial f}{\partial y} & \frac{\partial f}{\partial z}\\[2mm]
\frac{\partial g}{\partial x} & \frac{\partial g}{\partial y} & \frac{\partial g}{\partial z}
\end{pmatrix}
\end{align*}
在点$p$处的秩为$2$，则方程组$\eqref{eq:asfdsteyergrh}$在$p$的一个邻域内的解构成一条正则曲线，且可以表示为形如$\eqref{eq::-2jrj03j0wjo8f0jwfdsjge389hfgjp4e}$式的显式形式.

例如，若
\begin{align*}
\begin{vmatrix}
\frac{\partial f}{\partial y} & \frac{\partial f}{\partial z}\\[2mm]
\frac{\partial g}{\partial y} & \frac{\partial g}{\partial z}
\end{vmatrix}\neq 0,
\end{align*}
则存在$\delta>0$和唯一的函数
\begin{align*}
y(x):(x_0-\delta ,x_0+\delta )\rightarrow (y_0-\delta ,y_0+\delta ),
\quad z(x):(x_0-\delta ,x_0+\delta )\rightarrow (z_0-\delta ,z_0+\delta ),
\end{align*}
且$y(x),z(x)\in C^k(\mathbb{R}^3)$,使得
\begin{align*}
y=y(x),\quad z=z(x),\quad \forall x\in (x_0-\delta,x_0+\delta).
\end{align*}
此时曲线参数方程为$r(x)=(x,y(x),z(x))$. 
\end{theorem}
\begin{proof}
由\hyperref[Basis of Analytics-theorem:隐函数组定理多变量一般形式]{隐函数定理}立得$y=y(x),z=z(x)$的存在性,从而得到曲线的显式形式$r(x)=(x,y(x),z(x))$. 

\end{proof}

\begin{example}
将一个半径为$r$的圆盘在$Oxy$平面上沿$x$轴作无滑动的滚动，写出圆盘上与中心的距离为$a(0 < a \leqslant r)$的一点随圆盘的滚动所描出轨迹的方程.
\end{example}
\begin{solution}
设初始位置时，圆盘的圆心$M$坐标为$(0,r)$，滚动点$A$的坐标为$(0,r+a)$，
当$MA$转过的弧度为$\theta$时，记此时圆心为$M_\theta$，滚动点为$A_\theta$，则由几何关系可得
\begin{align*}
\overrightarrow{M_{\theta}A_{\theta}}=(a\sin\theta,a\cos\theta),\quad \overrightarrow{MM_{\theta}}=(r\theta,0).
\end{align*}
进而
\begin{align*}
\overrightarrow{OA_{\theta}}=\overrightarrow{OM}+\overrightarrow{MM_{\theta}}+\overrightarrow{M_{\theta}A_{\theta}}=(r\theta+a\sin\theta,r+a\cos\theta).
\end{align*}
故$A$点的轨迹方程为
\begin{align*}
\boldsymbol{r}(\theta)=(r\theta+a\sin\theta,r+a\cos\theta),
\end{align*}
其中$\theta$为参数.

\end{solution}

\begin{example}
求曲线$\bm{r}(t) = (2\sin t, t, \cos t)$在点$t = 0$和点$t = \frac{\pi}{2}$的切线的方程.
\end{example}
\begin{solution}
对$\bm{r}(t)$求导得其切向量为\begin{align*}
\bm{r}'(t)=(2\cos t,1,-\sin t)\neq\bm{0}.
\end{align*}
从而\begin{align*}
\bm{r}(0)=(0,0,1),\quad \bm{r}\left(\frac{\pi}{2}\right)=\left(2,\frac{\pi}{2},0\right),\quad \bm{r}'(0)=(2,1,0),\quad \bm{r}'\left(\frac{\pi}{2}\right)=(0,1,-1).
\end{align*}
故曲线$\bm{r}(t)$在点$t=0$和$t=\frac{\pi}{2}$处的切线方程分别为\begin{align*}
X_0(u)=\bm{r}(0)+u\bm{r}'(0)=(2u,u,1),\quad X_{\frac{\pi}{2}}(u)=\bm{r}\left(\frac{\pi}{2}\right)+u\bm{r}'\left(\frac{\pi}{2}\right)=\left(2,\frac{\pi}{2}+u,-u\right),
\end{align*}
其中$u$为参数.

\end{solution}

\begin{example}
设空间$E^3$中一条正则参数曲线$C$的方程是$\bm{r}(t)$，$p$是曲线$C$外的一个固定点，用$\rho(t)$表示点$p$到点$\bm{r}(t)$的距离. 证明：如果函数$\rho(t)$在$t_0$处达到它的极小值（或极大值），记$q = \bm{r}(t_0)$，则$\overrightarrow{pq} \perp \bm{r}'(t_0)$.
\end{example}
\begin{proof}
记$f(t)=\rho^2(t)=\left[r(t)-p\right]^2$，则
\begin{align*}
f'(t)=2\left[r(t)-p\right]r'(t).
\end{align*}
由题可知$t_0$是$f(t)$的极值点，故
\begin{align*}
f'(t_0)=2\left[r(t_0)-p\right]\cdot r'(t_0)=0.
\end{align*}
又$\overrightarrow{pq}\cdot r'(t_0)=\left[r(t_0)-p\right]\cdot r'(t_0)$，因此
\begin{align*}
\overrightarrow{pq}\cdot r'(t_0)=0\Longleftrightarrow \overrightarrow{pq}\bot r'(t_0).
\end{align*}

\end{proof}

\begin{example}
求曲线
\begin{align*}
\begin{cases}
x^2 + y^2 + z^2 = 1, & z \geqslant 0, \\
x^2 + y^2 = x
\end{cases}
\end{align*}
的参数方程.
\end{example}
\begin{solution}
{\color{blue}解法一:}由题设可知$z\in\left[0,1\right]$且\begin{align*}
x+z^2=1\Longrightarrow x=1-z^2,
\end{align*}进而\begin{align*}
x^2+y^2=\left(1-z^2\right)^2+y^2=1-z^2\Longrightarrow y=\pm\sqrt{1-z^2-\left(1-z^2\right)^2}=\pm z\sqrt{1-z^2}.
\end{align*}故曲线的参数方程为\begin{align*}
r(z)=\left(1-z^2,\pm z\sqrt{1-z^2},z\right),\quad z\in\left[0,1\right].
\end{align*}

{\color{blue}解法二:}由条件可设\begin{align*}
\begin{cases}
x=\frac{1}{2}+\frac{1}{2}\cos\theta\\
y=\frac{1}{2}\sin\theta
\end{cases},\quad \theta\in\left[0,2\pi\right].
\end{align*}故\begin{align*}
x^2+y^2+z^2=1\Longrightarrow z^2=1-\left(x^2+y^2\right)=\frac{1}{2}-\frac{1}{2}\cos\theta=\sin^2\frac{\theta}{2}\Longrightarrow z=\sin\frac{\theta}{2}.
\end{align*}因此曲线的参数方程为\begin{align*}
r(\theta)=\left(\frac{1}{2}+\frac{1}{2}\cos\theta,\frac{1}{2}\sin\theta,\sin\frac{\theta}{2}\right),\quad \theta\in\left[0,2\pi\right].
\end{align*}

\end{solution}

\begin{example}
设平面上一条正则参数曲线$C$的方程是$\bm{r}(t)(a \leqslant t \leqslant b)$，它与同一个平面上的一条直线$l$有公共点$p = \bm{r}(t_0)(a < t_0 < b)$，并且该曲线除点$p$外全部落在直线$l$的同一侧，证明：直线$l$是曲线$C$在点$p$的切线.
\end{example}
\begin{proof}
设直线$l$的法向量为$\overrightarrow{n}$，令$f(t)=(r(t)-r(t_0))\cdot\overrightarrow{n}$，则由条件可不妨设\begin{align*}
f(t)>0,\quad \forall t\in[a,b]\setminus\{t_0\}.
\end{align*}
又$f(t_0)=0$，故$t_0$为$f(t)$的极小值点，从而
\begin{align*}
f'(t_0)=r'(t_0)\cdot\overrightarrow{n}=0.
\end{align*}
因此$r'(t_0)$为直线$l$的方向向量，即直线$l$是曲线$C$在点$p$的切线.

\end{proof}

\begin{example}
设空间$E^3$中一条正则参数曲线$\bm{r}(t)$的切向量$\bm{r}'(t)$与一个固定的方向向量$\bm{a}$垂直，证明：该曲线落在一个平面内.
\end{example}
\begin{proof}
对$\forall t\in[a,b]$，由题设知$\boldsymbol{r}'(t)\cdot\boldsymbol{a}=0$，设$\boldsymbol{a}$的正交补空间为$\mathrm{span}\{\boldsymbol{b},\boldsymbol{c}\}$，则$\boldsymbol{r}'(t)\in\mathrm{span}\{\boldsymbol{b},\boldsymbol{c}\}$，进而存在不全为0的$\lambda(t),\mu(t)\in\mathbb{R}$，使得\begin{align*}
\boldsymbol{r}'(t)=\lambda(t)\boldsymbol{b}+\mu(t)\boldsymbol{c}.
\end{align*}
任取曲线$\boldsymbol{r}(t)$中一点$\boldsymbol{r}(t_0)$，再对上式两边同时积分得\begin{align*}
\boldsymbol{r}(t)-\boldsymbol{r}(t_0)=\int_{t_0}^t\boldsymbol{r}'(u)\mathrm{d}u=\boldsymbol{b}\int_{t_0}^t\lambda(u)\mathrm{d}u+\boldsymbol{c}\int_{t_0}^t\mu(u)\mathrm{d}u.
\end{align*}
再设平面$\pi$的参数方程为\begin{align*}
\boldsymbol{k}(u,v)=\boldsymbol{r}(t_0)+u\boldsymbol{b}+v\boldsymbol{c},\quad \forall u,v\in\mathbb{R}.
\end{align*}
于是对$\forall t\in[a,b]$，都有\begin{align*}
\boldsymbol{r}(t)=\boldsymbol{r}(t_0)+\boldsymbol{b}\int_{t_0}^t\lambda(u)\mathrm{d}u+\boldsymbol{c}\int_{t_0}^t\mu(u)\mathrm{d}u=\boldsymbol{k}\left(\int_{t_0}^t\lambda(u)\mathrm{d}u,\int_{t_0}^t\mu(u)\mathrm{d}u\right)\in\pi.
\end{align*}
故曲线$\boldsymbol{r}(t)$落在过点$\boldsymbol{r}(t_0)$且法向量为$\boldsymbol{a}$的平面$\pi$上.

\end{proof}
