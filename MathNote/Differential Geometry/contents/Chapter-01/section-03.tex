\section{曲线的曲率和Frenet标架}

\begin{theorem}\label{theorem:微分几何-定理3.1}
设$E^3$中的一条正则曲线的参数方程为$\bm{r}=\bm{r}(s)$,其中$s$是弧长参数.记$\bm{\alpha}(s)=\bm{r}'(s)$,用$\Delta\theta$表示切向量$\bm{\alpha}(s+\Delta s)$和$\bm{\alpha}(s)$之间的夹角,则$\bm{\alpha}(s)$为曲线$\bm{r}(s)$的单位切向量场,且
\begin{align*}
\lim\limits_{\Delta s\rightarrow 0}\left| \frac{\Delta \theta}{\Delta s} \right|=\left| \frac{\mathrm{d}\bm{\alpha}}{\mathrm{d}s} \right|.
\end{align*}
\end{theorem}
\begin{proof}
由\refthe{theorem:微分几何-定理2.1}可知,$\bm{\alpha}(s)$是曲线$\bm{r}(s)$的单位切向量场.
把曲线$C$上所有的单位切向量$\bm{\alpha}(s)$平行移动,使它们的起点都放在原点$O$处,则这些切向量的端点便描出单位球面上的一条曲线,于是切向量$\bm{\alpha}(s+\Delta s)$和$\bm{\alpha}(s)$之间的夹角$\Delta\theta$是在单位球面上从$\bm{\alpha}(s+\Delta s)$到$\bm{\alpha}(s)$的大圆弧的弧长(见\reffig{figure:曲线方向向量的转动}),
\begin{figure}[H]
\centering
\includegraphics[scale=0.3]{曲线方向向量的转动.png}
\caption{曲线方向向量的转动}
\label{figure:曲线方向向量的转动}
\end{figure}
而$|\bm{\alpha}(s+\Delta s)-\bm{\alpha}(s)|$正好是该角所对的弦长,所以由正弦定理可得
\begin{align*}
\frac{\left| \boldsymbol{\alpha }\left( s+\Delta s \right) -\boldsymbol{\alpha }\left( s \right) \right|}{\sin \Delta \theta}=\frac{1}{\sin \frac{\pi -\Delta \theta}{2}}\Longrightarrow \left| \boldsymbol{\alpha }\left( s+\Delta s \right) -\boldsymbol{\alpha }\left( s \right) \right|=\frac{\sin \Delta \theta}{\cos \frac{\Delta \theta}{2}}=2\left| \sin \frac{\Delta \theta}{2} \right|.
\end{align*}
于是
\begin{align*}
\left|\frac{\mathrm{d}\bm{\alpha}}{\mathrm{d}s}\right|
=\lim_{\Delta s\to 0}\frac{|\bm{\alpha}(s+\Delta s)-\bm{\alpha}(s)|}{|\Delta s|}
=\lim_{\Delta s\to 0}\frac{2\left|\sin\frac{\Delta\theta}{2}\right|}{|\Delta s|}
=\lim_{\Delta s\to 0}\left|\frac{\Delta\theta}{\Delta s}\right|.
\end{align*}

\end{proof}


\begin{definition}[曲率与曲率向量]
设$E^3$中的一条正则曲线的参数方程为$\bm{r}=\bm{r}(s)$,其中$s$是弧长参数.记$\bm{\alpha}(s)=\bm{r}'(s)$为曲线$\bm{r}(s)$的单位切向量场,命
\begin{align*}
\kappa \left( s \right) =\left| \frac{\mathrm{d}\boldsymbol{\alpha }}{\mathrm{d}s} \right|=|\bm{\alpha}'(s)|=\left| \boldsymbol{r}'' \left( s \right) \right|,
\end{align*}
称为$\kappa \left( s \right)$为曲线$\bm{r}(s)$在$s$处的\textbf{曲率},并且称$\frac{\mathrm{d}\bm{\alpha}}{\mathrm{d}s}$为该曲线的\textbf{曲率向量}.
\end{definition}

\begin{theorem}\label{theorem:正则曲线是直线等价于曲率为0}
一条正则曲线$C$是直线当且仅当它的曲率$\kappa(s)\equiv 0$.
\end{theorem}
\begin{proof}
设直线$C$的参数方程是$\bm{r}(s)=\bm{r}_0+\bm{\alpha}_0 s$,其中$\bm{\alpha}_0$是该直线的方向向量.

{\heiti 必要性:}
$\bm{r}'(s)=\bm{\alpha}_0$,$\bm{r}''(s)=\bm{0}$,故$\kappa(s)=|\bm{r}''(s)|\equiv 0$.

{\heiti 充分性:}由$\kappa(s)=\bm{r}''(s)\equiv 0$可知
\begin{align*}
\int_{s_0}^s \bm{r}''(s)\mathrm{d}s=0\Longrightarrow \bm{r}'(s)=\bm{r}'(s_0).
\end{align*}
再对上式两边积分得
\begin{align*}
\int_{s_0}^s \bm{r}'(s)\mathrm{d}s=\int_{s_0}^s \bm{r}'(s_0)\mathrm{d}s\Longrightarrow \bm{r}(s)=\bm{r}(s_0)+\bm{r}'(s_0)(s-s_0).
\end{align*}
故曲线$C$是一条直线.

\end{proof}



\begin{theorem}
设$E^3$中的一条正则曲线的参数方程为$\bm{r}=\bm{r}(s)$,其中$s$是弧长参数.把曲线$C$的单位切向量$\bm{\alpha}(s)$平行移动到原点$O$,其端点所描出的曲线称为曲线$C$的\textbf{切线像},它的参数方程是
\begin{align}
\bm{r} = \bm{\alpha}(s),
\label{eq:::-h39gj9e4ugjp39fds4}
\end{align}
切线像的弧长元素是
\begin{align*}
\mathrm{d}\tilde{s} = \kappa(s)\mathrm{d}s,
\end{align*}
进而
\begin{align*}
\kappa(s) = \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}.
\end{align*}
\end{theorem}
\begin{remark}
这就是说, 曲线的曲率$\kappa(s)$是曲线的切线像的弧长元素与曲线的弧长元素之比.
\end{remark}
\begin{proof}
由\eqref{eq:::-h39gj9e4ugjp39fds4}式可得
\begin{align*}
\mathrm{d}\tilde{s} = \left| \frac{\mathrm{d}\bm{\alpha}}{\mathrm{d}s} \right| \mathrm{d}s = \kappa(s)\mathrm{d}s,
\end{align*}
所以
\begin{align*}
\kappa(s) = \frac{\mathrm{d}\tilde{s}}{\mathrm{d}s}.
\end{align*}

\end{proof}

\begin{definition}[Frenet标架]\label{definition:Frenet标架的定义}
设一条正则曲线$C$的参数方程为$\bm{r}=\bm{r}(s)$,其中$s$为弧长参数，单位切向量场为$\bm{\alpha}(s)$，任取曲线$C$上一点$r(s)$,若曲率$\kappa(s) \neq 0$，则称单位向量
\begin{align*}
\bm{\beta}(s) = \frac{\bm{\alpha}'(s)}{\kappa(s)}
\end{align*}
为曲线$C$在点$r(s)$的\textbf{主法向量}，此时有
\[
\bm{\alpha}'(s) = \kappa(s)\bm{\beta}(s).
\]
进一步，定义单位向量
\begin{align*}
\bm{\gamma}(s) = \bm{\alpha}(s) \times \bm{\beta}(s)
\end{align*}
为曲线$C$在点$r(s)$的\textbf{次(副)法向量}。
称$\{\bm{r}(s);\bm{\alpha}(s),\bm{\beta}(s),\bm{\gamma}(s)\}$为曲线$C$在点$r(s)$的$\mathbf{Frenet}$\textbf{标架}。
\end{definition}
\begin{remark}
\begin{enumerate}
\item 曲线$C$在其上任意点的Frenet标架与表示曲线的笛卡尔坐标系选取无关,也不受曲线作保持定向的参数变换的影响.

\item 由$|\bm{\alpha}(s)|=1$，可得$\boldsymbol{\alpha }^2\left( s \right) =1$，再求导可得$\bm{\alpha}(s) \cdot \bm{\alpha}'(s)=0$，故$\bm{\alpha}'(s)$是曲线的法向量，从而主法向量$\bm{\beta}(s)$与切向量$\bm{\alpha}(s)$正交。

\item 次法向量$\bm{\gamma}(s)$由$\bm{\alpha}(s)$和$\bm{\beta}(s)$唯一确定，且三者构成右手单位正交标架。

\item 当$\kappa(s)=0$时，Frenet标架无定义；若在区间$(s_0-\varepsilon,s_0+\varepsilon)$内$\kappa(s)\equiv0$，则曲线段为直线，此时可选取两个正交的单位法向量$\bm{\beta},\bm{\gamma}$，使$\bm{\alpha},\bm{\beta},\bm{\gamma}$构成右手系并沿直线平行移动，得到直线的Frenet标架场。
\item 若$s_0$是$\kappa(s)$的孤立零点，当$s \to s_0\pm0$时Frenet标架场的极限存在且相同时，可将极限定义为标架场在$s_0$处的值,这就是说 Frenet 标架场可以延拓到 $\kappa(s)$ 的孤立零点 $s_0$. 如果在 $s \to s_0 \pm 0$ 时, Frenet 标架场的极限不是同一个, 则 Frenet 标架场不能够延拓到 $\kappa(s)$ 的孤立零点 $s_0$.
\end{enumerate}
\end{remark}

\begin{definition}\label{definition:Frenet标架定义}
设一条正则曲线$C$的参数方程为$\bm{r}=\bm{r}(s)$,其中$s$为弧长参数，单位切向量场为$\bm{\alpha}(s)$，在$\kappa(s)\neq0$的点，Frenet标架$\{\bm{r}(s);\bm{\alpha}(s),\bm{\beta}(s),\bm{\gamma}(s)\}$的三个坐标轴分别称为：
\begin{itemize}
\item \textbf{切线}：过$\bm{r}(s)$且以$\bm{\alpha}(s)$为方向向量的直线；
\item \textbf{主法线}：过$\bm{r}(s)$且以$\bm{\beta}(s)$为方向向量的直线；
\item \textbf{次法线}：过$\bm{r}(s)$且以$\bm{\gamma}(s)$为方向向量的直线。
\end{itemize}
三个坐标面分别为：
\begin{itemize}
\item \textbf{法平面}：过$\bm{r}(s)$且以$\bm{\alpha}(s)$为法向量的平面，方程为
\[
(\bm{X}-\bm{r}(s))\cdot\bm{\alpha}(s)=0;
\]
\item \textbf{从切平面}：过$\bm{r}(s)$且以$\bm{\beta}(s)$为法向量的平面，方程为
\[
(\bm{X}-\bm{r}(s))\cdot\bm{\beta}(s)=0;
\]
\item \textbf{密切平面}：过$\bm{r}(s)$且以$\bm{\gamma}(s)$为法向量的平面，方程为
\[
(\bm{X}-\bm{r}(s))\cdot\bm{\gamma}(s)=0,
\]
其中$\bm{X}$为平面上动点的向径。
\end{itemize}
\end{definition}
\begin{remark}
在曲率处处不为零的正则曲线上，Frenet标架场内在地确定了曲线的局部几何结构，此时曲线可视为$E^3$中所有右手单位正交标架构成的6维空间中的曲线.
\end{remark}

\begin{theorem}[曲线论基本公式]\label{theorem:曲线论基本公式}
\begin{enumerate}
\item 设正则曲线 $C$ 的参数方程为 $\bm{r} = \bm{r}(s)$，其中 $s$ 为弧长参数.
若 $\kappa(s) \neq 0$，则其单位切向量场与曲率分别为
\begin{align*}
\bm{\alpha}(s) = \bm{r}'(s), \quad 
\kappa(s) = |\bm{\alpha}'(s)| = |\bm{r}''(s)|.
\end{align*}
主法向量场与次法向量场分别为
\begin{align*}
\bm{\beta}(s) = \frac{\bm{r}''(s)}{|\bm{r}''(s)|}, \quad
\bm{\gamma}(s) = \bm{\alpha}(s) \times \bm{\beta}(s) = \frac{\bm{r}'(s) \times \bm{r}''(s)}{|\bm{r}''(s)|}.
\end{align*}

\item 设正则曲线 $C$ 的参数方程为 $\bm{r} = \bm{r}(t)$，其中 $t$ 为任意参数，则弧长元素满足
\[
\frac{\mathrm{d}s}{\mathrm{d}t} = |\bm{r}'(t)|.
\]
单位切向量场为
\[
\bm{\alpha}(t) = \frac{\bm{r}'(t)}{|\bm{r}'(t)|}.
\]
进一步，曲率、次法向量场与主法向量场分别为
\begin{gather*}
\kappa(t) = \frac{|\bm{r}'(t) \times \bm{r}''(t)|}{|\bm{r}'(t)|^3}, \quad
\bm{\gamma}(t) = \frac{\bm{r}'(t) \times \bm{r}''(t)}{|\bm{r}'(t) \times \bm{r}''(t)|}, \\
\bm{\beta}(t) = \bm{\gamma}(t) \times \bm{\alpha}(t) = \frac{|\bm{r}'(t)|}{|\bm{r}'(t) \times \bm{r}''(t)|}\bm{r}''(t) - \frac{\bm{r}'(t) \cdot \bm{r}''(t)}{|\bm{r}'(t)| \cdot |\bm{r}'(t) \times \bm{r}''(t)|}\bm{r}'(t).
\end{gather*}
\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}
\item 由弧长参数的定义，曲线的单位切向量场满足
\begin{align}\label{eq::fjw3890fj20qw3jfkqw0fjkw390f1}
\bm{\alpha}(s) = \bm{r}'(s).
\end{align}
对上式关于弧长参数 $s$ 求导，得
\begin{align*}
\bm{\alpha}'(s) = \bm{r}''(s).
\end{align*}
根据曲率的定义 $\kappa(s) = |\bm{\alpha}'(s)|$，代入可得
\begin{align*}
\kappa(s) = |\bm{r}''(s)|.
\end{align*}
当 $\kappa(s) \neq 0$ 时，由主法向量的定义得
\begin{align}\label{eq::fjw3890fj20qw3jfkqw0fjkw390f}
\bm{\beta}(s) = \frac{\bm{\alpha}'(s)}{\kappa(s)}=\frac{\bm{r}''(s)}{|\bm{r}''(s)|}.
\end{align}
再根据次法向量的定义 $\bm{\gamma}(s) = \bm{\alpha}(s) \times \bm{\beta}(s)$，将\eqref{eq::fjw3890fj20qw3jfkqw0fjkw390f1}\eqref{eq::fjw3890fj20qw3jfkqw0fjkw390f}式代入,得
\begin{align*}
\bm{\gamma}(s) = \frac{\bm{r}'(s) \times \bm{r}''(s)}{|\bm{r}''(s)|}.
\end{align*}

\item 首先由弧长公式 $s(t)=\int_{t_0}^{t}|\bm{r}'(\tau)|\,\mathrm{d}\tau$ 两边对 $t$ 求导得
\begin{align}\label{eq:--2389f89ej98jw34gehdf21.3}
\frac{\mathrm{d}s}{\mathrm{d}t}=|\bm{r}'(t)|.
\tag{3}
\end{align}
利用链式法则 $\bm{r}'(t)=\dfrac{\mathrm{d}\bm{r}}{\mathrm{d}s}\dfrac{\mathrm{d}s}{\mathrm{d}t}$ 及单位切向量 $\bm{\alpha}(t)=\dfrac{\mathrm{d}\bm{r}}{\mathrm{d}s}$，并代入\eqref{eq:--2389f89ej98jw34gehdf21.3}得
\begin{align}\label{eq:--2389f89ej98jw34gehdf21.4}
\bm{\alpha}(t)=\frac{\bm{r}'(t)}{|\bm{r}'(t)|}.
\end{align}
对 $\bm{\alpha}(t)$ 求导：
\[
\bm{\alpha}'(t)=\frac{\mathrm{d}\bm{\alpha}}{\mathrm{d}s}\frac{\mathrm{d}s}{\mathrm{d}t}
=\frac{\mathrm{d}\bm{\alpha}}{\mathrm{d}s}\,|\bm{r}'(t)|.
\]
由主法向量的定义知
\[
\frac{\mathrm{d}\bm{\alpha}}{\mathrm{d}s}=\kappa(t)\,\bm{\beta}(t),
\]
从而
\begin{align}\label{eq:--2389f89ej98jw34gehdf21.5}
\bm{\alpha}'(t)=\kappa(t)\,|\bm{r}'(t)|\,\bm{\beta}(t).
\end{align}
由\eqref{eq:--2389f89ej98jw34gehdf21.4}式知$\bm{r}'(t)=|\bm{r}'(t)|\,\bm{\alpha}(t)$,两边对 $t$ 求导：
\[
\bm{r}''(t)=\frac{\mathrm{d}}{\mathrm{d}t}\bigl(|\bm{r}'(t)|\,\bm{\alpha}(t)\bigr)
=|\bm{r}'(t)|\,\bm{\alpha}'(t).
\]
于是
\[
\bm{r}'(t)\times\bm{r}''(t)=|\bm{r}'(t)|\,\bm{\alpha}(t)\times
|\bm{r}'(t)|\,\bm{\alpha}'(t)
=|\bm{r}'(t)|^2\,\bm{\alpha}(t)\times\bm{\alpha}'(t),
\]
又由\eqref{eq:--2389f89ej98jw34gehdf21.5}式得
\begin{align*}
\bm{\alpha}(t)\times\bm{\alpha}'(t)
=\kappa(t)\,|\bm{r}'(t)|\,\bigl(\bm{\alpha}(t)\times\bm{\beta}(t)\bigr)
=\kappa(t)\,|\bm{r}'(t)|\,\bm{\gamma}(t).
\end{align*}
因此
\begin{align}\label{eq:--2389f89ej98jw34gehdf21.6}
\bm{r}'(t)\times\bm{r}''(t)=|\bm{r}'(t)|^2\cdot\kappa(t)\,|\bm{r}'(t)|\,\bm{\gamma}(t)
=\kappa(t)\,|\bm{r}'(t)|^3\,\bm{\gamma}(t).
\end{align}
对\eqref{eq:--2389f89ej98jw34gehdf21.6}式两边取模，利用 $|\bm{\gamma}(t)|=1$ 得
\[
|\bm{r}'(t)\times\bm{r}''(t)|=\kappa(t)\,|\bm{r}'(t)|^3,
\]
因此曲率公式为
\[
\kappa(t)=\frac{|\bm{r}'(t)\times\bm{r}''(t)|}{|\bm{r}'(t)|^3}.
\]
由 \eqref{eq:--2389f89ej98jw34gehdf21.6}式知 $\bm{r}'(t)\times\bm{r}''(t)$ 与 $\bm{\gamma}(t)$ 方向相同，单位化即得次法向量
\[
\bm{\gamma}(t)=\frac{\bm{r}'(t)\times\bm{r}''(t)}{|\bm{r}'(t)\times\bm{r}''(t)|}.
\]
最后，由 Frenet 标架的正交右手系关系知 $\bm{\beta}(t)=\bm{\gamma}(t)\times\bm{\alpha}(t)$，代入 $\bm{\gamma}(t),\bm{\alpha}(t)$ 的表达式并利用向量三重积公式得
\[
\boldsymbol{\beta }\left( t \right) =\frac{(\boldsymbol{r}'\left( t \right) \times \boldsymbol{r}'' \left( t \right) )\times \boldsymbol{r}'\left( t \right)}{|\boldsymbol{r}'\left( t \right) \times \boldsymbol{r}'' \left( t \right) |\,|\boldsymbol{r}'\left( t \right) |}=\frac{|\boldsymbol{r}'\left( t \right) |^2\,\boldsymbol{r}'' \left( t \right) -(\boldsymbol{r}'\left( t \right) \cdot \boldsymbol{r}'' \left( t \right) )\,\boldsymbol{r}'\left( t \right)}{|\boldsymbol{r}'\left( t \right) \times \boldsymbol{r}'' \left( t \right) |\,|\boldsymbol{r}'\left( t \right) |},
\]
整理即得
\[
\bm{\beta}(t) = \bm{\gamma}(t) \times \bm{\alpha}(t) = \frac{|\bm{r}'(t)|}{|\bm{r}'(t) \times \bm{r}''(t)|}\bm{r}''(t) - \frac{\bm{r}'(t) \cdot \bm{r}''(t)}{|\bm{r}'(t)| \cdot |\bm{r}'(t) \times \bm{r}''(t)|}\bm{r}'(t).
\]
\end{enumerate}

\end{proof}

\begin{example}\label{example:微分几何-例题2.1}
求圆螺旋线 $\bm{r} = (a\cos t, a\sin t, bt)$ 的曲率和它的 Frenet 标架, 其中 $a,b$ 是常数, 且 $a>0$ .
\end{example}
\begin{solution}
对圆螺旋线的参数方程求导数得到
\begin{gather*}
\bm{r}'(t) = (-a\sin t, a\cos t, b), \\
\bm{r}''(t) = (-a\cos t, -a\sin t, 0), \\
\bm{r}'(t) \times \bm{r}''(t) = (ab\sin t, -ab\cos t, a^2),
\end{gather*}
因此
\begin{align*}
\kappa(t) = \frac{|\bm{r}'(t) \times \bm{r}''(t)|}{|\bm{r}'(t)|^3} = \frac{a}{a^2 + b^2},
\end{align*}
并且
\begin{align*}
\bm{\alpha}(t) &= \frac{\bm{r}'(t)}{|\bm{r}'(t)|} = \left(-\frac{a}{\sqrt{a^2 + b^2}}\sin t, \frac{a}{\sqrt{a^2 + b^2}}\cos t, \frac{b}{\sqrt{a^2 + b^2}}\right), \\
\bm{\gamma}(t) &= \frac{\bm{r}'(t) \times \bm{r}''(t)}{|\bm{r}'(t) \times \bm{r}''(t)|} = \left(\frac{b}{\sqrt{a^2 + b^2}}\sin t, -\frac{b}{\sqrt{a^2 + b^2}}\cos t, \frac{a}{\sqrt{a^2 + b^2}}\right), \\
\bm{\beta}(t) &= \bm{\gamma}(t) \times \bm{\alpha}(t) = (-\cos t, -\sin t, 0).
\end{align*}
因为
\begin{align*}
\frac{\mathrm{d}s}{\mathrm{d}t} = |\bm{r}'(t)| = \sqrt{a^2 + b^2}\Longrightarrow s = \sqrt{a^2 + b^2}t,
\end{align*}
顺便得到该圆螺旋线以弧长 $s$ 为参数的参数方程是
\begin{align*}
\bm{r} = \left(a\cos\frac{s}{\sqrt{a^2 + b^2}}, a\sin\frac{s}{\sqrt{a^2 + b^2}}, \frac{bs}{\sqrt{a^2 + b^2}}\right).
\end{align*}

\end{solution}

\begin{example}
求曲线 $C$
\[
\begin{cases}
x^2 + y^2 + z^2 = 1, \\
x^2 + y^2 = x
\end{cases}
\]
在 $(0,0,1)$ 处的曲率 $\kappa$ 和 Frenet 标架.
\end{example}
\begin{remark}
曲线 $C$ 是球面和圆柱面的交线, 由两部分组成. 我们所考虑的点落在上半球面内. 解此题的方法有两种. 一种方法是把该曲线在点 $(0,0,1)$ 的邻域内的部分用参数方程表示出来, 然后按照\refexa{example:微分几何-例题2.1}的办法进行计算. 例如, 我们可以把该曲线用参数方程表示为
\begin{align*}
\bm{r}(t) = \left(\frac{1}{2} + \frac{1}{2}\cos t, \frac{1}{2}\sin t, \sqrt{\frac{1}{2} - \frac{1}{2}\cos t}\right),
\end{align*}
点 $(0,0,1)$ 对应于参数 $t = \pi$. 但是, 有时候用参数方程表示两个曲面的交线比较复杂, 涉及解函数方程. 因此, 我们在此介绍第二种方法.
\end{remark}
\begin{solution}
假定曲线的参数方程是
\begin{align*}
\bm{r}(s) = (x(s), y(s), z(s)),
\end{align*}

其中$Q_k(t)$是关于$t$的有理函数。于是
\begin{align*}
\lim\limits_{t\rightarrow 0^-}\bm{r}(t) &= \lim\limits_{t\rightarrow 0^-}\left( e^{-\frac{1}{t^2}},t,0 \right) = \left( 0,0,0 \right), \quad \lim\limits_{t\rightarrow 0^+}\bm{r}(t) = \lim\limits_{t\rightarrow 0^+}\left( 0,t,e^{-\frac{1}{t^2}} \right) = \left( 0,0,0 \right), \\
\lim\limits_{t\rightarrow 0^-}\bm{r}'(t) &= \lim\limits_{t\rightarrow 0^-}\left( \frac{2}{t^3}e^{-\frac{1}{t^2}},1,0 \right) = \left( 0,1,0 \right), \quad \lim\limits_{t\rightarrow 0^+}\bm{r}'(t) = \lim\limits_{t\rightarrow 0^+}\left( 0,1,\frac{2}{t^3}e^{-\frac{1}{t^2}} \right) = \left( 0,1,0 \right), \\
\lim\limits_{t\rightarrow 0^-}\bm{r}^{(k)}(t) &= \lim\limits_{t\rightarrow 0^-}\left( Q_k(t) e^{-\frac{1}{t^2}},0,0 \right) = \left( 0,0,0 \right), \quad \lim\limits_{t\rightarrow 0^+}\bm{r}^{(k)}(t) = \lim\limits_{t\rightarrow 0^+}\left( 0,0,Q_k(t) e^{-\frac{1}{t^2}} \right) = \left( 0,0,0 \right).
\end{align*}
故$\bm{r}(t)$在$(0,0,0)$点任意有限次连续可微，且
\begin{align*}
\bm{r}'(0) = \left( 0,1,0 \right) \ne \bm{0}, \quad \bm{r}''(0) = \bm{0}, \quad |\bm{r}'(0)| = 1.
\end{align*}
故这个曲线是正则参数曲线，从而$\kappa(0) = |\bm{r}''(0)| = 0$。再由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}，当$t<0$时，我们有
\begin{align*}
\bm{\alpha}_1(t) &= \frac{\bm{r}'(t)}{|\bm{r}'(t)|} = \frac{\left( \frac{2}{t^3}e^{-\frac{1}{t^2}},1,0 \right)}{\left| \left( \frac{2}{t^3}e^{-\frac{1}{t^2}},1,0 \right) \right|} = \left( \frac{2e^{-\frac{1}{t^2}}}{t^3\left( 1+\frac{4}{t^6}e^{-\frac{2}{t^2}} \right)^{\frac{1}{2}}},\frac{1}{\left( 1+\frac{4}{t^6}e^{-\frac{2}{t^2}} \right)^{\frac{1}{2}}},0 \right), \\
\bm{\gamma}_1(t) &= \frac{\bm{r}'(t) \times \bm{r}''(t)}{|\bm{r}'(t) \times \bm{r}''(t)|} = \frac{\left( 0,0,\frac{6t^2-4}{t^6}e^{-\frac{1}{t^2}} \right)}{\left| \left( 0,0,\frac{6t^2-4}{t^6}e^{-\frac{1}{t^2}} \right) \right|} = \left( 0,0,\sgn\left( \frac{6t^2-4}{t^6}e^{-\frac{1}{t^2}} \right) \right), \\
\bm{\beta}_1(t) &= \bm{\gamma}_1(t) \times \bm{\alpha}_1(t) = \left( -\frac{\sgn\left( \frac{6t^2-4}{t^6}e^{-\frac{1}{t^2}} \right)}{\left( 1+\frac{4}{t^6}e^{-\frac{2}{t^2}} \right)^{\frac{1}{2}}},\frac{\sgn\left( \frac{6t^2-4}{t^6}e^{-\frac{1}{t^2}} \right) \cdot 2e^{-\frac{1}{t^2}}}{t^3\left( 1+\frac{4}{t^6}e^{-\frac{2}{t^2}} \right)^{\frac{1}{2}}},0 \right).
\end{align*}
故这条曲线在$t<0$时的Frenet标架场为$\left\{ \bm{r}(t);\bm{\alpha}_1(t),\bm{\beta}_1(t),\bm{\gamma}_1(t) \right\}$，且此时
\begin{align*}
\lim\limits_{t\rightarrow 0^-}\bm{\alpha}_1(t) = \left( 0,0,0 \right), \quad \lim\limits_{t\rightarrow 0^-}\bm{\gamma}_1(t) = \left( 0,0,0 \right), \quad
\lim_{t\rightarrow 0^-} \boldsymbol{\beta }_1(t)=\left( 1,0,0 \right) .
\end{align*}
当$t>0$时，我们有
\begin{align*}
\bm{\alpha}_2(t) &= \frac{\bm{r}'(t)}{|\bm{r}'(t)|} = \frac{\left( 0,1,\frac{2}{t^3}e^{-\frac{1}{t^2}} \right)}{\left| \left( 0,1,\frac{2}{t^3}e^{-\frac{1}{t^2}} \right) \right|} = \left( 0,\frac{1}{\left( 1+\frac{4}{t^6}e^{-\frac{2}{t^2}} \right)^{\frac{1}{2}}},\frac{2e^{-\frac{1}{t^2}}}{t^3\left( 1+\frac{4}{t^6}e^{-\frac{2}{t^2}} \right)^{\frac{1}{2}}} \right), \\
\bm{\gamma}_2(t) &= \frac{\bm{r}'(t) \times \bm{r}''(t)}{|\bm{r}'(t) \times \bm{r}''(t)|} = \frac{\left( \frac{4-6t^2}{t^6}e^{-\frac{1}{t^2}},0,0 \right)}{\left| \left( \frac{4-6t^2}{t^6}e^{-\frac{1}{t^2}},0,0 \right) \right|} = \left( \sgn\left( \frac{4-6t^2}{t^6}e^{-\frac{1}{t^2}} \right) ,0,0 \right), \\
\bm{\beta}_2(t) &= \bm{\gamma}_2(t) \times \bm{\alpha}_2(t) = \left( 0,-\frac{\sgn\left( \frac{4-6t^2}{t^6}e^{-\frac{1}{t^2}} \right) \cdot 2e^{-\frac{1}{t^2}}}{t^3\left( 1+\frac{4}{t^6}e^{-\frac{2}{t^2}} \right)^{\frac{1}{2}}},\frac{\sgn\left( \frac{4-6t^2}{t^6}e^{-\frac{1}{t^2}} \right)}{\left( 1+\frac{4}{t^6}e^{-\frac{2}{t^2}} \right)^{\frac{1}{2}}} \right) .
\end{align*}
故这条曲线在$t>0$时的Frenet标架场为$\left\{ \bm{r}(t);\bm{\alpha}_2(t),\bm{\beta}_2(t),\bm{\gamma}_2(t) \right\}$，且此时
\begin{align*}
\lim\limits_{t\rightarrow 0^+}\bm{\alpha}_2(t) = \left( 0,0,0 \right), \quad \lim\limits_{t\rightarrow 0^+}\bm{\gamma}_2(t) = \left( 0,0,0 \right), \quad
\lim\limits_{t\rightarrow 0^+}\bm{\beta}_2(t) = \left( 0,1,0 \right) .
\end{align*}
综上可知，$\lim\limits_{t\rightarrow 0^+}\bm{\beta}_1(t) \ne \lim\limits_{t\rightarrow 0^-}\bm{\beta}_2(t)$。因此Frenet 标架在 $t \to 0^+$ 和 $t \to 0^-$ 时的极限是不同的.

\end{proof}

\begin{example}
证明: 若一条正则曲线在各点的切线都经过一个固定点, 则它必定是一条直线.
\end{example}
\begin{proof}
设正则曲线$C$以弧长$s$为参数的参数方程是$\bm{r}(s)$，其中$s\in[0,a]$,$a\in \mathbb{R}\cup\{\pm \infty\}$。则曲线$C$在$\bm{r}(s)$处的切线参数方程为
\begin{align*}
\bm{X}(u)=\bm{r}(s)+\bm{r}'(s)(u-s).
\end{align*}
由条件可设，曲线$C$在各点的切线都经过固定点$\bm{X}(u_0)$，即
\begin{align*}
\bm{r}(s)+\bm{r}'(s)(u_0-s)\equiv \bm{X}(u_0),\quad \forall s\in[0,a].
\end{align*}
对上式两边关于$s$求导得
\begin{align*}
\bm{r}''(s)(u_0-s)=0,\quad \forall s\in[0,a].
\end{align*}
故当$s\in [0,a]\setminus \{0,u_0\}$时,$\bm{r}''(s)=0$,再结合$\bm{r}''$的连续性可知
\begin{align*}
\bm{r}''(s)\equiv \bm{0}\Longleftrightarrow \kappa(s)=|\bm{r}''(s)|\equiv 0.
\end{align*}
由\refthe{theorem:正则曲线是直线等价于曲率为0}知曲线$C$是直线.

\end{proof}

\begin{example}
证明: 若一条正则曲线在各点的法平面都经过一个固定点, 则它必定落在一个球面上.
\end{example}
\begin{proof}
设正则曲线$C$以弧长$s$为参数的参数方程为$\bm{r}(s)$，其中$s\in[0,a]$，$a\in\mathbb{R}\cup\{\pm\infty\}$。则曲线$C$在点$\bm{r}(s)$处的法平面方程为
\begin{align*}
(\bm{X}-\bm{r}(s))\cdot\bm{\alpha}(s)=0,
\end{align*}
其中$\bm{X}=(x,y,z)\in\mathbb{R}^3$，$\bm{\alpha}(s)=\bm{r}'(s)$。
由条件可设曲线$C$在各点的法平面都过固定点$\bm{X}_0=(x_0,y_0,z_0)$，则
\begin{align*}
(\bm{X}_0-\bm{r}(s))\cdot\bm{r}'(s)=0,\quad \forall s\in[0,a].
\end{align*}
令$f(s)=|\bm{X}_0-\bm{r}(s)|^2$，则
\begin{align*}
f'(s)=-2(\bm{X}_0-\bm{r}(s))\cdot\bm{r}'(s)=0,\quad \forall s\in[0,a].
\end{align*}
故存在$R\in\mathbb{R}$，使得
\begin{align*}
f(s)=|\bm{X}_0-\bm{r}(s)|^2=R,\quad \forall s\in[0,a].
\end{align*}
即曲线$C$都落在以$\bm{X}_0$为球心，$R$为半径的球面上。

\end{proof}

\begin{example}
求圆螺旋线的切线和一个垂直于圆螺旋线的轴的固定平面的交点的轨迹.
\end{example}
\begin{solution}
不妨设圆螺旋线参数方程为
\begin{align}
\bm{r}(t)=(a\cos t,a\sin t,bt),
\end{align}
其中$t\in\mathbb{R}$为参数.\label{eq21.1}
则与其轴平行的方向向量为$\bm{n}=(0,0,1)$，从而可设垂直于这个圆螺旋线的轴的固定平面$\pi$的方程为
$\bm{X}\cdot\bm{n}=0$，其中$\bm{X}=(x,y,z)\in\mathbb{R}^3$.
由\eqref{eq21.1}式可得
\begin{align*}
\bm{r}'(t)=(-a\sin t,a\cos t,b),
\end{align*}
故这个圆螺线在点$\bm{r}(t)$处的切线$l_t$参数方程为
\begin{align*}
\bm{L}(u)=\bm{r}(t)+\bm{r}'(t)(u-t)=\left(a\left[\cos t-(u-t)\sin t\right],a\left[\sin t+(u-t)\cos t\right],bu\right),
\end{align*}
其中$u\in\mathbb{R}$为参数.
于是平面$\pi$与切线$l_t$的交点满足
\begin{align*}
\begin{cases}
\bm{L}(u)\cdot\bm{n}=0,\\
\bm{L}(u)=\left(a\left[\cos t-(u-t)\sin t\right],a\left[\sin t+(u-t)\cos t\right],bu\right),
\end{cases}
\end{align*}
解得$u=0$.从而所求交点的轨迹参数方程为
\begin{align*}
\bm{L}(u)=\left(a(\cos t+t\sin t),a(\sin t-t\cos t),0\right),
\end{align*}
其中$t\in\mathbb{R}$为参数.

\end{solution}
