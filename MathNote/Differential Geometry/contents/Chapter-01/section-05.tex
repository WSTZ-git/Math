\section{曲线论基本定理}

\begin{theorem}\label{theorem:两条曲率挠率相同的曲线只相差一个位置}
设$\bm{r} = \bm{r}_1(s)$和$\bm{r} = \bm{r}_2(s)$是$E^3$中两条以弧长$s$为参数的正则参数曲线，如果它们的曲率处处不为零，并且它们的曲率和挠率分别相等，即$\kappa_1(s) = \kappa_2(s)$，$\tau_1(s) = \tau_2(s)$，则有$E^3$中的一个刚体运动$\sigma$，它把曲线$\bm{r} = \bm{r}_1(s)$变为曲线$\bm{r} = \bm{r}_2(s)$。
\end{theorem}
\begin{proof}
由于这两条曲线的曲率处处不为零，因此沿这两条曲线有完全确定的Frenet标架。假定它们在$s=0$处的Frenet标架分别是$\{\bm{r}_1(0);\bm{\alpha}_1(0),\bm{\beta}_1(0),\bm{\gamma}_1(0)\}$和$\{\bm{r}_2(0);\bm{\alpha}_2(0),\bm{\beta}_2(0),\bm{\gamma}_2(0)\}$。因为它们都是右手单位正交标架，故由\refthe{theorem:刚体运动等距变换的充要条件}，在$E^3$中有一个刚体运动$\sigma$把后一个标架变成前一个标架。不妨把第二条曲线经过刚体运动$\sigma$得到的像仍然记为$\bm{r} = \bm{r}_2(s)$，那么由\refpro{proposition:曲线的曲率和挠率在刚体运动下不变}知曲线$\bm{r} = \bm{r}_1(s)$和$\bm{r} = \bm{r}_2(s)$在对应点仍旧有相同的曲率和挠率，并且在$s=0$处有相同的Frenet标架。我们要证明
\begin{align*}
\bm{r}_1(s) = \bm{r}_2(s),\quad \forall s.
\end{align*}
为此，首先定义函数
\begin{align}
f(s) = (\bm{\alpha}_1(s) - \bm{\alpha}_2(s))^2 + (\bm{\beta}_1(s) - \bm{\beta}_2(s))^2 + (\bm{\gamma}_1(s) - \bm{\gamma}_2(s))^2,
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.1}
\end{align}
且由假设得知$f(0) = 0$。对函数$f(s)$直接求导，利用\hyperref[theorem:微分几何-Frenet公式]{Frenet公式},并且注意到$\kappa_1(s) = \kappa_2(s)$，$\tau_1(s) = \tau_2(s)$，则得
\begin{align*}
\frac{1}{2}\frac{\mathrm{d}f(s)}{\mathrm{d}s} &= (\bm{\alpha}_1(s) - \bm{\alpha}_2(s)) \cdot (\kappa_1(s)\bm{\beta}_1(s) - \kappa_2(s)\bm{\beta}_2(s)) \\
&\quad + (\bm{\beta}_1(s) - \bm{\beta}_2(s)) \cdot (-\kappa_1(s)\bm{\alpha}_1(s) + \kappa_2(s)\bm{\alpha}_2(s)) \\
&\quad + (\bm{\beta}_1(s) - \bm{\beta}_2(s)) \cdot (\tau_1(s)\bm{\gamma}_1(s) - \tau_2(s)\bm{\gamma}_2(s)) \\
&\quad + (\bm{\gamma}_1(s) - \bm{\gamma}_2(s)) \cdot (-\tau_1(s)\bm{\beta}_1(s) + \tau_2(s)\bm{\beta}_2(s)) \\
&= \kappa_1(s)(\bm{\alpha}_1(s) - \bm{\alpha}_2(s)) \cdot (\bm{\beta}_1(s) - \bm{\beta}_2(s)) \\
&\quad - \kappa_1(s)(\bm{\beta}_1(s) - \bm{\beta}_2(s)) \cdot (\bm{\alpha}_1(s) - \bm{\alpha}_2(s)) \\
&\quad + \tau_1(s)(\bm{\beta}_1(s) - \bm{\beta}_2(s)) \cdot (\bm{\gamma}_1(s) - \bm{\gamma}_2(s)) \\
&\quad - \tau_1(s)(\bm{\gamma}_1(s) - \bm{\gamma}_2(s)) \cdot (\bm{\beta}_1(s) - \bm{\beta}_2(s)) \equiv 0,
\end{align*}
故$f(s) = f(0) = 0$，即
\begin{align}
\bm{\alpha}_1(s) = \bm{\alpha}_2(s),\quad \bm{\beta}_1(s) = \bm{\beta}_2(s),\quad \bm{\gamma}_1(s) = \bm{\gamma}_2(s).
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.2}
\end{align}
再定义函数
\begin{align}
g(s) = (\bm{r}_1(s) - \bm{r}_2(s))^2,\quad g(0) = 0,
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.3}
\end{align}
则
\begin{align*}
\frac{1}{2}\frac{\mathrm{d}g(s)}{\mathrm{d}s} = (\bm{r}_1(s) - \bm{r}_2(s)) \cdot (\bm{\alpha}_1(s) - \bm{\alpha}_2(s)) \equiv 0,
\end{align*}
所以$g(s) = g(0) = 0$，即$\bm{r}_1(s) = \bm{r}_2(s),\forall s$.

\end{proof}

\begin{corollary}
设$\bm{r} = \bm{r}_1(t)$和$\bm{r} = \bm{r}_2(u)$是$E^3$中两条正则参数曲线，其中$t,u$为任意参数，它们的曲率处处不为零。如果存在三次以上的连续可微函数$u = \lambda(t)$，$\lambda'(t) \neq 0$，使得这两条曲线的弧长函数、曲率函数和挠率函数之间有关系式
\begin{align}
s_1(t) = s_2(\lambda(t)),\quad \kappa_1(t) = \kappa_2(\lambda(t)),\quad \tau_1(t) = \tau_2(\lambda(t)),
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.4}
\end{align}
则有$E^3$中的一个刚体运动$\sigma$，它把曲线$\bm{r} = \bm{r}_1(t)$变为曲线$\bm{r} = \bm{r}_2(u)$，即曲线$\bm{r} = \bm{r}_2(\lambda(t))$是曲线$\bm{r} = \bm{r}_1(t)$在刚体运动$\sigma$下的像。
\end{corollary}
\begin{proof}
设曲线$\bm{r}_1(t)$的弧长参数为$s$，满足$\frac{\mathrm{d}s}{\mathrm{d}t}=|\bm{r}_1'(t)|$，且$s=s_1(t)=\int_0^t|\bm{r}_1'(u)|\mathrm{d}u$. 对曲线$\bm{r}_2(u)$，由$u=\lambda(t)$知$\frac{\mathrm{d}u}{\mathrm{d}t}=\lambda'(t)\neq0$. 由条件$s_1(t)=s_2(\lambda(t))$知
\begin{align*}
\frac{\mathrm{d}s_1}{\mathrm{d}t}=|\bm{r}_1'(t)|,\,\,\frac{\mathrm{d}s_2}{\mathrm{d}u}=|\bm{r}_2'(u)|\text{且}\frac{\mathrm{d}s_1}{\mathrm{d}t}=\frac{\mathrm{d}s_2}{\mathrm{d}u}\cdot\frac{\mathrm{d}u}{\mathrm{d}t},
\end{align*}
因此
\begin{align*}
|\bm{r}_1'(t)|=|\bm{r}_2'(\lambda(t))|\cdot\lambda'(t).
\end{align*}
令$t=t(s)$是$s=s_1(t)$的反函数，则$u=\lambda(t(s))$.再令$\bm{r}_2(s)=\bm{r}_2(\lambda(t(s)))$,由于
\begin{align*}
\frac{\mathrm{d}\bm{r}_2}{\mathrm{d}s}=\frac{\mathrm{d}\bm{r}_2}{\mathrm{d}u}\cdot\frac{\mathrm{d}u}{\mathrm{d}s}\text{且}\frac{\mathrm{d}u}{\mathrm{d}s}=\frac{\mathrm{d}u}{\mathrm{d}t}\cdot\frac{\mathrm{d}t}{\mathrm{d}s}=\lambda'(t)\cdot\frac{1}{|\bm{r}_1'(t)|}= \frac{1}{|\bm{r}_2'(u)|},
\end{align*}
故$\left|\frac{\mathrm{d}\bm{r}_2}{\mathrm{d}s}\right|=1$，因此由\refthe{theorem:微分几何-定理2.1}知$\bm{r}_2(s)$是以$s$为弧长参数的曲线.

由条件$\kappa_1(t)=\kappa_2(\lambda(t))$和$\tau_1(t)=\tau_2(\lambda(t))$，又曲率、挠率与曲线的参数变换无关，故
\[
\kappa_1(s)=\kappa_2(s),\qquad \tau_1(s)=\tau_2(s),\qquad \forall s.
\]
根据\refthe{theorem:两条曲率挠率相同的曲线只相差一个位置}，存在一个刚体运动$\sigma$使得$\sigma(\bm{r}_1(s))=\bm{r}_2(s)$对所有$s$成立. 于是对原参数$t$有
\[
\sigma \left( \boldsymbol{r}_1\left( t \right) \right) =\sigma \left( \boldsymbol{r}_1\left( s_1\left( t \right) \right) \right) =\sigma \left( \boldsymbol{r}_1\left( s \right) \right) =\boldsymbol{r}_2\left( s \right) =\boldsymbol{r}_2\left( \lambda \left( t\left( s \right) \right) \right) =\boldsymbol{r}_2\left( \lambda \left( t \right) \right) ,
\]
即曲线$\bm{r}_2(\lambda(t))$是$\bm{r}_1(t)$在刚体运动$\sigma$下的像.

\end{proof}

\begin{theorem}[曲线论基本定理]\label{theorem:微分几何-曲线论基本定理}
设$\kappa(s),\tau(s)$是在区间$[a,b]$上两个任意给定的连续可微函数，并且$\kappa(s) > 0$，则在空间$E^3$中存在正则参数曲线$\bm{r} = \bm{r}(s)$，$a \leqslant s \leqslant b$，以$s$为弧长参数，以给定的函数$\kappa(s),\tau(s)$为它的曲率和挠率，且这样的曲线在空间$E^3$中是完全确定的，其差异至多为曲线在空间中的位置不同(即存在一个刚体运动，将满足条件的曲线映成另一条满足条件的曲线)。
\end{theorem}
\begin{remark}
这个定理说明，函数$\kappa(s) > 0$与$\tau(s)$在空间$E^3$中不计位置的差异唯一地确定了一条曲线，因此它们可以看作是该曲线的方程，称为曲线的内在方程，或自然方程。从曲线的自然方程得到参数方程的过程，就是求解方程组\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}的过程。一般说来，该过程是比较难的。如果知道方程组\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}的一个特解，则\refthe{theorem:两条曲率挠率相同的曲线只相差一个位置}告诉我们，其余的解都是上述特解在$E^3$的刚体运动下的像。
\end{remark}
\begin{proof}
\refthe{theorem:两条曲率挠率相同的曲线只相差一个位置}已证明这样的曲线在空间$E^3$中确定到至多差一个位置的不同，因此我们只要证明这样的曲线的存在性。

为了能够用和式表示方程组，引进新的函数记号
\begin{align}
\begin{pmatrix}
a_{11}(s) & a_{12}(s) & a_{13}(s) \\
a_{21}(s) & a_{22}(s) & a_{23}(s) \\
a_{31}(s) & a_{32}(s) & a_{33}(s)
\end{pmatrix}
=
\begin{pmatrix}
0 & \kappa(s) & 0 \\
-\kappa(s) & 0 & \tau(s) \\
0 & -\tau(s) & 0
\end{pmatrix},
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.5}
\end{align}
并且用$\bm{r},\bm{e}_1,\bm{e}_2,\bm{e}_3$记写成向量形式的未知函数，每一个向量函数代表3个未知函数，共12个未知函数。考虑一阶常微分方程组
\begin{align}
\begin{cases}
\dfrac{\mathrm{d}\bm{r}}{\mathrm{d}s} = \bm{e}_1, \\
\dfrac{\mathrm{d}\bm{e}_i}{\mathrm{d}s} = \sum\limits_{j=1}^3 a_{ij}(s)\bm{e}_j,\quad 1 \leqslant i \leqslant 3.
\end{cases}
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}
\end{align}
这是由4个向量形式的线性齐次微分方程组组成的方程组（实际上有12个方程）。由于方程组的系数是连续可微函数，根据常微分方程理论，对于任意给定的一组初始值$\bm{r}^0,\bm{e}_1^0,\bm{e}_2^0,\bm{e}_3^0$，方程组\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}有唯一的一组解$\bm{r}(s),\bm{e}_1(s),\bm{e}_2(s),\bm{e}_3(s)$，$a \leqslant s \leqslant b$，满足初始条件
\begin{align}
\bm{r}(s_0) = \bm{r}^0,\quad \bm{e}_i(s_0) = \bm{e}_i^0,\quad 1 \leqslant i \leqslant 3,
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.7}
\end{align}
其中$s_0$是区间$[a,b]$中任意固定的一点。一般来说，这样的解并不会满足我们的要求。如果$\bm{r} = \bm{r}(s)$是我们所要的曲线，而$\{\bm{r}(s);\bm{e}_1(s),\bm{e}_2(s),\bm{e}_3(s)\}$是曲线的Frenet标架，则我们必须要求初始值$\{\bm{r}^0;\bm{e}_1^0,\bm{e}_2^0,\bm{e}_3^0\}$是右手单位正交标架，即它们要满足条件
\begin{align}
\begin{cases}
\bm{e}_i^0 \cdot \bm{e}_j^0 = \delta_{ij},\quad 1 \leqslant i,j \leqslant 3, \\
(\bm{e}_1^0,\bm{e}_2^0,\bm{e}_3^0) = 1.
\end{cases}
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.8}
\end{align}
我们要证明，当初始值满足条件\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.8}时，方程组\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}满足初始条件\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.7}的解给出的$\bm{r} = \bm{r}(s)$是一条正则曲线，以$s$为弧长参数，并以给定的函数$\kappa(s),\tau(s)$为它的曲率和挠率。为此，首先证明这组解$\{\bm{r}(s);\bm{e}_1(s),\bm{e}_2(s),\bm{e}_3(s)\}$是右手单位正交标架族。命
\begin{align}
g_{ij}(s) = \bm{e}_i(s) \cdot \bm{e}_j(s) - \delta_{ij},\quad 1 \leqslant i,j \leqslant 3.
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.9}
\end{align}
初始值所满足的条件\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.8}说明
\begin{align}
g_{ij}(s_0) = 0,\quad 1 \leqslant i,j \leqslant 3.
\label{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.10}
\end{align}
将\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.9}式求导，并且利用$\bm{e}_i(s)$满足方程组\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}得到
\begin{align*}
\frac{\mathrm{d}g_{ij}\left( s \right)}{\mathrm{d}s}=\frac{\mathrm{d}\boldsymbol{e}_i\left( s \right)}{\mathrm{d}s}\cdot \boldsymbol{e}_j\left( s \right) +\boldsymbol{e}_i\left( s \right) \cdot \frac{\mathrm{d}\boldsymbol{e}_j\left( s \right)}{\mathrm{d}s}=\sum_{k=1}^3{\left( a_{ik}e_k\left( s \right) \cdot e_j\left( s \right) +a_{jk}g_{ik}\left( s \right) e_i\left( s \right) \cdot e_k\left( s \right) \right)}.
\end{align*}
因为
\begin{align*}
a_{ij} + a_{ji} = 0,
\end{align*}
故$g_{ij}(s)$满足常微分方程组
\begin{align*}
\frac{\mathrm{d}g_{ij}(s)}{\mathrm{d}s} = \sum\limits_{k=1}^3 (a_{ik}g_{kj}(s) + a_{jk}g_{ik}(s)).
\end{align*}
这是线性齐次方程组，并且它在初始条件\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.10}下的解是唯一的，所以$g_{ij}(s)$只能是平凡解，即
\begin{align*}
g_{ij}(s) \equiv 0,\quad 1 \leqslant i,j \leqslant 3,
\end{align*}
也就是
\begin{align*}
\bm{e}_i(s) \cdot \bm{e}_j(s) = \delta_{ij},\quad 1 \leqslant i,j \leqslant 3,
\end{align*}
故$\{\bm{r}(s);\bm{e}_1(s),\bm{e}_2(s),\bm{e}_3(s)\}$是单位正交标架。这样，向量$\bm{e}_1(s),\bm{e}_2(s),\bm{e}_3(s)$的混合积$(\bm{e}_1(s),\bm{e}_2(s),\bm{e}_3(s))$的值是1或者-1。由于条件\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.8}知$(\bm{e}_1(s_0),\bm{e}_2(s_0),\bm{e}_3(s_0)) = 1$，故由解的连续性得到
\begin{align*}
(\bm{e}_1(s),\bm{e}_2(s),\bm{e}_3(s)) = 1,
\end{align*}
所以$\{\bm{r}(s);\bm{e}_1(s),\bm{e}_2(s),\bm{e}_3(s)\}$是右手单位正交标架族。

由方程\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}的第一式得知
\begin{align*}
\left|\frac{\mathrm{d}\bm{r}(s)}{\mathrm{d}s}\right| = |\bm{e}_1(s)| = 1,
\end{align*}
故$\bm{r}(s)$是正则参数曲线，$s$是弧长参数，$\bm{\alpha}(s) = \bm{e}_1(s)$。再由方程\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}的第二式得知
\begin{align*}
\frac{\mathrm{d}\bm{\alpha}(s)}{\mathrm{d}s} = \frac{\mathrm{d}\bm{e}_1(s)}{\mathrm{d}s} = \kappa(s)\bm{e}_2(s),
\end{align*}
因此
\begin{align*}
\kappa(s) = \left|\frac{\mathrm{d}\bm{\alpha}(s)}{\mathrm{d}s}\right|,\quad \bm{\beta}(s) = \bm{e}_2(s),
\end{align*}
即$\kappa(s)$是曲线$\bm{r} = \bm{r}(s)$的曲率，$\bm{e}_2(s)$是它的主法向量，于是
\begin{align*}
\bm{\gamma}(s) = \bm{\alpha}(s) \times \bm{\beta}(s) = \bm{e}_1(s) \times \bm{e}_2(s) = \bm{e}_3(s).
\end{align*}
由\eqref{eq:--3ur0jfdjgijiL:LIHFIOAWHFNASawf5.6}式的最后一式得到
\begin{align*}
\tau(s) = -\frac{\mathrm{d}\bm{e}_3(s)}{\mathrm{d}s} \cdot \bm{e}_2(s) = -\frac{\mathrm{d}\bm{\gamma}(s)}{\mathrm{d}s} \cdot \bm{\beta}(s),
\end{align*}
故$\tau(s)$是曲线$\bm{r} = \bm{r}(s)$的挠率。

\end{proof}

\begin{example}
求曲率和挠率分别是常数$\kappa_0 > 0$，$\tau_0$的曲线的参数方程。
\end{example}

\begin{solution}

已知圆螺旋线$\bm{r} = (a\cos t,a\sin t,bt)$（$a>0,b$是常数）的曲率和挠率分别是常数
\begin{align*}
\kappa = \frac{a}{a^2 + b^2},\quad \tau = \frac{b}{a^2 + b^2}.
\end{align*}
取$a,b$使得
\begin{align*}
\frac{a}{a^2 + b^2} = \kappa_0,\quad \frac{b}{a^2 + b^2} = \tau_0,
\end{align*}
也就是
\begin{align*}
a = \frac{\kappa_0}{\kappa_0^2 + \tau_0^2},\quad b = \frac{\tau_0}{\kappa_0^2 + \tau_0^2}.
\end{align*}
根据\refthe{theorem:曲率和挠率能确定一条正则曲线最多差一个位置}，曲率和挠率分别是常数$\kappa_0 > 0$，$\tau_0$的曲线必定是圆螺旋线。如果不计位置的差异，则它的参数方程是
\begin{align*}
\bm{r} = \left( \frac{\kappa_0}{\kappa_0^2 + \tau_0^2}\cos t,\ \frac{\kappa_0}{\kappa_0^2 + \tau_0^2}\sin t,\ \frac{\tau_0 t}{\kappa_0^2 + \tau_0^2} \right),
\end{align*}
或者是
\begin{align*}
\left( \frac{\kappa_0}{\kappa_0^2 + \tau_0^2}\cos\left( \sqrt{\kappa_0^2 + \tau_0^2}s \right),\ \frac{\kappa_0}{\kappa_0^2 + \tau_0^2}\sin\left( \sqrt{\kappa_0^2 + \tau_0^2}s \right),\ \frac{\tau_0 s}{\sqrt{\kappa_0^2 + \tau_0^2}} \right),
\end{align*}
其中$s$是弧长参数.

\end{solution}

\begin{definition}[定倾曲线]
如果一条曲线的切向量与一个固定的方向交成定角, 则称该曲线为\textbf{定倾曲线}, 或\textbf{一般螺线}. 
\end{definition}
\begin{remark}
这种曲线可以看作是落在一个柱面上与直母线成定角的曲线. 
\end{remark}

\begin{proposition}\label{proposition:曲线是一般螺线或定倾曲线的充要条件}
证明: 
\begin{enumerate}[(1)]
\item 一条曲率处处不为零的正则参数曲线$C$是定倾曲线当且仅当它的挠率与曲率之比是常数.

\item 一条曲率处处不为零的正则参数曲线$C$的所有主法线都平行于一个固定平面当且仅当它是一般螺线.
\end{enumerate}
\end{proposition}
\begin{remark}
根据\hyperref[definition:Frenet标架定义]{主法线的定义},只要考虑正则曲线的主法线都默认曲率不为零.
\end{remark}
\begin{remark}
证明(1)的充分性中构造$f(s)$的思路:待定满足要求的常向量$\bm{n}$，则存在$a\in\mathbb{R}$，使得$\bm{\alpha}(s)\cdot\bm{n}=a$。关于$s$求导并结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
$$\bm{\alpha}'(s)\cdot\bm{n}=\kappa(s)\bm{\beta}(s)\cdot\bm{n}=0\Longrightarrow\bm{\beta}(s)\cdot\bm{n}=0.$$
再对上式关于$s$求导并结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
$$\bm{\beta}'(s)\cdot\bm{n}=(-\kappa(s)\bm{\alpha}(s)+\tau(s)\bm{\gamma}(s))\cdot\bm{n}=0\Longrightarrow\left(\bm{\alpha}(s)-\frac{\tau(s)}{\kappa(s)}\bm{\gamma}(s)\right)\cdot\bm{n}=(\bm{\alpha}(s)-C\bm{\gamma}(s))\cdot\bm{n}=0.$$
故$\bm{n}$与$\bm{\beta}(s)$和$(\bm{\alpha}(s)-C\bm{\gamma}(s))$正交，因此直接猜测$\bm{n}=\bm{\beta}(s)\times(\bm{\alpha}(s)-C\bm{\gamma}(s))$，再验证即可。
\end{remark}
\begin{proof}\begin{enumerate}[(1)]
\item 设曲线$C$的参数方程为$\bm{r}=\bm{r}(s)$，其中$s$为弧长参数。其在各点的Frenet标架为$\{\bm{r}(s);\bm{\alpha}(s),\bm{\beta}(s),\bm{\gamma}(s)\}$，曲率为$\kappa(s)$，挠率为$\tau(s)$。

{\heiti 必要性:}设曲线$C$各点切向量与常向量$\bm{n}$成定角，则存在$a\in\mathbb{R}$，使得
$\bm{r}'(s)\cdot\bm{n}=\bm{\alpha}(s)\cdot\bm{n}=a.$
对上式关于$s$求导，结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}和$\kappa(s)\ne0$得
\begin{align}
\bm{r}''(s)\cdot\bm{n}=\bm{\alpha}'(s)\cdot\bm{n}=\kappa(s)\bm{\beta}(s)\cdot\bm{n}=0\Longrightarrow\bm{\beta}(s)\cdot\bm{n}=0.\label{eq21.11111}
\end{align}
于是$\bm{n}\in(\mathrm{span}\{\bm{\beta}(s)\})^\perp=\mathrm{span}\{\bm{\alpha}(s),\bm{\gamma}(s)\}$，从而存在$\lambda(s),\mu(s)\in\mathbb{R}^3$，使得
$\bm{n}=\lambda(s)\bm{\alpha}(s)+\mu(s)\bm{\gamma}(s).$
上式两边同时点乘$\bm{\alpha}(s)$得$\lambda(s)=a$。再对上式关于$s$求导，结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
\begin{align*}
0=a\bm{\alpha}'(s)+\mu'(s)\bm{\gamma}(s)+\mu(s)\bm{\gamma}'(s)=(a\kappa(s)-\mu(s)\tau(s))\bm{\beta}(s)+\mu'(s)\bm{\gamma}(s).
\end{align*}
又因为$\bm{\beta}(s),\bm{\gamma}(s)$线性无关，所以$\mu'(s)=a\kappa(s)-\mu(s)\tau(s)=0$，因此$\mu(s)$是常数，记$\mu(s)=b\in\mathbb{R}$。从而
\begin{align*}
\bm{n}=a\bm{\alpha}(s)+b\bm{\gamma}(s)\Longrightarrow\bm{\gamma}(s)\cdot\bm{n}=b\bm{\gamma}^2(s)=b.
\end{align*}
若$b=0$，则$\bm{n}=a\bm{\alpha}(s)$.又若$\bm{\alpha}(s)$为常向量，则此时曲线为直线,曲率为0,矛盾!因此$\bm{\alpha}(s)$不为常向量,这与$\bm{n}$为常向量矛盾!故$b\ne0$。再对\eqref{eq21.11111}式关于$s$求导，结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
\begin{align*}
\bm{\beta}'(s)\cdot\bm{n}=(-\kappa(s)\bm{\alpha}(s)+\tau(s)\bm{\gamma}(s))\cdot\bm{n}=0\Longrightarrow\frac{\tau(s)}{\kappa(s)}=\frac{\bm{\alpha}(s)\cdot\bm{n}}{\bm{\gamma}(s)\cdot\bm{n}}=\frac{a}{b}\in\mathbb{R}.
\end{align*}

{\heiti 充分性:}设$\frac{\tau(s)}{\kappa(s)}=C\in\mathbb{R}$，则$\tau(s)=C\kappa(s)$。令$f(s)=\bm{\beta}(s)\times(\bm{\alpha}(s)-C\bm{\gamma}(s))$，则由\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}可得
\begin{align*}
f'(s)&=\bm{\beta}'(s)\times(\bm{\alpha}(s)-C\bm{\gamma}(s))+\bm{\beta}(s)\times(\bm{\alpha}'(s)-C\bm{\gamma}'(s))\\
&=(-\kappa(s)\bm{\alpha}(s)+\tau(s)\bm{\gamma}(s))\times(\bm{\alpha}(s)-C\bm{\gamma}(s))+\bm{\beta}(s)\times(\kappa(s)\bm{\beta}(s)+C\tau(s)\bm{\beta}(s))\\
&=(C\kappa(s)-\tau(s))(\bm{\alpha}(s)\times\bm{\gamma}(s))=0.
\end{align*}
故$f(s)$是常向量，记$\bm{n}=f(s)=\bm{\beta}(s)\times(\bm{\alpha}(s)-C\bm{\gamma}(s))$，则
\begin{align*}
\bm{\alpha}(s)\cdot\bm{n}=\bm{\alpha}(s)\cdot\left[\bm{\beta}(s)\times(\bm{\alpha}(s)-C\bm{\gamma}(s))\right]
=\bm{\alpha}(s)\cdot(\bm{\beta}(s)\times\bm{\alpha}(s))-\bm{\alpha}(s)\cdot(\bm{\beta}(s)\times C\bm{\gamma}(s))=C\bm{\alpha}^2(s)=C,
\end{align*}
即曲线$C$各点切向量与常向量$\bm{n}$成定角。

\item 设曲线$C$的参数方程为$\bm{r}=\bm{r}(s)$，其中$s$为弧长参数。其在各点的Frenet标架为$\{\bm{r}(s);\bm{\alpha}(s),\bm{\beta}(s),\bm{\gamma}(s)\}$，曲率为$\kappa(s)$，挠率为$\tau(s)$.

{\heiti 必要性:}设曲线$C$的所有主法线都平行于法向量为$\bm{n}$的平面,则
\begin{align}\label{eq:203jrw34ur390}
\bm{\beta}(s)\cdot \bm{n}=0.
\end{align}
于是$\bm{n}\in(\mathrm{span}\{\bm{\beta}(s)\})^\perp=\mathrm{span}\{\bm{\alpha}(s),\bm{\gamma}(s)\}$，从而存在$\lambda(s),\mu(s)\in\mathbb{R}^3$，使得
$\bm{n}=\lambda(s)\bm{\alpha}(s)+\mu(s)\bm{\gamma}(s).$
上式两边同时点乘$\bm{\alpha}(s)$得$\lambda(s)=a$。再对上式关于$s$求导，结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
\begin{align*}
0=a\bm{\alpha}'(s)+\mu'(s)\bm{\gamma}(s)+\mu(s)\bm{\gamma}'(s)=(a\kappa(s)-\mu(s)\tau(s))\bm{\beta}(s)+\mu'(s)\bm{\gamma}(s).
\end{align*}
又因为$\bm{\beta}(s),\bm{\gamma}(s)$线性无关，所以$\mu'(s)=a\kappa(s)-\mu(s)\tau(s)=0$，因此$\mu(s)$是常数，记$\mu(s)=b\in\mathbb{R}$。从而
\begin{align*}
\bm{n}=a\bm{\alpha}(s)+b\bm{\gamma}(s)\Longrightarrow\bm{\gamma}(s)\cdot\bm{n}=b\bm{\gamma}^2(s)=b.
\end{align*}
若$b=0$，则$\bm{n}=a\bm{\alpha}(s)$.又若$\bm{\alpha}(s)$为常向量，则此时曲线为直线,曲率为0,矛盾!因此$\bm{\alpha}(s)$不为常向量,这与$\bm{n}$为常向量矛盾!故$b\ne0$。再对\eqref{eq:203jrw34ur390}式关于$s$求导，结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得
\begin{align*}
\bm{\beta}'(s)\cdot\bm{n}=(-\kappa(s)\bm{\alpha}(s)+\tau(s)\bm{\gamma}(s))\cdot\bm{n}=0\Longrightarrow\frac{\tau(s)}{\kappa(s)}=\frac{\bm{\alpha}(s)\cdot\bm{n}}{\bm{\gamma}(s)\cdot\bm{n}}=\frac{a}{b}\in\mathbb{R}.
\end{align*}
由(1)可知,此时曲线$C$是一般螺线.

{\heiti 充分性:}设曲线$C$各点切向量与常向量$\bm{n}$成定角，则存在$a\in\mathbb{R}$，使得
$\bm{r}'(s)\cdot\bm{n}=\bm{\alpha}(s)\cdot\bm{n}=a.$
对上式关于$s$求导，结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}和$\kappa(s)\ne0$得
\begin{align*}
\bm{r}''(s)\cdot\bm{n}=\bm{\alpha}'(s)\cdot\bm{n}=\kappa(s)\bm{\beta}(s)\cdot\bm{n}=0\Longrightarrow \bm{\beta}(s)\cdot\bm{n}=0.
\end{align*}
因此曲线$C$的所有主法向量都与法向量为$\bm{n}$的平面平行.
\end{enumerate}

\end{proof}

\begin{example}
设$\tau(s),\kappa(s) > 0$是两个连续可微函数, 满足关系式$\tau(s) = c \cdot \kappa(s)$, $c$是常数. 求曲线的参数方程$\bm{r}(s)$, 使它的弧长参数是$s$, 曲率是$\kappa(s)$, 挠率是$\tau(s)$.
\end{example}
\begin{remark}
实际上，所求曲线就是一般螺线方程.
\end{remark}
\begin{solution}

设满足条件的曲线的参数方程为$\bm{r}=\bm{r}(s)$，其中$s$为弧长参数。其在各点的Frenet标架为$\{\bm{r}(s);\bm{\alpha}(s),\bm{\beta}(s),\bm{\gamma}(s)\}$.
由\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}可知
\begin{align}\label{eq::Frenet1}
\begin{cases}
\frac{\mathrm{d}\boldsymbol{\alpha }}{\mathrm{d}s}=\kappa (s)\boldsymbol{\beta },\\
\frac{\mathrm{d}\boldsymbol{\beta }}{\mathrm{d}s}=-\kappa (s)\boldsymbol{\alpha }+\tau (s)\boldsymbol{\gamma },\\
\frac{\mathrm{d}\boldsymbol{\gamma }}{\mathrm{d}s}=-\tau (s)\boldsymbol{\beta }.\\
\end{cases}
\end{align}
考虑
\begin{align}\label{eq::92jf3aII}
\begin{cases}
\frac{1}{\kappa (s)}\frac{\mathrm{d}\boldsymbol{\alpha }}{\mathrm{d}s}=\boldsymbol{\beta },\\
\frac{1}{\kappa (s)}\frac{\mathrm{d}\boldsymbol{\beta }}{\mathrm{d}s}=-\boldsymbol{\alpha }+c\boldsymbol{\gamma },\\
\frac{1}{\kappa (s)}\frac{\mathrm{d}\boldsymbol{\gamma }}{\mathrm{d}s}=-c\boldsymbol{\beta }.\\
\end{cases}
\end{align}
令$t=t(s)=\int_0^s{\kappa (s)\,\mathrm{d}s}$，则$t=t(s)$是一个参数变换且$\kappa(s)=\frac{\mathrm{d}t}{\mathrm{d}s}$.由Frenet标架与参数变换无关，故方程\eqref{eq::92jf3aII}可化为
\begin{align}\label{eq::Frenet121234231}
\begin{cases}
\frac{1}{\kappa \left( s \right)}\frac{\mathrm{d}\boldsymbol{\alpha }}{\mathrm{d}s}=\frac{1}{\kappa \left( s \right)}\frac{\mathrm{d}\boldsymbol{\alpha }}{\mathrm{d}t}\cdot \frac{\mathrm{d}t}{\mathrm{d}s}=\frac{\mathrm{d}\boldsymbol{\alpha }}{\mathrm{d}t}=\boldsymbol{\beta },\\
\frac{1}{\kappa \left( s \right)}\frac{\mathrm{d}\boldsymbol{\beta }}{\mathrm{d}s}=\frac{1}{\kappa \left( s \right)}\frac{\mathrm{d}\boldsymbol{\beta }}{\mathrm{d}t}\cdot \frac{\mathrm{d}t}{\mathrm{d}s}=\frac{\mathrm{d}\boldsymbol{\beta }}{\mathrm{d}t}=-\boldsymbol{\alpha }+c\boldsymbol{\gamma },\\
\frac{1}{\kappa \left( s \right)}\frac{\mathrm{d}\boldsymbol{\gamma }}{\mathrm{d}s}=\frac{1}{\kappa \left( s \right)}\frac{\mathrm{d}\boldsymbol{\gamma }}{\mathrm{d}t}\cdot \frac{\mathrm{d}t}{\mathrm{d}s}=\frac{\mathrm{d}\boldsymbol{\gamma }}{\mathrm{d}t}=-c\boldsymbol{\beta }.\\
\end{cases}
\end{align}
对方程组\eqref{eq::Frenet121234231}中的第二式求导得
\begin{align}\label{eq:::-wj4ifm4j90jJIOJDW}
\frac{\mathrm{d}^2\boldsymbol{\beta }}{\mathrm{d}t^2}=-(c^2+1)\boldsymbol{\beta }.
\end{align}
由于$|\bm{\beta}|=1$，取
\[
\bm{\beta}(s)=\bm{\beta}(t(s)) = \left( \sin \left( t\sqrt{c^2+1} \right) ,\;0,\;\cos \left( t\sqrt{c^2+1} \right) \right) =\left( \sin \left( t\left( s \right) \sqrt{c^2+1} \right) ,\;0,\;\cos \left( t\left( s \right) \sqrt{c^2+1} \right) \right) ,
\]
则此时$\bm{\beta}$是\eqref{eq:::-wj4ifm4j90jJIOJDW}式的一个特解且$|\bm{\beta}|=1$.于是再由方程组\eqref{eq::Frenet121234231}中的第一式可得一个特解为
\begin{align*}
\boldsymbol{\alpha }\left( s \right) &=\boldsymbol{\alpha }\left( t\left( s \right) \right) =\int{\beta \,\mathrm{d}t}=\left( -\frac{1}{\sqrt{c^2+1}}\cos \left( t\sqrt{c^2+1}\, \right) ,\;\frac{c}{\sqrt{c^2+1}},\;\frac{1}{\sqrt{c^2+1}}\sin \left( t\sqrt{c^2+1}\, \right) \right) 
\\
&=\left( -\frac{1}{\sqrt{c^2+1}}\cos \left( t\left( s \right) \sqrt{c^2+1}\, \right) ,\;\frac{c}{\sqrt{c^2+1}},\;\frac{1}{\sqrt{c^2+1}}\sin \left( t\left( s \right) \sqrt{c^2+1}\, \right) \right) 
\end{align*}
从而
\begin{align*}
\boldsymbol{\gamma }\left( s \right) &=\boldsymbol{\gamma }\left( t\left( s \right) \right) =\boldsymbol{\alpha }\left( t\left( s \right) \right) \times \boldsymbol{\beta }\left( t\left( s \right) \right) =\left( \frac{c}{\sqrt{c^2+1}}\cos \left( t\sqrt{c^2+1}\, \right) ,\;\frac{1}{\sqrt{c^2+1}},\;-\frac{c}{\sqrt{c^2+1}}\sin \left( t\sqrt{c^2+1}\, \right) \right) 
\\
&=\left( \frac{c}{\sqrt{c^2+1}}\cos \left( t\left( s \right) \sqrt{c^2+1}\, \right) ,\;\frac{1}{\sqrt{c^2+1}},\;-\frac{c}{\sqrt{c^2+1}}\sin \left( t\left( s \right) \sqrt{c^2+1}\, \right) \right) .
\end{align*}
从而$\boldsymbol{\alpha }\left( s \right) ,\boldsymbol{\beta }\left( s \right) ,\boldsymbol{\gamma }\left( s \right)$构成两两单位正交的右手系，且满足Frenet公式\eqref{eq::Frenet1}.

下面求$\bm{r}(s)$.由
\[
\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}s}=\alpha =\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}\cdot \frac{\mathrm{d}t}{\mathrm{d}s}=\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}\cdot \kappa \left( s \right) ,
\]
知
\begin{align*}
\frac{\mathrm{d}\bm{r}}{\mathrm{d}t} = \frac{1}{\kappa(s)}\alpha.
\end{align*}
对上式两边积分可得上式的一个特解为
\begin{align*}
&\boldsymbol{r}\left( s \right) =\boldsymbol{r}\left( t\left( s \right) \right) =\left( -\frac{1}{\sqrt{c^2+1}}\int{\frac{\cos \left( t\sqrt{c^2+1} \right)}{\kappa \left( s \right)}\,\mathrm{d}t,\;\frac{c}{\sqrt{c^2+1}}\int{\frac{1}{\kappa \left( s \right)}\,\mathrm{d}t,\;\frac{1}{\sqrt{c^2+1}}\int{\frac{\sin \left( t\sqrt{c^2+1} \right)}{\kappa \left( s \right)}\,\mathrm{d}t}}} \right) 
\\
&=\left( -\frac{1}{\sqrt{c^2+1}}\int{\frac{\cos \left( t\left( s \right) \sqrt{c^2+1} \right)}{\kappa \left( s \right)}\cdot \frac{\mathrm{d}t}{\mathrm{d}s}\,\mathrm{d}s,\;\frac{c}{\sqrt{c^2+1}}\int{\frac{1}{\kappa \left( s \right)}\,\cdot \frac{\mathrm{d}t}{\mathrm{d}s}\,\mathrm{d}s,\;\frac{1}{\sqrt{c^2+1}}\int{\frac{\sin \left( t\left( s \right) \sqrt{c^2+1} \right)}{\kappa \left( s \right)}\cdot \frac{\mathrm{d}t}{\mathrm{d}s}\,\mathrm{d}s}}} \right) 
\\
&=\left( -\frac{1}{\sqrt{c^2+1}}\int{\cos \left( t\left( s \right) \sqrt{c^2+1} \right) \,\mathrm{d}s,\;\frac{c}{\sqrt{c^2+1}}s,\;\frac{1}{\sqrt{c^2+1}}\int{\sin \left( t\left( s \right) \sqrt{c^2+1} \right) \,\mathrm{d}s}} \right) .
\end{align*}
此时$\bm{r}(s)$满足$\frac{\mathrm{d}\bm{r}}{\mathrm{d}s}=\bm{\alpha}(s)$.

综上,$\left\{ \boldsymbol{r}\left( s \right) ;\boldsymbol{\alpha }\left( s \right) ,\boldsymbol{\beta }\left( s \right) ,\boldsymbol{\gamma }\left( s \right) \right\} $构成单位右手正交标架,且满足Frenet公式\eqref{eq::Frenet1}.
再由\hyperref[theorem:微分几何-曲线论基本定理]{曲线论基本定理}可知,满足条件的所有曲线与$\bm{r}(s)$只相差刚体运动.

\end{solution}

\boldsymbol{r}'(t)=3a\left(1-t^2,\;2t,\;1+t^2\right),\quad
\boldsymbol{r}''(t)=6a\left(-t,\;1,\;t\right),\quad
\boldsymbol{r}'''(t)=6a\left(-1,\;0,\;1\right),\\
\left|\boldsymbol{r}'(t)\right|=3\sqrt{2}a(1+t^2),\quad
\left|\boldsymbol{r}'(t)\times\boldsymbol{r}''(t)\right|=18\sqrt{2}a^2(1+t^2),\\
\boldsymbol{r}'(t)\times\boldsymbol{r}''(t)=18a^2\left(t^2-1,\;-2t,\;1+t^2\right),\quad
(\boldsymbol{r}'(t),\boldsymbol{r}''(t),\boldsymbol{r}'''(t))=216a^3.
\end{gather*}
于是由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}和\hyperref[theorem:微分几何-曲线挠率的计算公式]{曲线挠率的计算公式}可得，曲率和挠率分别为

\kappa(t)=\frac{\left|\boldsymbol{r}'(t)\times\boldsymbol{r}''(t)\right|}{\left|\boldsymbol{r}'(t)\right|^3}
= \frac{18\sqrt{2}a^2(1+t^2)}{\left(3\sqrt{2}a(1+t^2)\right)^3}

\tau(t)=\frac{(\boldsymbol{r}'(t),\boldsymbol{r}''(t),\boldsymbol{r}'''(t))}{\left|\boldsymbol{r}'(t)\times\boldsymbol{r}''(t)\right|^2}

\end{gather*}
因此 $\kappa(t)=\tau(t)$.

\boldsymbol{r}'''(t)=\left(0,\;0,\;\frac{1}{a^2}\right),\\

\end{gather*}

\boldsymbol{r}'(t)=\left(1-\frac{a^2}{t^2},\;1+\frac{a^2}{t^2},\;\frac{2a}{t}\right),\quad
\boldsymbol{r}''(t)=\left(\frac{2a^2}{t^3},\;-\frac{2a^2}{t^3},\;-\frac{2a}{t^2}\right),\\

于是由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}和\hyperref[theorem:微分几何-曲线挠率的计算公式]{曲线挠率的计算公式}可得,曲率和挠率分别为
\begin{gather*}
\kappa(t)=\frac{\left|\boldsymbol{r}'(t)\times\boldsymbol{r}''(t)\right|}{\left|\boldsymbol{r}'(t)\right|^3}
= \frac{\dfrac{2\sqrt{2}a(t^2+a^2)}{t^4}}{\left(\dfrac{\sqrt{2}(t^2+a^2)}{t^2}\right)^3}
= \frac{a t^2}{(t^2+a^2)^2},\\
\tau(t)=\frac{(\boldsymbol{r}'(t),\boldsymbol{r}''(t),\boldsymbol{r}'''(t))}{\left|\boldsymbol{r}'(t)\times\boldsymbol{r}''(t)\right|^2}
= \frac{\dfrac{8a^3}{t^6}}{\left(\dfrac{2\sqrt{2}a(t^2+a^2)}{t^4}\right)^2}
= \frac{a t^2}{(t^2+a^2)^2}.
\end{gather*}
因此 $\kappa(t)=\tau(t)$.
\end{enumerate}

\end{proof}

\boldsymbol{r}'(t)=\left( 1+\sqrt{3}\cos t,-2\sin t,\sqrt{3}-\cos t \right) ,\quad \boldsymbol{r}'' (t)=\left( -\sqrt{3}\sin t,-2\cos t,\sin t \right) ,

\end{proof}

\begin{example}
确定函数$\varphi(t)$使得曲线$\bm{r}(t) = (t, \sin t, \varphi(t))$的主法线都平行于$Oyz$平面.
\end{example}
\begin{solution}

设曲线 $\bm{r}(t) = (t, \sin t, \varphi(t))$，则
\begin{align}
\bm{r}'(t) = (1, \cos t, \varphi'(t)), \qquad 
\bm{r}''(t) = (0, -\sin t, \varphi''(t)).\label{eq::fi20903fj029jf90wjr03jf0w3jf}
\end{align}
由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}知,曲线的主法线方向与向量 $\bm{r}'(t) \times (\bm{r}'(t) \times \bm{r}''(t))$ 平行，而
\[
\bm{r}'(t) \times (\bm{r}'(t) \times \bm{r}''(t)) = (\bm{r}'(t)\cdot\bm{r}''(t))\bm{r}'(t) - (\bm{r}'(t)\cdot\bm{r}'(t))\bm{r}''(t).
\]
由\eqref{eq::fi20903fj029jf90wjr03jf0w3jf}式可知,该向量的$x$分量为
\[
(\bm{r}'(t)\cdot\bm{r}''(t)) \cdot 1 - (\bm{r}'(t)\cdot\bm{r}'(t)) \cdot 0 = \bm{r}'(t)\cdot\bm{r}''(t)= -\sin t\cos t + \varphi'(t)\varphi''(t).
\]
主法线平行于 $Oyz$ 平面当且仅当其 $x$ 分量为零，即
\[
\bm{r}'\cdot\bm{r}'' = 0.
\]
因此条件化为
\[
\varphi'(t)\varphi''(t) = \sin t\cos t = \frac{1}{2}\sin 2t.
\]
令 $u(t) = \varphi'(t)$，则 $u u' = \frac{1}{2}\sin 2t$，即 $\frac{1}{2}(u^2)' = \frac{1}{2}\sin 2t$，故
\[
(u^2)' = \sin 2t.
\]
积分得
\[
u^2 = \int \sin 2t\,dt = -\frac{1}{2}\cos 2t + C_0 = \frac{1}{2}(1 - \cos 2t) + \left(C_0 - \frac{1}{2}\right),\quad C\in \mathbb{R}.
\]
取 $C_0 = \frac{1}{2}$,则
\[
u^2 = \frac{1}{2}(1 - \cos 2t) = \sin^2 t,
\]
于是 $u(t) = \varphi'(t) = \pm \sin t$。积分得
\[
\varphi(t) = \mp \cos t + C,\text{其中}C\text{为任意常数}.
\]
因此，函数 $\varphi(t)$ 可取为
\[
\varphi(t) = -\cos t + C \quad\text{或}\quad \varphi(t) = \cos t + C,\quad C\in\mathbb{R}.
\]
此时曲线的主法线处处平行于 $Oyz$ 平面。

\end{solution}
