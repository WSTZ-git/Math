\section{可展曲面}


\begin{definition}
设$S$为直纹面，若$S$的切平面沿每一条直母线保持不变，则称$S$为\textbf{可展曲面}。
\end{definition}
\begin{remark}
典型可展曲面的参数形式：
\begin{enumerate}[(1)]
\item 柱面：$\bm r=\bm a(u)+v\bm l$，$\bm l$为非零常向量；
\item 锥面：$\bm r=\bm a_0+v\bm l(u)$，$\bm a_0$为常向量；
\item 空间曲线的切线面：$\bm r=\bm a(u)+v\bm a'(u)$。
\end{enumerate}
\begin{figure}[H]
\centering
\includegraphics[scale=0.3]{可展曲面.png}
\caption{可展曲面：柱面、锥面、切线面}
\label{figure:developablesurface}
\end{figure}
\end{remark}

\begin{theorem}
设直纹面$S:\bm r=\bm a(u)+v\bm l(u)$，则$S$为可展曲面的充分必要条件是
\begin{align}
(\bm a'(u),\bm l(u),\bm l'(u))=0.
\label{eq:developablecond}
\end{align}
\end{theorem}
\begin{remark}
单叶双曲面、双曲抛物面存在两组直母线，但均不满足可展条件；可展性与直纹面的参数选取无关，仅需验证一种参数表示即可。
\end{remark}
\begin{proof}
对直纹面求偏导得
\begin{align*}
\bm r_u=\bm a'(u)+v\bm l'(u),\quad \bm r_v=\bm l(u).
\end{align*}
曲面法向量为$\bm r_u\times\bm r_v=\big(\bm a'(u)+v\bm l'(u)\big)\times\bm l(u)$。

{\heiti 必要性}:
若$S$为可展曲面，则对直母线上任意两点$(u,v_1),(u,v_2)\;(v_1\neq v_2)$，法向量互相平行，即
$$\big((\bm a'+v_1\bm l')\times\bm l\big)\parallel\big((\bm a'+v_2\bm l')\times\bm l\big).$$
由双重向量积公式$(\bm a\times\bm b)\times\bm c=(\bm a\cdot\bm c)\bm b-(\bm b\cdot\bm c)\bm a$，计算得
\begin{align*}
0&=\big((\bm a'+v_1\bm l')\times\bm l\big)\times\big((\bm a'+v_2\bm l')\times\bm l\big)\\
&=(v_1-v_2)(\bm a',\bm l,\bm l')\bm l.
\end{align*}
因$(v_1-v_2)\bm l\neq\bm 0$，故$(\bm a',\bm l,\bm l')=0$。

{\heiti 充分性}:
上述推导可逆，故条件\eqref{eq:developablecond}成立时，$S$为可展曲面.

\end{proof}

\begin{theorem}
可展曲面在局部上是柱面、锥面、空间正则曲线的切线面，或三者沿直母线的光滑拼接。
\end{theorem}
\begin{proof}
设可展曲面$S:\bm r=\bm a(u)+v\bm l(u)$满足$(\bm a',\bm l,\bm l')=0$，分两种情形讨论：

\begin{enumerate}[(1)]
\item 若$\bm l(u)\times\bm l'(u)\equiv\bm 0$，则$\bm l(u)$方向固定，直母线互相平行，$S$为柱面。
\item 若$\bm l(u)\times\bm l'(u)\not\equiv\bm 0$，则$\bm l,\bm l'$线性无关，由混合积条件得$\bm a'=\alpha(u)\bm l+\beta(u)\bm l'$。
作准线变换$\bm b(u)=\bm a(u)-\beta(u)\bm l(u)$，求导得
$$\bm b'(u)=\big(\alpha(u)-\beta'(u)\big)\bm l(u).$$
\begin{enumerate}[(i)]
\item 若$\alpha(u)-\beta'(u)\equiv0$，则$\bm b'(u)\equiv\bm 0$，$\bm b=\bm b_0$为常向量，$S$的直母线过定点$\bm b_0$，故$S$为锥面$\bm r=\bm b_0+(v+\beta(u))\bm l(u)$；
\item 若$\alpha(u)-\beta'(u)\not\equiv0$，则$\bm b(u)$为正则曲线，此时
$$\bm r=\bm b(u)+\frac{v+\beta(u)}{\alpha(u)-\beta'(u)}\bm b'(u),$$
即$S$为正则曲线$\bm b(u)$的切线面。
\end{enumerate}
\end{enumerate}

在$\bm l\times\bm l'$或$\alpha-\beta'$的零点处，为三类曲面的拼接位置.

\end{proof}

\begin{theorem}
可展曲面在局部上可与平面建立\textbf{保长对应}。
\end{theorem}
\begin{proof}
仅需证明柱面、锥面、切线面可局部与平面保长对应。
\begin{enumerate}[(1)]
\item \textbf{柱面}:
设柱面$\bm r=\bm a(u)+v\bm l_0$，$|\bm l_0|=1$。作准线变换$\bm b(u)=\bm a(u)-\big(\bm a(u)\cdot\bm l_0\big)\bm l_0$，满足$\bm b'(u)\perp\bm l_0$。
取参数变换
\begin{align*}
\tilde u=\int_{u_0}^{u}|\bm b'(t)|\mathrm{d}t,\quad \tilde v=v-\bm a(u)\cdot\bm l_0,
\end{align*}
则柱面化为$\bm r=\tilde{\bm b}(\tilde u)+\tilde v\bm l_0$，满足$|\tilde{\bm b}'(\tilde u)|=1$，$\tilde{\bm b}'\perp\bm l_0$。
此时第一基本形式为
\begin{align*}
\mathrm{I}=(\mathrm{d}\tilde u)^2+(\mathrm{d}\tilde v)^2,
\end{align*}
与平面第一基本形式一致，故柱面与平面保长对应。

\item \textbf{锥面}:
设锥面$\bm r=\bm a_0+v\bm l(u)$，$|\bm l(u)|=1$。取参数变换
\begin{align*}
\tilde u=\int_{u_0}^{u}|\bm l'(t)|\mathrm{d}t,\quad \tilde v=v,
\end{align*}
则$\tilde u$为单位球面曲线$\bm l(\tilde u)$的弧长参数，此时第一基本形式为
$$\mathrm{I}=\tilde v^2(\mathrm{d}\tilde u)^2+(\mathrm{d}\tilde v)^2.$$
作平面极坐标变换$x=\tilde v\cos\tilde u,\;y=\tilde v\sin\tilde u$，则$\mathrm{I}=(\mathrm{d}x)^2+(\mathrm{d}y)^2$，故锥面与平面保长对应。

\item \textbf{切线面}:
设正则曲线$C:\bm r=\bm r(s)$（$s$为弧长参数）的切线面为$\bm r=\bm r(s)+t\bm\alpha(s)$，其中$\bm\alpha$为单位切向量。
计算第一基本量：$E=1+t^2\kappa^2,\;F=1,\;G=1$，第一基本形式为
$$\mathrm{I}=(1+t^2\kappa^2)(\mathrm{d}s)^2+2\mathrm{d}s\mathrm{d}t+(\mathrm{d}t)^2.$$
该形式仅依赖曲率$\kappa(s)$，与挠率无关。由曲线论基本定理，存在平面曲线与$C$有相同曲率函数，其切线面为平面区域，故原切线面与平面保长对应。
\end{enumerate}

\end{proof}

\begin{definition}
设$\{S_\alpha\}$为单参数正则曲面族，若正则曲面$S$满足：
\begin{enumerate}[(1)]
\item $S$上每一点属于某个$S_\alpha$，且$S$与$S_\alpha$在该点切平面相同；
\item 每个$S_\alpha$均与$S$在某点相切。
\end{enumerate}
则称$S$为曲面族$\{S_\alpha\}$的\textbf{包络}。
\end{definition}

\begin{proposition}
设单参数曲面族$F(x,y,z,\alpha)=0$，其包络$S$上的点满足方程组
\begin{align}
\begin{cases}
F(x,y,z,\alpha)=0,\\
F_\alpha(x,y,z,\alpha)=0.
\end{cases}
\label{eq:envelopesys}
\end{align}
消去参数$\alpha$得到判别曲面$\Phi(x,y,z)=0$；若判别曲面与原曲面族均正则，则该判别曲面即为包络。
\end{proposition}
\begin{proof}
包络上任意点$(x,y,z)$对应参数$\alpha(x,y,z)$，满足$F(x,y,z,\alpha(x,y,z))=0$。
包络与$S_\alpha$相切，故法向量平行，即$F_\alpha=0$。
联立得方程组\eqref{eq:envelopesys}，消去$\alpha$即得判别曲面。
正则性条件保证判别曲面即为包络.

\end{proof}

\begin{example}
求单参数平面族$x\cos\alpha+y\sin\alpha-z\sin\alpha=1$的包络。
\end{example}
\begin{solution}
令$F(x,y,z,\alpha)=x\cos\alpha+y\sin\alpha-z\sin\alpha-1$，则
$$F_\alpha(x,y,z,\alpha)=-x\sin\alpha+y\cos\alpha-z\cos\alpha.$$
联立$F=0,\;F_\alpha=0$，消去参数$\alpha$得
$$x^2+(y-z)^2=1,$$
为柱面（可展曲面），其参数方程为
$$x=\cos u,\quad y=\sin u+v,\quad z=v.$$

\end{solution}

\begin{example}

\begin{enumerate}[(1)]

\item 确定相应的可展曲面的类型.
\end{enumerate}
\end{example}

\begin{example}
确定下列曲面是不是可展曲面?
\begin{enumerate}[(1)]
\item $xyz=k^3$;
\item $xy=(z-k)^2$.
\end{enumerate}
\end{example}

\begin{example}
证明: 单参数直线族
\begin{align*}
y=ux-u^3,\quad z=u^3y-u^6
\end{align*}
生成一个可展曲面, 其中$u$是直线族的参数.
\end{example}

\begin{example}
证明: 由挠率不为零的正则曲线的主法线族和次法线族分别生成的直纹面都不是可展曲面.
\end{example}

\begin{example}
证明: 挠率不为零的正则曲线必有由其单参数法线族构成一个可展曲面.
\end{example}

\begin{example}
设曲线$C:\bm{r}=\bm{r}(s)$的Frenet标架为$\{\bm{r}(s);\bm{\alpha}(s),\bm{\beta}(s),\bm{\gamma}(s)\}$,$s$为弧长参数. 求定义在曲线上的向量场
\begin{align*}
\bm{l}(s)=\lambda(s)\bm{\alpha}(s)+\mu(s)\bm{\beta}(s),
\end{align*}
使得以$C$为准线、$\bm{l}(s)$为直母线的方向向量的直纹面是可展曲面.
\end{example}

\begin{example}
设$C$是直纹面$S$上与直母线处处正交的一条曲线, 由直纹面$S$沿曲线$C$的法线又构成另一个直纹面$\tilde{S}$. 证明: $\tilde{S}$是可展曲面的充分必要条件为$S$是可展曲面.
\end{example}

\end{example}
