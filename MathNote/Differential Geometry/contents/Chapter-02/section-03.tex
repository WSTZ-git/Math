\section{第一基本形式}

\begin{definition}
设正则参数曲面$S:\boldsymbol{r}=\boldsymbol{r}(u,v)$，对任意$p\in S$，曲面在$p$处的\textbf{切空间}$T_pS$是由切向量$\boldsymbol{r}_u(u,v),\boldsymbol{r}_v(u,v)$张成的二维实向量空间，且$T_pS\subseteq\mathbb{R}^3$。

对任意切向量$\mathrm{d}\boldsymbol{r}=\boldsymbol{r}_u\mathrm{d}u+\boldsymbol{r}_v\mathrm{d}v$，定义
\begin{align}
E(u,v)&=\boldsymbol{r}_u\cdot\boldsymbol{r}_u,\label{eq:E}\\
F(u,v)&=\boldsymbol{r}_u\cdot\boldsymbol{r}_v=\boldsymbol{r}_v\cdot\boldsymbol{r}_u,\label{eq:F}\\
G(u,v)&=\boldsymbol{r}_v\cdot\boldsymbol{r}_v,\label{eq:G}
\end{align}
称$E,F,G$为曲面$S$的\textbf{第一类基本量}，其构成的矩阵$\begin{pmatrix}E&F\\F&G\end{pmatrix}$为\textbf{度量矩阵}。

定义二次微分式
\begin{align}
\mathrm{I}&=\mathrm{d}\boldsymbol{r}\cdot\mathrm{d}\boldsymbol{r}=E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2,\label{eq:firstfundform}
\end{align}
称$\mathrm{I}$为曲面$S$的\textbf{第一基本形式}。

定义\textbf{面积元素}
\begin{align}
\mathrm{d}\sigma=\sqrt{EG-F^2}\mathrm{d}u\mathrm{d}v,\label{eq:areaelement}
\end{align}
曲面$S$的面积为
\begin{align}
A=\iint_D\sqrt{EG-F^2}\mathrm{d}u\mathrm{d}v,\label{eq:area}
\end{align}
其中$D$为曲面的参数定义域。
\end{definition}

\begin{proposition}
正则参数曲面的第一类基本量满足$E>0,G>0,EG-F^2>0$，即度量矩阵为正定矩阵。
\end{proposition}
\begin{proof}
由正则曲面条件，$\boldsymbol{r}_u,\boldsymbol{r}_v$线性无关。
\[
E=|\boldsymbol{r}_u|^2>0,\quad G=|\boldsymbol{r}_v|^2>0,
\]
\begin{align*}
EG-F^2&=|\boldsymbol{r}_u|^2|\boldsymbol{r}_v|^2-(\boldsymbol{r}_u\cdot\boldsymbol{r}_v)^2\\
&=|\boldsymbol{r}_u|^2|\boldsymbol{r}_v|^2\big(1-\cos^2\angle(\boldsymbol{r}_u,\boldsymbol{r}_v)\big)\\
&=|\boldsymbol{r}_u\times\boldsymbol{r}_v|^2>0.
\end{align*}

\end{proof}

\begin{proposition}
曲面的第一基本形式在容许参数变换下形式不变。
\end{proposition}
\begin{proof}
设容许参数变换$u=u(\tilde{u},\tilde{v}),v=v(\tilde{u},\tilde{v})$，其Jacobi矩阵为
\begin{align}
J=\begin{pmatrix}
\frac{\partial u}{\partial\tilde{u}}&\frac{\partial v}{\partial\tilde{u}}\\
\frac{\partial u}{\partial\tilde{v}}&\frac{\partial v}{\partial\tilde{v}}
\end{pmatrix},\label{eq:jacobi}
\end{align}
由链式法则，
\[
\begin{pmatrix}\boldsymbol{r}_{\tilde{u}}\\\boldsymbol{r}_{\tilde{v}}\end{pmatrix}=J\begin{pmatrix}\boldsymbol{r}_u\\\boldsymbol{r}_v\end{pmatrix},\quad (\mathrm{d}u,\mathrm{d}v)=(\mathrm{d}\tilde{u},\mathrm{d}\tilde{v})J.
\]
记新参数下第一类基本量矩阵为$\begin{pmatrix}\tilde{E}&\tilde{F}\\\tilde{F}&\tilde{G}\end{pmatrix}$，则
\[
\begin{pmatrix}\tilde{E}&\tilde{F}\\\tilde{F}&\tilde{G}\end{pmatrix}=J\begin{pmatrix}E&F\\F&G\end{pmatrix}J^\mathrm{T}.
\]
因此
\begin{align*}
\tilde{E}(\mathrm{d}\tilde{u})^2+2\tilde{F}\mathrm{d}\tilde{u}\mathrm{d}\tilde{v}+\tilde{G}(\mathrm{d}\tilde{v})^2
&=(\mathrm{d}\tilde{u},\mathrm{d}\tilde{v})\begin{pmatrix}\tilde{E}&\tilde{F}\\\tilde{F}&\tilde{G}\end{pmatrix}\begin{pmatrix}\mathrm{d}\tilde{u}\\\mathrm{d}\tilde{v}\end{pmatrix}\\
&=(\mathrm{d}u,\mathrm{d}v)\begin{pmatrix}E&F\\F&G\end{pmatrix}\begin{pmatrix}\mathrm{d}u\\\mathrm{d}v\end{pmatrix}\\
&=E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2,
\end{align*}
即第一基本形式形式不变。

\end{proof}

\begin{proposition}
设切向量$\mathrm{d}\boldsymbol{r}=\boldsymbol{r}_u\mathrm{d}u+\boldsymbol{r}_v\mathrm{d}v$，$\delta\boldsymbol{r}=\boldsymbol{r}_u\delta u+\boldsymbol{r}_v\delta v$，则
\begin{enumerate}[(1)]
\item 内积：$\mathrm{d}\boldsymbol{r}\cdot\delta\boldsymbol{r}=E\mathrm{d}u\delta u+F(\mathrm{d}u\delta v+\mathrm{d}v\delta u)+G\mathrm{d}v\delta v$；
\item 长度：$|\mathrm{d}\boldsymbol{r}|=\sqrt{E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2}=\sqrt{\mathrm{I}(\mathrm{d}\boldsymbol{r},\mathrm{d}\boldsymbol{r})}$；
\item 夹角余弦：
\[
\cos\angle(\mathrm{d}\boldsymbol{r},\delta\boldsymbol{r})=\frac{E\mathrm{d}u\delta u+F(\mathrm{d}u\delta v+\mathrm{d}v\delta u)+G\mathrm{d}v\delta v}{\sqrt{E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2}\sqrt{E(\delta u)^2+2F\delta u\delta v+G(\delta v)^2}};
\]
\item 正交充要条件：$E\mathrm{d}u\delta u+F(\mathrm{d}u\delta v+\mathrm{d}v\delta u)+G\mathrm{d}v\delta v=0$。
\end{enumerate}
\end{proposition}
\begin{proof}
(1) 由内积双线性，
\begin{align*}
\mathrm{d}\boldsymbol{r}\cdot\delta\boldsymbol{r}
&=(\boldsymbol{r}_u\mathrm{d}u+\boldsymbol{r}_v\mathrm{d}v)\cdot(\boldsymbol{r}_u\delta u+\boldsymbol{r}_v\delta v)\\
&=(\boldsymbol{r}_u\cdot\boldsymbol{r}_u)\mathrm{d}u\delta u+(\boldsymbol{r}_u\cdot\boldsymbol{r}_v)\mathrm{d}u\delta v+(\boldsymbol{r}_v\cdot\boldsymbol{r}_u)\mathrm{d}v\delta u+(\boldsymbol{r}_v\cdot\boldsymbol{r}_v)\mathrm{d}v\delta v\\
&=E\mathrm{d}u\delta u+F(\mathrm{d}u\delta v+\mathrm{d}v\delta u)+G\mathrm{d}v\delta v.
\end{align*}
(2)(3)(4)直接由内积、长度、夹角定义可得。

\end{proof}

\begin{theorem}
正则参数曲面$S:\boldsymbol{r}=\boldsymbol{r}(u,v)$上，参数曲线网为正交曲线网的充分必要条件是$F(u,v)\equiv0$。
\end{theorem}
\begin{proof}
参数曲线$u$-曲线切向量为$\boldsymbol{r}_u$，$v$-曲线切向量为$\boldsymbol{r}_v$，则
\[
\cos\angle(\boldsymbol{r}_u,\boldsymbol{r}_v)=\frac{\boldsymbol{r}_u\cdot\boldsymbol{r}_v}{|\boldsymbol{r}_u||\boldsymbol{r}_v|}=\frac{F}{\sqrt{EG}}.
\]
$\boldsymbol{r}_u\perp\boldsymbol{r}_v$当且仅当$\cos\angle(\boldsymbol{r}_u,\boldsymbol{r}_v)=0$，即$F\equiv0$。

\end{proof}

\begin{proposition}
设曲面$S$上曲线$u=u(t),v=v(t),a\leqslant t\leqslant b$，则曲线弧长为
\begin{align}
L=\int_a^b\sqrt{E\left(\frac{\mathrm{d}u}{\mathrm{d}t}\right)^2+2F\frac{\mathrm{d}u}{\mathrm{d}t}\frac{\mathrm{d}v}{\mathrm{d}t}+G\left(\frac{\mathrm{d}v}{\mathrm{d}t}\right)^2}\mathrm{d}t.\label{eq:arclength}
\end{align}
\end{proposition}
\begin{proof}
曲线切向量$\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}=\boldsymbol{r}_u\frac{\mathrm{d}u}{\mathrm{d}t}+\boldsymbol{r}_v\frac{\mathrm{d}v}{\mathrm{d}t}$，由切向量长度公式，
\[
\left|\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}\right|=\sqrt{E\left(\frac{\mathrm{d}u}{\mathrm{d}t}\right)^2+2F\frac{\mathrm{d}u}{\mathrm{d}t}\frac{\mathrm{d}v}{\mathrm{d}t}+G\left(\frac{\mathrm{d}v}{\mathrm{d}t}\right)^2},
\]
由弧长定义$L=\int_a^b\left|\frac{\mathrm{d}\boldsymbol{r}}{\mathrm{d}t}\right|\mathrm{d}t$即得。

\end{proof}

\begin{proposition}
曲面面积$\iint_D\sqrt{EG-F^2}\mathrm{d}u\mathrm{d}v$在容许参数变换下不变。
\end{proposition}
\begin{proof}
由$\tilde{E}\tilde{G}-\tilde{F}^2=(EG-F^2)\left(\frac{\partial u}{\partial\tilde{u}}\frac{\partial v}{\partial\tilde{v}}-\frac{\partial u}{\partial\tilde{v}}\frac{\partial v}{\partial\tilde{u}}\right)^2$，得
\[
\sqrt{\tilde{E}\tilde{G}-\tilde{F}^2}=\sqrt{EG-F^2}\left|\det J\right|,
\]
结合重积分变量替换公式，
\[
\iint_D\sqrt{EG-F^2}\mathrm{d}u\mathrm{d}v=\iint_{\tilde{D}}\sqrt{\tilde{E}\tilde{G}-\tilde{F}^2}\mathrm{d}\tilde{u}\mathrm{d}\tilde{v},
\]
故面积与参数选取无关。

\end{proof}

\begin{example}
求旋转面的第一基本形式。
\end{example}
\begin{solution}

旋转面参数方程为$\boldsymbol{r}(u,v)=\big(f(v)\cos u,f(v)\sin u,g(v)\big)$，其中$f(v)>0$。
计算偏导向量：
\begin{align*}
\boldsymbol{r}_u&=\big(-f(v)\sin u,f(v)\cos u,0\big),\\
\boldsymbol{r}_v&=\big(f'(v)\cos u,f'(v)\sin u,g'(v)\big).
\end{align*}
第一类基本量：
\begin{align*}
E&=\boldsymbol{r}_u\cdot\boldsymbol{r}_u=f^2(v),\\
F&=\boldsymbol{r}_u\cdot\boldsymbol{r}_v=0,\\
G&=\boldsymbol{r}_v\cdot\boldsymbol{r}_v=f'^2(v)+g'^2(v).
\end{align*}
因此第一基本形式为
\begin{align*}
\mathrm{I}=f^2(v)(\mathrm{d}u)^2+\big(f'^2(v)+g'^2(v)\big)(\mathrm{d}v)^2.
\end{align*}
此时$F\equiv0$，故旋转面的$u$-曲线与$v$-曲线构成正交参数曲线网。

\end{solution}


\begin{example}
求曲面上参数曲线的二等分角轨线所满足的微分方程。
\end{example}
\begin{solution}

设曲面第一基本形式为$\mathrm{I}=E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2$。
$u$-曲线方向向量为$(1,0)$，$v$-曲线方向向量为$(0,1)$，设二等分角轨线方向向量为$(\mathrm{d}u,\mathrm{d}v)$。
该方向与$u$-曲线夹角余弦为$\frac{E\mathrm{d}u+F\mathrm{d}v}{\sqrt{E}\sqrt{E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2}}$，
与$v$-曲线夹角余弦为$\frac{F\mathrm{d}u+G\mathrm{d}v}{\sqrt{G}\sqrt{E(\mathrm{d}u)^2+2F\mathrm{d}u\mathrm{d}v+G(\mathrm{d}v)^2}}$。
由二等分角条件，夹角余弦相等或互为相反数，故
\[
\frac{E\mathrm{d}u+F\mathrm{d}v}{\sqrt{E}}=\pm\frac{F\mathrm{d}u+G\mathrm{d}v}{\sqrt{G}}.
\]
结合$EG-F^2>0$化简，得
\[
\sqrt{E}\mathrm{d}u\pm\sqrt{G}\mathrm{d}v=0.
\]
几何上，$u$-曲线单位切向量为$\frac{\boldsymbol{r}_u}{\sqrt{E}}$，$v$-曲线单位切向量为$\frac{\boldsymbol{r}_v}{\sqrt{G}}$，角平分线方向为$\frac{\boldsymbol{r}_u}{\sqrt{E}}\mp\frac{\boldsymbol{r}_v}{\sqrt{G}}$，对应$(\mathrm{d}u,\mathrm{d}v)=\big(\frac{1}{\sqrt{E}},\mp\frac{1}{\sqrt{G}}\big)$，与上述微分方程一致。

\end{solution}

\begin{example}
在曲面
\begin{align*}
\boldsymbol{r}=\left(u\cos v,u\sin v,\frac{u^{2}}{2}\right)
\end{align*}
上考虑经过原点$O=(0,0,0)$的曲线$v=ku^{2}$.求该曲线从原点$O$到任意一点$u=t$的弧长.

设在曲面上一点$(u,v)$，由微分$\mathrm{d}u,\mathrm{d}v$的二次方程
\begin{align*}
P(u,v)(\mathrm{d}u)^{2}+2Q(u,v)\mathrm{d}u\mathrm{d}v+R(u,v)(\mathrm{d}v)^{2}=0
\end{align*}
确定了在该点的两个切方向.证明:这两个切方向彼此正交的充分必要条件是函数$P,Q,R$满足方程
\begin{align*}
ER-2FQ+GP=0,
\end{align*}
其中$E,F,G$是曲面的第一类基本量.

在球面上求与经线相交成固定角的轨线的方程.

考虑球面
\begin{align*}
\boldsymbol{r}=(a\cos u\cos v,a\cos u\sin v,a\sin u),\quad a>0
\end{align*}
上的曲线$u=v$.求该曲线分别与曲线$v=v_{0}$(常数)和曲线$u=u_{0}$(常数)的夹角.

已知曲面的第一基本形式为$\mathrm{I}=(\mathrm{d}u)^{2}+(u^{2}+a^{2})(\mathrm{d}v)^{2}$.
\begin{enumerate}[(1)]
\item 求曲线$C_{1}:u+v=0$与$C_{2}:u-v=0$的交角;
\item 求曲线$C_{1}:u=\frac{a}{2}v^{2}$,$C_{2}:u=-\frac{a}{2}v^{2}$和$C_{3}:v=1$所围成的曲边三角形的各个边长和各个内角;
\item 求曲线$C_{1}:u=av$,$C_{2}:u=-av$和$C_{3}:v=1$所围成的曲边三角形的面积.
\end{enumerate}
\end{example}
