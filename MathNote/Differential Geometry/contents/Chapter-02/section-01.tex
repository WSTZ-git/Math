\section{正则参数曲面}

\begin{definition}
所谓的\textbf{参数曲面} \( S \) 是指从 \( E^2 \) 的一个区域 \( D \)（即 \( E^2 \) 中的一个连通开子集）到空间 \( E^3 \) 的一个连续映射 \( S : D \to E^3 \). 在 \( E^2 \) 和 \( E^3 \) 中分别建立笛卡儿直角坐标系, 用 \( (u, v) \) 记 \( E^2 \) 中的点的坐标, 用 \( (x, y, z) \) 记 \( E^3 \) 中的点的坐标, 则参数曲面 \( S \) 的方程可以表示为
\begin{align*}
x = x(u, v),\quad y = y(u, v),\quad z = z(u, v),\quad (u, v) \in D,
\end{align*}
或者写成向量方程的形式
\begin{align*}
\bm{r} = \bm{r}(u, v) = (x(u, v), y(u, v), z(u, v)).
\end{align*}
通常假定函数 \( x(u, v), y(u, v), z(u, v) \) 有连续的 3 次以上的各阶偏导数. 自变量 \( u, v \) 称为曲面 \( S \) 的\textbf{参数}.并且还规定$\bm{r}_u\times \bm{r}_v$所指的一侧为曲面的\textbf{正侧}.
\end{definition}

\begin{definition}
对于曲面 \( S : \bm{r} = \bm{r}(u, v) \), 固定 \( u = u_0 \) 而让 \( v \) 变化得到的曲线 \( \bm{r}(u_0, v) \) 称为 \( \bm{u} \)-\textbf{曲线}; 固定 \( v = v_0 \) 得到的曲线 \( \bm{r}(u, v_0) \) 称为 \( \bm{v} \)-\textbf{曲线}. 这些曲线构成曲面上的\textbf{参数曲线网}. \( (u, v) \) 称为曲面上的\textbf{曲纹坐标}.
\end{definition}
\begin{remark}
在区域$D$上看,$u$-曲线和$v$-曲线分别是$E^2$中的坐标曲线.因此在直观上,参数曲面$S$是把$E^2$中的区域$D$经过伸缩、扭曲等变形后放置到$E^3$中的结果,此时$E^2$中的坐标曲线网就变成了曲面$S$上的参数曲线网(见\reffig{figure:正则参数曲线}).这就是说,从曲面$S$上看,$(u,v)$也可以作为曲面$S$上的点的坐标,也就是上面定义的曲面上的曲纹坐标.

\caption{正则参数曲线}
\label{figure:正则参数曲线}
\end{figure}

\end{remark}

\begin{definition}
设 \( S : \bm{r} = \bm{r}(u, v) \) 是参数曲面, 在点 \( p_0 = \bm{r}(u_0, v_0) \) 处的两个切向量为
\begin{align*}
\bm{r}_u(u_0, v_0) = \frac{\partial \bm{r}}{\partial u}\bigg|_{(u_0, v_0)},\quad \bm{r}_v(u_0, v_0) = \frac{\partial \bm{r}}{\partial v}\bigg|_{(u_0, v_0)}.
\end{align*}
如果 \( \bm{r}_u(u_0, v_0) \) 与 \( \bm{r}_v(u_0, v_0) \) 线性无关, 即 \( \bm{r}_u \times \bm{r}_v \big|_{(u_0, v_0)} \neq \bm{0} \), 则称曲面 \( S \) 在点 \( p_0 \) 是\textbf{正则的}. 处处是正则点且至少 3 次以上连续可微的参数曲面称为\textbf{正则参数曲面}.
\end{definition}

\begin{theorem}
设 \( S : \bm{r} = \bm{r}(u, v) \) 是正则参数曲面, 则对任意一点 \( (u_0, v_0) \in D \), 存在其邻域 \( U \subset D \), 使得曲面 \( S|_U \) 可以表示为 $\mathbf{Monge}$ \textbf{形式}
\begin{align*}
\bm{r} = (x, y, z(x, y)),
\end{align*}
其中 \( z = z(x, y) \) 是连续可微函数. 反之, 任何 Monge 形式的曲面都是正则的.
\end{theorem}
\begin{proof}

曲线在正则点处有
\begin{align*}
\boldsymbol{r}_u\times \boldsymbol{r}_v=\left( \frac{\partial \left( y,z \right)}{\partial \left( u,v \right)},\frac{\partial \left( z,x \right)}{\partial \left( u,v \right)},\frac{\partial \left( x,y \right)}{\partial \left( u,v \right)} \right) \ne 0.
\end{align*}
不妨设 \( \frac{\partial(x, y)}{\partial(u, v)}\big|_{(u_0, v_0)} \neq 0 \). 由\hyperref[Basis of Analytics-theorem:反函数存在定理]{反函数存在定理}, 存在邻域 \( U \) 使得 \( x = x(u, v), y = y(u, v) \) 有反函数 \( u = u(x, y), v = v(x, y) \). 代入 \( z = z(u, v) \) 得 \( z = z(u(x, y), v(x, y)) \), 记作 \( z(x, y) \), 于是曲面方程为 \( \bm{r} = (x, y, z(x, y)) \). 

反之,对于 Monge 形式, \( \bm{r}_x = (1, 0, z_x) \), \( \bm{r}_y = (0, 1, z_y) \), 从而 \( \bm{r}_x \times \bm{r}_y = (-z_x, -z_y, 1) \neq \bm{0} \), 故曲面正则.

\end{proof}

\begin{definition}
设正则参数曲面 \( S : \bm{r} = \bm{r}(u, v) \). 变换
\begin{align*}
\begin{cases}
u=u(\tilde{u},\tilde{v}),\\
v=v(\tilde{u},\tilde{v}),\\
\end{cases}
\end{align*}
称为\textbf{容许的参数变换}或\textbf{参数变换}, 如果它满足:
\begin{enumerate}[(1)]
\item \( u(\tilde{u}, \tilde{v}), v(\tilde{u}, \tilde{v}) \) 是 3 次以上连续可微函数;
\item \( \frac{\partial(u, v)}{\partial(\tilde{u}, \tilde{v})} \neq 0 \).
\end{enumerate}
\end{definition}

\begin{theorem}
在容许的参数变换
\begin{align*}
\begin{cases}
u=u(\tilde{u},\tilde{v}),\\
v=v(\tilde{u},\tilde{v}),\\
\end{cases}
\end{align*}
下, 正则参数曲面\( S : \bm{r} = \bm{r}(u, v) \)的正则性保持不变, 并且有
\begin{align*}
\bm{r}_{\tilde{u}} \times \bm{r}_{\tilde{v}} = \frac{\partial(u, v)}{\partial(\tilde{u}, \tilde{v})}\, (\bm{r}_u \times \bm{r}_v).
\end{align*}
特别地, 保持定向的充要条件是 \( \frac{\partial(u, v)}{\partial(\tilde{u}, \tilde{v})} > 0 \).
\end{theorem}
\begin{proof}

由链式法则,
\begin{align*}
\bm{r}_{\tilde{u}} = \bm{r}_u \frac{\partial u}{\partial \tilde{u}} + \bm{r}_v \frac{\partial v}{\partial \tilde{u}},\quad
\bm{r}_{\tilde{v}} = \bm{r}_u \frac{\partial u}{\partial \tilde{v}} + \bm{r}_v \frac{\partial v}{\partial \tilde{v}}.
\end{align*}
计算叉积得
\begin{align*}
\bm{r}_{\tilde{u}} \times \bm{r}_{\tilde{v}} = \frac{\partial(u, v)}{\partial(\tilde{u}, \tilde{v})}\, (\bm{r}_u \times \bm{r}_v).
\end{align*}
由容许参数变换定义, Jacobi 行列式非零, 故若 \( \bm{r}_u \times \bm{r}_v \neq \bm{0} \), 则 \( \bm{r}_{\tilde{u}} \times \bm{r}_{\tilde{v}} \neq \bm{0} \), 正则性保持. 曲面的正侧由 \( \bm{r}_u \times \bm{r}_v \) 的方向定义, 因此当 Jacobi 行列式为正时曲面的定向不变.

\end{proof}

\begin{definition}
设 \( S \subset E^3 \). 如果对任意一点 \( p \in S \), 存在 \( p \) 在 \( E^3 \) 中的邻域 \( V \) 以及 \( E^2 \) 中的区域 \( U \), 使得 \( U \) 与 \( V \cap S \) 之间有一个一一的、双向连续的对映 \( \bm{r} : U \to V \cap S\subset E^3 \), 并且 \( \bm{r} \) 本身是一个正则参数曲面, 则称 \( S \) 是 \( E^3 \) 中的一张\textbf{正则曲面}, 简称为\textbf{曲面}.
\end{definition}

\begin{theorem}\label{theorem:正则曲面两组曲纹坐标之间必有容许的参数变换}
设正则曲面 \( S \) 有两个正则参数表示
\begin{align*}
\bm{r}_1 : U_1 \to V_1 \cap S,\qquad \bm{r}_2 : U_2 \to V_2 \cap S,
\end{align*}
且 \( V_1 \cap V_2 \cap S \neq \varnothing \). 则在任意一点$p\in V_1\cap V_2\cap S$的附近的点都有两组曲纹坐标\( (u_1, v_1) \) 和 \( (u_2, v_2) \) ,而且它们之间必定有容许的参数变换.
\end{theorem}
\begin{proof}

设 \( \bm{r}_1(u_1, v_1) = (x_1(u_1, v_1), y_1(u_1, v_1), z_1(u_1, v_1)) \), \( \bm{r}_2(u_2, v_2) = (x_2(u_2, v_2), y_2(u_2, v_2), z_2(u_2, v_2)) \). 在$p$点附近有
\begin{align}\label{eq::09jf39j9IOFUEF}
x_1(u_1, v_1) = x_2(u_2, v_2),\quad y_1(u_1, v_1) = y_2(u_2, v_2),\quad z_1(u_1, v_1) = z_2(u_2, v_2).
\end{align}
由正则性知\( \frac{\partial(x_1, y_1)}{\partial(u_1, v_1)} \neq 0 \), 故由\hyperref[Basis of Analytics-theorem:反函数存在定理]{反函数存在定理}可知, 在点$p$对应的参数$(u_1^0,v_1^0)$的一个邻域内存在3次以上连续可微的反函数
\begin{align*}
u_1 = f(x_1, y_1), \quad v_1 = g(x_1, y_1).
\end{align*}
从而上式满足
\begin{align*}
f(x_1(u_1,v_1),y_1(u_1,v_1))\equiv u_1,\quad g(x_1(u_1,v_1),y_1(u_1,v_1))\equiv v_1.
\end{align*}
于是再结合\eqref{eq::09jf39j9IOFUEF}式可知
\begin{align}\label{eq::IAHFAZFJI0-ijw}
u_1 = f(x_2(u_2, v_2), y_2(u_2, v_2)),\quad v_1 = g(x_2(u_2, v_2), y_2(u_2, v_2)).
\end{align}
易见 \( u_1, v_1 \) 是 \( u_2, v_2 \) 的 3 次以上连续可微函数, 且
\begin{align*}
\frac{\partial(u_1, v_1)}{\partial(u_2, v_2)} = \frac{\partial(f, g)}{\partial(x, y)} \cdot \frac{\partial(x_2, y_2)}{\partial(u_2, v_2)} = \frac{\partial(x_2, y_2)}{\partial(u_2, v_2)} \left( \frac{\partial(x_1, y_1)}{\partial(u_1, v_1)} \right)^{-1} \neq 0,
\end{align*}
故变换\eqref{eq::IAHFAZFJI0-ijw}是\( (u_1, v_1) \) 和 \( (u_2, v_2) \)之间容许的参数变换.

\end{proof}

\begin{definition}
如果在正则曲面$S$的每一点的邻域内都能选定一个正则参数表示,则在邻域重叠部分的任意点$p$都有两组曲纹坐标$(u_1,v_1)$和$(u_2,v_2)$,由\refthe{theorem:正则曲面两组曲纹坐标之间必有容许的参数变换}知它们之间必存在容许的参数变换,记作
\begin{align*}
\begin{cases}
u_1=u_1\left( u_2,v_2 \right) ,\\
v_1=v_1\left( u_2,v_2 \right) ,\\
\end{cases}\, \text{其Jacobi行列式为}\frac{\partial \left( u_1,v_1 \right)}{\partial \left( u_2,v_2 \right)}.
\end{align*}
若对在邻域重叠部分的任意的点$p$,其两组曲纹坐标对应的Jacobi行列式$\frac{\partial \left( u_1,v_1 \right)}{\partial \left( u_2,v_2 \right)}$都恒大于零, 则称该正则曲面$S$是\textbf{可定向的}.
\end{definition}
\begin{remark}
在直观上看，每一个正则参数曲面是一个开的曲面片，它与$E^2$中的一个开区域是同胚的，而正则曲面是把一片片正则参数曲面粘起来的结果.我们着重研究的就是正则参数曲面以及它在容许参数变换下的不变量.
\end{remark}

\begin{example}[圆柱面]
圆柱面 \( x^2 + y^2 = a^2 \) 的一个正则参数表示为
\begin{align*}
\bm{r} = \bm{r}(u, v) = (a\cos u, a\sin u, bv),
\end{align*}
其中 \( a>0, b \) 是常数. 取 \( -\pi < u < \pi, -\infty < v < \infty \), 表示圆柱面除去直线 \( x=-a, y=0, z=bv \) 的部分; 取 \( 0<u<2\pi, -\infty<v<\infty \), 表示除去直线 \( x=a, y=0, z=bv \) 的部分. 这两片覆盖整个圆柱面, 在重叠区域有参数变换 \( \tilde{u} = u \) (当 \( 0<u<\pi \)) 或 \( \tilde{u} = u+2\pi \) (当 \( -\pi<u<0 \)), \( \tilde{v}=v \), 且 \( \frac{\partial(\tilde{u}, \tilde{v})}{\partial(u, v)} = 1 >0 \), 故圆柱面可定向.
\end{example}

\begin{example}[球面]
球面 \( x^2 + y^2 + z^2 = a^2 \) 的一个常用参数表示为
\begin{align*}
\bm{r} = \bm{r}(\theta, \varphi) = (a\cos\varphi\cos\theta, a\cos\varphi\sin\theta, a\sin\varphi),
\end{align*}
其中 \( a>0 \). 取 \( -\pi < \theta < \pi, -\frac{\pi}{2} < \varphi < \frac{\pi}{2} \), 表示球面除去从北极到南极、经过点 \( (-a,0,0) \) 的半个大圆周之后剩余的区域.
\end{example}

\begin{example}[旋转面]
设 \( C \) 是 \( Ozx \) 平面上的一条曲线, 参数方程为
\begin{align*}
x = f(v),\quad y = 0,\quad z = g(v),\quad a \leqslant v \leqslant b,
\end{align*}
且 \( f(v) > 0 \). 将它绕 \( z \) 轴旋转一周所得的旋转面的参数方程为
\begin{align*}
\bm{r} = \bm{r}(u, v) = (f(v)\cos u, f(v)\sin u, g(v)),
\end{align*}
其中 \( 0 < u < 2\pi, a < v < b \). 我们把旋转面上的\( u \)-曲线称为\textbf{纬线}(平行圆), \( v \)-曲线称为\textbf{经线}.
\end{example}

\begin{example}[正螺旋面]
正螺旋面的参数方程为
\begin{align*}
\bm{r} = \bm{r}(u, v) = (u\cos v, u\sin v, av),
\end{align*}
其中 \( a \) 是非零常数, \( -\infty < u < \infty, -\infty < v < \infty \). \( u \)-曲线是与 \( z \) 轴垂直且与之相交的直线, \( v \)-曲线是圆螺旋线.
\end{example}

\begin{definition}[直纹面]
\textbf{直纹面}是由一条直线在空间中运动所产生的曲面, 即单参数直线族所形成的曲面. 
\end{definition}

\begin{theorem}
曲面$S:\bm{r}=\bm{r}(u,v)$是直纹面的充要条件是:存在一条曲线\( \bm{r} = \bm{a}(u) \)和非零向量函数 \( \bm{l}(u) \neq \bm{0} \), 使得
\begin{align}\label{eq:::02u39j3j}
\bm{r} = \bm{r}(u, v) = \bm{a}(u) + v\,\bm{l}(u).
\end{align}
其中$\bm{l}(u)$称为\textbf{方向向量},曲线\( \bm{r} = \bm{a}(u) \)称为直纹面$S$的\textbf{准线},
而\( v \)-曲线称为直纹面$S$的\textbf{直母线}. 并且参数曲面\eqref{eq:::02u39j3j}正则的充要条件是
\begin{align*}
\bm{r}_u \times \bm{r}_v = (\bm{a}'(u) + v\,\bm{l}'(u)) \times \bm{l}(u) \neq \bm{0}.
\end{align*}
\end{theorem}
\begin{proof}

正则的充要条件只需注意到
\begin{align*}
\bm{r}_u(u,v)=\bm{a}'(u)+v\bm{l}'(u),\quad \bm{r}_v(u,v)=\bm{l}(u),
\end{align*}
即可.

\end{proof}

\begin{theorem}
设直纹面$S$由参数方程为$\bm{r}(u,v)=\bm{a}(u)+v\bm{l}(u)$，其中 $\bm{l}(u)\neq\bm{0}$。则对于任意连续可微函数 $\lambda(u)$，作准线变换 $\tilde{\bm{a}}(u)=\bm{a}(u)+\lambda(u)\bm{l}(u)$，得到的参数方程 $\tilde{\bm{r}}(u,v)=\tilde{\bm{a}}(u)+v\bm{l}(u)$ 也表示直纹面$S$。

若取
\begin{align}\label{eq::rj2jjJIOJFA}
\lambda(u)=-\frac{1}{|\bm{l}(u)|}\int_{u_0}^{u}\left(\bm{a}'(u)\cdot\frac{\bm{l}(u)}{|\bm{l}(u)|}\right)\mathrm{d}u,
\end{align}
则新准线 $\tilde{\bm{a}}(u)=\bm{a}(u)+\lambda(u)\bm{l}(u)$ 与直母线$\bm{l}(u)$垂直，即 $\tilde{\bm{a}}'(u)\cdot\bm{l}(u)=0$.

特别地,若 $\bm{l}(u)$ 已是单位向量函数，则\eqref{eq::rj2jjJIOJFA}式成为
\begin{align*}
\lambda(u) = -\int_{u_0}^{u}\bm{a}'(u)\cdot \bm{l}(u)\mathrm{d}u.
\end{align*}
\end{theorem}
\begin{proof}

对每个固定的$u\in\mathbb{R}$，令$v'=v+\lambda(u)$，则\begin{align*}
\tilde{r}(u,v)=\tilde{\bm{a}}(u)+v\bm{l}(u)=\bm{a}(u)+(v+\lambda(u))\bm{l}(u)=\bm{a}(u)+v'\bm{l}(u).
\end{align*}
因为此时$\lambda(u)$为固定的常数，且平移变换是$\mathbb{R}$上的双射，所以\begin{align*}
\left\{ \tilde{r}(u,v) \mid v\in\mathbb{R} \right\}=\left\{ \bm{a}(u)+v'\bm{l}(u) \mid v'\in\mathbb{R} \right\}=\left\{ \bm{a}(u)+v\bm{l}(u) \mid v\in\mathbb{R} \right\}=\left\{ \bm{r}(u,v) \mid v\in\mathbb{R} \right\}.
\end{align*}
再由$u$的任意性知\begin{align*}
\left\{ \tilde{r}(u,v) \mid u,v\in\mathbb{R} \right\}=\left\{ \bm{r}(u,v) \mid u,v\in\mathbb{R} \right\}.
\end{align*}
若取
\begin{align*}
\lambda(u)=-\frac{1}{|\bm{l}(u)|}\int_{u_0}^{u}\left(\bm{a}'(u)\cdot\frac{\bm{l}(u)}{|\bm{l}(u)|}\right)\mathrm{d}u,
\end{align*}
则
$\tilde{\bm{a}}'(u)=\bm{a}'(u)+\lambda'(u)\bm{l}(u)+\lambda(u)\bm{l}'(u).$
注意到$\bm{l}^2(u)=|\bm{l}(u)|^2$，求导得
\begin{align*}
2\bm{l}(u)\cdot\bm{l}'(u)=2|\bm{l}(u)|\left(|\bm{l}(u)|\right)'
\Longrightarrow
\bm{l}(u)\cdot\bm{l}'(u)=|\bm{l}(u)|\left(|\bm{l}(u)|\right)'.
\end{align*}
故\begin{align*}
\tilde{\bm{a}}'(u)\cdot\bm{l}(u)
&=\bm{a}'(u)\cdot\bm{l}(u)+\lambda'(u)|\bm{l}(u)|^2+\lambda(u)\bm{l}'(u)\cdot\bm{l}(u) \\
&=\bm{a}'(u)\cdot\bm{l}(u)+\lambda'(u)|\bm{l}(u)|^2+\lambda(u)|\bm{l}(u)|\left(|\bm{l}(u)|\right)' \\
&=|\bm{l}(u)|\left( \bm{a}'(u)\cdot\frac{\bm{l}(u)}{|\bm{l}(u)|}+\left( \lambda(u)|\bm{l}(u)| \right)' \right) \\
&=|\bm{l}(u)|\left( \bm{a}'(u)\cdot\frac{\bm{l}(u)}{|\bm{l}(u)|}+\left( -\int_{u_0}^u \frac{\bm{a}'(u)\cdot\bm{l}(u)}{|\bm{l}(u)|}\mathrm{d}u \right)' \right) \\
&=|\bm{l}(u)|\left( \bm{a}'(u)\cdot\frac{\bm{l}(u)}{|\bm{l}(u)|}-\frac{\bm{a}'(u)\cdot\bm{l}(u)}{|\bm{l}(u)|} \right) =0.
\end{align*}

\end{proof}

\begin{definition}
设直纹面$S$的参数方程为$\bm{r}(u,v)=\bm{a}(u)+v\bm{l}(u)$.
\begin{enumerate}[(1)]
\item 若方向向量$\bm{l}(u)$有固定指向,即所有直母线互相平行，此时$\bm{l}(u)\times\bm{l}'(u)=\bm{0}$，则称直纹面$S$为\textbf{柱面}.

\item 若所有直母线通过一个定点，即存在常向量 $\bm{r}_0$ 和连续可微函数 $\lambda(u)$ 使得
\begin{align*}
\bm{a}(u)+\lambda(u)\bm{l}(u)=\bm{r}_0.
\end{align*}
则直纹面$S$称为\textbf{锥面}.

\item 若 $\bm{a}'(u)=\bm{l}(u)$，则直母线是准线的切线,或者说,此时直纹面$S$是由曲线$\bm{r}=\bm{a}(u)$的切线形成的曲面,此时称直纹面$S$为曲线$\bm{r}=\bm{a}(u)$的\textbf{切线面}.
\end{enumerate}
\end{definition}
\begin{remark}
在锥面的情形，准线可以退化为定点 $\bm{r}_0$ 本身.
\end{remark}
