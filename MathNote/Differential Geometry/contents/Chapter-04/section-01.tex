\section{自然标架的运动公式}

\begin{definition}
采用\textbf{张量}记号，将正则参数曲面$S$的参数记为$u^1,u^2$，曲面参数方程为
\begin{align}
\boldsymbol{r}=\boldsymbol{r}(u^1,u^2),\label{eq:tensor_param}
\end{align}
记切向量
\begin{align}
\boldsymbol{r}_\alpha=\frac{\partial \boldsymbol{r}(u^1,u^2)}{\partial u^\alpha},\quad \alpha=1,2.\label{eq:tangent_tensor}
\end{align}
引入$\mathbf{Einstein}$\textbf{和式约定}：若单项式中同一指标字母出现两次，一次为上指标、一次为下指标，则该式表示对该指标取值$1,2$求和。希腊字母$\alpha,\beta,\gamma,\delta,\xi,\eta$等仅取$1,2$。

曲面的单位法向量为
\begin{align}
\boldsymbol{n}=\frac{\boldsymbol{r}_1\times\boldsymbol{r}_2}{|\boldsymbol{r}_1\times\boldsymbol{r}_2|}.\label{eq:normal_tensor}
\end{align}
曲面位置向量的微分为
\begin{align}
\mathrm{d}\boldsymbol{r}=\boldsymbol{r}_\alpha\mathrm{d}u^\alpha.\label{eq:dr_tensor}
\end{align}
\end{definition}

\begin{definition}
定义曲面的\textbf{第一类基本量}与\textbf{第二类基本量}的张量形式：
\begin{align}
\begin{cases}
g_{\alpha\beta}=\boldsymbol{r}_\alpha\cdot\boldsymbol{r}_\beta,\\
b_{\alpha\beta}=\boldsymbol{r}_{\alpha\beta}\cdot\boldsymbol{n}=-\boldsymbol{r}_\alpha\cdot\boldsymbol{n}_\beta=-\boldsymbol{r}_\beta\cdot\boldsymbol{n}_\alpha,
\end{cases}\label{eq:fundamental_tensor}
\end{align}
其中$\boldsymbol{r}_{\alpha\beta}=\frac{\partial}{\partial u^\beta}\left(\frac{\partial \boldsymbol{r}}{\partial u^\alpha}\right)$。
曲面的第一、第二基本形式为
\begin{align}
\mathrm{I}=g_{\alpha\beta}\mathrm{d}u^\alpha\mathrm{d}u^\beta,\quad \mathrm{II}=b_{\alpha\beta}\mathrm{d}u^\alpha\mathrm{d}u^\beta.\label{eq:form_tensor}
\end{align}
记度量矩阵行列式$g=\det(g_{\alpha\beta})=g_{11}g_{22}-(g_{12})^2$，第二类基本量行列式$b=\det(b_{\alpha\beta})=b_{11}b_{22}-(b_{12})^2$。

由于$(g_{\alpha\beta})$正定，存在逆矩阵$(g^{\alpha\beta})$，满足
\begin{align}
g^{\alpha\beta}g_{\beta\gamma}=\delta^\alpha_\gamma=
\begin{cases}
1,&\alpha=\gamma,\\
0,&\alpha\neq\gamma,
\end{cases}\label{eq:metric_inverse}
\end{align}
其中$\delta^\alpha_\gamma$为Kronecker记号，逆矩阵显式为
\begin{align}
\begin{pmatrix}
g^{11}&g^{12}\\
g^{21}&g^{22}
\end{pmatrix}
=\frac{1}{g}\begin{pmatrix}
g_{22}&-g_{12}\\
-g_{21}&g_{11}
\end{pmatrix}.\label{eq:metric_inv_form}
\end{align}
\end{definition}

\begin{theorem}
曲面$S$的自然标架场$\{\boldsymbol{r};\boldsymbol{r}_1,\boldsymbol{r}_2,\boldsymbol{n}\}$满足\textbf{运动公式}：
\begin{align}
\begin{cases}
\dfrac{\partial \boldsymbol{r}_\alpha}{\partial u^\beta}=\Gamma^\gamma_{\alpha\beta}\boldsymbol{r}_\gamma+b_{\alpha\beta}\boldsymbol{n},\\
\dfrac{\partial \boldsymbol{n}}{\partial u^\beta}=-b^\gamma_\beta\boldsymbol{r}_\gamma,
\end{cases}\label{eq:frame_motion}
\end{align}
其中$b^\gamma_\beta=b_{\beta\xi}g^{\xi\gamma}$为第二类基本量的升指标形式，$\Gamma^\gamma_{\alpha\beta}$为Christoffel记号。
第一式称为$\mathbf{Gauss}$\textbf{公式}，第二式称为$\mathbf{Weingarten}$\textbf{公式}，$(b^\gamma_\beta)$为Weingarten映射在自然基$\{\boldsymbol{r}_1,\boldsymbol{r}_2\}$下的矩阵。
\end{theorem}
\begin{proof}
由于$\boldsymbol{r}_1,\boldsymbol{r}_2,\boldsymbol{n}$线性无关，将$\frac{\partial \boldsymbol{r}_\alpha}{\partial u^\beta},\frac{\partial \boldsymbol{n}}{\partial u^\beta}$在该标架下展开：
\begin{align*}
\begin{cases}
\dfrac{\partial \boldsymbol{r}_\alpha}{\partial u^\beta}=\Gamma^\gamma_{\alpha\beta}\boldsymbol{r}_\gamma+C_{\alpha\beta}\boldsymbol{n},\\
\dfrac{\partial \boldsymbol{n}}{\partial u^\beta}=D^\gamma_\beta\boldsymbol{r}_\gamma+D_\beta\boldsymbol{n}.
\end{cases}
\end{align*}
两边与$\boldsymbol{n}$作内积得$C_{\alpha\beta}=\boldsymbol{r}_{\alpha\beta}\cdot\boldsymbol{n}=b_{\alpha\beta}$；
由$\boldsymbol{n}$为单位向量，$\boldsymbol{n}\cdot\frac{\partial \boldsymbol{n}}{\partial u^\beta}=0$，得$D_\beta=0$；
两边与$\boldsymbol{r}_\xi$作内积得$D^\gamma_\beta g_{\gamma\xi}=-b_{\beta\xi}$，故$D^\gamma_\beta=-b_{\beta\xi}g^{\xi\gamma}=-b^\gamma_\beta$。

\end{proof}

\begin{definition}
定义$\mathbf{Christoffel}$\textbf{记号}：
\begin{enumerate}[(1)]
\item 下指标Christoffel记号：$\Gamma_{\xi\alpha\beta}=\boldsymbol{r}_{\alpha\beta}\cdot\boldsymbol{r}_\xi$；
\item 上指标Christoffel记号：$\Gamma^\gamma_{\alpha\beta}=g^{\gamma\xi}\Gamma_{\xi\alpha\beta}$。
\end{enumerate}
\end{definition}

\begin{proposition}
Christoffel记号满足：
\begin{align}
\Gamma_{\gamma\alpha\beta}=\frac{1}{2}\left(\frac{\partial g_{\alpha\gamma}}{\partial u^\beta}+\frac{\partial g_{\gamma\beta}}{\partial u^\alpha}-\frac{\partial g_{\alpha\beta}}{\partial u^\gamma}\right),\label{eq:christoffel_low}
\end{align}
\begin{align}
\Gamma^\gamma_{\alpha\beta}=\frac{1}{2}g^{\gamma\xi}\left(\frac{\partial g_{\alpha\xi}}{\partial u^\beta}+\frac{\partial g_{\xi\beta}}{\partial u^\alpha}-\frac{\partial g_{\alpha\beta}}{\partial u^\xi}\right).\label{eq:christoffel_up}
\end{align}
Christoffel记号由第一类基本量及其一阶偏导数唯一确定，且关于后两个下标对称：$\Gamma^\gamma_{\alpha\beta}=\Gamma^\gamma_{\beta\alpha},\ \Gamma_{\xi\alpha\beta}=\Gamma_{\xi\beta\alpha}$。
\end{proposition}
\begin{proof}
对$g_{\alpha\beta}=\boldsymbol{r}_\alpha\cdot\boldsymbol{r}_\beta$求偏导得$\frac{\partial g_{\alpha\beta}}{\partial u^\gamma}=\Gamma_{\beta\alpha\gamma}+\Gamma_{\alpha\beta\gamma}$，轮换指标得到另外两式，联立即可解出\eqref{eq:christoffel_low}，结合指标升降关系即得\eqref{eq:christoffel_up}。由$\boldsymbol{r}_{\alpha\beta}=\boldsymbol{r}_{\beta\alpha}$，直接可得对称性。

\end{proof}

\begin{remark}
将张量记号还原为Gauss记号$(u^1=u,u^2=v,\ g_{11}=E,g_{12}=F,g_{22}=G)$，Christoffel记号显式为
\begin{align}
\begin{cases}
\Gamma_{111}=\frac{1}{2}\frac{\partial E}{\partial u},\quad \Gamma_{112}=\Gamma_{121}=\frac{1}{2}\frac{\partial E}{\partial v},\\
\Gamma_{122}=\frac{\partial F}{\partial v}-\frac{1}{2}\frac{\partial G}{\partial u},\quad \Gamma_{211}=\frac{\partial F}{\partial u}-\frac{1}{2}\frac{\partial E}{\partial v},\\
\Gamma_{212}=\Gamma_{221}=\frac{1}{2}\frac{\partial G}{\partial u},\quad \Gamma_{222}=\frac{1}{2}\frac{\partial G}{\partial v},
\end{cases}\label{eq:christoffel_gauss_low}
\end{align}
\begin{align}
\begin{cases}
\Gamma^1_{11}=\frac{1}{EG-F^2}\left(\frac{G}{2}\frac{\partial E}{\partial u}+\frac{F}{2}\frac{\partial E}{\partial v}-F\frac{\partial F}{\partial u}\right),\\
\Gamma^1_{12}=\Gamma^1_{21}=\frac{1}{EG-F^2}\left(\frac{G}{2}\frac{\partial E}{\partial v}-\frac{F}{2}\frac{\partial G}{\partial u}\right),\\
\Gamma^1_{22}=\frac{1}{EG-F^2}\left(G\frac{\partial F}{\partial v}-\frac{G}{2}\frac{\partial G}{\partial u}-\frac{F}{2}\frac{\partial G}{\partial v}\right),\\
\Gamma^2_{11}=\frac{1}{EG-F^2}\left(-\frac{F}{2}\frac{\partial E}{\partial u}-\frac{E}{2}\frac{\partial E}{\partial v}+E\frac{\partial F}{\partial u}\right),\\
\Gamma^2_{12}=\Gamma^2_{21}=\frac{1}{EG-F^2}\left(-\frac{F}{2}\frac{\partial E}{\partial v}+\frac{E}{2}\frac{\partial G}{\partial u}\right),\\
\Gamma^2_{22}=\frac{1}{EG-F^2}\left(-F\frac{\partial F}{\partial v}+\frac{F}{2}\frac{\partial G}{\partial u}+\frac{E}{2}\frac{\partial G}{\partial v}\right).
\end{cases}\label{eq:christoffel_gauss_up}
\end{align}
若取正交参数网$F=0$，公式简化为
\begin{align}
\begin{cases}
\Gamma^1_{11}=\frac{1}{2}\frac{\partial \ln E}{\partial u},\quad \Gamma^1_{12}=\Gamma^1_{21}=\frac{1}{2}\frac{\partial \ln E}{\partial v},\\
\Gamma^1_{22}=-\frac{1}{2E}\frac{\partial G}{\partial u},\quad \Gamma^2_{11}=-\frac{1}{2G}\frac{\partial E}{\partial v},\\
\Gamma^2_{12}=\Gamma^2_{21}=\frac{1}{2}\frac{\partial \ln G}{\partial u},\quad \Gamma^2_{22}=\frac{1}{2}\frac{\partial \ln G}{\partial v}.
\end{cases}\label{eq:christoffel_ortho}
\end{align}
\end{remark}

\begin{example}
求曲面$z=f(x,y)$的Christoffel记号。
\end{example}
\begin{solution}
曲面参数方程为$\boldsymbol{r}=(x,y,f(x,y))$，对应$u^1=x,\ u^2=y$，计算切向量与二阶偏导向量：
\begin{align*}
\boldsymbol{r}_x=(1,0,f_x),\quad \boldsymbol{r}_y=(0,1,f_y),\\
\boldsymbol{r}_{xx}=(0,0,f_{xx}),\quad \boldsymbol{r}_{xy}=(0,0,f_{xy}),\quad \boldsymbol{r}_{yy}=(0,0,f_{yy}).
\end{align*}
第一类基本量：
\begin{align*}
g_{11}=1+f_x^2,\quad g_{12}=f_xf_y,\quad g_{22}=1+f_y^2,\quad g=1+f_x^2+f_y^2,
\end{align*}
逆度量矩阵：
\begin{align*}
g^{11}=\frac{1+f_y^2}{1+f_x^2+f_y^2},\quad g^{12}=-\frac{f_xf_y}{1+f_x^2+f_y^2},\quad g^{22}=\frac{1+f_x^2}{1+f_x^2+f_y^2}.
\end{align*}
下指标Christoffel记号：
\begin{align*}
\Gamma_{111}=f_xf_{xx},\quad \Gamma_{112}=\Gamma_{121}=f_xf_{xy},\quad \Gamma_{122}=f_xf_{yy},\\
\Gamma_{211}=f_yf_{xx},\quad \Gamma_{212}=\Gamma_{221}=f_yf_{xy},\quad \Gamma_{222}=f_yf_{yy}.
\end{align*}
上指标Christoffel记号：
\begin{align*}
\Gamma^1_{11}=\frac{f_xf_{xx}}{1+f_x^2+f_y^2},\quad \Gamma^1_{12}=\Gamma^1_{21}=\frac{f_xf_{xy}}{1+f_x^2+f_y^2},\quad \Gamma^1_{22}=\frac{f_xf_{yy}}{1+f_x^2+f_y^2},\\
\Gamma^2_{11}=\frac{f_yf_{xx}}{1+f_x^2+f_y^2},\quad \Gamma^2_{12}=\Gamma^2_{21}=\frac{f_yf_{xy}}{1+f_x^2+f_y^2},\quad \Gamma^2_{22}=\frac{f_yf_{yy}}{1+f_x^2+f_y^2}.
\end{align*}

\end{solution}

\begin{example}
设容许的参数变换是$u^\alpha = u^\alpha(v^1,v^2)$，命$a^\alpha_\beta = \frac{\partial u^\alpha}{\partial v^\beta}$，假定$\det(a^\alpha_\beta) > 0$。用$g_{\alpha\beta},b_{\alpha\beta}$分别表示曲面$S$在参数系$(u^1,u^2)$下的第一类基本量和第二类基本量，用$\tilde{g}_{\alpha\beta},\tilde{b}_{\alpha\beta}$分别表示曲面在参数系$(v^1,v^2)$下的第一类基本量和第二类基本量，则
\begin{align*}
\tilde{g}_{\alpha\beta} = g_{\gamma\delta}a^\gamma_\alpha a^\delta_\beta,\quad \tilde{b}_{\alpha\beta} = b_{\gamma\delta}a^\gamma_\alpha a^\delta_\beta.
\end{align*}

\end{example}

\begin{example}
用$(g^{\alpha\beta})$表示$(g_{\alpha\beta})$的逆矩阵，用$(\tilde{g}^{\alpha\beta})$表示$(\tilde{g}_{\alpha\beta})$的逆矩阵，则在上题的参数变换下有
\begin{align*}
g^{\alpha\beta} = \tilde{g}^{\gamma\delta}a^\alpha_\gamma a^\beta_\delta.
\end{align*}

\end{example}

\begin{example}
如果用$\Gamma^\gamma_{\alpha\beta}$表示关于$g_{\alpha\beta}$的Christoffel记号，用$\tilde{\Gamma}^\gamma_{\alpha\beta}$表示关于$\tilde{g}_{\alpha\beta}$的Christoffel记号，则在第1题的参数变换下有
\begin{align*}
\tilde{\Gamma}^\gamma_{\alpha\beta} = \Gamma^\nu_{\lambda\mu}a^\lambda_\alpha a^\mu_\beta A^\gamma_\nu + \frac{\partial a^\nu_\alpha}{\partial v^\beta}A^\gamma_\nu,
\end{align*}
其中$(A^\gamma_\nu)$是$(a^\lambda_\alpha)$的逆矩阵，即$A^\alpha_\beta = \frac{\partial v^\alpha}{\partial u^\beta}$.

\end{example}

\begin{example}
曲面$S$的平均曲率$H$可以表示成
\begin{align*}
H = \frac{1}{2}b_{\alpha\beta}g^{\alpha\beta},
\end{align*}
并且$H$在第1题的参数变换下是不变的.

\end{example}

\begin{example}
下列恒等式成立:
\begin{enumerate}[(1)]
\item \begin{align*}
g^{\alpha\delta}\Gamma^\beta_{\delta\gamma} + g^{\beta\delta}\Gamma^\alpha_{\delta\gamma} = -\frac{\partial g^{\alpha\beta}}{\partial u^\gamma};
\end{align*}
\item \begin{align*}
\frac{\partial g_{\gamma\beta}}{\partial u^\alpha} - \frac{\partial g_{\alpha\gamma}}{\partial u^\beta} = g_{\beta\lambda}\Gamma^\lambda_{\gamma\alpha} - g_{\alpha\lambda}\Gamma^\lambda_{\gamma\beta};
\end{align*}
\item \begin{align*}
\Gamma^\beta_{\alpha\beta} = \frac{1}{2}\frac{\partial\ln g}{\partial u^\alpha},
\end{align*}
其中$g = g_{11}g_{22} - (g_{12})^2$.
\end{enumerate}

\end{example}
