\section{凸集与不动点}

\subsection{定义与基本性质}

\begin{definition}
设$\mathscr{X}$是线性空间，$E \subseteq \mathscr{X}$，称$E$为一\textbf{凸集}，如果
\begin{align*}
\lambda x + (1 - \lambda) y \in E \quad (\forall x, y \in E, \forall 0 \leqslant \lambda \leqslant 1).
\end{align*}
\end{definition}

\begin{proposition}
若$\{E_\lambda | \lambda \in \Lambda\}$是线性空间$\mathscr{X}$中的一族凸集，则$\bigcap\limits_{\lambda \in \Lambda} E_\lambda$也是凸集。
\end{proposition}

\begin{definition}
设$\mathscr{X}$是线性空间，$A \subseteq \mathscr{X}$。若$\{E_\lambda | \lambda \in \Lambda\}$为$\mathscr{X}$中包含$A$的一切凸集，那么称$\bigcap\limits_{\lambda \in \Lambda} E_\lambda$为$A$的\textbf{凸包}，并记作$\mathrm{co}(A)$。

又对$\forall n \in \mathbb{N}, x_1, x_2, \cdots, x_n \in A$，称$\sum\limits_{i=1}^n \lambda_i x_i$为$x_1, x_2, \cdots, x_n$的\textbf{凸组合}，是指其中系数满足$\lambda_i \geqslant 0, \sum\limits_{i=1}^n \lambda_i = 1$.
\end{definition}

\begin{proposition}
设$\mathscr{X}$是线性空间，$A \subseteq \mathscr{X}$，那么$A$的凸包是$A$中元素任意凸组合的全体，即
\begin{align}
\mathrm{co}(A) = \left\{ \sum\limits_{i=1}^n \lambda_i x_i \,\bigg|\, \sum\limits_{i=1}^n \lambda_i = 1, \lambda_i \geqslant 0, x_i \in A, i = 1, 2, \cdots, n, \forall n \in \mathbb{N} \right\}. \label{eq:HJJIOu3fs1.5.1}
\end{align}
\end{proposition}
\begin{proof}
若令$S$表示\eqref{eq:HJJIOu3fs1.5.1}式右端，则$A \subseteq S$而且$S$是凸集，从而$S \supseteq \mathrm{co}(A)$。反之，设$F$为包含$A$的任一凸集，那么$x_i \in F(i = 1, 2, \cdots, n)$，从而$\sum\limits_{i=1}^n \lambda_i x_i \in F$，即得$S \subseteq F$，从而$S \subseteq \mathrm{co}(A)$.

\end{proof}

\begin{definition}
设$\mathscr{X}$是线性空间，$C$是$\mathscr{X}$上含有$\theta$的凸子集，其中$\theta$是零向量.在$\mathscr{X}$上规定一个取值于$[0, \infty]$的函数
\begin{align}
P(x) = \inf \left\{ \lambda > 0 \,\bigg|\, \frac{x}{\lambda} \in C \right\} \quad (\forall x \in \mathscr{X}) \label{eq:HJJIOu3fs1.5.2}
\end{align}
与$C$对应，称函数$P$为$C$的$\mathbf{Minkowski}$\textbf{泛函}.
\end{definition}

\begin{proposition}
设$\mathscr{X}$是线性空间，$C$是$\mathscr{X}$上含有$\theta$的凸子集。若$P$为$C$的Minkowski泛函，则$P$具有下列性质:
\begin{enumerate}[(1)]
\item 正定性:$P(x) \in [0, \infty], P(\theta) = 0$；
\item 正齐次性:$P(\lambda x) = \lambda P(x) \quad (\forall x \in \mathscr{X}, \forall \lambda > 0)$；
\item 次可加性:$P(x + y) \leqslant P(x) + P(y) \quad (\forall x, y \in \mathscr{X})$.
\end{enumerate}
\end{proposition}
\begin{proof}
只有次可加性是需要验证的。不妨设$P(x), P(y)$有穷，对$\forall \varepsilon > 0$，取$\lambda_1 = P(x) +\frac{ \varepsilon}{2}, \lambda_2 = P(y) +\frac{ \varepsilon}{2}$，则有
\begin{align*}
\frac{x}{\lambda_1} \in C, \quad \frac{y}{\lambda_2} \in C.
\end{align*}
因为$C$是凸的，所以
\begin{align*}
\frac{x + y}{\lambda_1 + \lambda_2} = \frac{\lambda_1}{\lambda_1 + \lambda_2} \cdot \frac{x}{\lambda_1} + \frac{\lambda_2}{\lambda_1 + \lambda_2} \cdot \frac{y}{\lambda_2} \in C.
\end{align*}
这表明
\begin{align*}
P(x + y) \leqslant \lambda_1 + \lambda_2 = P(x) + P(y) + \varepsilon.
\end{align*}
由$\varepsilon > 0$的任意性得到次可加性.

\end{proof}

\begin{definition}
设$\mathscr{X}$是线性空间，凸集$C\subseteq \mathscr{X}$含有零向量$\theta$.
\begin{enumerate}[(1)]
\item 若$\forall x \in \mathscr{X}, \exists \lambda > 0$，使得$\frac{x}{\lambda} \in C$,则称$C$是\textbf{吸收的}.

\item 若$x \in C \implies -x \in C$,则称$C$是\textbf{对称的}.
\end{enumerate}
\end{definition}

\begin{proposition}
设$\mathscr{X}$是线性空间，凸集$C\subseteq \mathscr{X}$含有零向量$\theta$,则
\begin{enumerate}[(1)]
\item $C$是吸收凸集当且仅当其Minkowski泛函$P(x)$是实值函数；

\item 若$C$是对称凸集,则其Minkowski泛函$P(x)$是实齐次的，即
\begin{align*}
P(\alpha x) = |\alpha| P(x) \quad (\forall \alpha \in \mathbb{R}).
\end{align*}
\end{enumerate}
\end{proposition}

\begin{definition}
复线性空间$\mathscr{X}$的一个子集$C$称为是\textbf{均衡的}，是指
\begin{align*}
x \in C \implies \alpha x \in C \quad (\forall \alpha \in \mathbb{C}, |\alpha| = 1).
\end{align*}
\end{definition}

\begin{proposition}\label{proposition:泛函分析--命题1.5.10}
复线性空间$\mathscr{X}$上的任一个均衡吸收凸集$C$，决定了这空间上的一个半范数.
\end{proposition}

\begin{proposition}\label{proposition:泛函分析--命题1.5.11}
设$\mathscr{X}$是一个$B^*$空间，$C$是一个含有零向量$\theta$的闭凸集。如果$P(x)$是$C$的Minkowski泛函，那么$P(x)$下半连续，且有
\begin{align}
C = \{ x \in \mathscr{X} \mid P(x) \leqslant 1 \}. \label{eq:HJJIOu3fs1.5.3}
\end{align}
此外，如果$C$还是有界的，那么$P(x)$适合
\begin{align*}
P(x) = 0 \iff x = \theta.
\end{align*}
又若$C$以$\theta$为一内点，那么$C$是吸收的，并且$P(x)$还是一致连续的.
\end{proposition}

\begin{proof}
\begin{enumerate}[(1)]
\item $\forall \alpha > 0$，若$x \in \alpha C$，即$\frac{x}{\alpha} \in C$，由$P(x)$的定义便有$P(x) \leqslant \alpha$；反之，若$P(x) \leqslant \alpha$，则对$\forall n \in \mathbb{N}$，有
\begin{align*}
\frac{x}{\alpha + \frac{1}{n}} \in C, \quad \frac{x}{\alpha + \frac{1}{n}} \to \frac{x}{\alpha} \quad (n \to \infty).
\end{align*}
又因为$C$是闭的，所以$x \in \alpha C$。这就证得
\begin{align*}
\alpha C = \{ x \in \mathscr{X} \mid P(x) \leqslant \alpha \} \quad (\forall \alpha > 0). 
\end{align*}
特别地，令$\alpha = 1$即得\eqref{eq:HJJIOu3fs1.5.3}式。又因为$\forall \alpha > 0, \alpha C$是闭集，所以$P$是下半连续的.
\item 因为$\theta \in C$，所以$P(\theta) = 0$是显然的。在$C$有界假定下，即$\exists r > 0$，使得$C \subseteq B(\theta, r)$，于是
\begin{align*}
\forall x \in \mathscr{X} \setminus \{\theta\} \implies \frac{rx}{\|x\|} \notin C \implies P(x) \geqslant \frac{\|x\|}{r}.
\end{align*}
由此可见，当$P(x) = 0$时，$x = \theta$.
\item 若$C$以$\theta$为内点，即$\exists r > 0$，使得$B(\theta, r) \subseteq C$，那么
\begin{align*}
\frac{rx}{2\|x\|} \in C \quad (\forall x \in \mathscr{X} \setminus \{\theta\}).
\end{align*}
因此$C$是吸收的，并且有$P(x) \leqslant \frac{2\|x\|}{r} (\forall x \in \mathscr{X})$。于是
\begin{align*}
|P(x) - P(y)| \leqslant \max(P(x - y), P(y - x)) \leqslant \frac{2}{r} \|x - y\| \quad (\forall x, y \in \mathscr{X}),
\end{align*}
从而$P(x)$是一致连续的.
\end{enumerate}

\end{proof}

\begin{corollary}
若$C$是$\mathbb{R}^n$中的一个紧凸子集，则必存在正整数$m \leqslant n$，使得$C$同胚于$\mathbb{R}^m$中的单位球.
\end{corollary}
\begin{proof}
\begin{enumerate}[(1)]
\item 用$E$表示包含$C$的最小闭线性子流形。设其维数是$m(\leqslant n)$。于是在$C$上必有$m + 1$个向量$e_1, e_2, \cdots, e_m, e_{m+1}$，使得$e_i - e_{m+1}(i = 1, 2, \cdots, m)$是线性无关的.
\item 令
\begin{align*}
e_0 = \frac{1}{m + 1} \sum\limits_{i=1}^{m+1} e_i.
\end{align*}
因为$e_0 \in C \subseteq E$，所以$E - e_0$是一个$m$维线性子空间。于是对$\forall y \in E$，存在唯一的表示
\begin{align}
y = \sum\limits_{i=1}^m \mu_i (e_i - e_0) + e_0, \label{eq:HJJIOu3fs1.5.5}
\end{align}
并在$E - e_0$上可引进一个等价范数
\begin{align}
\|z\| = \left( \sum\limits_{i=1}^m |\mu_i|^2 \right)^{\frac{1}{2}} \quad (z = y - e_0, y \in E). \label{eq:HJJIOu3fs1.5.6}
\end{align}
\item 我们要证：当\eqref{eq:HJJIOu3fs1.5.6}式所表示的$\|z\|$足够小时，蕴含\eqref{eq:HJJIOu3fs1.5.5}式所表示的$y \in C$。事实上，因为
\begin{align}
y = \sum\limits_{i=1}^m \mu_i e_i + \left( 1 - \sum\limits_{j=1}^m \mu_j \right) e_0 
= \sum\limits_{i=1}^m \left[ \mu_i + \frac{1}{m + 1} \left( 1 - \sum\limits_{j=1}^m \mu_j \right) \right] e_i 
+ \frac{1}{m + 1} \left( 1 - \sum\limits_{j=1}^m \mu_j \right) e_{m+1},
\label{eq:HJJIOu3fs1.5.7}
\end{align}
当$|\mu_j|(j = 1, 2, \cdots, m)$足够小时，\eqref{eq:HJJIOu3fs1.5.7}式右端各项系数都是正的，并且各项系数的总和满足
\begin{align*}
\sum\limits_{i=1}^m \left[ \mu_i + \frac{1}{m + 1} \left( 1 - \sum\limits_{j=1}^m \mu_j \right) \right] + \frac{1}{m + 1} \left( 1 - \sum\limits_{j=1}^m \mu_j \right) = 1,
\end{align*}
所以$y \in \mathrm{co}\{e_1, e_2, \cdots, e_m, e_{m+1}\} \subseteq C$.
\item 在$E - e_0$上，$C - e_0$是一个以$\theta$为内点的有界闭凸集，它的Minkowski泛函$P(z)$是$E - e_0$上的一个一致连续、正齐次、次可加泛函，适合$P(z) = 0 \iff z = \theta$。应用\refthe{theorem:泛函分析--定理1.4.22}，$\exists$常数$C_1, C_2 > 0$，使得
\begin{align*}
C_1 \|z\| \leqslant P(z) \leqslant C_2 \|z\| \quad (\forall z \in E - e_0).
\end{align*}
设$B^m(\theta, 1)$是$E - e_0$中的单位球，若令
\begin{align*}
\varphi(z) = \begin{cases}
e_0 + \frac{\|z\| z}{P(z)}, & z \neq \theta, \\
0, & z = \theta,
\end{cases}
\end{align*}
则$\varphi: B^m(\theta, 1) \to C$是一个在上同胚.
\end{enumerate}

\end{proof}

\subsection{Brouwer与Schauder不动点定理}

\begin{theorem}[Brouwer不动点定理]\label{theorem:泛函分析---Brouwer不动点定理}
设$B$是$\mathbb{R}^n$中的闭单位球，又设$T: B \to B$是一个连续映射，那么$T$必有一个不动点$x \in B$.
\end{theorem}

\begin{corollary}\label{corollary:泛函分析--推论1.5.14}
设$C$是$\mathbb{R}^n$中的一个紧凸子集，$T: C \to C$是连续的，则$T$必有一个在$C$上的不动点.
\end{corollary}
\begin{proof}
由于$C$与$\mathbb{R}^m(m \leqslant n)$中的一个单位球同胚，记此同胚为$\varphi: B^m(\theta, 1) \to C$。考察映射
\begin{align*}
T_\varphi = \varphi^{-1} \circ T \circ \varphi.
\end{align*}
显然$T_\varphi: B^m(\theta, 1) \to B^m(\theta, 1)$。对$T_\varphi$应用Brouwer不动点定理，存在$x \in B^m(\theta, 1)$使得$T_\varphi x = x$。由此得到$y = \varphi x \in C$是$T$的不动点.

\end{proof}

\begin{theorem}[Schauder不动点定理]\label{theorem:泛函分析---Schauder不动点定理}
设$C$是$B^*$空间$\mathscr{X}$中的一个闭凸子集，$T: C \to C$连续且$T(C)$列紧，则$T$在$C$上必有一个不动点.
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item 因为$T(C)$是列紧集，所以对$\forall n \in \mathbb{N}$，存在$\frac{1}{n}$网$N_n = \{y_1, y_2, \cdots, y_{r_n}\}$，即
\begin{align*}
T(C) \subseteq \bigcup\limits_{i=1}^{r_n} B\left(y_i, \frac{1}{n}\right) \quad (y_i \in T(C), i = 1, 2, \cdots, r_n).
\end{align*}
记$E_n = \mathrm{span}N_n$，即$E_n$为由$N_n$张成的有穷维线性子空间.
\item 做$T(C) \to \mathrm{co}(N_n)$的映射$I_n$如下:
\begin{align}
I_n(y) = \sum\limits_{i=1}^{r_n} y_i \lambda_i(y) \quad (\forall y \in T(C)), \label{eq:HJJIOu3fs1.5.8}
\end{align}
其中
\begin{align*}
\lambda_i(y) = \frac{m_i(y)}{\sum\limits_{i=1}^{r_n} m_i(y)},
\end{align*}
\begin{align*}
m_i(y) = \begin{cases}
1 - n\|y - y_i\|, & y \in B\left(y_i, \frac{1}{n}\right), \\
0, & y \notin B\left(y_i, \frac{1}{n}\right).
\end{cases}
\end{align*}
因为$m_i(y) \geqslant 0$，并且$\forall y \in T(C), \exists i_0(1 \leqslant i_0 \leqslant r_n)$，使得$y \in B\left(y_{i_0}, \frac{1}{n}\right)$，于是$m_{i_0}(y) > 0$，所以$\sum\limits_{i=1}^{r_n} m_i(y) > 0(\forall y \in T(C))$。因此，$\lambda_i(y)(1 \leqslant i \leqslant r_n)$有定义并满足
\begin{align}
\lambda_i(y) \geqslant 0 \quad (1 \leqslant i \leqslant r_n), \quad \sum\limits_{i=1}^{r_n} \lambda_i(y) = 1. \label{eq:HJJIOu3fs1.5.9}
\end{align}
于是$I_n$在$T(C)$上有定义，并且\eqref{eq:HJJIOu3fs1.5.8}式与\eqref{eq:HJJIOu3fs1.5.9}式蕴含$I_n(y)$是$N_n$元素的凸组合，从而$I_n(y) \in \mathrm{co}(N_n)$。此外还有
\begin{align*}
\|I_n y - y\| &= \left\| \sum\limits_{i=1}^{r_n} y_i \lambda_i(y) - \sum\limits_{i=1}^{r_n} y \lambda_i(y) \right\| \\
&= \left\| \sum\limits_{i=1}^{r_n} (y_i - y) \lambda_i(y) \right\| \leqslant \sum\limits_{i=1}^{r_n} \|y_i - y\| \lambda_i(y) \\
&= \sum_{\substack{y \in B(y_i, \frac{1}{n})}} \|y_i - y\| \lambda_i(y) + \sum_{\substack{y \notin B(y_i, \frac{1}{n})}} \|y_i - y\| \lambda_i(y) \\
&< \frac{1}{n} + 0 = \frac{1}{n}.
\end{align*}
\item 注意到$T: C \to C, N_n \subseteq T(C)$，而$C$是凸的，所以$\mathrm{co}(N_n) \subseteq C$。令$T_n \triangleq I_n \circ T$，那么$T_n: \mathrm{co}(N_n) \to \mathrm{co}(N_n)$。又注意到$\mathrm{co}(N_n)$是$E_n$中的一个有界闭凸子集，应用\refcor{corollary:泛函分析--推论1.5.14}，$\exists x_n \in \mathrm{co}(N_n) \subseteq C$，使得
\begin{align*}
T_n x_n = x_n. 
\end{align*}
又因为$T(C)$是列紧集而$C$是闭集，故存在子列$\{n_k\}\subseteq\mathbb{N}$及$x\in C$，使得
\begin{align}
Tx_{n_k}\to x\quad (k\to\infty).
\label{eq:schauder_subseq}
\end{align}
结合逼近序列$\{T_n\}$的性质，有
\begin{align*}
\|x_n - x\| &= \|T_nx_n - x\| \\
&= \|I_nTx_n - Tx_n + Tx_n - x\| \\
&\leqslant \|I_nTx_n - Tx_n\| + \|Tx_n - x\| \\
&< \frac{1}{n} + \|Tx_n - x\|\quad (\forall n\in\mathbb{N}).
\end{align*}
联合\eqref{eq:schauder_subseq}式与上式可得$x_{n_k}\to x\ (k\to\infty)$。再由$T$的连续性及\eqref{eq:schauder_subseq}式，得$Tx=x$.
\end{enumerate}
\end{proof}

\begin{definition}
设$\mathscr{X}$是$B^*$空间，$E\subseteq\mathscr{X}$，称映射$T:E\to\mathscr{X}$是\textbf{紧的}，如果$T$连续且将$E$中的任意有界集映为$\mathscr{X}$中的列紧集。
\end{definition}

\begin{corollary}\label{corollary:泛函分析--推论1.5.121}
设$C$是$B^*$空间$\mathscr{X}$中的有界闭凸子集，$T:C\to C$是紧映射，则$T$在$C$上必有不动点。
\end{corollary}

\subsection{应用}

\begin{theorem}[Caratheodory定理]\label{theorem:Caratheodory定理}
设函数$f(t,x):\mathbb{R}\times\mathbb{R}\to\mathbb{R}$在$[-h,h]\times[\xi-b,\xi+b]$上二元连续，且存在常数$M>0$使得$|f(t,x)|\leqslant M$对所有$(t,x)\in[-h,h]\times[\xi-b,\xi+b]$成立。若$h<\frac{b}{M}$，则初值问题
\begin{align*}
\begin{cases}
\dot{x}(t)=f(t,x(t)), \\
x(0)=\xi
\end{cases}
\end{align*}
在$[-h,h]$上存在解$x(t)$.
\end{theorem}

\begin{proof}
考虑空间$C[-h,h]$中的闭球$\overline{B}(\xi,b)=\{x\in C[-h,h]:\|x-\xi\|_{C[-h,h]}\leqslant b\}$，定义映射$T:\overline{B}(\xi,b)\to C[-h,h]$为
\begin{align*}
(Tx)(t)=\xi+\int_{0}^{t}f(\tau,x(\tau))\mathrm{d}\tau.
\end{align*}

首先证明$T$映$\overline{B}(\xi,b)$到自身。对任意$x\in\overline{B}(\xi,b)$，有
\begin{align*}
\|Tx-\xi\|_{C[-h,h]}=\sup_{t\in[-h,h]}\left|\int_{0}^{t}f(\tau,x(\tau))\mathrm{d}\tau\right|\leqslant Mh.
\end{align*}
当$h\leqslant\frac{b}{M}$时，$\|Tx-\xi\|_{C[-h,h]}\leqslant b$，故$T(\overline{B}(\xi,b))\subseteq\overline{B}(\xi,b)$.

其次证明$T$是紧映射。一方面，对任意$t,t'\in[-h,h]$及$x\in\overline{B}(\xi,b)$，有
\begin{align*}
|(Tx)(t)-(Tx)(t')|=\left|\int_{t'}^{t}f(\tau,x(\tau))\mathrm{d}\tau\right|\leqslant M|t-t'|,
\end{align*}
且$|(Tx)(t)|\leqslant|\xi|+Mh$，故$T(\overline{B}(\xi,b))$是一致有界且等度连续的。由Arzelà-Ascoli定理，$T(\overline{B}(\xi,b))$是列紧集。另一方面，$T$的连续性可由$f$的连续性及积分的连续性直接得到。因此$T$是紧映射。

由\refcor{corollary:泛函分析--推论1.5.121}，$T$在$\overline{B}(\xi,b)$上存在不动点$x(t)$，该不动点即为初值问题的解.

\end{proof}


\begin{proposition}\label{proposition:泛函分析--例题1.5.1}
设$\mathscr{X}$是$B^*$空间，$E$是以$\theta$为内点的真凸子集，$P$是由$E$产生的Minkowski泛函，求证:
\begin{enumerate}[(1)]
\item\label{proposition:泛函分析--例题1.5.1-1} $x \in \mathring{E} \iff P(x) < 1$;
\item\label{proposition:泛函分析--例题1.5.1-2} $\overline{\mathring{E}} = \overline{E}$.
\end{enumerate}
\end{proposition}

\begin{example}
求证: 在$B$空间中, 列紧集的凸包是列紧集.
