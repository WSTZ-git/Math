\section{Hahn-Banach定理}

\subsection{线性泛函的延拓定理}

\begin{theorem}[实Hahn-Banach定理]\label{theorem:泛函分析--实Hahn-Banach定理}
设$\mathscr{X}$是实线性空间，$p$是定义在$\mathscr{X}$上的次线性泛函，$\mathscr{X}_0$是$\mathscr{X}$的实线性子空间，$f_0$是$\mathscr{X}_0$上的实线性泛函并满足$f_0(x) \leqslant p(x) \ (\forall x \in \mathscr{X}_0)$。那么$\mathscr{X}$上必有一个实线性泛函$f$，满足：
\begin{enumerate}[(1)]
\item 受$p$控制条件:$f(x) \leqslant p(x) \ (\forall x \in \mathscr{X})$；

\item 延拓条件:$f(x) = f_0(x) \ (\forall x \in \mathscr{X}_0)$。
\end{enumerate}
\end{theorem}
\begin{proof}
对$\forall y_0 \in \mathscr{X} \setminus \mathscr{X}_0$，记$\mathscr{X}_1 \triangleq \{x + \alpha y_0 \mid x \in \mathscr{X}_0, \alpha \in \mathbb{R}\}$。首先，先证明可以将$f_0$延拓到$\mathscr{X}_1$,设延拓后的线性泛函记为$f_1$，则
\begin{align}\label{eqIOUHIO323847237894728947892}
f_1(x + \alpha y_0) = f_0(x) + \alpha f_1(y_0) \quad (\forall x \in \mathscr{X}_0, \forall \alpha \in \mathbb{R}).
\end{align}
问题的关键是确定$f_1(y_0)$的值。由于$f_1$需满足受$p$控制条件，故
\begin{align}
f_1(x + \alpha y_0) \leqslant p(x + \alpha y_0) \quad (\forall x \in \mathscr{X}_0, \forall \alpha \in \mathbb{R}). \label{eq::--HJiojio32340uroj2}
\end{align}
当$\alpha=0$时上式取等.当$\alpha >0$时，令$z=-\frac{x}{\alpha}$，则$x+\alpha y_0=\alpha \left( y_0-z \right)$。从而由\eqref{eq::--HJiojio32340uroj2}式知对$\forall z\in \mathscr{X}_0$,有
$$\alpha f_1\left( y_0-z \right) =f_1\left( \alpha \left( y_0-z \right) \right) \leqslant p\left( \alpha \left( y_0-z \right) \right) =\alpha p\left( y_0-z \right) \Longleftrightarrow f_1\left( y_0-z \right) \leqslant p\left( y_0-z \right) .$$
当$\alpha <0$时，令$y=-\frac{x}{\alpha}$，则$x+\alpha y_0=-\alpha \left( -y_0+y \right)$。从而由\eqref{eq::--HJiojio32340uroj2}式知对$\forall y\in \mathscr{X}_0$,有
$$-\alpha f_1\left( -y_0+y \right) =f_1\left( -\alpha \left( -y_0+y \right) \right) \leqslant p\left( -\alpha \left( -y_0+y \right) \right) =-\alpha p\left( -y_0+y \right) \Longleftrightarrow f_1\left( -y_0+y \right) \leqslant p\left( -y_0+y \right) .$$
故\eqref{eq::--HJiojio32340uroj2}式等价于
\begin{align*}
\begin{cases}
f_1(y_0 - z) \leqslant p(y_0 - z), & \forall z \in \mathscr{X}_0, \\
f_1(-y_0 + y) \leqslant p(-y_0 + y), & \forall y \in \mathscr{X}_0,
\end{cases}
\end{align*}
即
\begin{align*}
f_0(y) - p(-y_0 + y) \leqslant f_1(y_0) \leqslant f_0(z) + p(y_0 - z) \quad (\forall y, z \in \mathscr{X}_0).
\end{align*}
因此，为取到适合\eqref{eq::--HJiojio32340uroj2}式的$f_1(y_0)$当且仅当
\begin{align}
\sup_{y \in \mathscr{X}_0} \{f_0(y) - p(-y_0 + y)\} \leqslant \inf_{z \in \mathscr{X}_0} \{f_0(z) + p(y_0 - z)\}. \label{eq::--HJiojio32340uroj3}
\end{align}
对任意$y, z \in \mathscr{X}_0$，由条件有
\begin{align*}
f_0(y) - f_0(z) = f_0(y - z) \leqslant p(y - z) 
\leqslant p(y - y_0) + p(y_0 - z).
\end{align*}
故
\begin{align}
f_0(y) - p(-y_0 + y) \leqslant f_0(z) + p(y_0 - z) \quad (\forall y, z \in \mathscr{X}_0). \label{eq::--HJiojio32340uroj4}
\end{align}
显然\eqref{eq::--HJiojio32340uroj4}式蕴含\eqref{eq::--HJiojio32340uroj3}式。任取$f_1(y_0)$为\eqref{eq::--HJiojio32340uroj3}式两端的中间值,再结合\eqref{eqIOUHIO323847237894728947892}式即可得到$f_0$在$\mathscr{X}_1$上的延拓$f_1$且满足\eqref{eq::--HJiojio32340uroj2}式。由于\eqref{eq::--HJiojio32340uroj3}式两端未必相等，中间值$f_1(y_0)$的取法一般不唯一，因此这种延拓也不一定唯一.

接下来，利用\hyperref[lemma:泛函分析--Zorn引理]{Zorn引理}将$f_0$逐步延拓到整个$\mathscr{X}$。令
\begin{align*}
\mathscr{F} \triangleq \left\{ (\mathscr{X}_\Delta, f_\Delta) \,\middle| \,
\begin{aligned}
&\mathscr{X}_0 \subseteq \mathscr{X}_\Delta \subseteq \mathscr{X}; \\
&\forall x \in \mathscr{X}_0 \Rightarrow f_\Delta(x) = f_0(x); \\
&\forall x \in \mathscr{X}_\Delta \Rightarrow f_\Delta(x) \leqslant p(x)
\end{aligned}
\right\}.
\end{align*}
显然$(\mathscr{X}_1,f_1)\in \mathscr{F}$,故$\mathscr{F}\neq \varnothing$.
在$\mathscr{F}$中引入序关系：$(\mathscr{X}_{\Delta_1}, f_{\Delta_1}) \prec (\mathscr{X}_{\Delta_2}, f_{\Delta_2})$当且仅当$\mathscr{X}_{\Delta_1} \subseteq \mathscr{X}_{\Delta_2}$，且$f_{\Delta_1}(x) = f_{\Delta_2}(x) \ (\forall x \in \mathscr{X}_{\Delta_1})$。于是$\mathscr{F}$成为半序集。设$M$是$\mathscr{F}$中的任一全序子集，令
\begin{align*}
\mathscr{X}_M \triangleq \bigcup_{(\mathscr{X}_\Delta, f_\Delta) \in M} \mathscr{X}_\Delta,
\end{align*}
并定义
\begin{align*}
f_M(x) = f_\Delta(x) \quad (\forall x \in \mathscr{X}_\Delta, (\mathscr{X}_\Delta, f_\Delta) \in M).
\end{align*}
由于$M$是全序子集，易验证$\mathscr{X}_M$是$\mathscr{X}$的包含$\mathscr{X}_0$的子空间，且$f_M$在$\mathscr{X}_M$上是唯一确定的，满足$f_M(x) \leqslant p(x)$。因此$(\mathscr{X}_M, f_M) \in \mathscr{F}$且是$M$的一个上界。依\hyperref[lemma:泛函分析--Zorn引理]{Zorn引理}，$\mathscr{F}$本身存在极大元，记为$(\mathscr{X}_\Lambda, f_\Lambda)$。

用反证法证明$\mathscr{X}_\Lambda = \mathscr{X}$。若不然，根据第一段的证明，可构造出$(\widetilde{\mathscr{X}}_\Lambda, \widetilde{f}_\Lambda) \in \mathscr{F}$，使得$\mathscr{X}_\Lambda \subset \widetilde{\mathscr{X}}_\Lambda$且$\mathscr{X}_\Lambda \neq \widetilde{\mathscr{X}}_\Lambda$，从而$(\widetilde{\mathscr{X}}_\Lambda, \widetilde{f}_\Lambda) \succ (\mathscr{X}_\Lambda, f_\Lambda)$且$(\widetilde{\mathscr{X}}_\Lambda, \widetilde{f}_\Lambda) \neq (\mathscr{X}_\Lambda, f_\Lambda)$，这与$(\mathscr{X}_\Lambda, f_\Lambda)$的极大性矛盾。因此$\mathscr{X}_\Lambda = \mathscr{X}$，取$f = f_\Lambda$即为所求.

\end{proof}

\begin{theorem}[复Hahn-Banach定理]\label{theorem:泛函分析--复Hahn-Banach定理}
设$\mathscr{X}$是复线性空间，$p$是$\mathscr{X}$上的半范数，$\mathscr{X}_0$是$\mathscr{X}$的线性子空间，$f_0$是$\mathscr{X}_0$上的线性泛函，并满足$|f_0(x)| \leqslant p(x), \forall x \in \mathscr{X}_0$，那么$\mathscr{X}$上必有一个线性泛函$f$满足:
\begin{enumerate}[(1)]
\item $|f(x)| \leqslant p(x) \ (\forall x \in \mathscr{X})$;
\item $f(x) = f_0(x) \ (\forall x \in \mathscr{X}_0)$.
\end{enumerate}
\end{theorem}
\begin{proof}
把$\mathscr{X}$看成实线性空间，相应把$\mathscr{X}_0$也看成是实线性子空间，令
\begin{align*}
g_0(x) \triangleq \operatorname{Re} f_0(x) \quad (\forall x \in \mathscr{X}_0),
\end{align*}
则有$g_0(x) \leqslant p(x) \ (\forall x \in \mathscr{X}_0)$。根据\hyperref[theorem:泛函分析--实Hahn-Banach定理]{实Hahn-Banach定理}，存在$\mathscr{X}$上的实线性泛函$g$，使得
\begin{align}
g(x) = g_0(x) \quad (\forall x \in \mathscr{X}_0), \label{eq::--HJiojio32340uroj5}
\end{align}
且
\begin{align}
g(x) \leqslant p(x) \quad (\forall x \in \mathscr{X}). \label{eq::--HJiojio32340uroj6}
\end{align}
定义
\begin{align}
f(x) \triangleq g(x) - \mathrm{i} g(\mathrm{i}x) \quad (\forall x \in \mathscr{X}). \label{eq::--HJiojio32340uroj7}
\end{align}
依\eqref{eq::--HJiojio32340uroj5}式，有
\begin{align*}
f(x) = g_0(x) - \mathrm{i} g_0(\mathrm{i}x) 
= \operatorname{Re} f_0(x) + \mathrm{i} \operatorname{Im} f_0(x) = f_0(x) \quad (\forall x \in \mathscr{X}_0).
\end{align*}
又
\begin{align*}
f(\mathrm{i}x)=g(\mathrm{i}x)-\mathrm{i}g(-x)=g(\mathrm{i}x)+\mathrm{i}g(x)=-\mathrm{i}\left[ g(\mathrm{i}x)+\mathrm{i}g(x) \right] =\mathrm{i[}g(x)-\mathrm{i}g(\mathrm{i}x)]=\mathrm{i}f(x)\quad (\forall x\in \mathscr{X} ),
\end{align*}
故$f$也是复线性的。

接下来证明$|f(x)|$受$p(x)$控制。若$f(x) = 0$，结论显然成立。若$f(x) \neq 0$，令$\theta \triangleq \arg f(x)$，依\eqref{eq::--HJiojio32340uroj6}式，有
\begin{align*}
|f(x)| = e^{-\mathrm{i}\theta} f(x) = f(e^{-\mathrm{i}\theta} x) 
= g(e^{-\mathrm{i}\theta} x) \leqslant p(e^{-\mathrm{i}\theta} x) = p(x) \quad (\forall x \in \mathscr{X}),
\end{align*}
其中第三个等号是因为正数$f(e^{-\mathrm{i}\theta} x) = |f(x)|$的虚部为0.

\end{proof}

\begin{theorem}
若复线性空间$\mathscr{X}$中含有某一个均衡吸收真凸子集$C$,则复线性空间$\mathscr{X}$上至少有一个非零线性泛函.
\end{theorem}
\begin{proof}
由\refpro{proposition:泛函分析--命题1.5.10}知$C$的Minkowski泛函$p(x)$是空间$\mathscr{X}$的一个半范数.
因为$C$是真子集，所以任取$x_0\in \mathscr{X} \setminus C$，则由$C$是吸收的知$\theta \in C$，故$x_0\neq \theta$。记$\mathscr{X} _0\triangleq \mathrm{span}\{ x_0 \}$，定义$\mathscr{X} _0$上的线性泛函
\begin{align*}
f_0\left( \alpha x_0 \right) \triangleq \alpha \cdot p\left( x_0 \right) ,\forall \alpha \in \mathbb{C}.
\end{align*}
对$\forall \alpha x_0\in \mathscr{X} _0$，有
\begin{align*}
\left| f_0\left( \alpha x_0 \right) \right|=\left| \alpha \right|p\left( x_0 \right) =p\left( \alpha x_0 \right).
\end{align*}
由\hyperref[theorem:泛函分析--复Hahn-Banach定理]{复Hahn-Banach定理}知，存在$\mathscr{X}$上的线性泛函$f$满足
\begin{align*}
\left| f\left( x \right) \right|\leqslant p\left( x \right) \left( \forall x\in \mathscr{X} \right) \quad \text{且} \quad f\left( x \right) =f_0\left( x \right) \left( \forall x\in \mathscr{X} _0 \right).
\end{align*}
由\refpro{proposition:泛函分析--命题1.5.11}知$C=\left\{ x\in \mathscr{X} \mid p\left( x \right) \leqslant 1 \right\}$，故由$x_0\notin C$知$p\left( x_0 \right) >1$，从而
\begin{align*}
f\left( x_0 \right) =f_0\left( x_0 \right) =p\left( x_0 \right) >1.
\end{align*}
因此$f$是$\mathscr{X}$上的非零线性泛函。
\end{proof}

\begin{definition}[保范延拓]
设$\mathscr{X}$是$B^*$空间，$\mathscr{X}_0$是$\mathscr{X}$的线性子空间，$f_0$是定义在$\mathscr{X}_0$上的有界线性泛函，若有$\mathscr{X}$上的有界线性泛函$f$满足：
\begin{enumerate}[(1)]
\item 延拓条件:$f(x) = f_0(x) \ (\forall x \in \mathscr{X}_0)$；
\item 保范条件:$\|f\| = \|f_0\|_0$，其中$\|f_0\|_0$表示$f_0$在$\mathscr{X}_0$上的范数。
\end{enumerate}
则称$f$为$f_0$的\textbf{保范延拓}。
\end{definition}

\begin{theorem}[Hahn-Banach定理]\label{theorem:泛函分析--(哈恩-巴拿赫)Hahn-Banach定理}
设$\mathscr{X}$是$B^*$空间，$\mathscr{X}_0$是$\mathscr{X}$的线性子空间，$f_0$是定义在$\mathscr{X}_0$上的有界线性泛函，则在$\mathscr{X}$上必有有界线性泛函$f$满足：
\begin{enumerate}[(1)]
\item 延拓条件:$f(x) = f_0(x) \ (\forall x \in \mathscr{X}_0)$；
\item 保范条件:$\|f\| = \|f_0\|_0$，其中$\|f_0\|_0$表示$f_0$在$\mathscr{X}_0$上的范数。
\end{enumerate}
即在$\mathscr{X}$上必有$f_0$的保范延拓$f$.
\end{theorem}
\begin{proof}
在$\mathscr{X}$上定义$p(x) \triangleq \|f_0\|_0 \cdot \|x\|$，则$p(x)$是$\mathscr{X}$上的半范数。根据\hyperref[theorem:泛函分析--复Hahn-Banach定理]{复Hahn-Banach定理}，存在$\mathscr{X}$上的线性泛函$f(x)$，满足
\begin{align}
f(x) = f_0(x) \quad (\forall x \in \mathscr{X}_0), \label{eq::--HJiojio32340uroj8}
\end{align}
及
\begin{align}
|f(x)| \leqslant p(x) = \|f_0\|_0 \cdot \|x\| \quad (\forall x \in \mathscr{X}). \label{eq::--HJiojio32340uroj9}
\end{align}
由泛函范数的定义，\eqref{eq::--HJiojio32340uroj9}式蕴含$\|f\| \leqslant \|f_0\|_0$。又由\eqref{eq::--HJiojio32340uroj8}式,有
\begin{align*}
\parallel f_0\parallel _0=\underset{x\in \mathscr{X} _0\backslash \left\{ \theta \right\}}{\mathrm{sup}}\frac{\left\| f_0\left( x \right) \right\|}{\left\| x \right\|}=\underset{x\in \mathscr{X} _0\backslash \left\{ \theta \right\}}{\mathrm{sup}}\frac{\left\| f\left( x \right) \right\|}{\left\| x \right\|}\leqslant \underset{x\in \mathscr{X} \backslash \left\{ \theta \right\}}{\mathrm{sup}}\frac{\left\| f\left( x \right) \right\|}{\left\| x \right\|}=\left\| f \right\| .
\end{align*}
因此$\|f\| = \|f_0\|_0$.

\end{proof}

\begin{corollary}\label{corollary:泛函分析--推论2.4.6}
设$\mathscr{X}$是$B^*$空间，则$\forall x_0 \in \mathscr{X} \setminus \{\theta\}$，必$\exists f \in \mathscr{X}^*$，使得
\begin{align*}
f(x_0) = \|x_0\|, \quad \text{且} \quad \|f\| = 1.
\end{align*}
进而对$\forall x_1, x_2 \in \mathscr{X}$，若$x_1 \neq x_2$，则必存在$\mathscr{X}$上的非零连续线性泛函$f$，使得
\begin{align*}
f(x_0) = \|x_0\|, \quad \|f\| = 1,\quad f(x_1) \neq f(x_2).
\end{align*}
进而$x_0 = \theta$当且仅当$\forall f \in \mathscr{X}^*$蕴含$f(x_0) = 0$。
\end{corollary}
\begin{remark}
这个推论表明:每个$B^*$空间必有足够多的连续线性泛函。
\end{remark}
\begin{proof}
任给$x_1, x_2 \in \mathscr{X}$，若$x_1 \neq x_2$，则$x_0 \triangleq x_1 - x_2 \neq \theta$。令$\mathscr{X}_0 \triangleq \{\lambda x_0 \mid \lambda \in \mathbb{C}\}$，并在$\mathscr{X}_0$上定义
\begin{align*}
f_0(\lambda x_0) = \lambda \|x_0\| \quad (\forall \lambda \in \mathbb{C}).
\end{align*}
则$f_0(x_0) = \|x_0\|$且$\|f_0\|_0 = 1$,其中$\|f_0\|_0$表示$f_0$在$\mathscr{X}_0$上的范数。依\hyperref[theorem:泛函分析--(哈恩-巴拿赫)Hahn-Banach定理]{Hahn-Banach定理}，存在$\mathscr{X}$上的有界线性泛函$f$，使得
\begin{align*}
f(x_0) = f_0(x_0) = \|x_0\|, \quad \|f\| = \|f_0\|_0 = 1.
\end{align*}
又由\refpro{proposition:线性算子有界当且仅当连续}知$f$是连续的.并且$\mathscr{X}$上的这个非零连续线性泛函$f$可以分辨$x_1, x_2$,即
\begin{align*}
f(x_1) - f(x_2) = f(x_1 - x_2) = f(x_0) \neq 0.
\end{align*}

\end{proof}

\begin{theorem}\label{theorem:泛函分析--零值泛函延拓定理}
设$\mathscr{X}$是$B^*$空间，$M$是$\mathscr{X}$的线性子空间。若$x_0 \in \mathscr{X}$，且$d \triangleq \rho(x_0, M) > 0$，则必$\exists f \in \mathscr{X}^*$适合条件：
\begin{enumerate}[(1)]
\item $f(x) = 0 \ (\forall x \in M)$;
\item $f(x_0) = d$;
\item $\|f\| = 1$.
\end{enumerate}
\end{theorem}
\begin{proof}
考虑$\mathscr{X}_0 \triangleq \{x = x' + \alpha x_0 \mid x' \in M, \alpha \in \mathbb{K}\}$，$\forall x \in \mathscr{X}_0$，定义
\begin{align*}
f_0(x) = \alpha d,\quad \forall x\in \mathscr{X}_0.
\end{align*}
显然，$f_0$适合条件(1), (2)。又若$x = x' + \alpha x_0 \ (x' \in M, \alpha \neq 0)$，则
\begin{align*}
|f_0(x)|=|\alpha |d=|\alpha |\rho (x_0,M)\leqslant |\alpha |\left\| x_0-\left( -\frac{x\prime}{\alpha} \right) \right\| =|\alpha |\left\| \frac{x\prime}{\alpha}+x_0 \right\| =\parallel x\prime +\alpha x_0\parallel =\parallel x\parallel .
\end{align*}
因此$\|f_0\| \leqslant 1$。依\hyperref[theorem:泛函分析--(哈恩-巴拿赫)Hahn-Banach定理]{Hahn-Banach定理}，将$f_0$保范延拓为$f \in \mathscr{X}^*$，则$f$满足条件(1), (2)及$\|f\| \leqslant 1$。又因为$f \in \mathscr{X}^*$且满足条件(2)，故由\refpro{proposition:有界线性算子的基本不等式}可知
\begin{align*}
|f(x)| = \| f(x) \| \leqslant \| f \| \| x \| \ (\forall x\in \mathscr{X}).
\end{align*}
因为$f$满足条件$(1)$，所以对$\forall y\in M$，有
\begin{align*}
|f(x_0)| = |f(x_0-y+y)| = |f(x_0-y)+f(y)| = |f(x_0-y)| \leqslant \| f \| \| x_0-y \|.
\end{align*}
故
\begin{align*}
d = |f(x_0)| \leqslant \| f \| \inf\limits_{y\in M} \| x_0-y \| = \| f \| \rho(x_0,M) = \| f \| d \Longrightarrow \| f \| \geqslant 1.
\end{align*}
于是$\|f\| = 1$.

\end{proof}

\begin{corollary}
设$M$是$B^*$空间$\mathscr{X}$的一个子集，又设$x_0$是$\mathscr{X}$中的任一个非零元素。那么$x_0 \in \overline{\operatorname{span} M}$的充要条件是：对$\forall f \in \mathscr{X}^*$，有
\begin{align*}
f(x) = 0 \ (\forall x \in M) \implies f(x_0) = 0.
\end{align*}
\end{corollary}
\begin{remark}
若$M = \{x_1, x_2, \cdots, x_n, \cdots\}$，是否能用形如$\sum_{i=1}^n c_i x_i$的线性组合的序列极限去逼近给定的元素$x_0$？本推论给出了这种逼近存在的一个充要条件：对所有的在$x_1, x_2, \cdots, x_n, \cdots$上为0的连续线性泛函$f$都有$f(x_0) = 0$.
\end{remark}
\begin{proof}
必要性是显然的，用反证法证明充分性。倘若$x_0 \notin \overline{\operatorname{span} M}$，那么$d \triangleq \rho(x_0, \overline{\operatorname{span} M}) > 0$。依\refthe{theorem:泛函分析--零值泛函延拓定理}，$\exists f \in \mathscr{X}^*$，使得$f(x) = 0 \ (\forall x \in M)$，并且$f(x_0) = d > 0$。这与充分性假定矛盾，故充分性成立.

\end{proof}


\subsection{几何形式——凸集分离定理}

\begin{theorem}
平面上两个互不相交的凸集$A$与$B$，$A \cap B = \varnothing$，有一条重要的几何性质：存在一条直线$l$分离$A$与$B$，即存在直线$l$使$A$与$B$各在$l$的一侧。
\end{theorem}
\begin{remark}
在一般的线性空间$\mathscr{X}$中，这条几何性质有相应的推广。为简单起见，今后总假定$\mathscr{X}$是实的，$\mathscr{X}$上的线性泛函也取实值。

\end{remark}


\begin{definition}[极大线性子空间]
在线性空间$\mathscr{X}$中，$\mathscr{X}$的线性子空间$M$称为是\textbf{极大线性子空间}，如果对于任何一个以$M$为真子集的线性子空间$M_1$必有$M_1 = \mathscr{X}$。
\end{definition}
\begin{remark}
在$\mathscr{X}$上相应于平面上过原点的直线的概念是极大线性子空间的概念.
\end{remark}

\begin{proposition}\label{proposition:极大线性子空间诱导空间直和分解}
$M$是极大线性子空间的充要条件是，$M$是线性真子空间，并且$\forall x_0 \in \mathscr{X} \setminus M$有
\begin{align*}
\mathscr{X} = \{\lambda x_0 \mid \lambda \in \mathbb{R}\} \oplus M.
\end{align*}
\end{proposition}
\begin{proof}
必要性由极大线性子空间定义是显然的。为了证明充分性，设$M_1$是以$M$为真子集的线性子空间，则$\exists x_0 \in M_1 \setminus M$。于是有$\lambda x_0 \in M_1 \ (\forall \lambda \in \mathbb{R})$及$M \subseteq M_1$，从而
\begin{align*}
\mathscr{X} = \{\lambda x_0 \mid \lambda \in \mathbb{R}\} \oplus M \subseteq M_1,
\end{align*}
即得$\mathscr{X} = M_1$。于是$M$是极大线性子空间.

\end{proof}

\begin{definition}
线性空间$\mathscr{X}$的极大线性子空间$M$对向量$x_0 \in \mathscr{X}$的平移
\begin{align*}
L \triangleq x_0 + M
\end{align*}
称为\textbf{极大线性流形}，或简称\textbf{超平面}.
若$M$是$B^*$空间$\mathscr{X}$ 的极大线性子空间且$M$是闭的，则 $L$ 称为\textbf{闭超平面}。
\end{definition}
\begin{remark}
超平面是平面上一般直线概念的推广。
\end{remark}

\begin{theorem}\label{theorem:超平面是闭的当且仅当存在非零连续泛函}
$L$是线性（$B^*$）空间$\mathscr{X}$上的一个（闭）超平面当且仅当存在非零（连续）线性泛函$f$及$r \in \mathbb{R}$，使得
\begin{align*}
L=H_{f}^{r}\triangleq \left\{ x\in \mathscr{X} \mid f\left( x \right) =r \right\} .
\end{align*}
\end{theorem}
\begin{proof}
{\heiti 必要性:}设 $L$ 是 $\mathscr{X}$ 中的超平面。由定义，存在极大线性子空间 $M\subset\mathscr{X}$ 和向量 $x_0\in\mathscr{X}\setminus M$，使得 $L = x_0 + M$。由于 $M$ 是极大的，故由\refpro{proposition:极大线性子空间诱导空间直和分解}知$\mathscr{X}$ 可分解为直和 $\mathscr{X} = M \oplus \operatorname{span}\{x_0\}$。于是任意 $x\in\mathscr{X}$ 可唯一地表示为
\[
x = \lambda x_0 + y, \qquad \lambda\in\mathbb{R},\; y\in M.
\]
定义$f:\mathscr{X}\to\mathbb{R},x\mapsto \lambda$.显然 $f$ 是线性泛函，且 $f(x_0)=1$，$M = \ker f = H_f^0$。于是
\[
L = x_0 + M = \{x\in\mathscr{X}\mid f(x)=1\} = H_f^1.
\]
若$L=H_f^1$是闭的,则由\rrefpro{proposition:泛函分析--命题1.1231.31.3}{proposition:泛函分析--命题1.1231.31.3-3}知$f$连续.

{\heiti 充分性:}首先考虑 $H_f^0 = \ker f = \{ x \in \mathscr{X} \mid f(x) = 0 \}$。由于 $f$ 是线性的，$\ker f$ 是 $\mathscr{X}$ 的线性子空间。因为 $f \neq 0$，存在 $x_1 \in \mathscr{X}$ 使得 $f(x_1) \neq 0$。对$\forall x \in \mathscr{X}$，有$x - \frac{f(x)}{f(x_1)} x_1 \in \ker f$，因为
\[
f(x - \frac{f(x)}{f(x_1)} x_1) = f(x) - \frac{f(x)}{f(x_1)} f(x_1) = 0.
\]
于是
\[
x = \frac{f(x)}{f(x_1)} x_1 + (x - \frac{f(x)}{f(x_1)} x_1) \in \operatorname{span}\{x_1\} + \ker f.
\]
显然 $\operatorname{span}\{x_1\} \cap \ker f = \{0\}$，故 $\mathscr{X} = \operatorname{span}\{x_1\} \oplus \ker f$。由\refpro{proposition:极大线性子空间诱导空间直和分解}知$\ker f$ 是极大线性子空间.

现在考虑 $L = H_f^r$。取 $x_0 = r x_1$，则 $f(x_0) = r$。对任意 $x \in L$，有 $f(x) = r$，于是 $f(x - x_0) = 0$，即 $x - x_0 \in \ker f$，所以 $x \in x_0 + \ker f$。反之，若 $x = x_0 + y$ 其中 $y \in \ker f$，则 $f(x) = f(x_0) + f(y) = r$，故 $x \in L$。因此 $L = x_0 + \ker f$，又$\ker f$是极大线性子空间,故$L$是一个超平面。

若 $f$ 是连续的，则
\begin{align*}
x_n\in \mathrm{ker}f\left( \forall n\in \mathbb{N} \right) \text{且}x_n\rightarrow x\left( n\rightarrow \infty \right) \Longrightarrow f\left( x_n \right) =0\left( \forall n\in \mathbb{N} \right) \text{且}f\left( x_n \right) \rightarrow f\left( x \right) \left( n\rightarrow \infty \right) .
\end{align*}
由极限的唯一性知$f(x)=0$,因此$x\in \ker f$.故$\ker f$是闭子空间，因此 $L$ 是闭超平面。

\end{proof}

\begin{definition}
设 $\mathscr{X}$ 是实线性空间，$f: \mathscr{X} \to \mathbb{R}$ 是非零线性泛函，$r \in \mathbb{R}$。由\refthe{theorem:超平面是闭的当且仅当存在非零连续泛函}可定义超平面
\[
H_f^r = \{ x \in \mathscr{X} \mid f(x) = r \}.
\]
称集合 $E \subset \mathscr{X}$ 位于超平面 $H_f^r$ 的\textbf{一侧}，如果
\[
\forall x \in E,\quad f(x) \leqslant r \quad \text{或} \quad \forall x \in E,\quad f(x) \geqslant r.
\]

称超平面 $H_f^r$ \textbf{分离} 集合 $E, F \subset \mathscr{X}$，如果
\[
\forall x \in E,\; f(x) \leqslant r \quad \text{且} \quad \forall y \in F,\; f(y) \geqslant r,
\]
或者
\[
\forall x \in E,\; f(x) \geqslant r \quad \text{且} \quad \forall y \in F,\; f(y) \leqslant r.
\]

在上述分离的定义中，如果将不等号 $\le$ 和 $\ge$ 分别替换为 $<$ 和 $>$，
则称超平面 $H_f^r$ \textbf{严格分离} $E$ 与 $F$。
\end{definition}

\begin{theorem}[Hahn-Banach定理的几何形式]\label{theorem:Hahn-Banach定理的几何形式}
设$\mathscr{X}$是实$B^*$空间,$x_1\in \mathscr{X}$,$E$是$\mathscr{X}$上以$x_1$为内点的真凸子集,又设$x_0 \notin E$，则必存在一个超平面$H_f^r$分离$x_0$与$E$,并且超平面$H_f^r$还是闭的。
\end{theorem}
\begin{remark}
本定理对于无穷维空间$\mathscr{X}$，$E$有内点这一条是不能省略的.
\end{remark}
\begin{proof}
因为只要通过适当平移，总可以把任一点变为$\theta$点，所以不妨设$x_1=\theta$.
因为 $E$ 是以 $\theta$ 为内点的真凸子集，所以可定义 $E$ 的 Minkowski 泛函
\[
p(x) = \inf\{\lambda > 0 \mid x \in \lambda E\}, \quad x \in \mathscr{X}.
\]
则由\refpro{proposition:泛函分析--命题1.5.11}和\refpro{proposition:Minkowski泛函的基本性质}知$p$ 是一个非零的连续次线性泛函，且满足
\begin{align}\label{eq:p_le_1}
p(x) \leqslant 1,\quad\forall x \in E.
\end{align}
又由\refpro{proposition:泛函分析--命题1.5.11}知$E=\{x\in \mathscr{X}\mid P(x)\leqslant 1\}$及$x_0 \notin E$，可推出 $p(x_0) \geqslant 1$。

现在构造分离超平面。考虑由 $x_0$ 张成的一维子空间
\[
\mathscr{X}_0 = \{\lambda x_0 \mid \lambda \in \mathbb{R}\}.
\]
在 $\mathscr{X}_0$ 上定义线性泛函 $f_0$ 为
\[
f_0(\lambda x_0) = \lambda p(x_0), \quad \forall \lambda \in \mathbb{R}.
\]
显然 $f_0$ 是线性的，并且对任意 $x = \lambda x_0 \in \mathscr{X}_0$，有
\[
f_0(x) = \lambda p(x_0) \leqslant p(\lambda x_0) = p(x),
\]
这里利用了 $p$ 的齐次性。于是 $f_0$ 被 $p$ 控制。

根据\hyperref[theorem:泛函分析--实Hahn-Banach定理]{实Hahn-Banach定理}，存在 $\mathscr{X}$ 上的线性泛函 $f$，使得
\begin{align}
f|_{\mathscr{X}_0} = f_0, \quad \text{且} \quad f(x) \leqslant p(x),\ \forall x \in \mathscr{X}.\label{eq::FHJOIWJ421423}
\end{align}
特别地，
\begin{align}\label{eq:f_x0}
f(x_0) = f_0(x_0) = p(x_0) \geqslant 1,
\end{align}
并且对任意 $x \in E$，由 \eqref{eq:p_le_1} 和 $f(x) \leqslant p(x)$ 得
\begin{align}\label{eq:f_E}
f(x) \leqslant 1.
\end{align}
因此，对于超平面 $H_f^1 = \{x \in \mathscr{X} \mid f(x) = 1\}$，我们有：
\begin{enumerate}[(i)]
\item $x_0$ 满足 $f(x_0) \geqslant 1$，即 $x_0$ 位于 $H_f^1$ 的某一侧；

\item 所有 $x \in E$ 满足 $f(x) \leqslant 1$，即 $E$ 含于闭半空间 $\{x \mid f(x) \leqslant 1\}$ 中。
\end{enumerate}
从而 $H_f^1$ 分离点 $x_0$ 与凸集 $E$。

事实上，由\eqref{eq::FHJOIWJ421423}式推出
\begin{align*}
|f(x)| \leqslant \max(p(x), p(-x)) \quad (\forall x \in \mathscr{X}).
\end{align*}
因此，$p(x)$的连续性蕴含$f$在$\theta$点连续。又因为$f$是线性的，所以由\refpro{proposition:线性算子在零元处连续等价于全空间连续}知$f$在整个$\mathscr{X}$上连续.故由\refthe{theorem:超平面是闭的当且仅当存在非零连续泛函}知$H_f^r$还是闭的.
\end{proof}

\begin{theorem}[凸集分离定理]\label{theorem:泛函分析--凸集分离定理}
设$E_1$和$E_2$是$B^*$空间中两个非空凸集，$\mathring{E}_1 \cap E_2 = \varnothing$,$E_1$有内点，则$\exists s \in \mathbb{R}$及非零连续线性泛函$f$，使得超平面$H_f^s$分离$E_1$和$E_2$。换句话说，存在一个非零连续线性泛函$f$，使得
\begin{align*}
f(x) \leqslant s \quad (\forall x \in E_1),
\quad 
f(x) \geqslant s \quad (\forall x \in E_2).
\end{align*}
这也等价于
\begin{align*}
\underset{x\in E_1}{\mathrm{sup}}f\left( x \right) \leqslant \underset{x\in E_2}{\mathrm{inf}}f\left( x \right) .
\end{align*}
\end{theorem}
\begin{remark}
显然定理中条件 $\mathring{E}_1 \cap E_2 = \varnothing$ 改成 $E_1 \cap E_2 = \varnothing$ 也成立.
\end{remark}
\begin{proof}
首先考虑 $E_1$ 本身与 $E_2$ 不交的情形，即 $E_1 \cap E_2 = \varnothing$。此时令
\[
E = E_1 + (-E_2) = \{y - z \mid y \in E_1,\; z \in E_2\}.
\]
由于 $E_1, E_2$ 非空凸，$E$ 非空凸；又 $E_1$ 有内点，平移 $E_1 - z$（$z \in E_2$ 固定）也有内点，而 $E$ 包含这些平移的并，故 $E$ 也有内点。若 $\theta \in E$，则存在 $y\in E_1$, $z\in E_2$ 使 $y=z$，与 $E_1\cap E_2=\varnothing$ 矛盾，因此 $\theta \notin E$。

对凸集 $E$ 和点 $\theta\notin E$ 应用\hyperref[theorem:Hahn-Banach定理的几何形式]{Hahn-Banach定理的几何形式}，存在非零连续线性泛函 $f$ 及实数 $r$ 使得超平面 $H_f^r$ 分离 $E$ 与 $\theta$。不妨设
\[
f(x) \leqslant r \quad (\forall x \in E),\qquad f(\theta) \geqslant r.
\]
因为 $f(\theta)=0$，得 $0 \geqslant r$，从而 $f(x) \leqslant 0$ 对所有 $x\in E$ 成立。于是对任意 $y\in E_1$, $z\in E_2$ 有 $f(y)-f(z)=f(y-z)\leqslant 0$，即
\[
f(y) \leqslant f(z).
\]
令 $\alpha = \sup_{y\in E_1} f(y)$, $\beta = \inf_{z\in E_2} f(z)$，则 $\alpha \leqslant \beta$。任取 $s$ 满足 $\alpha \leqslant s \leqslant \beta$，则
\[
f(y) \leqslant s \quad (\forall y\in E_1), \qquad f(z) \geqslant s \quad (\forall z\in E_2).
\]
这就证明了 $E_1\cap E_2=\varnothing$ 时的结论。

现在回到定理的假设：$\mathring{E}_1 \cap E_2 = \varnothing$。因为只要通过适当平移，总可以把任一点变为$\theta$点，所以不妨设$\mathring{E}_1$的内点就是$\theta$.
由于 $\mathring{E}_1$ 是开凸集，且 $\mathring{E}_1$ 与 $E_2$ 不交，将已证结论应用于凸集 $\mathring{E}_1$ 与 $E_2$，存在非零连续线性泛函 $f$ 和实数 $s$ 使得
\begin{equation}\label{eq:sep1}
f(x) \leqslant s \quad (\forall x \in \mathring{E}_1), \qquad 
f(y) \geqslant s \quad (\forall y \in E_2).
\end{equation}
由 $f$ 的连续性，不等式 $f(x) \leqslant s$ 对 $x\in \overline{\mathring{E}_1}$ 也成立。由\rrefpro{proposition:泛函分析--例题1.5.1}{proposition:泛函分析--例题1.5.1-2}，有 $\overline{\mathring{E}_1} = \overline{E_1}$。而 $E_1 \subset \overline{E_1}$，故对任意 $x\in E_1$ 仍有 $f(x) \leqslant s$。结合 \eqref{eq:sep1} 中第二个不等式，即得
\[
f(x) \leqslant s \quad (\forall x \in E_1), \qquad 
f(y) \geqslant s \quad (\forall y \in E_2).
\]
因此超平面 $H_f^s$ 分离 $E_1$ 与 $E_2$。
\end{proof}

\begin{corollary}[Ascoli定理]\label{corollary:泛函分析--Ascoli定理}
设$E$是实$B^*$空间$\mathscr{X}$中的闭凸集，则$\forall x_0 \in \mathscr{X} \setminus E$，$\exists f \in \mathscr{X}^*$及$\alpha \in \mathbb{R}$，适合
\begin{align}
f(x) < \alpha < f(x_0) \quad (\forall x \in E).
\label{eq::--AFAAAFHUH8234-9}
\end{align}
\end{corollary}
\begin{proof}
因为$x_0 \in \mathscr{X} \setminus E$及$E$是闭集，所以$\exists \delta > 0$，使得
\begin{align*}
B(x_0, \delta) \subset \mathscr{X} \setminus E,
\end{align*}
而$B(x_0, \delta)$是有内点的凸集。对$E$和$B(x_0, \delta)$应用\hyperref[theorem:泛函分析--凸集分离定理]{凸集分离定理}，存在非零连续线性泛函$f$，适合
\begin{align}
\sup\limits_{x \in E} f(x) \leqslant \inf\limits_{y \in B(x_0, \delta)} f(y).
\label{eq::--AFAAAFHUH8234-10}
\end{align}
进一步可以证明
\begin{align}
\inf\limits_{y \in B(x_0, \delta)} f(y) < f(x_0).
\label{eq::--AFAAAFHUH8234-11}
\end{align}
事实上，倘若\eqref{eq::--AFAAAFHUH8234-11}式不成立，那么
\begin{align*}
f(y) \geqslant f(x_0) \quad (\forall y \in B(x_0, \delta)).
\end{align*}
这表明$f(x_0)$是$f(y)$在$B(x_0, \delta)$中的极小值，再由\refpro{proposition:泛函分析--例题2.1.9}可知,这与$f$的非零线性矛盾!于是\eqref{eq::--AFAAAFHUH8234-11}式成立。任取\eqref{eq::--AFAAAFHUH8234-11}式两端的中间值$\alpha \in \mathbb{R}$，并由\eqref{eq::--AFAAAFHUH8234-10}式即得\eqref{eq::--AFAAAFHUH8234-9}式.

\end{proof}

\begin{corollary}[Mazur定理]\label{corollary:泛函分析--Mazur定理}
设$E$是$B^*$空间$\mathscr{X}$上的一个有内点的闭凸集，$F$是$\mathscr{X}$上的一个线性流形，又设$\mathring{E} \cap F = \varnothing$，那么存在一个包含$F$的闭超平面$L$，使$E$在$L$的一侧.即存在$\mathscr{X}$上的非零连续线性泛函$f$及$s \in \mathbb{R}$，使得
\begin{align*}
f(x) \leqslant s \quad (\forall x \in E), \quad f(x) = s \quad (\forall x \in F).
\end{align*}
\end{corollary}
\begin{proof}
设$F = x_0 + \mathscr{X}_0$,其中$x_0 \in \mathscr{X}$,$\mathscr{X}_0$是$\mathscr{X}$的线性子空间。由\hyperref[theorem:泛函分析--凸集分离定理]{凸集分离定理}，存在$H_f^r$分离$E$与$F$，即
\begin{align}
f(E)\leqslant r,\quad f\left( x_0 \right) +f\left( \mathscr{X} _0 \right) =f(x_0+\mathscr{X} _0)\geqslant r.
\label{eq::--AFAAAFHUH8234-12}
\end{align}
记$r_0 \triangleq r - f(x_0)$，便有$f(x) \geqslant r_0 \ (\forall x \in \mathscr{X}_0)$。又由$f$是线性的，及$\mathscr{X}_0$是线性子空间知
\begin{align*}
f(tx)=tf(x)\geqslant r_0,\quad \forall t\in\mathbb{R},\,x\in\mathscr{X}_0.
\end{align*}
若存在$x_1\in\mathscr{X}_0$，使得$f(x_1)\ne 0$，则由上式可知
\begin{align*}
tf(x_1)\geqslant r_0,\quad \forall t\in\mathbb{R}.
\end{align*}
当$f(x_1)>0$时，令$t\to-\infty$得$tf(x_1)\to-\infty$，与下界$r_0$矛盾! 当$f(x_1)<0$时，令$t\to+\infty$得$tf(x_1)\to-\infty$，与下界$r_0$矛盾!
故
\begin{align*}
f(x) \equiv 0 \quad (\forall x \in \mathscr{X}_0),
\end{align*}
即有$\mathscr{X}_0 \subseteq H_f^0$，从而$F \subseteq x_0 + H_f^0 = H_f^s$，其中$s \triangleq f(x_0)$。再由\eqref{eq::--AFAAAFHUH8234-12}式推出
\begin{align*}
f(E) \leqslant r\leqslant f(x_0+\theta)=f(x_0)=s.
\end{align*}
于是$L \triangleq H_f^s$便为所求.

\end{proof}

\begin{definition}[承托超平面]
设$E$是实$B^*$空间$\mathscr{X}$中的凸集，超平面$L = H_f^r$称为凸集$E$在点$x_0$的\textbf{承托超平面}，是指$E$在$L$的一侧，且$\overline{E}$与$L$有公共点$x_0$。换句话说，
\begin{align*}
f(x) \leqslant r = f(x_0) \quad (\forall x \in E),
\end{align*}
或
\begin{align*}
f(x) \geqslant r = f(x_0) \quad (\forall x \in E).
\end{align*}
\end{definition}

\begin{example}
设$\mathscr{X}$是$B^*$空间，$E = \{ x \in \mathscr{X} \mid \|x\| \leqslant r \}$，$\|x_0\| = r$，那么$E$在$x_0$有一个承托超平面.
\end{example}
\begin{proof}

\subsubsection{抽象可微函数的中值定理}

\begin{definition}[抽象函数的微商]
设 $\mathscr{Y}$ 是 $B^*$ 空间，$f: (a,b) \to \mathscr{Y}$ 称为数值变数 $t$ 的\textbf{抽象函数}。对于 $t \in (a,b)$,若极限
\[
\lim_{\Delta t \to 0} \frac{f(t + \Delta t) - f(t)}{\Delta t}
\]
在 $\mathscr{Y}$ 中存在，则称该极限为 $f$ 在点 $t$ 处的\textbf{微商}或\textbf{导数}，记为 $f'(t)$。
若 $f$ 在区间 $(a,b)$ 内的每一点都存在微商，则称 $f$ 在 $(a,b)$ 上\textbf{可微}。
\end{definition}

\begin{theorem}
设抽象函数$f: (a,b) \to \mathscr{Y}$在$(a,b)$内可微，那么对$\forall t_1, t_2 \in (a,b)$，$\exists \theta \in (0,1)$，使得
\begin{align}
\|f(t_2) - f(t_1)\| \leqslant \|f'(\theta t_2 + (1 - \theta)t_1)\| \cdot |t_2 - t_1|.
\label{eq::--AFAAAFHUH8234-13}
\end{align}
\end{theorem}
\begin{proof}
由\refcor{corollary:泛函分析--推论2.4.6}，$\exists y^* \in \mathscr{Y}^*$，使得$\|y^*\| = 1$，且
\begin{align}
\langle y^*, f(t_2) - f(t_1) \rangle = \|f(t_2) - f(t_1)\|.
\label{eq::--AFAAAFHUH8234-14}
\end{align}
令$\varphi(\eta) = \langle y^*, f(t_1 + \eta(t_2 - t_1)) \rangle$，那么$\varphi(\eta)$是在$[0,1]$上连续，在$(0,1)$内可微的实函数，且
\begin{align*}
\varphi'(\eta) = \langle y^*, f'(t_1 + \eta(t_2 - t_1))(t_2 - t_1) \rangle.
\end{align*}
对$\varphi(\eta)$应用微分中值公式，即得
\begin{align}
\varphi(1) - \varphi(0) = \varphi'(\theta)  
= \langle y^*, f'(t_1 + \theta(t_2 - t_1))(t_2 - t_1) \rangle,
\label{eq::--AFAAAFHUH8234-15}
\end{align}
其中$0 < \theta < 1$。联合\eqref{eq::--AFAAAFHUH8234-14}与\eqref{eq::--AFAAAFHUH8234-15}便得
\begin{align*}
\|f(t_2) - f(t_1)\| = \varphi(1) - \varphi(0) 
\leqslant \|y^*\| \cdot \|f'(t_1 + \theta(t_2 - t_1))\| \cdot |t_2 - t_1|,
\end{align*}
即得\eqref{eq::--AFAAAFHUH8234-13}式.

\end{proof}

\subsubsection{凸规划问题的Lagrange乘子}

\begin{definition}[上方图]
设$X$为线性空间,$f:X\to [-\infty,+\infty]$为广义实值函数,则称
\begin{align*}
\mathrm{epi}(f) \triangleq \{ (x,t) \in C \times \mathbb{R} \mid f(x) \leqslant t \}
\end{align*}
为$f$的\textbf{上方图}.
\end{definition}

\begin{definition}[凸泛函]
设$\mathscr{X}$是一个线性空间，$C \subseteq \mathscr{X}$是一个凸集。称$f: C \to \mathbb{R}$是一个\textbf{凸泛函}，是指$f$满足
\begin{align*}
f(\lambda x + (1 - \lambda)y) \leqslant \lambda f(x) + (1 - \lambda)f(y) \quad (\forall x, y \in C, \forall \lambda \in (0,1)).
\end{align*}
\end{definition}
\begin{remark}
这个定义可以等价地表达为：$f$的上方图
\begin{align*}
\mathrm{epi}(f) = \{ (x,t) \in C \times \mathbb{R} \mid f(x) \leqslant t \}
\end{align*}
是$C \times \mathbb{R}$中的凸集.
\end{remark}

\begin{theorem}[Kuhn-Tucker定理]\label{theorem:Kuhn-Tucker定理}
设$\mathscr{X}$是一个线性空间，$C$是$\mathscr{X}$的一个凸子集。又设$f, g_1, \cdots, g_n$是$C$上的凸泛函，且存在$\widehat{x} \in C$满足
\begin{align*}
g_i(\widehat{x}) < 0 \quad (i=1,2,\cdots,n).
\end{align*}
若$x_0$是问题$(P)$的解，则必存在实数$\lambda_1, \lambda_2, \cdots, \lambda_n \geqslant 0$，适合
\begin{align*}
f(x_0) = \min \left\{ f(x) + \sum\limits_{i=1}^n \lambda_i g_i(x) \,\bigg|\, \forall x \in C \right\},
\end{align*}
以及
\begin{align*}
\lambda_i g_i(x_0) = 0 \quad (i=1,2,\cdots,n).
\end{align*}
\end{theorem}
\begin{proof}
设 $x_0$ 是问题 $(P)$ 的最优解。构造 $\mathbb{R}^{n+1}$ 中的两个集合:
\begin{align*}
E &\triangleq \left\{ (t_0, t_1, \cdots, t_n) \in \mathbb{R}^{n+1} \;\middle|\; t_0 \leqslant f(x_0);\; t_i \leqslant 0 \ (i=1,2,\cdots,n) \right\}, \\
F &\triangleq \left\{ (t_0, t_1, \cdots, t_n) \in \mathbb{R}^{n+1} \;\middle|\; \exists x \in C,\; t_0 \geqslant f(x),\; t_i \geqslant g_i(x) \ (i=1,2,\cdots,n) \right\}.
\end{align*}
由 $f, g_1, \cdots, g_n$ 的凸性知 $F$ 是凸集；$E$ 是凸集且有内点，其内部为
\[
\mathring{E} = \left\{ (t_0, \cdots, t_n) \mid t_0 < f(x_0),\; t_i < 0 \ (i=1,\cdots,n) \right\}.
\]
由于 $x_0$ 是 $(P)$ 的解，$\mathring{E} \cap F = \varnothing$。应用\hyperref[theorem:泛函分析--凸集分离定理]{凸集分离定理}，存在非零向量 $(\wideparen{\lambda}_0, \wideparen{\lambda}_1, \cdots, \wideparen{\lambda}_n) \in \mathbb{R}^{n+1}$ 使得
\begin{align}\label{eq:sep-ineq}
\wideparen{\lambda}_0 f(x_0) + \sum_{i=1}^n \wideparen{\lambda}_i g_i(x_0) \leqslant \wideparen{\lambda}_0 (f(x)+\xi_0) + \sum_{i=1}^n \wideparen{\lambda}_i (g_i(x)+\xi_i)
\end{align}
对所有 $x \in C$ 及所有 $\xi_i \geqslant 0$（$i=0,1,\cdots,n$）成立。由 $\xi_i$ 的任意非负性可推出 $\wideparen{\lambda}_i \geqslant 0$（$i=0,1,\cdots,n$）。特别地，取 $\xi_i=0$ 得
\begin{align}\label{eq:lambda-ineq}
\wideparen{\lambda}_0 f(x_0) + \sum_{i=1}^n \wideparen{\lambda}_i g_i(x_0) \leqslant \wideparen{\lambda}_0 f(x) + \sum_{i=1}^n \wideparen{\lambda}_i g_i(x) \quad (\forall x \in C).
\end{align}
在 \eqref{eq:lambda-ineq} 中取 $x = x_0$，得 $\sum_{i=1}^n \wideparen{\lambda}_i g_i(x_0) \geqslant 0$。又由 $g_i(x_0) \leqslant 0$ 和 $\wideparen{\lambda}_i \geqslant 0$ 知 $\sum_{i=1}^n \wideparen{\lambda}_i g_i(x_0) \leqslant 0$，因此
\begin{align}\label{eq:complementary}
\sum_{i=1}^n \wideparen{\lambda}_i g_i(x_0) = 0\Longrightarrow  \wideparen{\lambda}_i g_i(x_0)=0 \ (i=1,\cdots,n).
\end{align}

断言 $\wideparen{\lambda}_0 > 0$。倘若不然，$\wideparen{\lambda}_0 = 0$。由\eqref{eq:sep-ineq}式和\eqref{eq:complementary}式便有
\begin{align}
\sum\limits_{i=1}^n \wideparen{\lambda}_i g_i(\widehat{x}) \geqslant 0.
\label{eq::--AFAAAFHUH8234-23}
\end{align}
因为$(\wideparen{\lambda}_0, \wideparen{\lambda}_1, \cdots, \wideparen{\lambda}_n) \neq \theta$，所以$(\wideparen{\lambda}_1, \cdots, \wideparen{\lambda}_n) \neq (0,\cdots,0)$。又$\wideparen{\lambda}_i \geqslant 0 \ (i=1,\cdots,n)$，联合\eqref{eq::--AFAAAFHUH8234-22}式便有
\begin{align*}
\sum\limits_{i=1}^n \wideparen{\lambda}_i g_i(\widehat{x}) < 0.
\end{align*}
这与\eqref{eq::--AFAAAFHUH8234-23}式矛盾！

令 $\lambda_i = \frac{\wideparen{\lambda}_i}{\wideparen{\lambda}_0}$（$i=1,\cdots,n$），则 $\lambda_i \geqslant 0$。将 \eqref{eq:lambda-ineq} 两边除以 $\wideparen{\lambda}_0$ 得
\[
f(x_0) \leqslant f(x) + \sum_{i=1}^n \lambda_i g_i(x) \quad (\forall x \in C).
\]
取 $x = x_0$ 时由 \eqref{eq:complementary} 知等号成立，故
\[
f(x_0) = \min_{x \in C} \left\{ f(x) + \sum_{i=1}^n \lambda_i g_i(x) \right\},
\]
且 $\lambda_i g_i(x_0)=0$。
\end{proof}

\begin{definition}
设 $f: \mathscr{X} \to \mathbb{R}$ 是凸泛函，$\forall x_0 \in \mathscr{X}$，称集合
\begin{align*}
\partial f(x_0) \triangleq \{ x^* \in \mathscr{X}^* \mid \langle x^*, x - x_0 \rangle + f(x_0) \leqslant f(x) \ (\forall x \in \mathscr{X}) \}.
\end{align*}
为 $f$ 在 $x_0$ 点的\textbf{次微分},$\partial f(x_0)$ 中的元素 $x^*$ 称为 $f$ 在 $x_0$ 点的\textbf{次梯度}.
\end{definition}

\begin{theorem}
若 $f: \mathscr{X} \to \mathbb{R}$ 是凸泛函，且在 $x_0 \in \mathscr{X}$ 连续，则 $\partial f(x_0) \neq \varnothing$。
\end{theorem}
\begin{proof}
在空间 $\mathscr{X} \times \mathbb{R}$ 上，考察凸集 $\mathrm{epi}(f)$ 与单点集 $\{(x_0, f(x_0))\}$。因 $f$ 在 $x_0$ 连续，故 $\mathrm{epi}(f)$ 存在内点 $(x_0, f(x_0)+1)$，且
\begin{align*}
\{(x_0, f(x_0))\} \cap (\mathrm{epi}(f))^\circ = \varnothing.
\end{align*}
由\hyperref[theorem:泛函分析--凸集分离定理]{凸集分离定理}，存在非零元 $(x^*, \xi) \in \mathscr{X}^* \times \mathbb{R}$ 分离 $\mathrm{epi}(f)$ 与 $\{(x_0, f(x_0))\}$，即
\begin{align}
\langle x^*, x_0 \rangle + \xi f(x_0) \leqslant \langle x^*, x \rangle + \xi t \quad (\forall (x,t) \in \mathrm{epi}(f)).
\label{eq:sep}
\end{align}
令 $x = x_0$，$t = f(x_0) + s$（$\forall s > 0$），可得 $\xi \geqslant 0$。若 $\xi = 0$，则由 \eqref{eq:sep} 得
\begin{align*}
\langle x^*, x_0 - x \rangle \leqslant 0 \quad (\forall x \in \mathscr{X}).
\end{align*}
即 $x^* = \theta$，与 $(x^*, \xi)$ 非零矛盾，故 $\xi > 0$。令 $x_0^* = -\dfrac{x^*}{\xi}$，则 $x_0^* \in \partial f(x_0)$。

\end{proof}
