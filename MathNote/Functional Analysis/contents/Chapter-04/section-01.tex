\section{广义函数的概念}

\subsection{基本空间$\mathscr{D}(\Omega)$}

\begin{definition}
设$\Omega\subseteq\mathbb{R}^n$为开集，$u\in C(\overline{\Omega})$，称集合
$$F=\{x\in\Omega\mid u(x)\neq0\}$$
关于$\Omega$的闭包为$u$关于$\Omega$的\textbf{支集}，记作$\mathrm{supp}(u)$。即连续函数$u$的支集是满足在此集合外$u$恒为0的、关于$\Omega$的最小闭集。
\end{definition}

\begin{definition}
设$\Omega\subseteq\mathbb{R}^n$为开集，对整数$k\geqslant0$（允许$k=\infty$），$C_0^k(\Omega)$表示支集在$\Omega$内紧的全体$C^k(\overline{\Omega})$函数构成的集合,显然满足
\begin{align*}
C_0^\infty(\Omega)\subseteq\cdots\subseteq C_0^{k+1}(\Omega)\subseteq C_0^k(\Omega)\subseteq\cdots\subseteq C_0^0(\Omega).
\end{align*}
\end{definition}

\begin{example}
设
\begin{align}
j(x)=
\begin{cases}
C_n\mathrm{e}^{-\frac{1}{1-|x|^2}},&|x|<1,\\
0,&|x|\geqslant1,
\end{cases}\label{eq:molify_base}
\end{align}
其中
$$C_n=\left(\int_{|x|\leqslant1}\mathrm{e}^{-\frac{1}{1-|x|^2}}\mathrm{d}x\right)^{-1},$$
$C_n$为仅依赖空间维数的常数，则$j(x)\in C_0^\infty(\mathbb{R}^n)$，且
$$\int_{\mathbb{R}^n}j(x)\mathrm{d}x=1.$$
对任意$\delta>0$，定义
\begin{align}
j_\delta(x)=\frac{1}{\delta^n}j\left(\frac{x}{\delta}\right),\label{eq:molify_delta}
\end{align}
\end{example}

\begin{proposition}
设$\Omega\subseteq\mathbb{R}^n$为开集，$u(x)$为可积函数，且在$\Omega$的紧子集$K$外恒为0，则当$\delta>0$充分小时，函数
\begin{align}
u_\delta(x)=\int_{\Omega}u(y)j_\delta(x-y)\mathrm{d}y\label{eq:convolution_molify}
\end{align}
满足$u_\delta\in C_0^\infty(\Omega)$。
\end{proposition}

\begin{proof}
记$K_\delta=\{x\in\mathbb{R}^n\mid\mathrm{dist}(x,K)\leqslant\delta\}$，当$\delta$充分小时，$K_\delta\subseteq\Omega$且$u_\delta(x)=0,\forall x\notin K_\delta$，且
\begin{align}
\partial^\alpha u_\delta(x)=\int_{\Omega}u(y)\partial^\alpha j_\delta(x-y)\mathrm{d}y,\quad\forall x\in K_\delta.\label{eq:diff_convolution}
\end{align}
以多重指标$\alpha_0=(1,0,\cdots,0)$为例，
\begin{align*}
\partial^{\alpha_0}u_\delta(x)&=\lim\limits_{h\to0}\int_{\Omega}\frac{1}{h}\left[j_\delta(x+he_1-y)-j_\delta(x-y)\right]u(y)\mathrm{d}y\\
&=\lim\limits_{h\to0}\int_{\Omega}\partial^{\alpha_0}j_\delta(x+\theta he_1-y)u(y)\mathrm{d}y,
\end{align*}
其中$\theta=\theta(x,y)\in(0,1)$，$e_1=(1,0,\cdots,0)\in\mathbb{R}^n$。
由$j_\delta$的连续可微性，存在常数$M_{\alpha_0}$使得$|\partial^{\alpha_0}j_\delta(z)|\leqslant M_{\alpha_0},\forall z\in\mathbb{R}^n$。
由\hyperref[Real Analysis-theorem:控制收敛定理]{Lebesgue控制收敛定理}，
\begin{align*}
\partial^{\alpha_0}u_\delta(x)=\int_{\Omega}\lim\limits_{h\to0}\partial^{\alpha_0}j_\delta(x+\theta he_1-y)u(y)\mathrm{d}y
=\int_{\Omega}\partial^{\alpha_0}j_\delta(x-y)u(y)\mathrm{d}y.
\end{align*}
对任意多重指标$\alpha$重复上述步骤，可得\eqref{eq:diff_convolution}.

\end{proof}

\begin{theorem}\label{theorem:磨光逼近定理}
设$\Omega\subseteq\mathbb{R}^n$为开集，若$u\in C_0^k(\Omega)$，则
$$\|u_\delta(x)-u(x)\|_{C^k(\overline{\Omega})}\to0\quad(\delta\to0).$$
\end{theorem}
\begin{proof}
将$u(x)$零延拓至$\mathbb{R}^n$。对任意多重指标$\alpha=(\alpha_1,\cdots,\alpha_n)$，$|\alpha|\leqslant k$，
\begin{align*}
\partial_x^\alpha u_\delta(x)&=\int_{\mathbb{R}^n}u(y)\partial_x^\alpha j_\delta(x-y)\mathrm{d}y
=(-1)^{|\alpha|}\int_{\mathbb{R}^n}u(y)\partial_y^\alpha j_\delta(x-y)\mathrm{d}y\\
&=\int_{\mathbb{R}^n}\partial_y^\alpha u(y)j_\delta(x-y)\mathrm{d}y=\int_{\mathbb{R}^n}\partial_y^\alpha u(x-\delta y)j(y)\mathrm{d}y,
\end{align*}
因此
$$|\partial^\alpha u_\delta(x)-\partial^\alpha u(x)|\leqslant\int_{\mathbb{R}^n}|\partial^\alpha u(x-\delta y)-\partial^\alpha u(x)|j(y)\mathrm{d}y.$$
注意到$j(y)=0,|y|\geqslant1$，且$\partial^\alpha u(z)$在集合
$$(\mathrm{supp}(u))_1=\{x\in\mathbb{R}^n\mid\mathrm{dist}(x,\mathrm{supp}(u))\leqslant1\}$$
上一致连续。
对任意$\varepsilon>0$，存在$0<\delta_0<\frac{1}{2}$，当$0<\delta<\delta_0$时，
$$|\partial^\alpha u(x-\delta y)-\partial^\alpha u(x)|<\varepsilon,\quad\forall x\in\mathbb{R}^n,|y|\leqslant1.$$
故
$$\sup\limits_{x\in\mathbb{R}^n}|\partial^\alpha u_\delta(x)-\partial^\alpha u(x)|<\varepsilon\int_{\mathbb{R}^n}j(y)\mathrm{d}y=\varepsilon.
$$
\end{proof}

\begin{corollary}
设$\Omega\subseteq\mathbb{R}^n$为开集，$\mu$为$\Omega$上的完全可加测度，若
$$\int_{\Omega}\varphi\mathrm{d}\mu=0,\quad\forall\varphi\in C_0^\infty(\Omega),$$
则
$$\int_{\Omega}\varphi\mathrm{d}\mu=0,\quad\forall\varphi\in C_0^0(\Omega).$$
\end{corollary}

\begin{definition}
设$\Omega\subseteq\mathbb{R}^n$为开集，在$C_0^\infty(\Omega)$上定义收敛性：序列$\{\varphi_j\}$收敛于$\varphi_0$，若
\begin{enumerate}[(1)]
\item 存在关于$\Omega$的紧集$K\subseteq\Omega$，使得$\mathrm{supp}(\varphi_j)\subseteq K,\forall j=0,1,2,\cdots$；
\item 对任意多重指标$\alpha=(\alpha_1,\cdots,\alpha_n)$，有
$$\max\limits_{x\in K}|\partial^\alpha\varphi_j(x)-\partial^\alpha\varphi_0(x)|\to0\quad(j\to\infty).$$
\end{enumerate}
赋予上述收敛性的线性空间$C_0^\infty(\Omega)$称为\textbf{基本空间}$\mathscr{D}(\Omega)$。
\end{definition}
\begin{remark}
仅在$\mathscr{D}(\Omega)$上引入上述收敛性，该收敛性无法由任意范数或准范数导出，故$\mathscr{D}(\Omega)$不是$B^*$空间。
\end{remark}

\begin{proposition}
设$\Omega\subseteq\mathbb{R}^n$为开集，$\mathscr{D}(\Omega)$是序列完备的，即若$\{\varphi_j\}_{j=0}^\infty$为基本列，满足
\begin{enumerate}[(1)]
\item 存在公共紧支集$K$，使得$\mathrm{supp}(\varphi_j)\subseteq K$；
\item 对任意$\varepsilon>0$、任意多重指标$\alpha$，存在$N=N(\varepsilon,\alpha)\in\mathbb{N}$，当$m,n>N$时，
$$\max\limits_{x\in K}|\partial^\alpha\varphi_n(x)-\partial^\alpha\varphi_m(x)|<\varepsilon,$$
\end{enumerate}
则存在$\varphi_0\in\mathscr{D}(\Omega)$，使得$\varphi_j\to\varphi_0\quad(j\to\infty)$。
\end{proposition}


\subsection{广义函数的定义和基本性质}

\begin{definition}
设$\Omega\subseteq\mathbb{R}^n$为开集，$\mathscr{D}(\Omega)$上的全体连续线性泛函称为\textbf{广义函数}，即泛函$f:\mathscr{D}(\Omega)\to\mathbb{R}$满足
\begin{enumerate}[(1)]
\item 线性性：
\begin{align*}
\langle f,\lambda_1\varphi_1+\lambda_2\varphi_2\rangle=\lambda_1\langle f,\varphi_1\rangle+\lambda_2\langle f,\varphi_2\rangle,\quad\forall\varphi_1,\varphi_2\in\mathscr{D}(\Omega),\forall\lambda_1,\lambda_2\in\mathbb{R};
\end{align*}
\item 连续性：对任意$\{\varphi_j\}\subseteq\mathscr{D}(\Omega)$，若$\varphi_j\to\varphi_0(\mathscr{D}(\Omega))$，则
$$\langle f,\varphi_j\rangle\to\langle f,\varphi_0\rangle\quad(j\to\infty).$$
\end{enumerate}
全体广义函数构成的集合记作$\mathscr{D}'(\Omega)$。
\end{definition}

\begin{examples}
\begin{enumerate}[(1)]
\item 设$\theta\in\Omega$，定义$\delta$函数：
$$\langle\delta,\varphi\rangle=\varphi(\theta),\quad\forall\varphi\in\mathscr{D}(\Omega).$$
$\delta$是线性的；若$\varphi_j\to\varphi_0(\mathscr{D}(\Omega))$，则$|\varphi_j(\theta)-\varphi_0(\theta)|\to0$，故$\langle\delta,\varphi_j\rangle=\varphi_j(\theta)\to\varphi_0(\theta)=\langle\delta,\varphi_0\rangle$，因此$\delta$为广义函数。

\item 对任意多重指标$\alpha=(\alpha_1,\cdots,\alpha_n)$，定义
$$\langle\delta^{(\alpha)},\varphi\rangle=(-1)^{|\alpha|}(\partial^\alpha\varphi)(\theta),\quad\forall\varphi\in\mathscr{D}(\Omega),$$
则$\delta^{(\alpha)}$为广义函数。

\item 设$f(x)$是$\Omega$上一个局部可积，即对任意关于$\Omega$的紧集$K$，积分
$$
\int_K|f(x)|\mathrm{d}x<\infty.
$$
记作$f(x)\in L^1_{\mathrm{loc}}(\Omega)$，定义
\begin{align}
\langle f,\varphi\rangle=\int_{\Omega}f(x)\varphi(x)\mathrm{d}x,\quad\forall\varphi\in\mathscr{D}(\Omega).\label{eq:loc_func_to_dist}
\end{align}
线性性显然；若$\varphi_j\to\varphi_0(\mathscr{D}(\Omega))$，则存在紧集$K\subseteq\Omega$，使得$\mathrm{supp}(\varphi_j)\subseteq K$且$\max\limits_{x\in K}|\varphi_j(x)-\varphi_0(x)|\to0$，由Lebesgue控制收敛定理，
$$|\langle f,\varphi_j\rangle-\langle f,\varphi_0\rangle|\leqslant\int_K|f(x)|\cdot|\varphi_j(x)-\varphi_0(x)|\mathrm{d}x\to0\quad(j\to\infty),$$
故$f$对应广义函数。

\item 设$\mu$为$\Omega$上的完全可加测度，定义
$$\langle f,\varphi\rangle=\int_{\Omega}\varphi(x)\mathrm{d}\mu(x),\quad\forall\varphi\in\mathscr{D}(\Omega),$$
该式定义一个广义函数，对应关系为一一对应。

\item 设$\Omega=(0,1)$，定义
$$\langle f,\varphi\rangle=\sum\limits_{j=1}^\infty\varphi^{(j)}\left(\frac{1}{j}\right),\quad\forall\varphi\in\mathscr{D}(\Omega),$$
则$f$为广义函数.
\end{enumerate}
\end{examples}

\begin{enumerate}[(1)]

\item 

\item 线性条件显然，再验证连续性: 若$\varphi_j \to \varphi_0(\mathscr{D}(\Omega))$，则存在紧集$K\subset \Omega$，使得
$\mathrm{supp}(\varphi_j) \subset K$，且$\max\limits_{x\in K}|\varphi_j(x)-\varphi_0(x)| \to 0\ (j\to\infty)$。

\begin{remark}
将几乎处处相等的局部可积函数视为同一函数，则映射$f(x)\mapsto f\ (L^1_{\mathrm{loc}}(\Omega)\to\mathscr{D}'(\Omega))$是一一映射。
若$f\in L^1_{\mathrm{loc}}(\Omega)$且$\int_{\Omega}f\cdot\varphi\mathrm{d}x=0,\forall\varphi\in\mathscr{D}(\Omega)$，则$f(x)=0\ \mathrm{a.e.}$于$\Omega$。
只需证对任意闭球$B(x_0,\delta)\subseteq\Omega$，$f(x)=0\ \mathrm{a.e.}$于$B(x_0,\delta)$。
定义
$$\widetilde{f}(x)=
\begin{cases}
\mathrm{sign}f(x),&x\in B(x_0,\delta),\\
0,&x\notin B(x_0,\delta),
\end{cases}$$
则$\widetilde{f}\in L^1(\Omega)$，且在紧集$B(x_0,\delta)$外恒为0。
取磨光函数$\widetilde{f}_\delta(x)=\int_{\Omega}\widetilde{f}(y)j_\delta(x-y)\mathrm{d}y$，当$\delta>0$充分小时，$\widetilde{f}_\delta\in\mathscr{D}(\Omega)$且$\|\widetilde{f}_\delta-\widetilde{f}\|_{L^1(\Omega)}\to0\ (\delta\to0)$。
由Riesz表示定理与Lebesgue控制收敛定理，
\begin{align*}
\int_{B(x_0,\delta)}|f(x)|\mathrm{d}x=\int_{\Omega}f(x)\cdot\widetilde{f}(x)\mathrm{d}x
=\lim\limits_{\delta_i\to0}\int_{\Omega}f(x)\cdot\widetilde{f}_{\delta_i}(x)\mathrm{d}x=0,
\end{align*}
故$f(x)=0\ \mathrm{a.e.}$于$B(x_0,\delta)$.
\end{remark}
\begin{remark}
并非所有函数均可视为广义函数，不可测函数不能对应广义函数。广义函数是局部可积函数的推广，局部可积函数按\eqref{eq:loc_func_to_dist}对应广义函数，但该对应不是满射，$\mathscr{D}'(\Omega)$包含更多元素。
\end{remark}

\begin{theorem}
设$\Omega\subseteq\mathbb{R}^n$为开集，$f\in\mathscr{D}'(\Omega)$当且仅当：对任意关于$\Omega$的紧集$K$，存在常数$C$与非负整数$m$，使得
\begin{align}
|\langle f,\varphi\rangle|\leqslant C\sum\limits_{|\alpha|\leqslant m}\sup\limits_{x\in K}|\partial^\alpha\varphi(x)|,\quad\forall\varphi\in\mathscr{D}(\Omega),\mathrm{supp}(\varphi)\subseteq K.\label{eq:dist_equi}
\end{align}
\end{theorem}
\begin{proof}
充分性显然。下证必要性，采用反证法。
若结论不成立，则存在紧集$K$，使得\eqref{eq:dist_equi}不成立。
由齐次性，对任意$j\in\mathbb{N}$，存在$\varphi_j\in\mathscr{D}(\Omega)$，满足$\mathrm{supp}(\varphi_j)\subseteq K$，
$$\sup\limits_{x\in K}|\partial^\alpha\varphi_j|\leqslant\frac{1}{j},\quad\forall|\alpha|\leqslant j,$$
且$\langle f,\varphi_j\rangle=1$。
故$\{\varphi_j\}$在$\mathscr{D}(\Omega)$中收敛于0，因此$\langle f,\varphi_j\rangle\to0\ (j\to\infty)$，矛盾.

\end{proof}

\subsection{广义函数的收敛性}

\begin{definition}
设$\Omega\subseteq\mathbb{R}^n$为开集，在$\mathscr{D}'(\Omega)$上定义加法与数乘：
$$\langle(\lambda_1f_1+\lambda_2f_2),\varphi\rangle=\lambda_1\langle f_1,\varphi\rangle+\lambda_2\langle f_2,\varphi\rangle,\quad\forall\varphi\in\mathscr{D}(\Omega),\forall\lambda_1,\lambda_2\in\mathbb{R},$$
则$\mathscr{D}'(\Omega)$为线性空间。
称$\{f_j\}\subseteq\mathscr{D}'(\Omega)$\textbf{*弱收敛}到$f_0\in\mathscr{D}'(\Omega)$，若
$$\langle f_j,\varphi\rangle\to\langle f_0,\varphi\rangle\quad(j\to\infty),\quad\forall\varphi\in\mathscr{D}(\Omega).$$
\end{definition}

\begin{examples}
\begin{enumerate}[(1)]
\item 在$\mathbb{R}$上，取$f_j(x)=\frac{1}{\pi}\cdot\frac{\sin jx}{x}\ (j=1,2,\cdots)$，则$f_j\in L^1_{\mathrm{loc}}(\mathbb{R})$，可视为广义函数列，满足$f_j\to\delta\ (\mathscr{D}'(\mathbb{R}))\ (j\to\infty)$.

\item 取\eqref{eq:molify_delta}中定义的$j_\delta(x)$，当$\delta\to0$时，$j_\delta$作为广义函数列收敛到$\delta\ (\mathscr{D}'(\mathbb{R}^n))$；对$\delta_{x_0}$满足$\langle\delta_{x_0},\varphi\rangle=\varphi(x_0),\forall\varphi\in\mathscr{D}(\mathbb{R}^n)$，有$j_\delta(x-x_0)\to\delta_{x_0}\ (\delta\to0)$.

\item 设$\{f_j(x)\}$为$\Omega$上的局部可积函数列，若对任意关于$\Omega$的紧集$K$，存在常数$M_K$使得$|f_j(x)|\leqslant M_K,\forall x\in K,\forall j$，且$f_j(x)\to f_0(x)\ (j\to\infty)\ \mathrm{a.e.}$于$\Omega$，则广义函数列$\langle f_j,\varphi\rangle=\int_{\Omega}f_j(x)\varphi(x)\mathrm{d}x$弱收敛到$f_0$.

\item 设$f_1(x)=\mathrm{e}^{-\pi|x|^2}$，$f_j(x)=j^{\frac{n}{2}}f_1(\sqrt{j}x)=j^{\frac{n}{2}}\mathrm{e}^{-j\pi|x|^2}\ (j=2,3,\cdots)$，则
$$\langle f_j,\varphi\rangle\to\langle\delta,\varphi\rangle\quad(j\to\infty),\quad\forall\varphi\in\mathscr{D}(\mathbb{R}^n).$$
\end{enumerate}
\end{examples}
\begin{proof}
\begin{enumerate}[(1)]
\item 由$\lim\limits_{T\to\infty}\frac{1}{\pi}\int_{-T}^T\frac{\sin jx}{x}\mathrm{d}x=1$。
对任意$\varphi\in\mathscr{D}(\mathbb{R})$，存在$T_0>0$使得$\mathrm{supp}(\varphi)\subseteq[-T_0,T_0]$。
当$T>T_0$时，$\int_{-\infty}^\infty f_j(x)\varphi(x)\mathrm{d}x=\int_{-T}^T f_j(x)\varphi(x)\mathrm{d}x$。
对任意$\varepsilon>0$，取$T_1$充分大，当$T>T_1$时，
$$\left|\frac{1}{\pi}\int_{-T}^T\frac{\sin jx}{x}\mathrm{d}x-1\right|<\frac{\varepsilon}{2}.$$
当$T>\max\{T_0,T_1\}$时，
\begin{align*}
|\langle f_j,\varphi\rangle-\varphi(0)|&\leqslant\left|\frac{1}{\pi}\int_{-T}^T\frac{\sin jx}{x}[\varphi(x)-\varphi(0)]\mathrm{d}x\right|+\frac{\varepsilon}{2}|\varphi(0)|\\
&=\frac{1}{\pi}\left|\int_{0}^T\sin jx\frac{\varphi(x)+\varphi(-x)-2\varphi(0)}{x}\mathrm{d}x\right|+\frac{\varepsilon}{2}|\varphi(0)|.
\end{align*}
固定$T$，由Riemann-Lebesgue引理，存在正整数$n_0$，当$j>n_0$时，
$$\frac{1}{\pi}\left|\int_{0}^T\sin jx\frac{\varphi(x)+\varphi(-x)-2\varphi(0)}{x}\mathrm{d}x\right|<\frac{\varepsilon}{2},$$
故$\langle f_j,\varphi\rangle\to\varphi(0)=\langle\delta,\varphi\rangle\ (j\to\infty),\forall\varphi\in\mathscr{D}(\mathbb{R})$.

\item 直接应用\refthe{theorem:磨光逼近定理}.

\item 由\hyperref[Real Analysis-theorem:控制收敛定理]{Lebesgue控制收敛定理}直接可得.

\item 
\end{enumerate}
\end{proof}

\begin{note}
提示

(1) 若$u\in L^p(\Omega)(1<p<\infty)$，$u_\delta(x)$是按(4.1.3)式构造的函数，则按Young不等式，便有
\begin{align*}
\|u_\delta\|_p \leqslant \|u\|_p.
\end{align*}

(2) 用Luzin定理证明$C_0^0(\Omega)$在$L^p(\Omega)$中稠密。

(3) 用典型的$\varepsilon/3$论证法。从$u\in L^p(\Omega)$出发，为了找到$u_\delta\in C_0^\infty(\Omega)$，使得它逼近$u$的误差
\begin{align*}
\|u_\delta - u\|_p < \varepsilon.
\end{align*}
我们可以分三个步骤进行，每步引进的误差各小于$\varepsilon/3$。
第一步：找$\varphi\in C_0^0(\Omega)$，使得$\|u - \varphi\|_p < \varepsilon/3$；
第二步：找$\varphi_\delta\in C_0^\infty(\Omega)$，使得$\|\varphi - \varphi_\delta\|_p < \varepsilon/3$；
第三步：找$u_\delta\in C_0^\infty(\Omega)$，使得$\|\varphi_\delta - u_\delta\|_p < \varepsilon/3$。
\end{note}
