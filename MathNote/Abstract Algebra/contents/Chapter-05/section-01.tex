\section{自由模}

设$R$是幺环，在集合
\begin{align}
R^{(n)}=\underbrace{R\times R\times\cdots\times R}_{n\text{个}}=\{(a_1,\cdots,a_n)\mid a_i\in R,i=1,2,\cdots,n\},
\end{align}
中，对$x=(x_1,x_2,\cdots,x_n)\in R^{(n)}$，记$x$的第$i$个分量$x_i$为$\mathrm{ent}_i x$，即$\mathrm{ent}_i x=x_i$。定义运算：
\begin{align}
\mathrm{ent}_i(x+y)=\mathrm{ent}_i x+\mathrm{ent}_i y,\quad 1\leqslant i\leqslant n,\label{eq:5.1.1}
\end{align}
\begin{align}
\mathrm{ent}_i(ax)=a\mathrm{ent}_i x,\quad x\in R^{(n)},\ 1\leqslant i\leqslant n,\label{eq:5.1.2}
\end{align}
可验证$R^{(n)}$是左$R$-模。

取$e_i\in R^{(n)}$满足$\mathrm{ent}_j e_i=\delta_{ij}$，则$\forall x\in R^{(n)}$，有
\begin{align*}
x=\sum\limits_{i=1}^n (\mathrm{ent}_i x)e_i.
\end{align*}
由此可得：
\begin{enumerate}[(1)]
\item $e_1,e_2,\cdots,e_n$生成$R^{(n)}$；
\item $\sum\limits_{i=1}^n x_i e_i=0$当且仅当$x_i=0,\ 1\leqslant i\leqslant n$。
\end{enumerate}

\begin{definition}
幺环$R$上的模$\mathscr{M}$若与$R$-模$R^{(n)}$同构，则称为\textbf{秩}$n$的\textbf{自由模}。
\end{definition}

\begin{theorem}
$R$-模$\mathscr{M}$是秩$n$的自由模当且仅当存在$\mathscr{M}$中的$n$个元素$u_1,u_2,\cdots,u_n$满足
\begin{enumerate}[(1)]
\item $u_1,u_2,\cdots,u_n$生成$\mathscr{M}$，即
\begin{align}
\mathscr{M}=Ru_1+Ru_2+\cdots+Ru_n;\label{eq:5.1.3}
\end{align}
\item 对任意$x\in\mathscr{M}$，$x=\sum\limits_{i=1}^n x_i u_i,\ x_i\in R$的表示法唯一，即
\begin{align}
\sum\limits_{i=1}^n x_i u_i=0 \iff x_i=0,\ 1\leqslant i\leqslant n.\label{eq:5.1.4}
\end{align}
\end{enumerate}
此时称$u_1,u_2,\cdots,u_n$为$\mathscr{M}$的一组\textbf{基}。
\end{theorem}

\begin{proof}
若$\mathscr{M}$为秩$n$的自由模，则存在$R^{(n)}$到$\mathscr{M}$的模同构$\sigma$。令$u_i=\sigma(e_i),\ 1\leqslant i\leqslant n$，于是
\begin{align*}
\sigma\left(\sum\limits_{i=1}^n x_i e_i\right)=\sum\limits_{i=1}^n x_i \sigma(e_i)=\sum\limits_{i=1}^n x_i u_i,
\end{align*}
因而$u_1,u_2,\cdots,u_n$生成$\mathscr{M}$。

若$\sum\limits_{i=1}^n x_i u_i=0$，则$\sigma\left(\sum\limits_{i=1}^n x_i e_i\right)=0$，由$\sigma$是单射得$\sum\limits_{i=1}^n x_i e_i=0$，故$x_i=0,\ 1\leqslant i\leqslant n$。

反之，设$u_1,u_2,\cdots,u_n$满足条件\eqref{eq:5.1.3}与\eqref{eq:5.1.4}，作映射$\eta:\mathscr{M}\to R^{(n)}$，
\begin{align*}
\eta\left(\sum\limits_{i=1}^n x_i u_i\right)=\sum\limits_{i=1}^n x_i e_i,
\end{align*}
可验证$\eta$是模同构，即$\mathscr{M}$是秩$n$的自由模.

\end{proof}

\begin{theorem}
设$R$是幺环，$\mathscr{M}$是$R$-模，$u_1,u_2,\cdots,u_n$是$\mathscr{M}$中$n$个元素，则$\mathscr{M}$为秩$n$的自由模且$u_1,u_2,\cdots,u_n$为$\mathscr{M}$的基的充分必要条件是：对任一$R$-模$\mathscr{M}'$及其中$n$个元素$v_1,v_2,\cdots,v_n$，存在唯一的$\mathscr{M}$到$\mathscr{M}'$的模同态$\eta$，使得
\begin{align*}
\eta(u_i)=v_i,\quad i=1,\cdots,n.
\end{align*}
\end{theorem}

\begin{proof}
必要性：设$u_1,\cdots,u_n$为自由模$\mathscr{M}$的基，作映射
\begin{align*}
\eta\left(\sum\limits_{i=1}^n x_i u_i\right)=\sum\limits_{i=1}^n x_i v_i,
\end{align*}
易验证$\eta$是模同态且$\eta(u_i)=v_i,\ 1\leqslant i\leqslant n$。
若另有同态$\eta'$满足$\eta'(u_i)=v_i$，则
\begin{align*}
\eta'\left(\sum\limits_{i=1}^n x_i u_i\right)=\sum\limits_{i=1}^n x_i \eta'(u_i)=\sum\limits_{i=1}^n x_i v_i=\eta\left(\sum\limits_{i=1}^n x_i u_i\right),
\end{align*}
故$\eta'=\eta$，唯一性得证。

\caption{}

\end{figure}

充分性：令$\mathscr{M}'=\langle u_1,u_2,\cdots,u_n\rangle$，则存在唯一模同态$\eta:\mathscr{M}\to\mathscr{M}'$满足$\eta(u_i)=u_i,\ 1\leqslant i\leqslant n$。
设$\theta:\mathscr{M}'\to\mathscr{M}$为嵌入映射，则$\theta\circ\eta:\mathscr{M}\to\mathscr{M}$是模同态，且$\theta\circ\eta(u_i)=u_i$。
由唯一性得$\theta\circ\eta=\mathrm{id}$，故$\eta(\mathscr{M})=\mathscr{M}'=\mathscr{M}$。

\end{figure}

作模同态$\sigma:\mathscr{M}\to R^{(n)}$满足$\sigma(u_i)=e_i,\ 1\leqslant i\leqslant n$，则对任意$x=\sum\limits_{i=1}^n x_i u_i\in\mathscr{M}$，
\begin{align*}
\sigma(x)=\sum\limits_{i=1}^n x_i \sigma(u_i)=\sum\limits_{i=1}^n x_i e_i,
\end{align*}
故$\sigma$是满同态，且$\sigma(x)=0$当且仅当$x=0$，即$\sigma$为同构，因此$\mathscr{M}$是秩$n$的自由模.

\end{proof}

\begin{definition}
设$R$是交换幺环，$\mathscr{M}$是秩$n$的自由$R$-模，$u_1,u_2,\cdots,u_n$为$\mathscr{M}$的一组基。对任意$x\in\mathscr{M}$，存在唯一$x_1,x_2,\cdots,x_n\in R$使得$x=\sum\limits_{i=1}^n x_i u_i$，称$x_1,x_2,\cdots,x_n$为$x$在基$u_1,u_2,\cdots,u_n$下的\textbf{坐标}，记为
\begin{align*}
\mathrm{crd}(x;u_1,u_2,\cdots,u_n)=(x_1,x_2,\cdots,x_n)'=\begin{pmatrix}x_1\\x_2\\\vdots\\x_n\end{pmatrix}.
\end{align*}
不混淆时简记为$\mathrm{crd}(x)$，且有
\begin{align*}
x=(u_1,u_2,\cdots,u_n)\mathrm{crd}(x).
\end{align*}
\end{definition}

\begin{theorem}
设$R$是交换幺环，$\mathscr{M}$是自由$R$-模，$u_1,u_2,\cdots,u_n$与$v_1,v_2,\cdots,v_m$是$\mathscr{M}$的两组基，则$m=n$。
\end{theorem}

\begin{proof}
因$u_1,\cdots,u_n$为基，存在$R$上$n\times m$矩阵$A$，满足$\mathrm{col}_j A=\mathrm{crd}(v_j;u_1,\cdots,u_n),\ 1\leqslant j\leqslant m$，故
\begin{align*}
(v_1,v_2,\cdots,v_m)=(u_1,u_2,\cdots,u_n)A.
\end{align*}
同理，存在$m\times n$矩阵$B$满足$\mathrm{col}_i B=\mathrm{crd}(u_i;v_1,\cdots,v_m),\ 1\leqslant i\leqslant n$，故
\begin{align*}
(u_1,u_2,\cdots,u_n)=(v_1,v_2,\cdots,v_m)B=(u_1,u_2,\cdots,u_n)AB,
\end{align*}
得$AB=I_n$。同理$BA=I_m$。

假设$n<m$，作$m$阶方阵$A_1=\begin{pmatrix}A\\O\end{pmatrix}$，$B_1=(B\ O)$，则
\begin{align*}
B_1 A_1=BA=I_m.
\end{align*}
交换环上方阵行列式满足$\det(B_1)\cdot\det(A_1)=1$，但$\det(A_1)=\det(B_1)=0$，矛盾，故$n\geqslant m$。同理$m\geqslant n$，即$m=n$.

\end{proof}

\begin{corollary}
$R^{(n)}\cong R^{(m)}$当且仅当$m=n$。
\end{corollary}

\begin{corollary}
交换幺环$R$上的两个自由模$\mathscr{M},\mathscr{M}'$是模同构的当且仅当$\mathscr{M}$与$\mathscr{M}'$的秩相同。
\end{corollary}

\begin{corollary}
设$u_1,u_2,\cdots,u_n$与$v_1,v_2,\cdots,v_n$为自由$R$-模的两组基，则存在$R$上$n$阶方阵$A,B$，使得
\begin{align*}
(v_1,v_2,\cdots,v_n)=(u_1,u_2,\cdots,u_n)A,\quad (u_1,u_2,\cdots,u_n)=(v_1,v_2,\cdots,v_n)B,
\end{align*}
且$AB=BA=I_n$。对任意$x\in\mathscr{M}$，有
\begin{align*}
\mathrm{crd}(x;v_1,\cdots,v_n)=B\mathrm{crd}(x;u_1,\cdots,u_n),\quad \mathrm{crd}(x;u_1,\cdots,u_n)=A\mathrm{crd}(x;v_1,\cdots,v_n).
\end{align*}
称$A$为从基$u_1,\cdots,u_n$到$v_1,\cdots,v_n$的\textbf{过渡矩阵}，记作$T\begin{pmatrix}u_1\ u_2\ \cdots\ u_n\\v_1\ v_2\ \cdots\ v_n\end{pmatrix}$；$B$为反向过渡矩阵。
\end{corollary}

\begin{example}
交换环上的自由模与域上线性空间存在本质差异：域上$n$维线性空间中任意$n$个线性无关元素必为一组基，但该性质在自由模中一般不成立。
取$\mathbb Z$为秩$1$的自由$\mathbb Z$-模，$1$是基，$2$是“线性无关”的，但$2$不是基。
\end{example}

\begin{definition}
交换幺环$R$上全体$n$阶方阵的集合$R^{n\times n}$记为$M_n(R)$，关于矩阵加法与乘法构成环，称为$R$上的$n$阶\textbf{方阵环}。
\end{definition}

\begin{theorem}
设$R$是交换幺环，$\mathscr{M}$是秩$n$的自由$R$-模，则$\mathscr{M}$的自同态环$\mathrm{End}_R\mathscr{M}$与$M_n(R)$同构。
\end{theorem}

\begin{proof}
取$\mathscr{M}$的一组基$u_1,u_2,\cdots,u_n$。对任意$\eta\in\mathrm{End}_R\mathscr{M}$，取$n$阶方阵$M(\eta)$满足$\mathrm{col}_j M(\eta)=\mathrm{crd}(\eta(u_j);u_1,\cdots,u_n),\ 1\leqslant j\leqslant n$，则映射$\eta\mapsto M(\eta)$是$\mathrm{End}_R\mathscr{M}$到$M_n(R)$的一一映射。

对任意$A\in M_n(R)$，存在$v_1,\cdots,v_n\in\mathscr{M}$使得$\mathrm{crd}(v_j;u_1,\cdots,u_n)=\mathrm{col}_j A$，由自由模泛性质，存在唯一$\eta\in\mathrm{End}_R\mathscr{M}$满足$\eta(u_j)=v_j$，故$M(\eta)=A$。

对$\eta,\xi\in\mathrm{End}_R\mathscr{M}$，由$(\eta+\xi)(u_j)=\eta(u_j)+\xi(u_j)$，得$M(\eta+\xi)=M(\eta)+M(\xi)$。
又
\begin{align*}
(\eta\xi)(u_j)&=\eta(\xi(u_j))=\eta\left((u_1,\cdots,u_n)\mathrm{col}_j M(\xi)\right)\\
&=(\eta(u_1),\cdots,\eta(u_n))\mathrm{col}_j M(\xi)\\
&=(u_1,\cdots,u_n)M(\eta)\mathrm{col}_j M(\xi),
\end{align*}
故$\mathrm{col}_j(M(\eta\xi))=M(\eta)\mathrm{col}_j M(\xi)=\mathrm{col}_j(M(\eta)M(\xi))$，即$M(\eta\xi)=M(\eta)M(\xi)$。

因此$\eta\mapsto M(\eta)$是环同构.

\end{proof}

\begin{remark}
方阵$M(\eta;u_1,u_2,\cdots,u_n)$称为自同态$\eta$在基$u_1,u_2,\cdots,u_n$下的矩阵。
\end{remark}

\begin{theorem}
设$R$为交换幺环，$\mathscr{M}$是秩$n$的自由$R$-模，$u_1,u_2,\cdots,u_n$为$\mathscr{M}$的一组基，$\eta\in\mathrm{End}_R\mathscr{M}$，则下列条件等价：
\begin{enumerate}[(1)]
\item $\eta$是$\mathscr{M}$的自同构；
\item $\eta(u_1),\eta(u_2),\cdots,\eta(u_n)$也是$\mathscr{M}$的一组基；
\item $M(\eta;u_1,\cdots,u_n)$是$M_n(R)$中的可逆矩阵，即存在$B\in M_n(R)$使得$BM(\eta)=M(\eta)B=I_n$；
\item $\det(M(\eta))$是$R$中的可逆元素。
\end{enumerate}
\end{theorem}

\begin{definition}
$M_n(R)$中可逆矩阵的全体构成乘法群，称为$R$上的$n$阶\textbf{线性群}，记作$GL(n,R)$或$L_n(R)$。
\end{definition}
