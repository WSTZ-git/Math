\section{凸集与不动点}

\subsection{定义与基本性质}

\begin{definition}
设$\mathscr{X}$是线性空间，$E \subseteq \mathscr{X}$，称$E$为一\textbf{凸集}，如果
\begin{align*}
\lambda x + (1 - \lambda) y \in E \quad (\forall x, y \in E, \forall 0 \leqslant \lambda \leqslant 1).
\end{align*}
\end{definition}

\begin{proposition}\label{proposition:凸集任意交仍是凸集}
若$\{E_\lambda | \lambda \in \Lambda\}$是线性空间$\mathscr{X}$中的一族凸集，则$\bigcap\limits_{\lambda \in \Lambda} E_\lambda$也是凸集。
\end{proposition}
\begin{proof}

定义自明.
\end{proof}

\begin{definition}
设$\mathscr{X}$是线性空间，$A \subseteq \mathscr{X}$。若$\{E_\lambda | \lambda \in \Lambda\}$为$\mathscr{X}$中包含$A$的一切凸集，那么称$\bigcap\limits_{\lambda \in \Lambda} E_\lambda$为$A$的\textbf{凸包}，并记作$\mathrm{co}(A)$。

又对$\forall n \in \mathbb{N}, x_1, x_2, \cdots, x_n \in A$，称$\sum\limits_{i=1}^n \lambda_i x_i$为$x_1, x_2, \cdots, x_n$的\textbf{凸组合}，是指其中系数满足$\lambda_i \geqslant 0, \sum\limits_{i=1}^n \lambda_i = 1$.
\end{definition}
\begin{remark}
由\refpro{proposition:凸集任意交仍是凸集}知任意集合的凸包也是凸集.
\end{remark}

\begin{proposition}
\begin{enumerate}[(1)]
\item 设 $C$ 是实（或复）线性空间 $\mathscr{X}$ 中的凸集。则对任意整数 $n \geqslant 2$，任意点 $x_1, x_2, \cdots, x_n \in C$，以及任意非负实数 $\lambda_1, \lambda_2, \cdots, \lambda_n$ 满足 $\sum_{i=1}^n \lambda_i = 1$，其凸组合
\[
\sum_{i=1}^n \lambda_i x_i\in C.
\]

\item 设$\mathscr{X}$是线性空间，$A \subseteq \mathscr{X}$，那么$A$的凸包是$A$中元素任意凸组合的全体，即
\begin{align}
\mathrm{co}(A) = \left\{ \sum\limits_{i=1}^n \lambda_i x_i \,\bigg|\, \sum\limits_{i=1}^n \lambda_i = 1, \lambda_i \geqslant 0, x_i \in A, i = 1, 2, \cdots, n, \forall n \in \mathbb{N} \right\}. \label{eq:HJJIOu3fs1.5.1}
\end{align}
\end{enumerate}
\end{proposition}
\begin{proof}
\begin{enumerate}[(1)]
\item 由凸集的定义知 $n=2$ 时命题成立。

假设对某个整数 $n \geqslant 3$，任意 $n-1$ 个点的凸组合均属于 $C$。现考虑 $n$ 个点 $x_1, \cdots, x_n \in C$ 及系数 $\lambda_1, \cdots, \lambda_n \geqslant 0$ 满足 $\sum_{i=1}^n \lambda_i = 1$。若 $\lambda_n = 1$，则其余系数均为 $0$，凸组合即 $x_n \in C$，结论平凡成立。否则 $\lambda_n < 1$。记
$$
\mu \triangleq \sum_{i=1}^{n-1} \lambda_i = 1 - \lambda_n > 0.
$$
对 $i = 1, \cdots, n-1$, 令 $\alpha_i \triangleq \frac{\lambda_i}{\mu}$,则 $\alpha_i \geqslant 0$ 且 $\sum_{i=1}^{n-1} \alpha_i = 1$。由归纳假设知$y \triangleq \sum_{i=1}^{n-1} \alpha_i x_i\in C$.
此时原凸组合可写为
\[
\sum_{i=1}^n \lambda_i x_i = \mu y + \lambda_n x_n.
\]
由于 $\mu, \lambda_n \geqslant 0$ 且 $\mu + \lambda_n = 1$,$x_n,y \in C$ ,因此 $\sum_{i=1}^n \lambda_i x_i \in C$.故由数学归纳法知结论成立.

\item 若令$S$表示\eqref{eq:HJJIOu3fs1.5.1}式右端，则$A \subseteq S$而且$S$是凸集，从而$S \supseteq \mathrm{co}(A)$。反之，设$F$为包含$A$的任一凸集，那么$x_i \in F(i = 1, 2, \cdots, n)$，从而由(1)$\sum\limits_{i=1}^n \lambda_i x_i \in F$，即得$S \subseteq F$，从而$S \subseteq \mathrm{co}(A)$.
\end{enumerate}

\end{proof}

\begin{definition}
设$\mathscr{X}$是线性空间，$C$是$\mathscr{X}$上含有$\theta$的凸子集，其中$\theta$是零向量.在$\mathscr{X}$上规定一个取值于$[0, \infty]$的函数
\begin{align}
P(x) = \inf \left\{ \lambda > 0 \,\bigg|\, \frac{x}{\lambda} \in C \right\} \quad (\forall x \in \mathscr{X}) \label{eq:HJJIOu3fs1.5.2}
\end{align}
与$C$对应，称函数$P$为$C$的$\mathbf{Minkowski}$\textbf{泛函}.
\end{definition}

\begin{proposition}\label{proposition:Minkowski泛函的基本性质}
设$\mathscr{X}$是线性空间，$C$是$\mathscr{X}$上含有$\theta$的凸子集。若$P$为$C$的Minkowski泛函，则$P$具有下列性质:
\begin{enumerate}[(1)]
\item\label{proposition:Minkowski泛函的基本性质-1} $P(x) \in [0, \infty]$, $P(\theta) = 0$,并且$P(x)=+\infty$当且仅当$\frac{x}{\lambda}\notin C,\forall \lambda >0$;

\item\label{proposition:Minkowski泛函的基本性质-2} 正齐次性:$P(\lambda x) = \lambda P(x) \quad (\forall x \in \mathscr{X}, \forall \lambda > 0)$；

\item\label{proposition:Minkowski泛函的基本性质-3} 次可加性:$P(x + y) \leqslant P(x) + P(y) \quad (\forall x, y \in \mathscr{X})$.进而对$\forall x,y\in \mathscr{X}$,有
\begin{gather*}
P\left( x \right) -P\left( y \right) \leqslant P\left( x-y \right) ,\,\, P\left( y \right) -P\left( x \right) \leqslant P\left( y-x \right) ,
\\
\left| P\left( x \right) -P\left( y \right) \right|\leqslant \max \left\{ P\left( x-y \right) ,P\left( y-x \right) \right\} .
\end{gather*}
\end{enumerate}
\end{proposition}
\begin{proof}

(1)和(2)是显然的.只有(3)是需要验证的。不妨设$P(x), P(y)$有穷，对$\forall \varepsilon > 0$，取$\lambda_1 = P(x) +\frac{ \varepsilon}{2}, \lambda_2 = P(y) +\frac{ \varepsilon}{2}$，使得
\begin{align*}
\frac{x}{\lambda_1} \in C, \quad \frac{y}{\lambda_2} \in C.
\end{align*}
因为$C$是凸的，所以
\begin{align*}
\frac{x + y}{\lambda_1 + \lambda_2} = \frac{\lambda_1}{\lambda_1 + \lambda_2} \cdot \frac{x}{\lambda_1} + \frac{\lambda_2}{\lambda_1 + \lambda_2} \cdot \frac{y}{\lambda_2} \in C.
\end{align*}
这表明
\begin{align*}
P(x + y) \leqslant \lambda_1 + \lambda_2 = P(x) + P(y) + \varepsilon.
\end{align*}
由$\varepsilon > 0$的任意性得到次可加性.再由次可加性可得
\begin{align*}
P\left( x \right) =P\left( x-y \right) +P\left( y \right) \Longrightarrow P\left( x \right) -P\left( y \right) \leqslant P\left( x-y \right) .
\end{align*}
交换$x,y$即得$P\left( y \right) -P\left( x \right) \leqslant P\left( y-x \right)$,故
\begin{align*}
\left| P\left( x \right) -P\left( y \right) \right|\leqslant \max \left\{ P\left( x-y \right) ,P\left( y-x \right) \right\} .
\end{align*}

\end{proof}

\begin{definition}
设$\mathscr{X}$是线性空间，凸集$C\subseteq \mathscr{X}$含有零向量$\theta$.
\begin{enumerate}[(1)]
\item 若$\forall x \in \mathscr{X}, \exists \lambda > 0$，使得$\frac{x}{\lambda} \in C$,则称$C$是\textbf{吸收的}.

\item 若$x \in C \implies -x \in C$,则称$C$是\textbf{对称的}.
\end{enumerate}
\end{definition}

\begin{proposition}\label{proposition:吸收凸集和对称凸集的等价}
设$\mathscr{X}$是线性空间，凸集$C\subseteq \mathscr{X}$含有零向量$\theta$,则
\begin{enumerate}[(1)]
\item $C$是吸收凸集当且仅当其Minkowski泛函$P(x)$是实值函数,即$P(x)\in [0,+\infty),\forall x\in C$；

\item 若$C$是对称凸集,则其Minkowski泛函$P(x)$是实齐次的，即
\begin{align*}
P(\alpha x) = |\alpha| P(x) \quad (\forall \alpha \in \mathbb{R}).
\end{align*}
\end{enumerate}
\end{proposition}
\begin{proof}
\begin{enumerate}[(1)]
\item 由\rrefpro{proposition:Minkowski泛函的基本性质}{proposition:Minkowski泛函的基本性质-1}和吸收凸集的定义知
\begin{align*}
C\text{不是吸收凸集}\Longleftrightarrow \text{存在}x\in C,\text{使得}\frac{x}{\lambda}\notin C,\forall \lambda >0\Longleftrightarrow \text{存在}x\in C,\text{使得}\frac{x}{\lambda}P\left( x \right) =+\infty .
\end{align*}
故$C$是吸收凸集当且仅当$P(x)\in [0,+\infty),\forall x\in C$.

\item 当$\alpha \geqslant 0$时，由\refpro{proposition:Minkowski泛函的基本性质}知结论成立。下设$\alpha <0$，记$\beta =-\alpha$，则由$C$的对称性可得
\begin{align*}
P(\alpha x) = P(-\beta x) = \inf\left\{ \lambda >0 \mid -\frac{\beta x}{\lambda}\in C \right\} = \inf\left\{ \lambda >0 \mid \frac{\beta x}{\lambda}\in C \right\} = P(\beta x) = \beta P(x) = |\alpha| P(x).
\end{align*}
\end{enumerate}
\end{proof}

\begin{definition}
复线性空间$\mathscr{X}$的一个子集$C$称为是\textbf{均衡的}，是指
\begin{align*}
x \in C \implies \alpha x \in C \quad (\forall \alpha \in \mathbb{C}, |\alpha| = 1).
\end{align*}
\end{definition}
\begin{remark}
均衡凸集显然也是对称凸集.
\end{remark}

\begin{proposition}\label{proposition:泛函分析--命题1.5.10}
复线性空间$\mathscr{X}$上的任一个均衡吸收凸集$C$，决定了这空间上的一个半范数,即凸集$C$的Minkowski泛函.
\end{proposition}
\begin{proof}

由\refpro{proposition:Minkowski泛函的基本性质}和\refpro{proposition:吸收凸集和对称凸集的等价}可知,凸集$C$的Minkowski泛函就是这个空间$\mathscr{X}$上的一个半范数.

\end{proof}

\begin{proposition}\label{proposition:泛函分析--命题1.5.11}
设$\mathscr{X}$是一个$B^*$空间，$C$是一个含有零向量$\theta$的闭凸集。如果$P(x)$是$C$的Minkowski泛函，那么$P(x)$下半连续，且有
\begin{align}
C = \{ x \in \mathscr{X} \mid P(x) \leqslant 1 \}. \label{eq:HJJIOu3fs1.5.3}
\end{align}
此外，如果$C$还是有界的，那么$P(x)$适合
\begin{align*}
P(x) = 0 \iff x = \theta.
\end{align*}
又若$C$以$\theta$为在空间$\mathscr{X}$上的一个内点，那么$C$是吸收的，并且$P(x)$还是Lipschitz连续的,进而也是一致连续的,更是连续的.
\end{proposition}
\begin{proof}
\begin{enumerate}[(1)]
\item $\forall \alpha > 0$，若$x \in \alpha C$，即$\frac{x}{\alpha} \in C$，由$P(x)$的定义便有$P(x) \leqslant \alpha$；反之，若$P(x) \leqslant \alpha$，则由下确界定义知对$\forall n \in \mathbb{N}$,存在$\lambda_0>0$,使得
\begin{align*}
P\left( x \right) \leqslant \lambda _0<P\left( x \right) +\frac{1}{n}\leqslant \alpha +\frac{1}{n},\quad \frac{x}{\lambda _0}\in C.
\end{align*}
于是由$C$是凸集知
\begin{align*}
\frac{x}{\alpha +\frac{1}{n}}=\frac{\lambda _0}{\alpha +\frac{1}{n}}\cdot \frac{x}{\lambda _0}+\left( 1-\frac{\lambda _0}{\alpha +\frac{1}{n}} \right) \cdot \theta \in C,\quad \frac{x}{\alpha +\frac{1}{n}}\rightarrow \frac{x}{\alpha}\quad (n\rightarrow \infty ).
\end{align*}
又因为$C$是闭的，所以$x \in \alpha C$。这就证得
\begin{align*}
x\in \alpha C\iff P(x)\leqslant \alpha\quad (\forall \alpha >0).
\end{align*}
即
\begin{align*}
\alpha C = \{ x \in \mathscr{X} \mid P(x) \leqslant \alpha \} \quad (\forall \alpha > 0). 
\end{align*}
特别地，令$\alpha = 1$即得\eqref{eq:HJJIOu3fs1.5.3}式。又因为$\forall \alpha > 0, \alpha C$是闭集，所以其补集$\{x\in \mathscr{X}\mid P(x)>\alpha\}$是开集，即$P$是下半连续的.

\item 由\rrefpro{proposition:Minkowski泛函的基本性质}{proposition:Minkowski泛函的基本性质-1}知$P(\theta) = 0$。在$C$有界假定下，即$\exists r > 0$，使得$C \subseteq B(\theta, r)$，于是
\begin{align*}
\forall x \in \mathscr{X} \setminus \{\theta\} \implies \frac{rx}{\|x\|} \notin C \implies P(x) \geqslant \frac{\|x\|}{r}.
\end{align*}
由此可见，当$P(x) = 0$时，$x = \theta$.
\item 若$C$以$\theta$为内点，即$\exists r > 0$，使得$B(\theta, r) \subseteq C$，那么
\begin{align*}
\frac{rx}{2\|x\|} \in C \quad (\forall x \in \mathscr{X} \setminus \{\theta\}).
\end{align*}
因此$C$是吸收的，并且有$P(x) \leqslant \frac{2\|x\|}{r} (\forall x \in \mathscr{X})$。于是由\rrefpro{proposition:Minkowski泛函的基本性质}{proposition:Minkowski泛函的基本性质-3}可得
\begin{align*}
|P(x) - P(y)| \leqslant \max(P(x - y), P(y - x)) \leqslant \frac{2}{r} \|x - y\| \quad (\forall x, y \in \mathscr{X}),
\end{align*}
从而$P(x)$是Lipschitz连续的.
\end{enumerate}

\end{proof}

\begin{corollary}\label{corollary:紧凸子集同胚单位球}
若$C$是$\mathbb{R}^n$中的一个紧凸子集，则必存在正整数$m \leqslant n$，使得$C$同胚于$\mathbb{R}^m$中的单位球.
\end{corollary}
\begin{note}
事实上,先取定$C$上的一组基$\{u_1,\cdots,u_m\}$.任取$e_{m+1}\in C$,再令$e_i=u_i+e_{m+1},i=1,\cdots,m$即可得到下述证明需要的$\{e_1,\cdots,e_{m+1}\}$.
\end{note}
\begin{proof}
\begin{enumerate}[(1)]
\item 因为$C$是$\mathbb{R}^n$中的紧集,所以$C$是有界闭集.用$E$表示包含$C$的最小线性子流形,则$E$也是有界闭集.设$E$的维数是$m(\leqslant n)$。于是在$C$上必有$m + 1$个向量$e_1, e_2, \cdots, e_m, e_{m+1}$，使得$e_i - e_{m+1}(i = 1, 2, \cdots, m)$是线性无关的.
\item 令
\begin{align*}
e_0 = \frac{1}{m + 1} \sum\limits_{i=1}^{m+1} e_i.
\end{align*}
由$C$是凸集知$e_0 \in C \subseteq E$，再由$E$是$n$维线性子流形知$E - e_0$是一个$m$维线性子空间。又因为$e_i - e_{m+1}(i = 1, 2, \cdots, m)$是线性无关的,所以$\{e_1-e_{m+1},\cdots,e_m-e_{m+1}\}$构成$E$的一组基.于是对$\forall y \in E$，存在唯一的表示
\begin{align}
y = \sum\limits_{i=1}^m \mu_i (e_i - e_0) + e_0, \label{eq:HJJIOu3fs1.5.5}
\end{align}
并在$E - e_0$上可引进一个范数
\begin{align*}
\|z\| = \left( \sum\limits_{i=1}^m |\mu_i|^2 \right)^{\frac{1}{2}} \quad (z = y - e_0, y \in E). 
\end{align*}
下面都在$B^*$空间$(E-e_0,\|\cdot\|)$上考虑.

\item 任取$\delta \in \left( 0,\frac{1}{m+1} \right)$，记$B_E\left( e_0,\delta \right) =B\left( e_0,\delta \right) \cap E$，则对$\forall y\in B_E\left( e_0,\delta \right) \subseteq E$，由\eqref{eq:HJJIOu3fs1.5.5}式知$y$存在唯一表示\begin{align*}
y&=\sum\limits_{i=1}^m \mu_i (e_i-e_0)+e_0=\sum\limits_{i=1}^m \mu_i e_i+\left( 1-\sum\limits_{j=1}^m \mu_j \right) e_0\\
&=\sum\limits_{i=1}^m \left[ \mu_i+\frac{1}{m+1}\left( 1-\sum\limits_{j=1}^m \mu_j \right) \right] e_i+\frac{1}{m+1}\left( 1-\sum\limits_{j=1}^m \mu_j \right) e_{m+1}.
\end{align*}
由$\delta \in \left( 0,\frac{1}{m+1} \right)$及$y\in B\left( \theta ,\delta \right)$知上式右端各项系数都是正数，且各项系数的总和满足\begin{align*}
\sum\limits_{i=1}^m \left[ \mu_i+\frac{1}{m+1}\left( 1-\sum\limits_{j=1}^m \mu_j \right) \right]+\frac{1}{m+1}\left( 1-\sum\limits_{j=1}^m \mu_j \right) =1.
\end{align*}
故$y\in \mathrm{co}\left\{ e_1,e_2,\cdots ,e_m,e_{m+1} \right\} \subseteq C$。因此\begin{align*}
B_E\left( e_0,\delta \right) =B\left( e_0,\delta \right) \cap E\subseteq C\Longrightarrow B_E\left( \theta ,\delta \right) \lhd B\left( \theta ,\delta \right) \cap \left( E-e_0 \right) \subseteq C-e_0,
\end{align*}
又$B_E\left( \theta ,\delta \right)$是在$E-e_0$的子空间拓扑上的开集，故$\theta$是$C-e_0$在空间$E-e_0$上的一个内点.进而$\theta \in C$.

\item 由(1)(3)知,在$E - e_0$上，$C - e_0$是一个以$\theta$为内点的有界闭凸集.由\refpro{proposition:泛函分析--命题1.5.11}知$C-e_0$的Minkowski泛函$P(z)$是$E - e_0$上的一个一致连续、正齐次、次可加泛函，适合$P(z) = 0 \iff z = \theta$.应用\refthe{theorem:泛函分析--定理1.4.22}，$\exists$常数$C_1, C_2 > 0$，使得
\begin{align}\label{eq:::2isejroIORU}
C_1 \|z\| \leqslant P(z) \leqslant C_2 \|z\| \quad (\forall z \in E - e_0).
\end{align}

\item 设$B^m(\theta, 1)$是$E - e_0$中的单位球，若令
\begin{align*}
\varphi(z) = \begin{cases}
e_0 + \frac{\|z\| z}{P(z)}, & z \neq \theta, \\
e_0, & z = \theta,
\end{cases}
\end{align*}
\begin{enumerate}[(1)]
\item 显然$\varphi(\theta)=e_0\in C$。现对$\forall z\in B^m(\theta,1)\setminus\theta$，记$w=\frac{\|z\|}{P(z)}z$，则\begin{align*}
P(w)=P\left(\frac{\|z\|}{P(z)}z\right)=\frac{\|z\|}{P(z)}P(z)=\|z\|\leqslant 1\Longrightarrow w\in\left\{x\in E-e_0\mid P(x)\leqslant 1\right\}.
\end{align*}
又由\refpro{proposition:泛函分析--命题1.5.11}知\begin{align}\label{eq::::::----jhriuo23jFJIOEJ31.0}
C-e_0=\left\{x\in E-e_0\mid P(x)\leqslant 1\right\},
\end{align}
所以$w\in C-e_0$，进而$\varphi(z)=w+e_0\in C$。故$\varphi\left(B^m(\theta,1)\right)\subseteq C$，因此$\varphi$是$B^m(\theta,1)\rightarrow C$上的良定义的映射。

\item 显然由$\varphi(z_1)=\varphi(z_2)=e_0$必有$z_1=z_2=\theta$。若$\varphi(z_1)=\varphi(z_2)\ne e_0$，则\begin{align}\label{eq::::::----jhriuo23jFJIOEJ31.1}
\frac{\|z_1\|}{P(z_1)}z_1=\frac{\|z_2\|}{P(z_2)}z_2\ne 0.
\end{align}
对\eqref{eq::::::----jhriuo23jFJIOEJ31.1}式两边同时取$P$得\begin{align}\label{eq::::::----jhriuo23jFJIOEJ31.2}
\|z_1\|=P\left(\frac{\|z_1\|}{P(z_1)}z_1\right)=P\left(\frac{\|z_2\|}{P(z_2)}z_2\right)=\|z_2\|.
\end{align}
再对\eqref{eq::::::----jhriuo23jFJIOEJ31.1}式两边同时取范数得\begin{align}\label{eq::::::----jhriuo23jFJIOEJ31.3}
\frac{\|z_1\|^2}{P(z_1)}=\frac{\|z_2\|^2}{P(z_2)}\Longrightarrow P(z_1)=P(z_2).
\end{align}
将\eqref{eq::::::----jhriuo23jFJIOEJ31.2}、\eqref{eq::::::----jhriuo23jFJIOEJ31.3}式代入\eqref{eq::::::----jhriuo23jFJIOEJ31.1}式得$z_1=z_2$。故$\varphi$是单射。

\item 任取$y\in C$，记$y_0=y-e_0\in C-e_0$，则由\eqref{eq::::::----jhriuo23jFJIOEJ31.0}式知\begin{align}\label{eq::::::----jhriuo23jFJIOEJ31.4}
y_0\in C-e_0=\left\{x\in E-e_0\mid P(x)\leqslant 1\right\}\Longrightarrow P(y_0)\leqslant 1.
\end{align}
若$y_0=\theta$，则$y=e_0=\varphi(\theta)$。若$y_0\ne\theta$，取$z=\frac{P(y_0)}{\|y_0\|}y_0$，则由\eqref{eq::::::----jhriuo23jFJIOEJ31.4}式知\begin{align*}
\|z\|=\left\|\frac{P(y_0)}{\|y_0\|}y_0\right\|=P(y_0)\leqslant 1\Longrightarrow z\in B^m(\theta,1),
\end{align*}
\begin{align*}
\varphi(z)&=e_0+\frac{\|z\|}{P(z)}z=e_0+\frac{\left\|\frac{P(y_0)}{\|y_0\|}y_0\right\|}{P\left(\frac{P(y_0)}{\|y_0\|}y_0\right)}\cdot\frac{P(y_0)}{\|y_0\|}y_0\\
&=e_0+\frac{P(y_0)}{\frac{P(y_0)}{\|y_0\|}P(y_0)}\cdot\frac{P(y_0)}{\|y_0\|}y_0=e_0+y_0=y.
\end{align*}
故$\varphi$是满射。

\item 当$z\ne\theta$时，由$(4)$和\rrefpro{proposition:范数定义的基本推论}{proposition:范数定义的基本推论-3}知$\|z\|,P(z)$都是连续的，且$P(z)>0$，故此时$\varphi(z)=e_0+\frac{\|z\|}{P(z)}z$也连续。
对$\forall z\ne\theta$，由\eqref{eq:::2isejroIORU}式知\begin{align*}
\|\varphi(z)-e_0\|=\frac{\|z\|}{P(z)}\|z\|\leqslant\frac{\|z\|}{C_1\|z\|}=\frac{\|z\|}{C_1}\rightarrow 0,\quad z\rightarrow\theta.
\end{align*}
故$\varphi$在$\theta$处也连续，综上$\varphi$在$B^m(\theta,1)$上连续。

\item 定义\begin{align*}
\psi(z)=
\begin{cases}
\frac{P(z-e_0)}{\|z-e_0\|}(z-e_0),&z\ne e_0,\\
\theta,&z=e_0,
\end{cases}
\end{align*}
则\begin{align*}
\psi \left( \varphi \left( z \right) \right) =\begin{cases}
\frac{P\left( \frac{\left\| z \right\|}{P\left( z \right)}z \right)}{\left\| \frac{\left\| z \right\|}{P\left( z \right)}z \right\|}\frac{\left\| z \right\|}{P\left( z \right)}z,z\ne e_0,\\
\psi \left( e_0 \right) ,z=\theta\\
\end{cases}=\begin{cases}
\frac{\left\| z \right\|}{\frac{\left\| z \right\| ^2}{P\left( z \right)}}\frac{\left\| z \right\|}{P\left( z \right)}z,z\ne e_0,\\
\theta ,z=\theta\\
\end{cases}=\begin{cases}
z,z\ne e_0,\\
\theta ,z=\theta .\\
\end{cases}
\end{align*}
\begin{align*}
\varphi(\psi(z))&=
\begin{cases}
e_0+\frac{\left\|\frac{P(z-e_0)}{\|z-e_0\|}(z-e_0)\right\|}{P\left(\frac{P(z-e_0)}{\|z-e_0\|}(z-e_0)\right)}\cdot\frac{P(z-e_0)}{\|z-e_0\|}(z-e_0),&z\ne e_0,\\
\varphi(\theta),&z=e_0
\end{cases}\\
&=
\begin{cases}
e_0+\frac{P(z-e_0)}{\frac{P(z-e_0)}{\|z-e_0\|}P(z-e_0)}\cdot\frac{P(z-e_0)}{\|z-e_0\|}(z-e_0),&z\ne e_0,\\
\theta,&z=e_0
\end{cases}\\
&=
\begin{cases}
z,&z\ne e_0,\\
\theta,&z=e_0.
\end{cases}
\end{align*}
故$\psi=\varphi^{-1}$。当$z\ne e_0$时，由$(4)$和\rrefpro{proposition:范数定义的基本推论}{proposition:范数定义的基本推论-3}知$\|z\|,P(z)$都是连续的，故此时$\psi(z)$也连续。
对$\forall z\ne e_0$，由\eqref{eq:::2isejroIORU}式知\begin{align*}
\|\psi(z)\|=P(z-e_0)\leqslant C_2\|z-e_0\|\rightarrow 0,\quad z\rightarrow e_0.
\end{align*}
故$\psi$在$e_0$处也连续，综上$\psi$在$C$上连续。
\end{enumerate}

综上可知$\varphi: B^m(\theta, 1) \to C$是一个在上同胚.
\end{enumerate}
\end{proof}

\subsection{Brouwer与Schauder不动点定理}

\begin{theorem}[Brouwer不动点定理]\label{theorem:泛函分析---Brouwer不动点定理}
设$B$是$\mathbb{R}^n$中的闭单位球，又设$T: B \to B$是一个连续映射，那么$T$必有一个不动点$x \in B$.
\end{theorem}

\begin{corollary}\label{corollary:泛函分析--推论1.5.14}
设$C$是$\mathbb{R}^n$中的一个紧凸子集，$T: C \to C$是连续的，则$T$必有一个在$C$上的不动点.
\end{corollary}
\begin{proof}

由\refcor{corollary:紧凸子集同胚单位球}知$C$与$\mathbb{R}^m(m \leqslant n)$中的一个单位球同胚，记此同胚为$\varphi: B^m(\theta, 1) \to C$。考察映射
\begin{align*}
T_\varphi = \varphi^{-1} \circ T \circ \varphi.
\end{align*}
显然$T_\varphi: B^m(\theta, 1) \to B^m(\theta, 1)$是连续映射。对$T_\varphi$应用\hyperref[theorem:泛函分析---Brouwer不动点定理]{Brouwer不动点定理}，存在$x \in B^m(\theta, 1)$使得$T_\varphi x = x$。由此得到$y = \varphi x \in C$是$T$的不动点.

\end{proof}

\begin{theorem}[Schauder不动点定理]\label{theorem:泛函分析---Schauder不动点定理}
设$C$是$B^*$空间$\mathscr{X}$中的一个闭凸子集，$T: C \to C$连续且$T(C)$列紧，则$T$在$C$上必有一个不动点.
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item 因为$T(C)$是列紧集，所以对$\forall n \in \mathbb{N}$，存在$\frac{1}{n}$网$N_n = \{y_1, y_2, \cdots, y_{r_n}\}$,由\refpro{proposition:泛函分析--网的等价定义}知
\begin{align*}
T(C) \subseteq \bigcup\limits_{i=1}^{r_n} B\left(y_i, \frac{1}{n}\right) \quad (y_i \in T(C), i = 1, 2, \cdots, r_n).
\end{align*}
记$E_n = \mathrm{span}N_n$，即$E_n$为由$N_n$张成的有穷维线性子空间.
\item 做$T(C) \to \mathrm{co}(N_n)$的映射$I_n$如下:
\begin{align}
I_n(y) = \sum\limits_{i=1}^{r_n} y_i \lambda_i(y) \quad (\forall y \in T(C)), \label{eq:HJJIOu3fs1.5.8}
\end{align}
其中
\begin{align*}
\lambda_i(y) = \frac{m_i(y)}{\sum\limits_{i=1}^{r_n} m_i(y)},
\quad
m_i(y) = \begin{cases}
1 - n\|y - y_i\|, & y \in B\left(y_i, \frac{1}{n}\right), \\
0, & y \notin B\left(y_i, \frac{1}{n}\right).
\end{cases}
\end{align*}
因为$m_i(y) \geqslant 0$，并且$\forall y \in T(C), \exists i_0(1 \leqslant i_0 \leqslant r_n)$，使得$y \in B\left(y_{i_0}, \frac{1}{n}\right)$，于是$m_{i_0}(y) > 0$，所以$\sum\limits_{i=1}^{r_n} m_i(y) > 0(\forall y \in T(C))$。因此，$\lambda_i(y)(1 \leqslant i \leqslant r_n)$有定义并满足
\begin{align}
\lambda_i(y) \geqslant 0 \quad (1 \leqslant i \leqslant r_n), \quad \sum\limits_{i=1}^{r_n} \lambda_i(y) = 1. \label{eq:HJJIOu3fs1.5.9}
\end{align}
于是$I_n$在$T(C)$上有定义，并且\eqref{eq:HJJIOu3fs1.5.8}式与\eqref{eq:HJJIOu3fs1.5.9}式蕴含$I_n(y)$是$N_n$元素的凸组合，从而$I_n(y) \in \mathrm{co}(N_n)$,故$I_n$是良定义的.此外还有
\begin{align*}
\|I_n y - y\| &= \left\| \sum\limits_{i=1}^{r_n} y_i \lambda_i(y) - \sum\limits_{i=1}^{r_n} y \lambda_i(y) \right\| \\
&= \left\| \sum\limits_{i=1}^{r_n} (y_i - y) \lambda_i(y) \right\| \leqslant \sum\limits_{i=1}^{r_n} \|y_i - y\| \lambda_i(y) \\
&= \sum_{\substack{y \in B(y_i, \frac{1}{n})}} \|y_i - y\| \lambda_i(y) + \sum_{\substack{y \notin B(y_i, \frac{1}{n})}} \|y_i - y\| \lambda_i(y) \\
&<\frac{1}{n}\sum_{i=1}^{r_n}{\lambda _i\left( y \right)}+0=\frac{1}{n}+0=\frac{1}{n}.
\end{align*}
\item 注意到$T: C \to C, N_n \subseteq T(C)$，而$C$是凸的，所以$\mathrm{co}(N_n) \subseteq C$。令$T_n \triangleq I_n \circ T$，那么$T_n: \mathrm{co}(N_n) \to \mathrm{co}(N_n)$。又注意到$\mathrm{co}(N_n)$是有限维线性空间$E_n$中的一个有界闭凸子集，故由\refcor{corollary:有限维线性空间等价于任意有界集列紧}知$\mathrm{co}(N_n)$是列紧的,又因为$\mathrm{co}(N_n)$是闭紧,所以由\rrefthe{theorem:列紧的基本性质}{theorem:列紧的基本性质-2}知$\mathrm{co}(N_n)$是自列紧的,因此由\refthe{theorem:紧集等价于自列紧}知$\mathrm{co}(N_n)$是紧的.应用\refcor{corollary:泛函分析--推论1.5.14}，$\exists x_n \in \mathrm{co}(N_n) \subseteq C$，使得
\begin{align*}
T_n x_n = x_n. 
\end{align*}
又因为$T(C)$是列紧集而$C$是闭集，故存在子列$\{n_k\}\subseteq\mathbb{N}$及$x\in C$，使得
\begin{align}
Tx_{n_k}\to x\quad (k\to\infty).
\label{eq:schauder_subseq}
\end{align}
结合逼近序列$\{T_n\}$的性质，有
\begin{align*}
\|x_n - x\| &= \|T_nx_n - x\| 
= \|I_nTx_n - Tx_n + Tx_n - x\| \\
&\leqslant \|I_nTx_n - Tx_n\| + \|Tx_n - x\| 
< \frac{1}{n} + \|Tx_n - x\|\quad (\forall n\in\mathbb{N}).
\end{align*}
联合\eqref{eq:schauder_subseq}式与上式可得$x_{n_k}\to x\ (k\to\infty)$。再由$T$的连续性及\eqref{eq:schauder_subseq}式，得$Tx=x$.
\end{enumerate}
\end{proof}

\begin{definition}
设$\mathscr{X}$是$B^*$空间，$E\subseteq\mathscr{X}$，称映射$T:E\to\mathscr{X}$是\textbf{紧的}，如果$T$连续且将$E$中的任意有界集映为$\mathscr{X}$中的列紧集。
\end{definition}

\begin{corollary}\label{corollary:泛函分析--推论1.5.121}
设$C$是$B^*$空间$\mathscr{X}$中的有界闭凸子集，$T:C\to C$是紧映射，则$T$在$C$上必有不动点。
\end{corollary}
\begin{proof}

由$T$是紧映射知$T$连续且$T(C)$是列紧的,故由\hyperref[theorem:泛函分析---Schauder不动点定理]{Schauder不动点定理}知$T$在$C$上必有不动点.
\end{proof}

\begin{theorem}[Caratheodory定理]\label{theorem:Caratheodory定理}
设函数$f(t,x):\mathbb{R}\times\mathbb{R}\to\mathbb{R}$在$[-h,h]\times[\xi-b,\xi+b]$上二元连续，且存在常数$M>0$使得$|f(t,x)|\leqslant M$对所有$(t,x)\in[-h,h]\times[\xi-b,\xi+b]$成立。若$h<\frac{b}{M}$，则初值问题
\begin{align*}
\begin{cases}
x'(t)=f(t,x(t)), \\
x(0)=\xi
\end{cases}
\end{align*}
在$[-h,h]$上存在解$x(t)$.
\end{theorem}
\begin{proof}

由\rrefpro{proposition:常见F空间}{proposition:常见F空间-1}知$(C[-h,h],\|\cdot\|)$是$B$空间,其中
\begin{align*}
\|u\|=\max_{x\in M}|u(x)|,\quad \forall u\in C[-h,h].
\end{align*}
考虑空间$C[-h,h]$中的闭球$\overline{B}(\xi,b)=\{x\in C[-h,h]:\|x-\xi\|_{C[-h,h]}\leqslant b\}$，定义映射$T:\overline{B}(\xi,b)\to C[-h,h]$为
\begin{align*}
(Tx)(t)=\xi+\int_{0}^{t}f(\tau,x(\tau))\mathrm{d}\tau.
\end{align*}

首先证明$T$映$\overline{B}(\xi,b)$到自身。对任意$x\in\overline{B}(\xi,b)$，有
\begin{align}\label{eq::FIOUSAHIO242}
\|Tx-\xi\|_{C[-h,h]}=\max_{t\in[-h,h]}\left|\int_{0}^{t}f(\tau,x(\tau))\mathrm{d}\tau\right|\leqslant Mh.
\end{align}
当$h\leqslant\frac{b}{M}$时，$\|Tx-\xi\|_{C[-h,h]}\leqslant b$，故$T(\overline{B}(\xi,b))\subseteq\overline{B}(\xi,b)$.

一方面，对任意$t,t'\in[-h,h]$及$x\in\overline{B}(\xi,b)$，有
\begin{align*}
|(Tx)(t)-(Tx)(t')|=\left|\int_{t'}^{t}f(\tau,x(\tau))\mathrm{d}\tau\right|\leqslant M|t-t'|,
\end{align*}
且由\eqref{eq::FIOUSAHIO242}式可知
\begin{align*}
|(Tx)(t)|\leqslant|\xi|+Mh.
\end{align*}
故$T(\overline{B}(\xi,b))$是一致有界且等度连续的。由\hyperref[theorem:泛函分析--Arzelà-Ascoli定理]{Arzelà-Ascoli定理}，$T(\overline{B}(\xi,b))$是列紧集。另一方面，$T$的连续性可由$f$的连续性及积分的连续性直接得到。

又$\overline{B}(\xi,b)$是$\mathbb{R}$中的闭凸集,故由\hyperref[theorem:泛函分析---Schauder不动点定理]{Schauder不动点定理}知，$T$在$\overline{B}(\xi,b)$上存在不动点$x(t)$，该不动点即为初值问题的解.

\end{proof}
