\section{分裂域$\,\,$正规扩张}

\begin{definition}
设$F[x]$是域$F$的一元多项式环，$f(x)\in F[x]$，$\deg f(x)=n$。若$F$的扩域$K$满足下列条件：
\begin{enumerate}[(1)]
\item $f(x)$在$K[x]$内可分解为一次因式之积，
\begin{align}
f(x)=c(x-\alpha_1)(x-\alpha_2)\cdots(x-\alpha_n);\label{eq:split-fact}
\end{align}
\item \begin{align}K=F(\alpha_1,\alpha_2,\cdots,\alpha_n),\label{eq:split-gen}
\end{align}
\end{enumerate}
则称$K$是$f(x)$的\textbf{分裂域}.
\end{definition}

\begin{theorem}
设$f(x)$是域$F$上的一元多项式且$\deg f(x)>0$，则$f(x)$的分裂域存在。
\end{theorem}
\begin{proof}

对$\deg f(x)$用数学归纳法。
当$\deg f(x)=1$时，$F$本身就是$f(x)$的分裂域。

设$\deg f(x)=n>1$，$p(x)$是$f(x)$的一个不可约因子，则存在$F$的代数扩张$F_1=F(\alpha_1)$，其中$\alpha_1$满足$p(\alpha_1)=0$，故$f(\alpha_1)=0$。
$f(x)$作为$F_1[x]$内的多项式有分解
\begin{align*}
f(x)=(x-\alpha_1)f_1(x),\quad \deg f_1(x)=\deg f(x)-1,
\end{align*}
取$f_1(x)$对$F_1$的分裂域
\begin{align*}
K=F_1(\alpha_2,\alpha_3,\cdots,\alpha_n),
\end{align*}
显然$f(x)\in K[x]$且可分解为一次因式之积，同时
\begin{align*}
K=F_1(\alpha_2,\alpha_3,\cdots,\alpha_n)=F(\alpha_1,\alpha_2,\cdots,\alpha_n),
\end{align*}
故$K$是$f(x)$对$F$的分裂域。

\end{proof}

\begin{corollary}
设$K$是$f(x)\in F[x]$的分裂域且$\deg f(x)=n$，则$[K:F]\leqslant n!$。
\end{corollary}

\begin{proof}

因$[F(\alpha_1):F]\leqslant n$，结合分裂域存在性的归纳证明可得。

\end{proof}

\begin{corollary}
设$K$是$f(x)\in F[x]$的分裂域，又$E$是$F$与$K$的中间域，则$K$也是$f(x)\in E[x]$的分裂域。
\end{corollary}

\begin{proof}

若$K=F(\alpha_1,\alpha_2,\cdots,\alpha_n)$，则有
\begin{align*}
K=F(\alpha_1,\alpha_2,\cdots,\alpha_n)\subseteq E(\alpha_1,\alpha_2,\cdots,\alpha_n)\subseteq K,
\end{align*}
故$K=E(\alpha_1,\alpha_2,\cdots,\alpha_n)$。

\end{proof}

\begin{definition}
设$\sigma$是环$R_1$到环$R_2$上的同构，又$K_i$是$R_i$的扩张，$i=1,2$。若$\eta$是$K_1$到$K_2$上的同构且满足$\eta|_{R_1}=\sigma$，则称$\eta$是$\sigma$的\textbf{开拓}。
特别地，若$K_1,K_2$都是域$F$的扩域，$\eta$是$K_1$到$K_2$上的$F$同构，则$\eta$是$\mathrm{id}_F$的开拓。
\end{definition}

\begin{theorem}
设$\sigma$是$F$到$\overline{F}$上的域同构，则
\begin{enumerate}[(1)]
\item $\sigma$可开拓为$F[x]$到$\overline{F}[x]$上的同构，仍记为$\sigma$，$p(x)\in F[x]$不可约当且仅当$\sigma p(x)\in\overline{F}[x]$不可约；
\item 又若$K,\overline{K}$分别为$F,\overline{F}$的扩域且$p(x)\in F[x]$不可约，而$\alpha\in K,\overline{\alpha}\in\overline{K}$分别为$p(x),\sigma p(x)$的根，则$\sigma$可开拓为$F(\alpha)$到$\overline{F}(\overline{\alpha})$上的同构$\overline{\sigma}$，使得$\overline{\sigma}(\alpha)=\overline{\alpha}$。
\end{enumerate}
\end{theorem}

\begin{proof}

(1) 设$f(x)\in F[x]$且$f(x)=\sum\limits_{i=0}^{n}a_ix^i$，定义
\begin{align*}
\sigma(f(x))=\sum\limits_{i=0}^{n}\sigma(a_i)x^i,
\end{align*}
该映射为$F[x]$到$\overline{F}[x]$的同构，故$p(x)\in F[x]$不可约当且仅当$\sigma p(x)\in\overline{F}[x]$不可约。

(2) 因$p(x)\in F[x]$不可约，故$\sigma p(x)\in\overline{F}[x]$不可约。设$\pi$是$F[x]$到$F[x]/\langle p(x)\rangle$的自然映射，$\overline{\pi}$是$\overline{F}[x]$到$\overline{F}[x]/\langle\sigma p(x)\rangle$的自然映射，则$\overline{\pi}\sigma$是$F[x]$到$\overline{F}[x]/\langle\sigma p(x)\rangle$的同态，且$\ker(\overline{\pi}\sigma)=\langle p(x)\rangle$。
于是存在$F[x]/\langle p(x)\rangle$到$\overline{F}[x]/\langle\sigma p(x)\rangle$的同构$\overline{\sigma}'$，使得$\overline{\sigma}'\pi=\overline{\pi}\sigma$。

\caption{}
\label{figure:iso-ext1}
\end{figure}
存在$F[x]/\langle p(x)\rangle$到$F(\alpha)$的同构$\eta$，使$\eta(a)=a(\forall a\in F),\eta(\pi(x))=\alpha$；同时存在$\overline{F}[x]/\langle\sigma p(x)\rangle$到$\overline{F}(\overline{\alpha})$的同构$\overline{\eta}$，使得$\overline{\eta}(\overline{a})=\overline{a}\ (\forall\overline{a}\in\overline{F}),\overline{\eta}(\overline{\pi}(x))=\overline{\alpha}$。
令$\overline{\sigma}=\overline{\eta}\overline{\sigma}'\eta^{-1}$，则$\overline{\sigma}$是$F(\alpha)$到$\overline{F}(\overline{\alpha})$的同构，且$\forall a\in F$，有$\overline{\sigma}(a)=\sigma(a)$，$\overline{\sigma}(\alpha)=\overline{\alpha}$。

\end{figure}

\end{proof}

\begin{theorem}
设$\sigma$是域$F$到域$\overline{F}$上的同构，$\sigma$开拓为$F[x]$到$\overline{F}[x]$上的同构仍记为$\sigma$。又设$E,\overline{E}$分别为$f(x)\in F[x]$和$\sigma f(x)\in\overline{F}[x]$的分裂域，则$\sigma$可开拓为$E$到$\overline{E}$的同构。
$\sigma$的不同开拓的个数$n_\sigma\leqslant[\overline{E}:\overline{F}]$，而且当且仅当$\sigma f(x)$中每一个不可约因子在$\overline{E}$中不同根的个数恰为此因子的次数时等号成立。
\end{theorem}

\begin{proof}

对$\deg f(x)$用数学归纳法。当$\deg f(x)=1$时有$\deg(\sigma f(x))=1$，此时$E=F,\overline{E}=\overline{F}$，定理成立。

设$\deg f(x)=n-1$时定理成立。当$\deg f(x)=n$时，取$f(x)$的不可约因子$p(x)$，因$E$是$f(x)$的分裂域，故存在$\alpha_1\in E$，使$p(\alpha_1)=0$；由上一定理，$\sigma p(x)$是$\sigma f(x)$的不可约因子，又$\overline{E}$是$\sigma f(x)$的分裂域，故存在$\overline{\alpha_1}\in\overline{E}$，使$\sigma p(\overline{\alpha_1})=0$。
$\sigma$可开拓为$F(\alpha_1)$到$\overline{F}(\overline{\alpha_1})$的同构$\sigma_1$且$\sigma_1(\alpha_1)=\overline{\alpha_1}$。此时$f(x)=(x-\alpha_1)f_1(x)$，$\deg f_1(x)=n-1$，故$\sigma_1 f(x)=(x-\overline{\alpha_1})\sigma_1 f_1(x)$。
由归纳假设，$\sigma_1$可开拓为$E$到$\overline{E}$的同构，即$\sigma$可开拓为$E$到$\overline{E}$的同构。

设$\overline{\sigma}$是$\sigma$的开拓，则$\overline{\sigma}(\alpha_1)$是$\sigma p(x)$的根，$\sigma$在$F(\alpha_1)$上开拓的个数$k_1\leqslant\deg p(x)=[F(\alpha_1):F]$，等号成立当且仅当$\sigma p(x)$在$\overline{E}$中无重根。对固定的$\sigma_1$，其开拓为$E$到$\overline{E}$的同构个数$k_2\leqslant[E:F(\alpha_1)]=[\overline{E}:\overline{F}(\overline{\alpha_1})]$，故
\begin{align*}
n_\sigma\leqslant k_1k_2\leqslant[F(\alpha_1):F][E:F(\alpha_1)]=[E:F].
\end{align*}
当$\sigma f(x)$的每个不可约因子在$\overline{E}$中无重根时，上述不等式取等号。

\end{proof}

\begin{corollary}
设$F$为域，$f(x)\in F[x],\deg f(x)>0$，则$f(x)$的任意两个分裂域$E,\overline{E}$是$F$同构的。特别地，当$E$与$\overline{E}$都是$F$的同一扩域$K$的子域时，即$F\subseteq E\subseteq K$，$F\subseteq\overline{E}\subseteq K$，则$E=\overline{E}$。
\end{corollary}

\begin{proof}

在上述定理中取$F=\overline{F}$与$\sigma=\mathrm{id}_F$即得前一结论。
若$E,\overline{E}\subseteq K$，设$f(x)=(x-\alpha_1)(x-\alpha_2)\cdots(x-\alpha_n)=(x-\beta_1)(x-\beta_2)\cdots(x-\beta_n)$，由因式分解唯一性，存在$\tau\in S_n$使得$\alpha_i=\beta_{\tau(i)}$，故
\begin{align*}
E=F(\alpha_1,\alpha_2,\cdots,\alpha_n)=F(\beta_{\tau(1)},\beta_{\tau(2)},\cdots,\beta_{\tau(n)})=\overline{E}.
\end{align*}

\end{proof}

\begin{corollary}
设$f(x)\in F[x]$的分裂域是$E$，则$E$的不同的$F$自同构的个数不超过$[E:F]$；当且仅当$f(x)$的不可约因子在$E$内无重根时，此个数恰为$[E:F]$。
\end{corollary}

\begin{proof}

在上述定理中取$F=\overline{F}$与$E=\overline{E},\sigma=\mathrm{id}_F$即得。

\end{proof}

\begin{corollary}
设$K$为域$F$的扩域，$E$是$f(x)\in F[x]$的分裂域且$E\subseteq K$，则对$K$的任意$F$自同构$\sigma$有$\sigma E=E$。
\end{corollary}

\begin{proof}

将$\sigma$在$K[x]$上的开拓仍记为$\sigma$，$f(x)$在$K[x]$中有分解$f(x)=(x-\alpha_1)(x-\alpha_2)\cdots(x-\alpha_n)$，$\alpha_i\in E\subseteq K$。
因$\sigma|_F=\mathrm{id}_F$，故
\begin{align*}
f(x)=\sigma f(x)=(x-\sigma(\alpha_1))(x-\sigma(\alpha_2))\cdots(x-\sigma(\alpha_n)),
\end{align*}
于是存在$\tau\in S_n$使得$\sigma(\alpha_i)=\alpha_{\tau(i)}$，因此
\begin{align*}
\sigma E=\sigma(F(\alpha_1,\alpha_2,\cdots,\alpha_n))=F(\alpha_{\tau(1)},\alpha_{\tau(2)},\cdots,\alpha_{\tau(n)})=E.
\end{align*}

\end{proof}

\begin{example}
设$F$为一个域，求$x^2+ax+b\ (a,b\in F)$的分裂域$E$。
\end{example}

\begin{solution}

若$x^2+ax+b$在$F[x]$中可约，则$E=F$。
若$x^2+ax+b$不可约，则其在$F[x]/\langle x^2+ax+b\rangle$中有根$\alpha_1=x+\langle x^2+ax+b\rangle$，且
\begin{align*}
x^2+ax+b=(x-\alpha_1)(x-\alpha_2),
\end{align*}
故$E=F(\alpha_1,\alpha_2)=F(\alpha_1)$，此时$[E:F]=2$。

\end{solution}

\begin{example}
求$(x^2-2)(x^2-3)\in\mathbb{Q}[x]$的分裂域。
\end{example}

\begin{solution}

$x^2-2,x^2-3$在$\mathbb{Q}[x]$中不可约，$\mathbb{Q}(\sqrt{2})$是$x^2-2$的分裂域，$\mathbb{Q}(\sqrt{3})$是$x^2-3$的分裂域，故$\mathbb{Q}(\sqrt{2},\sqrt{3})$是$(x^2-2)(x^2-3)$的分裂域。

\end{solution}

\begin{remark}
$\mathbb{Q}(\sqrt{2},\sqrt{3})\neq\mathbb{Q}(\sqrt{2})$，即$\sqrt{3}\notin\mathbb{Q}(\sqrt{2})$。
反证：若$\sqrt{3}\in\mathbb{Q}(\sqrt{2})$，则存在$a,b\in\mathbb{Q}$使得$0=(a+b\sqrt{2})^2-3=a^2+2b^2-3-2ab\sqrt{2}$，故$a^2+2b^2-3=2ab=0$。
若$a=0$，则$2b^2-3=0$，与$x^2-\frac{3}{2}$在$\mathbb{Q}[x]$不可约矛盾；若$b=0$，则$a^2-3=0$，与$x^2-3$在$\mathbb{Q}[x]$不可约矛盾。
因此$[\mathbb{Q}(\sqrt{2},\sqrt{3}):\mathbb{Q}]=4$。
\end{remark}

\begin{example}
求$x^p-1\in\mathbb{Q}[x]$的分裂域，其中$p$是素数。
\end{example}

\begin{solution}

因$x^p-1=(x-1)(x^{p-1}+x^{p-2}+\cdots+x+1)$，$p$为素数，故$x^{p-1}+x^{p-2}+\cdots+x+1$在$\mathbb{Q}[x]$中不可约。
设$\alpha$是它的一个根，则$\mathbb{Q}(\alpha)=\mathbb{Q}(\alpha,\alpha^2,\cdots,\alpha^{p-1})$，且$(\alpha^k)^p=1\ (1\leqslant k\leqslant p-1)$，故$\alpha,\alpha^2,\cdots,\alpha^{p-1}$都是$x^{p-1}+x^{p-2}+\cdots+x+1$的根。
于是$\mathbb{Q}(\alpha)$是$x^p-1$的分裂域，且$[\mathbb{Q}(\alpha):\mathbb{Q}]=p-1$。

\end{solution}

\begin{example}
求$x^3-2\in\mathbb{Q}[x]$的分裂域$E$。
\end{example}

\begin{solution}

$x^3-2$在$\mathbb{Q}[x]$中不可约，设$\alpha_1=\sqrt[3]{2}$是其一个根，则$[\mathbb{Q}(\sqrt[3]{2}):\mathbb{Q}]=3$，且
\begin{align*}
x^3-2=(x-\alpha_1)(x^2+\alpha_1x+\alpha_1^2)
\end{align*}
在$\mathbb{Q}(\sqrt[3]{2})[x]$中成立。
设$\alpha_2$是$x^2+\alpha_1x+\alpha_1^2$的一个根，令$w=\frac{\alpha_2}{\alpha_1}$，则
\begin{align*}
w^2+w+1=\frac{\alpha_2^2}{\alpha_1^2}+\frac{\alpha_2}{\alpha_1}+1
=\frac{1}{\alpha_1^2}(\alpha_2^2+\alpha_1\alpha_2+\alpha_1^2)=0,
\end{align*}
即$w$是$x^2+x+1$的根。$x^2+x+1$在$\mathbb{Q}[x]$中不可约，故$[\mathbb{Q}(w):\mathbb{Q}]=2$，因此$[E:\mathbb{Q}]=6$且$E=\mathbb{Q}(\alpha_1,\alpha_2)=\mathbb{Q}(\sqrt[3]{2},w)$。

\end{solution}

\begin{example}
求$x^3+x+1\in\mathbb{Z}_2[x]$的分裂域$E$。
\end{example}

\begin{solution}

$x^3+x+1$在$\mathbb{Z}_2$中无根，故不可约。设$\alpha$是它的一个根，则$[\mathbb{Z}_2(\alpha):\mathbb{Z}_2]=3$。
$\mathbb{Z}_2$的特征为$2$，故$\forall\beta\in E,\beta=-\beta$。又$\alpha^2\neq\alpha$，且$(\alpha^2)^3+\alpha^2+1=(\alpha^3+\alpha+1)^2=0$，故在$\mathbb{Z}_2(\alpha)[x]$中有
\begin{align*}
x^3+x+1=(x+\alpha)(x+\alpha^2)(x+\alpha+\alpha^2),
\end{align*}
因而$E=\mathbb{Z}_2(\alpha)$。

\end{solution}

\begin{definition}
域$F$的一个代数扩张$K$称为$F$的\textbf{正规扩张}，若$F[x]$中任意不可约多项式$p(x)$在$K$中有一个根，则$p(x)$在$K[x]$中可分解为一次因式之积，即$p(x)$的所有根都在$K$中。
\end{definition}

\begin{theorem}
域$F$的有限扩张$K$为$F$的正规扩张当且仅当$K$是$F[x]$中一个多项式的分裂域。
\end{theorem}

\begin{proof}

必要性：因$K$是$F$的有限扩张，故存在$\alpha_1,\alpha_2,\cdots,\alpha_r\in K$使得$K=F(\alpha_1,\alpha_2,\cdots,\alpha_r)$。令$f(x)=\mathrm{Irr}(\alpha_1,F)\mathrm{Irr}(\alpha_2,F)\cdots\mathrm{Irr}(\alpha_r,F)$。
由$K$是正规扩张，$\mathrm{Irr}(\alpha_i,F)$在$K$中有根$\alpha_i$，故$f(x)$在$K[x]$中可分解为一次因式之积。令$E=F(\beta_1,\beta_2,\cdots,\beta_n)$为$f(x)$的分裂域，则$E=K$，即$K$是$f(x)$的分裂域。

充分性：设$K$是$f(x)\in F[x]$的分裂域，则$[K:F]<+\infty$，$K$是代数扩张。取$F[x]$中不可约多项式$p(x)$，若存在$\alpha\in K$使得$p(\alpha)=0$，令$E$为$p(x)$在$K$上的分裂域，则$E\supseteq K\supseteq F$且$K(\alpha)=K$。
令$g(x)=p(x)f(x)$，则$E$是$g(x)\in F[x]$的分裂域。任取$p(x)$的根$\beta\in E$，存在$F(\alpha)$到$F(\beta)$的$F$同构$\tau$使得$\tau(\alpha)=\beta$。由推论，$\tau$可开拓为$E$的$F$自同构，仍记为$\tau$；由推论，$\tau(K)=K$。
因$\alpha\in K$，故$\beta=\tau(\alpha)\in K$，即$p(x)$的根均在$K$中，故$K$为$F$的正规扩张。

\end{proof}
