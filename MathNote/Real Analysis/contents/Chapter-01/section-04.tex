\section{$\mathbb{R}^n$中的基本点集:闭集$\cdot$开集$\cdot$Borel集$\cdot$Cantor集}

\subsection{闭集和开集}

\begin{lemma}
$\mathbb{R}^n$中的子集族
\[
\mathscr{T}=\{G\mid \forall x\in G,\exists \delta=\delta(x)>0,\text{s.t. }B(x;\delta)\subseteq G\}
\]
具有3条性质：
\begin{enumerate}[(1)]
\item $\varnothing,\mathbb{R}^n\in\mathscr{T}$;
\item 若$G_1,G_2\in\mathscr{T}$,则$G_1\cap G_2\in\mathscr{T}$;
\item 若$G_\alpha\in\mathscr{T},\alpha\in\Gamma$(指标集),则$\bigcup_{\alpha\in\Gamma}G_\alpha\in\mathscr{T}$,或者, $\forall\mathscr{G}'\subseteq\mathscr{T}$,必有$\bigcup_{G\in\mathscr{G}'}G\in\mathscr{T}$.
\end{enumerate}
由此可知$\mathscr{T}$为$\mathbb{R}^n$上的一个拓扑,$(\mathbb{R}^n,\mathscr{T})$成为$\mathbb{R}^n$上的一个拓扑空间.我们将$\mathscr{T}$称为$\mathbb{R}^n$上的\textbf{通常拓扑}.自然,$G\in\mathscr{T}$为该拓扑空间$(\mathbb{R}^n,\mathscr{T})$上的一个\textbf{开集},有时也简称为$\mathbb{R}^n$上的开集.而开集$G$的余(补)集$G^c$称为该拓扑空间的一个\textbf{闭集},有时也简称为$\mathbb{R}^n$上的闭集.
记$E=E\cup E'$,则称\(\overline{E}\)为\(E\)的\textbf{闭包}.

若集合$A,B$满足\(A\subseteq B\)且\(\overline{A}=B\)，则称\(A\)在\(B\)中\textbf{稠密}，或称\(A\)是\(B\)的\textbf{稠密子集}.
\end{lemma}
\begin{remark}
易见,\(\mathbb{R}^n\)本身与空集\(\varnothing\)是开集；
闭集的补集是开集；\(\mathbb{R}^n\)中的开矩体是开集；开球体$B(x;\delta)$为开集,闭球体$\overline{B(x;\delta)}$和球面$S(x;\delta)$都为闭集.
\end{remark}
\begin{proof}
\begin{enumerate}[(1)]
\item 因为$\varnothing$不含点,它自然满足$\mathscr{T}$中的条件,故$\varnothing\in\mathscr{T}$.对$\forall x\in\mathbb{R}^n$,必有$B(x;1)\subseteq\mathbb{R}^n$,所以$\mathbb{R}^n\in\mathscr{T}$.

\item \one 由(1),$G_1\cap G_2=\varnothing\in\mathscr{T}$.
\two $G_1\cap G_2\neq\varnothing$,则对$\forall x\in G_1\cap G_2$,必有$x\in G_1,x\in G_2$.于是,$\exists\delta_i>0$,s.t. $B(x;\delta_i)\subseteq G_i$, $i=1,2$.记$\delta=\min\{\delta_1,\delta_2\}>0$,则$B(x;\delta)\subseteq B(x;\delta_1)\cap B(x;\delta_2)\subseteq G_1\cap G_2$.从而,$G_1\cap G_2\in\mathscr{T}$.

\item 对$\forall x\in\bigcup_{\alpha\in\Gamma}G_\alpha$,必有$\alpha_0\in\Gamma$,s.t. $x\in G_{\alpha_0}\in\mathscr{T}$,所以,$\exists\delta>0$,s.t. $B(x;\delta)\subseteq G_{\alpha_0}\subseteq\bigcup_{\alpha\in\Gamma}G_\alpha$.这就证明了$\bigcup_{\alpha\in\Gamma}G_\alpha\in\mathscr{T}$.
\end{enumerate}


\end{proof}

\begin{theorem}[闭集的基本性质]\label{theorem:引理1.4.1-实变函数}
设
\[
\mathscr{F}=\{F\subseteq\mathbb{R}^n\mid F^c\in\mathscr{T}\}
\]
为所有闭集形成的$(\mathbb{R}^n,\mathscr{T})$的闭集族,它也具有以下性质:
\begin{enumerate}[(1)]
\item\label{theorem:引理1.4.1-实变函数-1} $\varnothing,\mathbb{R}^n\in\mathscr{F}$;

\item\label{theorem:引理1.4.1-实变函数-2} 若$F_1,F_2\in\mathscr{F}$,则$F_1\cup F_2\in\mathscr{F}$,进而有限多个闭集的并集仍是闭集;

\item\label{theorem:引理1.4.1-实变函数-3} 若$F_\alpha\in\mathscr{F},\alpha\in\Gamma$,则$\bigcap_{\alpha\in\Gamma}F_\alpha\in\mathscr{F}$,或者, $\forall\mathscr{F}'\subseteq\mathscr{F}$,必有$\bigcap_{F\in\mathscr{F}'}F\in\mathscr{F}$.
\end{enumerate}
\end{theorem}
\begin{remark}
无穷多个闭集的并集不一定是闭集。例如，令
\[F_k=\left[\frac{1}{k + 1},\frac{1}{k}\right]\subseteq\mathbb{R}\quad(k = 1,2,\cdots),\]
则有\(\bigcup_{k = 1}^{\infty}F_k=(0,1]\)。此例还说明
\[\left[ 0,1 \right] =\overline{\bigcup_{k=1}^{\infty}{F_k}}\ne \bigcup_{k=1}^{\infty}{\overline{F_k}}=\left( 0,1 \right] .\]
\end{remark}
\begin{proof}
\begin{enumerate}[(1)]
\item 显然,由$\varnothing^c=\mathbb{R}^n\in\mathscr{T}$, $(\mathbb{R}^n)^c=\varnothing\in\mathscr{T}$立知$\varnothing,\mathbb{R}^n\in\mathscr{F}$.

\item 根据$F_i^c\in\mathscr{T}$, $i=1,2$,De Morgan公式以及\rreflem{theorem:引理1.4.1-实变函数}{theorem:引理1.4.1-实变函数-2},有
\[
(F_1\cup F_2)^c=F_1^c\cap F_2^c\in\mathscr{T}.
\]
因此,$F_1\cup F_2\in\mathscr{F}$.

\item 根据$F_\alpha^c\in\mathscr{T}$, $\alpha\in\Gamma$,De Morgan公式以及\rreflem{theorem:引理1.4.1-实变函数}{theorem:引理1.4.1-实变函数-3},有
\[
\left(\bigcap_{\alpha\in\Gamma}F_\alpha\right)^c=\bigcup_{\alpha\in\Gamma}F_\alpha^c\in\mathscr{T}.
\]
因此,$\bigcap_{\alpha\in\Gamma}F_\alpha\in\mathscr{F}$.
由归纳法立知,若$F_1,F_2,\cdots,F_m\in\mathscr{F}$,则$\bigcup_{k=1}^m F_k\in\mathscr{F}$.
\end{enumerate}

\end{proof}

\begin{theorem}\label{theorem:Rn中闭集的充要条件}
设$E\subseteq\mathbb{R}^n$,则以下结论等价:
\begin{enumerate}[(1)]
\item $E$为闭集

\item\label{theorem:Rn中闭集的充要条件-2} $E'\subseteq E$

\item $\overline{E}=E$

\item\label{theorem:Rn中闭集的充要条件-4} 对$\forall\,x^k\in E,x^k\to x(k\to+\infty)$,必有$x\in E$.
\end{enumerate}
\end{theorem}
\begin{proof}
$(1)\Rightarrow(2)$因为$E$为闭集,所以$E^c$为开集.于是,对$\forall\,x\in E^c$,由于$E^c$为$x$的开邻域,且$(E^c\setminus\{x\})\cap E=\varnothing$,所以$x\notin E'$.从而,$E'\subseteq E$.

$(1)\Leftarrow(2)$设$E'\subseteq E$.对$\forall\,x\in E^c$,则$x\notin E'$,即存在$x$的开邻域$G$,使$(G\setminus\{x\})\cap E=\varnothing$.由于$x\in E^c$,故$x\notin E$.从而,$G\cap E=\varnothing,x\in G\subseteq E^c$.这就证明了$E^c$为开集,而$E$为闭集.

$(2)\Rightarrow(3)$由$E\subseteq\overline{E}=E\cup E'\subseteq E\cup E=E$知$\overline{E}=E$.

$(2)\Leftarrow(3)$由$E\cup E'=\overline{E}=E$立知$E'\subseteq E$.

$(1)\Rightarrow(4)$因为$E$为闭集,故$E^c$为开集.(反证)假设$x\notin E$,即$x\in E^c$.$E^c$为$x$的开邻域.又因$x^k\in E$,故$x^k\notin E^c(\forall\,k\in\mathbb{N})$.再由题设$x^k\to x(k\to+\infty)$,则$\exists\,K\in\mathbb{N}$,当$k>K$时,$x^k\in E^c$,矛盾.这就证明了$x\in E$.

$(1)\Leftarrow(4)$(反证)假设$E$不为闭集,从而$E^c$不为开集.于是,$\exists\,x\in E^c$,对$\forall\,k\in\mathbb{N}$,$B\left(x;\,\frac{1}{k}\right)$中必有$x^k\in E$.显然,$d(x^k,x)=\|x^k-x\|<\frac{1}{k}\to0(k\to+\infty)$,即$x^k\to x(k\to+\infty)$.由(4),$x\in E$,这与$x\in E^c$(从而$x\notin E$)相矛盾.

\end{proof}

\begin{theorem}[开集的基本性质]\label{theorem:开集的运算性质}
\begin{enumerate}[(1)]
\item 若\(\{G_{\alpha}:\alpha\in I\}\)是\(\mathbb{R}^n\)中的一个开集族，则其并集\(G = \bigcup_{\alpha\in I}G_{\alpha}\)是开集；

\item 若\(G_k(k = 1,2,\cdots,m)\)是\(\mathbb{R}^n\)中的开集，则其交集\(G=\bigcap_{k = 1}^{m}G_k\)是开集（有限个开集的交集是开集）；

\item 若\(G\)是\(\mathbb{R}^n\)中的非空点集，则\(G\)是开集的充分必要条件是，对于\(G\)中任一点\(x\)，存在\(\delta>0\)，使得\(B(x,\delta)\subseteq G\)。
\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item 由定义知\(G_{\alpha}^c(\alpha\in I)\)是闭集，且有\(G^c=\bigcap_{\alpha\in I}G_{\alpha}^c\)。根据闭集的性质可知\(G^c\)是闭集，即\(G\)是开集。

\item 由定义知\(G_k^c(k = 1,2,\cdots,m)\)是闭集，且有\(G^c=\bigcup_{k = 1}^{m}G_k^c\)。根据闭集的性质可知\(G^c\)是闭集，即\(G\)是开集。

\item 若\(G\)是开集且\(x\in G\)，则由于\(G^c\)是闭集以及\(x\notin G^c\)，可知存在\(\delta>0\)，使得\(B(x,\delta)\subseteq G\)。

反之，若对\(G\)中的任一点\(x\)，存在\(\delta>0\)，使得\(B(x,\delta)\subseteq G\)，则
\[B(x,\delta)\cap G^c=\varnothing,\]
从而\(x\)不是\(G^c\)的极限点，即\(G^c\)的极限点含于\(G^c\)。这说明\(G^c\)是闭集，即\(G\)是开集。
\end{enumerate}

\end{proof}

\begin{proposition}
\begin{enumerate}[(1)]
\item 设$E_i \subseteq \mathbb{R}^n, i=1,2,\cdots,k$. 证明:
\begin{align*}
\bigcup_{i=1}^k E_i' = \left( \bigcup_{i=1}^k E_i \right)'; \quad \bigcup_{i=1}^k \overline{E_i} = \overline{\bigcup_{i=1}^k E_i}.
\end{align*}

\item 设$E_i \subseteq \mathbb{R}^n, i=1,2,\cdots$. 证明:
\begin{align*}
\bigcup_{i=1}^\infty E_i' \subseteq \left( \bigcup_{i=1}^\infty E_i \right)'; \quad \bigcup_{i=1}^\infty \overline{E_i} \subseteq \overline{\bigcup_{i=1}^\infty E_i}.
\end{align*}
但是下面两式不成立,并给出反例.
\begin{align*}
\bigcup_{i=1}^\infty E_i' = \left( \bigcup_{i=1}^\infty E_i \right)'; \quad \bigcup_{i=1}^\infty \overline{E_i} = \overline{\bigcup_{i=1}^\infty E_i}.
\end{align*}
\end{enumerate}
\end{proposition}
\begin{proof}
\begin{enumerate}[(1)]
\item 因为$E_j \subseteq \bigcup_{i=1}^k E_i, j=1,2,\cdots,k$, 故$E_j' \subseteq \left( \bigcup_{i=1}^k E_i \right)', j=1,2,\cdots,k$. 于是
\begin{align*}
\bigcup_{i=1}^k E_i' = \bigcup_{j=1}^k E_j' \subseteq \left( \bigcup_{i=1}^k E_i \right)'.
\end{align*}
另一方面, 如果$x \notin \bigcup_{i=1}^k E_i'$, 则$x \notin E_i', i=1,2,\cdots,k$, 故由\rrefthe{theorem:聚点的充要条件--实变函数}{theorem:聚点的充要条件--实变函数-2}知,存在$x$的开邻域$U_{E_i}, i=1,2,\cdots,k$, s.t.
\begin{align*}
U_{E_i} \cap (E_i \setminus \{x\}) = \varnothing, \quad i=1,2,
\end{align*}
显然, $\bigcap_{i=1}^k U_{E_i}$为$x$的开邻域, 且
\begin{align*}
\left( \bigcap_{i=1}^k U_{E_i} \right) \cap \left( \bigcup_{i=1}^k E_i \setminus \{x\} \right) = \varnothing,
\end{align*}
所以, $x \notin \left( \bigcup_{i=1}^k E_i \right)'.$ 这就表明了
\begin{align*}
\bigcup_{i=1}^k E_i' \supseteq \left( \bigcup_{i=1}^k E_i \right)'
\end{align*}
综合上述得到
\begin{align*}
\bigcup_{i=1}^k E_i' = \left( \bigcup_{i=1}^k E_i \right)'.
\end{align*}
由此推得
\begin{align*}
\bigcup_{i=1}^k \overline{E_i} = \bigcup_{i=1}^k (E_i \cup E_i') = \left( \bigcup_{i=1}^k E_i \right) \cup \left( \bigcup_{i=1}^k E_i' \right) \stackrel{上式}{=} \left( \bigcup_{i=1}^k E_i \right) \cup \left( \bigcup_{i=1}^k E_i \right)' = \overline{\bigcup_{i=1}^k E_i}.
\end{align*}

\item 因为$E_j \subseteq \bigcup_{i=1}^\infty E_i, j=1,2,\cdots$, 故$E_j' \subseteq \left( \bigcup_{i=1}^\infty E_i \right)', j=1,2,\cdots$. 于是
\begin{align*}
\bigcup_{i=1}^\infty E_i' = \bigcup_{j=1}^\infty E_j' \subseteq \left( \bigcup_{i=1}^\infty E_i \right)';
\end{align*}

\begin{align*}
\bigcup_{i=1}^\infty \overline{E_i} &= \bigcup_{i=1}^\infty (E_i \cup E_i') = \left( \bigcup_{i=1}^\infty E_i \right) \cup \left( \bigcup_{i=1}^\infty E_i' \right) \\
&\stackrel{上式}{\subseteq} \left( \bigcup_{i=1}^\infty E_i \right) \cup \left( \bigcup_{i=1}^\infty E_i \right)' = \overline{\bigcup_{i=1}^\infty E_i}.
\end{align*}

但是, 一般来说, 下面两式并不成立:
\begin{align*}
\bigcup_{i=1}^\infty E_i' \supseteq \left( \bigcup_{i=1}^\infty E_i \right)'; \quad \bigcup_{i=1}^\infty \overline{E_i} \supseteq \overline{\bigcup_{i=1}^\infty E_i}.
\end{align*}
因而
\begin{align*}
\bigcup_{i=1}^\infty E_i' = \left( \bigcup_{i=1}^\infty E_i \right)'; \quad \bigcup_{i=1}^\infty \overline{E_i} = \overline{\bigcup_{i=1}^\infty E_i}.
\end{align*}
两式也并不成立.

反例1: 设可数集$\mathbb{Q}^n = \{P_i \mid i \in \mathbb{N}\}, E_i = \{P_i\}$为独点集, 则
\begin{align*}
\bigcup_{i=1}^\infty E_i' = \bigcup_{i=1}^\infty \varnothing = \varnothing \not\supseteq \mathbb{R}^n = (\mathbb{Q}^n)' = \left( \bigcup_{i=1}^\infty E_i \right)';
\end{align*}
\begin{align*}
\bigcup_{i=1}^\infty \overline{E_i} = \bigcup_{i=1}^\infty \{P_i\} = \mathbb{Q}^n \not\supseteq \mathbb{R}^n = \overline{\mathbb{Q}^n} = \overline{\bigcup_{i=1}^\infty E_i}.
\end{align*}

反例2: 设独点集$E_i = \left\{ \left( \frac{1}{i},0,\cdots,0 \right) \right\} \subseteq \mathbb{R}^n, i \in \mathbb{N}$. 则
\begin{align*}
\bigcup_{i=1}^{\infty}{E_i'} &=\bigcup_{i=1}^{\infty}{\varnothing}=\varnothing \nsupseteq \{0=(0,0,\cdots ,0)\mid 0\in \mathbb{R} ^n\}
\\
&=\left\{ \left( \frac{1}{i},0,\cdots ,0 \right) \mid i\in \mathbb{N} \right\}' =\left( \bigcup_{i=1}^{\infty}{E_i} \right)' ;
\end{align*}
\begin{align*}
\bigcup_{i=1}^\infty \overline{E_i} &= \bigcup_{i=1}^\infty \left\{ \left( \frac{1}{i},0,\cdots,0 \right) \right\} \not\supseteq \{ \mathbf{0} = (0,0,\cdots,0) \mid \mathbf{0} \in \mathbb{R}^n \} \cup \left( \bigcup_{i=1}^\infty \left\{ \left( \frac{1}{i},0,\cdots,0 \right) \right\} \right) \\
&= \overline{\left\{ \left( \frac{1}{i},0,\cdots,0 \right) \mid i \in \mathbb{N} \right\}} = \overline{\bigcup_{i=1}^\infty E_i}.
\end{align*}
\end{enumerate}

\end{proof}

\begin{theorem}[Cantor闭集套定理]\label{theorem:实变函数--Cantor闭集套定理}
若\(\{F_k\}\)是\(\mathbb{R}^n\)中的非空有界闭集列，且满足\(F_1\supseteq F_2\supseteq\cdots\supseteq F_k\supseteq\cdots\)，则
\begin{align*}
\bigcap_{k = 1}^{\infty}F_k\neq\varnothing.
\end{align*}
若$F_k$的直径$\mathrm{diam}F_k=\sup\{d(x',x'')=\|x'-x''\|\mid x',x''\in F_k\}\to0(k\to+\infty)$,则存在唯一的$ x^0\in\bigcap\limits_{k=1}^{\infty}F_k$.
\end{theorem}
\begin{remark}
Cantor闭集套定理中的“有界”条件不可缺.反例：$F_k=[k,+\infty)^n$,$\bigcap\limits_{k=1}^{\infty}F_k=\varnothing$.
\end{remark}
\begin{proof}
若$\{F_k\}$中有无穷个相同的集合,由于
\begin{align*}
F_1\supseteq F_2\supseteq\cdots\supseteq F_k\supseteq\cdots,
\end{align*}
故$\exists\,k_0\in\mathbb{N}$,s.t.当$k\geqslant k_0$时,有$F_k=F_{k_0}$.此时
\begin{align*}
\bigcap\limits_{k=1}^{\infty}F_k=F_{k_0}\neq\varnothing.
\end{align*}
现在不妨设对$\forall\,k\in\mathbb{N}$,$F_{k+1}$为$F_k$的真子集,则
\begin{align*}
F_k\setminus F_{k+1}\neq\varnothing,\forall\,k\in\mathbb{N}.
\end{align*}
我们选$x^k\in F_k\setminus F_{k+1},k\in\mathbb{N}$,则$\{x^k\}$为$\mathbb{R}^n$中的有界互异点列.根据\hyperref[theorem:R^n中的Bolzano - Weierstrass定理]{Bolzano-Weierstrass定理}可知,必$\exists\,x^{k_i}\to x\in\mathbb{R}^n(i\to+\infty)$.由于$\{F_k\}$为递降闭集列,则$x\in F_k,k=1,2,\cdots$,即$x\in\bigcap\limits_{k=1}^{\infty}F_k$.

再证惟一性.若$x^0,y^0\in\bigcap\limits_{k=1}^{\infty}F_k$,则$x^0,y^0\in F_k,\forall\,k\in\mathbb{N}$,且
\begin{align*}
0\leqslant\|x^0-y^0\|\leqslant\mathrm{diam}F_k\to0(k\to+\infty),
\end{align*}
所以,$0\leqslant\|x^0-y^0\|\leqslant0,\|x^0-y^0\|=0,x^0=y^0$.

\end{proof}

\begin{definition}
设\(E\subseteq\mathbb{R}^n\)，\(\Gamma\)是\(\mathbb{R}^n\)中的一个开集族。若对任意的\(x\in E\)，存在\(G\in\Gamma\)，使得\(x\in G\)，则称\(\Gamma\)为\(E\)的一个\textbf{开覆盖}。

设\(\Gamma\)是\(E\)的一个开覆盖。若\(\Gamma'\subseteq\Gamma\)仍是\(E\)的一个开覆盖，则称\(\Gamma'\)为\(\Gamma\)（关于\(E\)）的一个\textbf{子覆盖}.

如果\(E\)的任一开覆盖均包含有限子覆盖，我们就称\(E\)为\textbf{紧(致)集}.
\end{definition}

\begin{theorem}[Heine - Borel有限子覆盖定理]\label{theorem:Heine - Borel有限子覆盖定理}
$E\subseteq\mathbb{R}^n$为紧集$\iff E\subseteq\mathbb{R}^n$为有界闭集.
\end{theorem}
\begin{remark}
在上述定理中，有界的条件是不能缺的。例如，在\(\mathbb{R}^1\)中对自然数集作开覆盖\(\{(n-\frac{1}{2},n+\frac{1}{2})\}\)就不存在有限子覆盖。同样，闭集的条件也是不能缺的。例如，在\(\mathbb{R}\)中对点集\(\{1,\frac{1}{2},\cdots,\frac{1}{n},\cdots\}\)作开覆盖
\[\left\{\left(\frac{1}{n}-\frac{1}{2n},\frac{1}{n}+\frac{1}{2n}\right)\right\}\quad(n = 1,2,\cdots),\]
就不存在有限子覆盖。
\end{remark}
\begin{proof}
$(\Rightarrow)$显然,$\mu=\{B(\mathbf{0};\,k)\mid k\in\mathbb{N}\}$为紧集$E$的一个开覆盖,故必有有限子覆盖$\{B(\mathbf{0};k_1),B(\mathbf{0};k_2),\cdots,B(\mathbf{0};k_l)\}$.令$k_0=\max\{k_1,k_2,\cdots,k_l\}$,则$B(\mathbf{0};\,k_0)\supseteq E$,从而$E$有界.

$\forall\,\mathbf{x}^0\in E'$,若$\mathbf{x}^0\notin E$,则
\begin{align*}
\nu=\left\{B\left(\mathbf{x};\,\frac{d(\mathbf{x},\mathbf{x}^0)}{2}\right)\mid \mathbf{x}\in E\right\}
\end{align*}
为紧集$E$的一个开覆盖,故必有有限子覆盖$\left\{B\left(\mathbf{x}^i;\,\frac{d(\mathbf{x}^i,\mathbf{x}^0)}{2}\right)\mid i=1,2,\cdots,l\right\}$.于是,$\mathbf{x}^0\notin E'$,这显然与已知$\mathbf{x}^0\in E'$相矛盾.

$(\Leftarrow)$设$E\subseteq\mathbb{R}^n$为有界闭集.取闭正方体$I\supseteq E$.又设$\mu$为$E$的任一开覆盖.(反证)假设$\mu$无有限子覆盖,将$I$ $2^n$等分,必有一等分$I_1$(闭正方体)使得$\mu$关于$I_1\cap E$无有限子覆盖.再将$I_1$ $2^n$等分,必有一等分$I_2$(闭正方体)使得$\mu$关于$I_2\cap E$无有限子覆盖.依次类推得到一个有界递降闭集列$\{I_k\cap E\mid k\in\mathbb{N}\}$.根据\hyperref[theorem:实变函数--Cantor闭集套定理]{Cantor闭集套定理},$\exists\,\mathbf{x}^0\in\bigcap\limits_{k=1}^{\infty}(I_k\cap E)$.由$E$为闭集,显然,$\mathbf{x}^0\in\overline{E}=E$.因为$\mu$为$E$的开覆盖,故必有开集$G_0\in\mu$,使得$\mathbf{x}^0\in G_0$.于是,存在充分大的$k_0\in\mathbb{N}$,使得$\mathbf{x}^0\in I_{k_0}\subset G_0$.由此看出,一个$G_0\in\mu$就将$I_{k_0}$(从而$I_{k_0}\cap E$)覆盖住了,这与$I_{k_0}$的构造,$I_{k_0}\cap E$不被$\mu$中有限个元素覆盖相矛盾.这就证明了$\mu$有有限子覆盖,从而$E$为紧集.

\end{proof}

\begin{lemma}\label{lemma:实变函数--引理1.4.6}
\begin{enumerate}[(1)]
\item\label{lemma:实变函数--引理1.4.6-1} $(\mathbb{R}^n,\mathcal{T})$为$T_2$或Hausdorff空间(即$\forall\,\boldsymbol{p},\boldsymbol{q}\in\mathbb{R}^n,\boldsymbol{p}\neq\boldsymbol{q}$,必有$\boldsymbol{p}$的开邻域$U_{\boldsymbol{p}}$和$\boldsymbol{q}$的开邻域$U_{\boldsymbol{q}}$,使得$U_{\boldsymbol{p}}\cap U_{\boldsymbol{q}}=\varnothing$).

\item\label{lemma:实变函数--引理1.4.6-2} $(\mathbb{R}^n,\mathcal{T})$为$A_2$(具有可数拓扑基)空间,即存在至多可数个集组成的开集族$\mu$,使得对$\forall\,G\in\mathcal{T}$,有
\begin{align*}
G=\bigcup_{\substack{U\in\mu\\U\subset\mu}}U(G\text{为}\mu\text{中若干元素的并}).
\end{align*}
此时,称$\mu$为$(\mathbb{R}^n,\mathcal{T})$的\textbf{可数拓扑基}.
\end{enumerate}
\end{lemma}
\begin{proof}
\begin{enumerate}[(1)]
\item 令
\begin{align*}
U_{\boldsymbol{p}}=B\left(\boldsymbol{p};\,\frac{d(\boldsymbol{p},\boldsymbol{q})}{2}\right),\quad U_{\boldsymbol{q}}=B\left(\boldsymbol{q};\,\frac{d(\boldsymbol{p},\boldsymbol{q})}{2}\right),
\end{align*}
则$U_{\boldsymbol{p}}$与$U_{\boldsymbol{q}}$分别为$\boldsymbol{p}$与$\boldsymbol{q}$的不相交的开邻域,因此$(\mathbb{R}^n,\mathcal{T})$为$T_2$空间.

\item 显然,
\begin{align*}
\mu=\left\{B\left(\boldsymbol{x};\,\frac{1}{k}\right)\mid \boldsymbol{x}\in\mathbb{Q}^n\subset\mathbb{R}^n,\ k\in\mathbb{N}\right\}
\end{align*}
为可数族,且对$\forall\,G\in\mathcal{T},\forall\,\boldsymbol{x}\in G,\exists\,\delta>0$,s.t.$B(\boldsymbol{x};\,\delta)\subseteq G$.

现取有理点$\boldsymbol{x}'\in\mathbb{Q}^n$,s.t.$\|\boldsymbol{x}'-\boldsymbol{x}\|<\frac{1}{k}$,其中$k>\frac{2}{\delta}$(即$\frac{1}{k}<\frac{\delta}{2}$).从而,有
\begin{align*}
\boldsymbol{x}\in B\left(\boldsymbol{x}';\,\frac{1}{k}\right)\subseteq B(\boldsymbol{x};\,\delta)\subseteq G.
\end{align*}
易见
\begin{align*}
\bigcup_{\boldsymbol{x}\in G}B\left(\boldsymbol{x}';\,\frac{1}{k}\right)=G.
\end{align*}
因而,$\mu$为$(\mathbb{R}^n,\mathcal{T})$的可数拓扑基,从而$(\mathbb{R}^n,\mathcal{T})$为$A_2$空间.
\end{enumerate}

\end{proof}

\begin{theorem}[Lindelöf定理]\label{theorem:Lindelöf定理}
$\mathbb{R}^n$中的点集$E$的任何开覆盖$\nu$都含有一个至多可数的子覆盖.
\end{theorem}
\begin{proof}
根据\reflem{lemma:实变函数--引理1.4.6},必有$(\mathbb{R}^n,\mathcal{T})$的可数拓扑基$\mu$.显然
\begin{align*}
\mu'=\{B\mid B\in\mu,\exists\,G\in\nu,\text{s.t.}\ B\subset G\}
\end{align*}
为$\mu$的一个子族.

对于$\forall\,B'\in\mu'$,取定$\nu$的一个元素$G(B')\supset B'$.并令
\begin{align*}
\nu'=\{G(B')\mid B'\in\mu'\}\subseteq\nu.
\end{align*}
因为$\mu$至多可数,从而$\mu'$和$\nu'$也都至多可数.

还需证明的是$\nu'$为$\nu$关于$E$的一个子覆盖.因而,$\nu$有一个至多可数的子覆盖$\nu'$.事实上,对$\forall\,\boldsymbol{x}\in E,\exists\,G\in\nu$,s.t.$\boldsymbol{x}\in G$.因为$\mu$为$(\mathbb{R}^n,\mathcal{T})$的一个可数拓扑基,$\exists\,B\in\mu$,s.t.$\boldsymbol{x}\in B\subseteq G$.从$\mu'$的定义知,$B\in\mu'$.于是,$\boldsymbol{x}\in B\subseteq G(B)\in\nu'$.这就证明了$\nu'$为$\nu$关于$E$的一个子覆盖.

\end{proof}

\begin{definition}[内点与边界点]\label{definition:内点与边界点}
设\(E\subseteq\mathbb{R}^n\)。对\(x\in E\)，若存在\(\delta>0\)，使得\(B(x,\delta)\subseteq E\)，则称\(x\)为\(E\)的\textbf{内点}。\(E\)的内点全体记为\(\mathring{E}\)或$E^{\circ}$，称为\(E\)的\textbf{内核}或\textbf{内点集}。$E^c$的内点称为$E$的\textbf{外点},而$(E^c)^{\circ}$称为$E$的\textbf{外点集}.如果在$x\in \mathbb{R}^n$的任何开邻域中既含$E$的点,又含$E^c$的点,则称\(x\)为\(E\)的\textbf{边界点}。$E$的边界点全体记为\(\partial E\)或$E^{\mathrm{b}}$。
\end{definition}
\begin{note}
显然有
\begin{align*}
\partial E=\partial E^c,\quad \mathbb{R} ^n=E^{\circ}\cup \left( E^c \right) ^{\circ}\cup \partial E.
\end{align*}
其中$E^{\circ},\left( E^c \right) ^{\circ},\partial E$为$\mathbb{R}^n$的三个互不相同的部分.
\end{note}

\begin{theorem}
设$E \subseteq \mathbb{R}^n$,则
\begin{enumerate}[(1)]
\item $E$为开集$\iff E=E^\circ$.

\item  $E^\circ$与$(E^c)^\circ$都为开集, $E$的边界$\partial E = E^{\mathrm{b}}$都为闭集.
\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item $(\Rightarrow)$设$E$为开集,则对$\forall\,\boldsymbol{x}\in E$,取$\boldsymbol{x}$的开邻域$G_{\boldsymbol{x}}=E$,则$G_{\boldsymbol{x}}=E\subseteq E$,故$\boldsymbol{x}$为$E$的内点,从而$E=E^\circ$.

$(\Leftarrow)$事实上,$\forall\,\boldsymbol{x}\in E^\circ$,则$\exists\,\boldsymbol{x}$的开邻域$G_{\boldsymbol{x}}\subseteq E$,则$\boldsymbol{x}\in G_{\boldsymbol{x}}\subseteq E^\circ$.从而
\begin{align*}
E^\circ=\bigcup_{\boldsymbol{x}\in E^\circ}\{\boldsymbol{x}\}\subseteq\bigcup_{\boldsymbol{x}\in E^\circ}G_{\boldsymbol{x}}\subseteq E^\circ,
\end{align*}
于是
\begin{align*}
E^\circ=\bigcup_{\boldsymbol{x}\in E^\circ}G_{\boldsymbol{x}},
\end{align*}
故$E=E^\circ$为开集.

\item {\color{blue}证法一:}
对$\forall\ x \in E^\circ$, 则存在$x$的开邻域$B \subseteq E$. 由于$\forall\ y \in B$, $B$为$y$的开邻域, $B \subseteq E$, 从而$y \in E^\circ$, $B \subseteq E^\circ$. 因此, $E^\circ$为开集. 同理, $(E^c)^\circ$为开集.于是$E^\circ \cup (E^c)^\circ$为开集. 故$\partial E = E^{\mathrm{b}} = \mathbb{R}^n \setminus (E^\circ \cup (E^c)^\circ)$为闭集.

{\color{blue}证法二:}
对$\forall\ x \in E^\circ$, 则存在开球$B(x; r) \subseteq E$. 由于$\forall\ y \in B(x; r)$, 则$r - d(x,y) > 0$, 且$B(y; r - d(x,y)) \subseteq B(x; r) \subseteq E$, 从而$y \in E^\circ$, $B(x; r) \subseteq E^\circ$, $E^\circ$为开集. 同理, $(E^c)^\circ$也为开集. 根据闭集的定义, $\partial E = E^{\mathrm{b}} = \mathbb{R}^n \setminus (E^\circ \cup (E^c)^\circ)$为闭集.

{\color{blue}证法三:}
$\forall\ x \in (\partial E)'$, $x$的任何开邻域$U$, 则$\exists\ y \neq x$, s.t. $y \in \partial E$. 根据边界点定义, $\exists\ u \in E \cap U$, $v \in E^c \cap U$, 故$x \in \partial E$. 从而, $(\partial E)' \subseteq \partial E$. 根据\rrefthe{theorem:Rn中闭集的充要条件}{theorem:Rn中闭集的充要条件-1}, $\partial E$为闭集.

{\color{blue}证法四:}
$\forall\ x_n \in \partial E, n \in \mathbb{N}$, $\lim\limits_{n \to +\infty} x_n = x$及$x$的任何开邻域$U$, $\exists\ N \in \mathbb{N}$, 当$n > N$时, $x_n \in U$. 根据边界点的定义, $\exists\ u \in E \cap U$, $v \in E^c \cap U$, 故$x \in \partial E$. 从而, 由\rrefthe{theorem:Rn中闭集的充要条件}{theorem:Rn中闭集的充要条件-4}知, 在度量空间$(\mathbb{R}^n, \mathcal{T}_{d})$中, $\partial E$为闭集.
\end{enumerate}

\end{proof}

\begin{definition}[构成区间]
设$G$为$\mathbb{R}^1$上的开集,若开区间$(\alpha,\beta)\subseteq G$且$\alpha,\beta \notin G$,则称$(\alpha,\beta)$为$G$的一个构成区间.
\end{definition}
\begin{note}
例如,开集$(0,1)\cup (2,3)$的构成区间是$(0,1)$和$(2,3)$.
\end{note}

\begin{theorem}[$\mathbb{R}^n$中的开集构造定理]\label{theorem:开集构造定理}
\begin{enumerate}[(1)]
\item\label{theorem:开集构造定理-1} $\mathbb{R}^1$中非空开集$G$是至多可数个互不相交的(形如$(-\infty,a),(a,b),(b,+\infty)$的)开区间的并集.显然,反之亦真.

\item\label{theorem:开集构造定理-2} $\mathbb{R}^n$中的非空开集$G$是可数个互不相交的半开半闭$n$维正方体的并集.
\end{enumerate}
\end{theorem}
\begin{note}
由\hyperref[theorem:开集构造定理]{开集构造定理\ref{theorem:开集构造定理-1}}还可知,$\mathbb{R}^1$中非空开集都可以写成至多可数个构成区间的并.
\end{note}
\begin{proof}
\begin{enumerate}[(1)]
\item 设$G\subset\mathbb{R}^1$为开集.对$\forall\,a\in G,\exists\,\delta>0$,s.t.$(a-\delta,a+\delta)\subset G$,令
\begin{align*}
a'=\inf\{x\mid (x,a)\subset G\},\quad a''=\sup\{x\mid (a,x)\subset G\}
\end{align*}
(这里$a'$可为$-\infty$,$a''$可为$+\infty$).显然,$a'<a<a''$且$(a',a'')\subset G$.这是因为对$\forall\,z\in(a',a'')$.不妨设$a'<z<a$.根据$\inf$的定义,必$\exists\,x\in(a',z)$且有$(x,a)\subset G$,故$z\in(x,a)\subset G$.因此开区间$(a',a'')$为$G$的一个构成区间,记作$I_a$.

如果$I_a=(a',a''),I_b=(b',b'')$为$G$的构成区间,则可以证明它们或者重合或者不相交(绝不会相交而不重合).为此,不妨设$a<b$.若
\begin{align*}
I_a\cap I_b\neq\varnothing,
\end{align*}
则有$b'<a''$.于是,令
\begin{align*}
c'=\min\{a',b'\},\quad c''=\max\{a'',b''\},
\end{align*}
则$(c',c'')=(a',a'')\cup(b',b'')$.取$x\in I_a\cap I_b$,则$I_x=(c',c'')$为构成区间,且
\begin{align*}
(c',c'')=(a',a'')=(b',b'').
\end{align*}
最后,由\refpro{Set Theory-proposition:实轴上互不相交的开区间至多有可列个}知,$G$的互不相交的构成区间是至多可数的.

\item 先将$\mathbb{R}^n$用格点(坐标皆为整数)分为可数个边长为1的半开闭正方体,其全体为$\Gamma_0$.再将$\Gamma_0$中的每个正方体的每一边二等分,则每个正方体就可分为$2^n$个边长为$\frac{1}{2}$的半开闭正方体,记$\Gamma_0$中如此作成的子正方体的全体为$\Gamma_1$.继续按此方法二等分下去,可得到边长为$\frac{1}{2^k}$的半开闭小正方体族组成$\Gamma_k$,且此种正方体是$\Gamma_{k+1}$中相应的$2^k$个互不相交的边长为$\frac{1}{2^{k+1}}$的更小的半开闭正方体的并集.

现将$\Gamma_0$中凡含于$G$内的半开闭正方体取出来,记其全体为$\mathscr{H}_0$.再将$\Gamma_1$中含于
\begin{align*}
G\setminus\bigcup_{J\in\mathscr{H}_0}J
\end{align*}
($J$为半开闭二进方体)内的半开闭正方体取出来,记其全体为$\mathscr{H}_1$.依次类推,$\mathscr{H}_k$为$\Gamma_k$中含于
\begin{align*}
G\setminus\bigcup_{i=1}^{k-1}\bigcup_{J\in\mathscr{H}_i}J
\end{align*}
内的半开闭正方体的全体.显然,由$\mathscr{H}_k(k=0,1,2,\cdots)$中的半开闭正方体构成之集是可数的.

因为$G$为开集,所以对$\forall\,\boldsymbol{x}\in G,\exists\,\delta>0$,s.t.$B(\boldsymbol{x};\,\delta)\subset G$.而$\Gamma_k$中半开闭正方体之直径$\frac{\sqrt{2}}{2^k}\to0(k\to+\infty)$.从而,$\boldsymbol{x}$最终必落入某个$\Gamma_{k_0}$中的半开闭正方体中.这说明
\begin{align*}
G=\bigcup_{i=1}^{\infty}\bigcup_{J\in\mathscr{H}_i}J.
\end{align*}
其中$J$为半开闭二进方体.
\end{enumerate}

\end{proof}


\subsection{点集之间的距离}

\begin{definition}
设$E\subseteq\mathbb{R}^n$,$f:E\to\mathbb{R}$为实函数,$\boldsymbol{x}^0\in E$.如果对$\forall\,\varepsilon>0,\exists\,\delta>0$,s.t.当$\boldsymbol{x}\in E\cap B(\boldsymbol{x}^0;\delta)$时,有
\begin{align*}
|f(\boldsymbol{x})-f(\boldsymbol{x}^0)|<\varepsilon,
\end{align*}
则称$f$\textbf{在点$\boldsymbol{x}^0$处连续};称$\boldsymbol{x}^0$为$f$的一个\textbf{连续点}(当$\boldsymbol{x}^0\in E-E'$,即$\boldsymbol{x}^0$为$E$的孤立点时,对充分小的$\delta$,$\boldsymbol{x}\in E\cap B(\boldsymbol{x}^0;\delta)$时,有$|f(\boldsymbol{x})-f(\boldsymbol{x}^0)|=|f(\boldsymbol{x}^0)-f(\boldsymbol{x}^0)|=0<\varepsilon$).如果$E$中任一点皆为$f$的连续点,则称$f$\textbf{在$E$上连续}.并记$C^0(E,\mathbb{R})=C(E,\mathbb{R})$(简记为$C^0(E)=C(E)$)为$E$上\textbf{连续函数的全体}.

若对$\forall\,\varepsilon>0,\exists\,\delta>0$,当$\boldsymbol{x}',\boldsymbol{x}''\in E,d(\boldsymbol{x}',\boldsymbol{x}'')=\|\boldsymbol{x}'-\boldsymbol{x}''\|<\delta$时,有
\begin{align*}
|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|<\varepsilon.
\end{align*}
则称$f$\textbf{在$E$上一致连续}.
\end{definition}

\begin{definition}
设$E\subseteq\mathbb{R}^n$,$f_k:E\to\mathbb{R},k\in \mathbb{N}$都为实函数,若对$\forall\,\varepsilon>0,\forall\,\boldsymbol{x}\in E,\exists\,K=K(\varepsilon,\boldsymbol{x})\in\mathbb{N}$,当$k>K$时,有
\begin{align*}
|f_k(\boldsymbol{x})-f(\boldsymbol{x})|<\varepsilon.
\end{align*}
则称$\{f_k\}$\textbf{在$E$上(逐点)收敛}.

若$f:E\to\mathbb{R}$为实函数,对$\forall\,\varepsilon>0,\exists\,K=K(\varepsilon)\in\mathbb{N}$,当$k>K$时,有
\begin{align*}
|f_k(\boldsymbol{x})-f(\boldsymbol{x})|<\varepsilon,\forall\,\boldsymbol{x}\in E.
\end{align*}
则称$\{f_k\}$\textbf{在$E$上一致收敛到}$f$.记为$f_k\stackrel{\text{一致}}{\to}f.$
\end{definition}

\begin{definition}
设 \(x\in\mathbb{R}^n\)，\(E\) 是 \(\mathbb{R}^n\) 中的非空点集，称
\begin{align*}
d(x,E)=\inf\{|x - y|: y\in E\}
\end{align*}
为点 \(x\) 到 \(E\) 的\textbf{距离}；若 \(E_1\)，\(E_2\) 是 \(\mathbb{R}^n\) 中的非空点集，称
\begin{align*}
d(E_1,E_2)=\inf\{|x - y|: x\in E_1, y\in E_2\}
\end{align*}
为 \(E_1\) 与 \(E_2\) 之间的\textbf{距离}。也可等价地定义为
\[
\inf\{d(x,E_2): x\in E_1\} \quad \text{或} \quad \inf\{d(E_1,y): y\in E_2\}.
\] 
\end{definition}

\begin{proposition}[点集间的距离的性质]\label{proposition:点集间的距离的性质}
\begin{enumerate}[(1)]
\item 设$E_1,E_2,\cdots,E_n,F$是$\mathbb{R}^n$中$n+1$个非空点集,则
\begin{align*}
d\left( F,\bigcup_{i=1}^n{E_i} \right) =\underset{i=1,2,\cdots ,n}{\min}\left\{ d\left( F,E_i \right) \right\}.
\end{align*}

\item 设$E_1,E_2,\cdots,E_n$是$\mathbb{R}^n$中$n$个非空点集,若$d(E_i,E_j)>0(i\ne j)$,则$E_i\cap E_j=\varnothing (i\ne j)$.

\item 设$E_1,E_2,\cdots,E_n$是$\mathbb{R}^n$中$n$个非空闭集,若$E_i\cap E_j=\varnothing (i\ne j)$,则$d(E_i,E_j)>0(i\ne j)$.
\end{enumerate}
\end{proposition}
\begin{remark}
若(3)中去掉$E_i$是闭集这个条件,则结论不一定成立(例如,两个开球相切).
\end{remark}
\begin{proof}
\begin{enumerate}[(1)]
\item 由$\bigcup_{i=1}^nE_i\supset E_i$（$i=1,2,\cdots,n$）可知
\begin{align*}
\left\{(x,y)|x\in F,y\in\bigcup_{i=1}^nE_i\right\}\supset\left\{(x,y)|x\in F,y\in E_i\right\}\quad(i=1,2,\cdots,n).
\end{align*}
因此
\begin{align*}
d\left(F,\bigcup_{i=1}^nE_i\right)=\inf\left\{d(x,y)|x\in F,y\in\bigcup_{i=1}^nE_i\right\}\geqslant\inf\left\{d(x,y)|x\in F,y\in E_i\right\}=d(F,E_i)\quad(i=1,2,\cdots,n).
\end{align*}
故
\begin{align*}
d\left(F,\bigcup_{i=1}^nE_i\right)\geqslant\min_{i=1,2,\cdots,n}\left\{d(F,E_i)\right\}.
\end{align*}
对$\forall x\in F,y\in\bigcup_{i=1}^nE_i$，都存在$j\in\{1,2,\cdots,n\}$，使得$y\in E_j$。于是
\begin{align*}
d(x,y)\geqslant d(F,E_j)\geqslant\min_{i=1,2,\cdots,n}\left\{d(F,E_i)\right\}.
\end{align*}
故$\min_{i=1,2,\cdots,n}\left\{d(F,E_i)\right\}$是$\left\{d(x,y)|x\in F,y\in\bigcup_{i=1}^nE_i\right\}$的一个下界。因此
\begin{align*}
d\left(F,\bigcup_{i=1}^nE_i\right)=\inf\left\{d(x,y)|x\in F,y\in\bigcup_{i=1}^nE_i\right\}\geqslant\min_{i=1,2,\cdots,n}\left\{d(F,E_i)\right\}.
\end{align*}
综上，$d\left(F,\bigcup_{i=1}^nE_i\right)=\min_{i=1,2,\cdots,n}\left\{d(F,E_i)\right\}$。

\item 反证,假设存在$i\ne j$,使得$E_i\cap E_j\ne \varnothing,$则任取$x_0\in E_i\cap E_j$,又由$d(E_i,E_j)>0$可知,对$\forall x\in E_i,y\in E_j$,都有$d(x,y)\geqslant  d(E_i,E_j)>0$.这与$d(x_0,x_0)=0,x_0\in E_i\cap E_j$矛盾!

\item 反证,假设存在$i\ne j$,使得$d(E_i,E_j)=0$.由$d(E_i,E_j) = \inf\{d(x,y) \mid x \in E_i, y \in E_j\}$及下确界的定义可知，对$\forall n \in \mathbb{N}$，存在$x_n \in E_i$，$y_n \in E_j$，使得$d(x_n,y_n) < \frac{1}{n}$。从而$\lim_{n \to \infty} d(x_n,y_n) = 0$，因此
\begin{align*}
\lim_{n \to \infty} x_n = \lim_{n \to \infty} y_n = c.
\end{align*}
再由$E_i$，$E_j$都是闭集可知$c \in E_i \cap E_j$，这与$E_i \cap E_j = \varnothing$矛盾！
\end{enumerate}

\end{proof}

\begin{example}
在 \(\mathbb{R}^2\) 中作点集
\begin{align*}
E_1 &= \{x = (\xi,\eta): -\infty < \xi < +\infty, \eta = 0\},\\
E_2 &= \{y = (\xi,\eta): \xi\cdot\eta = 1\},
\end{align*}
则 \(d(E_1,E_2) = 0\)。
\end{example}
\begin{proof}
事实上，当我们取 \(x = (\xi,0)\in E_1\) 且 \(y = (\xi,\eta)\in E_2\) 时，由
\begin{align*}
d(E_1,E_2) \leqslant d(x,y) = |\eta| = \frac{1}{|\xi|}
\end{align*}
可知，对任给的 \(\varepsilon > 0\)，只需 \(|\xi|\) 充分大，就有 \(d(E_1,E_2) < \varepsilon\)。由此得
\begin{align*}
d(E_1,E_2) = 0.
\end{align*}

显然，若 \(x\in E\)，则 \(d(x,E) = 0\)。但反之，若 \(d(x,E) = 0\)，则 \(x\) 不一定属于 \(E\)。不过在 \(x\notin E\) 时，必有 \(x\in E'\)。 

\end{proof}

\begin{theorem}\label{theorem:非空闭集关于点集之间距离的基本定理}
若 \(F\subseteq\mathbb{R}^n\) 是非空闭集，且 \(x_0\in\mathbb{R}^n\)，则存在 \(y_0\in F\)，有
\[
|x_0 - y_0| = d(x_0,F).
\]
\end{theorem}
\begin{proof}
作闭球 \(\overline{B}=\overline{B}(x_0,\delta)\)，使得 \(\overline{B}\cap F\) 不是空集。显然
\[
d(x_0,F)=d(x_0,\overline{B}\cap F).
\]
\(\overline{B}\cap F\) 是有界闭集，而 \(|x_0 - y|\) 看作定义在 \(\overline{B}\cap F\) 上的 \(y\) 的函数是连续的，故它在 \(\overline{B}\cap F\) 上达到最小值，即存在 \(y_0\in\overline{B}\cap F\)，使得
\[
|x_0 - y_0| = \inf\{|x_0 - y|: y\in\overline{B}\cap F\},
\]
从而有 \(|x_0 - y_0| = d(x_0,F)\)。

\end{proof}

\begin{theorem}
若 \(E\) 是 \(\mathbb{R}^n\) 中非空点集，则 \(d(x,E)\) 作为 \(x\) 的函数在 \(\mathbb{R}^n\) 上是一致连续的。
\end{theorem}
\begin{proof}
考虑 \(\mathbb{R}^n\) 中的两点 \(x,y\)。根据 \(d(y,E)\) 的定义，对任给的 \(\varepsilon>0\)，必存在 \(z\in E\)，使得 \(|y - z|<d(y,E)+\varepsilon\)，从而有
\begin{align*}
d(x,E)&\leqslant|x - z|\leqslant|x - y|+|y - z|\\
&<|x - y|+d(y,E)+\varepsilon.
\end{align*}
由 \(\varepsilon\) 的任意性可知
\[
d(x,E)-d(y,E)\leqslant|x - y|.
\]
同理可证 \(d(y,E)-d(x,E)\leqslant|x - y|\)。这说明
\[
|d(x,E)-d(y,E)|\leqslant|x - y|.
\]

\end{proof}

\begin{corollary}
若 \(F_1,F_2\) 是 \(\mathbb{R}^n\) 中的两个非空闭集且其中至少有一个是有界的，则存在 \(x_1\in F_1\)，\(x_2\in F_2\)，使得
\[
|x_1 - x_2| = d(F_1,F_2).
\] 
\end{corollary}


\subsection{Borel集}

\begin{definition}[$\boldsymbol{F}_{\boldsymbol{\sigma }},\boldsymbol{G}_{\boldsymbol{\delta }}$集]\label{definition:F_sigma,G_delta集}
若 $E \subseteq \mathbb{R}^n$ 是至多可数个闭集的并集，则称 $E$ 为 $\boldsymbol{F}_{\boldsymbol{\sigma }}$\textbf{(型)集}；

若 $E \subseteq \mathbb{R}^n$ 是至多可数个开集的交集，则称 $E$ 为 $\boldsymbol{G}_{\boldsymbol{\sigma }}$\textbf{(型)集}。

至多可数个$F_{\sigma}$集的交集称为$\boldsymbol{F_{\sigma\delta}}$\textbf{集};至多可数个$G_{\delta}$集的并集称为$\boldsymbol{G_{\delta\sigma}}$\textbf{集}.
\end{definition}

\begin{proposition}
设$E\subseteq \mathbb{R}^n$,则
\begin{enumerate}[(1)]
\item $E\text{为}G_{\delta}\text{集}\iff E^c\text{为}F_{\sigma}\text{集}.$

\item $\mathbb{R}^n$中每个闭集为$G_\delta$集; 每个开集为$F_\sigma$集.

\item 至多可数个$G_{\delta}$集的交仍是$G_{\delta}$集;至多可数个$F_{\sigma}$集的交仍是$F_{\sigma}$集.
\end{enumerate}
\end{proposition}
\begin{proof}
\begin{enumerate}[(1)]
\item 由De Morgan公式立知
\begin{align*}
E\text{为}G_{\delta}\text{集}\iff E^c\text{为}F_{\sigma}\text{集}.
\end{align*}

\item {\color{blue}证法一:}
设$F$为闭集, 作$G_n = \left\{ x \in \mathbb{R}^n \mid d(x,F) < \frac{1}{n} \right\}$, 显然$G_n$为开集. 易知, $F \subseteq \bigcap_{n=1}^\infty G_n$. 又若$x \in \bigcap_{n=1}^\infty G_n$, 则对$\forall\ n \in \mathbb{N}$, 有$x \in G_n$, 即$d(x,F) < \frac{1}{n}$, 从而$d(x,F) = 0$. 注意到$F$为闭集, 故由\refthe{theorem:非空闭集关于点集之间距离的基本定理}知$\exists\ y \in F$, s.t. $d(x,y) = d(x,F) = 0$. 由此知$x = y \in F$. 这说明$\bigcap_{n=1}^\infty G_n \subseteq F$. 综合上述, 有$F = \bigcap_{n=1}^\infty G_n$, 从而, 闭集$F$为$G_\delta$集.

设$G$为开集, 则$F = \mathbb{R}^n \setminus G$为闭集, 由上知$F = \bigcap_{n=1}^\infty G_n$, $G_n(n \in \mathbb{N})$为开集, 从而$G_n^c$为闭集, 且有
\begin{align*}
G = F^c = \left( \bigcap_{n=1}^\infty G_n \right)^c \xlongequal{\mathrm{De}\, \text{Morgan公式}} \bigcup_{n=1}^\infty G_n^c,
\end{align*}
即开集$G$为$F_\sigma$集.

{\color{blue}证法二:}
根据\hyperref[theorem:开集构造定理]{$\mathbb{R}^n$中的开集构造定理}中的证明知, 记含于开集$G$中的边长为1的闭正方体的全体为$\mathscr{H}_0$; 含于$G$中的边长为$\frac{1}{2}$的闭正方体的全体$\mathscr{H}_1$; $\cdots$; 含于$G$中的边长为$\frac{1}{2^n}$的闭正方体的全体为$\mathscr{H}_n$; $\cdots$. 显然, $F_n = \bigcup_{A \in \mathscr{H}_n} A$为闭集, 且$G = \bigcup_{n=1}^\infty F_n$为$F_\sigma$集.

设$F$为闭集, 则$G = \mathbb{R}^n \setminus F$为开集, 由上知$G = \bigcup_{n=1}^\infty F_n$, $F_n(n \in \mathbb{N})$为闭集, 从而$F_n^c$为开集, 且有
\begin{align*}
F=G^c=\left( \bigcup_{n=1}^{\infty}{F_n} \right) ^c\xlongequal{\mathrm{De}\, \text{Morgan公式}}\bigcap_{n=1}^{\infty}{F_{n}^{c}},
\end{align*}
即闭集$F$为$G_\delta$集.

\item 设开集$G=\bigcap\limits_{i=1}^{\infty}G_i$为$G_{\delta}$集,闭集$F=\bigcup\limits_{i=1}^{\infty}F_i$为$F_{\sigma}$集,则至多可数个$G_{\delta}$集的交$\bigcap\limits_{i=1}^{\infty}\left(\bigcap\limits_{j=1}^{\infty}G_{ij}\right)=\bigcap\limits_{i,j=1}^{\infty}G_{ij}$仍为$G_{\delta}$集(其中$G_{ij}$都为开集)；至多可数个$F_{\sigma}$集的并$\bigcup\limits_{i=1}^{\infty}\left(\bigcup\limits_{j=1}^{\infty}F_{ij}\right)=\bigcup\limits_{i,j=1}^{\infty}F_{ij}$仍为$F_{\sigma}$集(其中$F_{ij}$都为闭集).
\end{enumerate}

\end{proof}

\begin{example}
记 $\mathbb{R}^n$ 中全体有理点为 $\{r_k\}$，则有理点集
\begin{align*}
\bigcup_{k = 1}^{\infty} \{r_k\}
\end{align*}
为 $F_{\sigma}$ 集。 
\end{example}

\begin{example}\label{example:例12}
设$f_k\in C(\mathbb{R}^n)$ ($k = 1,2,\cdots$)，且有
\begin{align*}
\lim_{k\to\infty}f_k(x)=f(x), \quad x\in\mathbb{R}^n,
\end{align*}
则$f(x)$的连续点集
\begin{align*}
\bigcap_{m = 1}^{\infty}\bigcup_{k = 1}^{\infty}\mathring{E}_k\left(\frac{1}{m}\right)
\end{align*}
是$G_{\delta}$型集，其中$E_k(\varepsilon)=\{x\in\mathbb{R}^n: |f_k(x)-f(x)|\leqslant\varepsilon\}$.

(i) 设$x_0\in\mathbb{R}^n$是$f(x)$的连续点. 由题设知，对任意$\varepsilon>0$，存在$k_0$，使得$|f_{k_0}(x_0)-f(x_0)|<\varepsilon/3$，且存在$\delta>0$，使得

|f(x)-f(x_0)|<\varepsilon/3, \quad |f_{k_0}(x)-f_{k_0}(x_0)|<\varepsilon/3, \quad x\in U(x_0,\delta),
\end{align*}
从而对$x\in U(x_0,\delta)$，有

\end{align*}

\end{align*}

\begin{theorem}\label{theorem:Borel集的势}
设$\mathscr{R}_0=\left\{\bigcup\limits_{i=1}^{n}(a_i,b_i]\mid -\infty<a_i\leqslant b_i<+\infty\right\}$,则
\begin{align*}
\overline{\mathscr{B}}=\overline{\mathscr{A}_{\sigma}(\mathscr{R}_0)}=\overline{\mathscr{R}_{\sigma}(\mathscr{R}_0)}=\aleph.
\end{align*}
更一般地,设$\mathscr{R}$为$\mathbb{R}$上的集类,其势$\overline{\mathscr{R}}=\aleph$,则包含$\mathscr{R}$的最小$\sigma$环$\mathscr{R}_{\sigma}(\mathscr{R})$有$\overline{\mathscr{R}_{\sigma}(\mathscr{R})}=\aleph$.

又显然有$\mathscr{A} _{\sigma}(\mathscr{R} _0),\mathscr{R} _{\sigma}(\mathscr{R} _0)\subseteq \mathscr{B}$,进而\begin{align*}
\mathscr{B} =\mathscr{A} _{\sigma}(\mathscr{R} _0)=\mathscr{R} _{\sigma}(\mathscr{R} _0).
\end{align*}
\end{theorem}
\begin{proof}
第1步,最小良序集的构造.

由良序定理,任何集合都可良序化.将实数集$\mathbb{R}$良序化(用$b\leqslant a$表示$b$先$a$后),仍记为$\mathbb{R}$.令$A=\{a\in\mathbb{R}\mid$至多可数个$b\in\mathbb{R}$满足$b\leqslant a\}$.设$\mathbb{R}$的最小元为$a_0$,则$a_0\in A$,因此,$A\neq\varnothing$.下证$\aleph_0<\overline{A}\leqslant\aleph$.(反证)若$\overline{A}\leqslant\aleph_0$,则$\mathbb{R}-A\neq\varnothing$.因$\mathbb{R}$为良序集,故$\mathbb{R}-A$中有最小元$a_1$.显然,由$A$的定义知$a_1\in A$,这与$a_1\in\mathbb{R}-A,a_1\notin A$相矛盾.称$A$为最小良序集.

第2步,$\mathscr{R}_{\sigma}(\mathscr{R})$的结构.

利用超限归纳法构造集族$\{\mathscr{R}_a\mid a\in A\}$.令$\mathscr{R}_{a_0}=\mathscr{R}$.设对$\lambda\in A$,$\mathscr{R}$已定义,则对$\lambda$的后继元$\lambda+1$,定义
\begin{align*}
\mathscr{R}_{\lambda+1}=\left\{\bigcup\limits_{i=1}^{\infty}E_i,E_1-E_2\mid E_i\in\mathscr{R}_{\lambda},i=1,2,\cdots\right\}.
\end{align*}
如此对每个$a\in A$,$\mathscr{R}_a$均可定义好.下证$\bigcup\limits_{a\in A}\mathscr{R}_a$为$\sigma$环.

(i) 任取$G_n\in\bigcup\limits_{a\in A}\mathscr{R}_a$,则必有$\mathscr{R}_{a_n}(a_n\in A)$,s.t.$G_n\in\mathscr{R}_{a_n},n=1,2,\cdots$.因$A$不可数,而
\begin{align*}
\{c\in\mathbb{R}\mid \exists n_0\in\mathbb{N},\text{s.t.}c\leqslant a_{n_0}\}
\end{align*}
为至多可数集,故
\begin{align*}
A-\{c\in\mathbb{R}\mid \exists n_0\in\mathbb{N},\text{s.t.}c\leqslant a_{n_0}\}\neq\varnothing,
\end{align*}
它必有最小元$b\in A$.显然,$\mathscr{R}_{a_n}\subset\mathscr{R}_b,n=1,2,\cdots$,
\begin{align*}
\bigcup\limits_{n=1}^{\infty}G_n\in\mathscr{R}_{b+1}\subset\bigcup\limits_{a\in A}\mathscr{R}_a(\text{注意}b+1\in A).
\end{align*}

(ii) 任取$G_i,G_j\in\bigcup\limits_{a\in A}\mathscr{R}_a,G_i\in\mathscr{R}_{a_i},G_j\in\mathscr{R}_{a_j}$.不妨设$a_i\leqslant a_j$.于是,$G_i,G_j\in\mathscr{R}_{a_j}$,故$G_i-G_j\in\mathscr{R}_{a_j+1}\subset\bigcup\limits_{a\in A}\mathscr{R}_a$.

综上所述知$\bigcup\limits_{a\in A}\mathscr{R}_a$为$\sigma$环.因此,由$\mathscr{R}_{\sigma}(\mathscr{R})$的最小性知
\begin{align*}
\mathscr{R}_{\sigma}(\mathscr{R})\subset\bigcup\limits_{a\in A}\mathscr{R}_a.
\end{align*}

另一方面,从$\bigcup\limits_{a\in A}\mathscr{R}_a$的构造步骤立知
\begin{align*}
\bigcup\limits_{a\in A}\mathscr{R}_a\subset\mathscr{R}_{\sigma}(\mathscr{R}).
\end{align*}
所以
\begin{align*}
\mathscr{R}_{\sigma}(\mathscr{R})=\bigcup\limits_{a\in A}\mathscr{R}_a.
\end{align*}

第3步,$\overline{\mathscr{R}_{\sigma}(\mathscr{R})}=\aleph$.

仍用超限归纳法可证,对$\forall a\in A$,$\overline{\mathscr{R}_a}=\aleph$.事实上,显然有$\overline{\mathscr{R}_{a_0}}=\overline{\mathscr{R}}=\aleph$.设对$\lambda\in A$,有$\overline{\mathscr{R}_{\lambda}}=\aleph$,则根据第2步中$\mathscr{R}_{\lambda}$构造$\mathscr{R}_{\lambda+1}$的过程可看出$\overline{\mathscr{R}_{\lambda+1}}=\aleph$.

因为$\mathscr{R}\subset\mathscr{R}_{\sigma}(\mathscr{R})=\bigcup\limits_{a\in A}\mathscr{R}_a$,所以
\begin{align*}
\aleph=\overline{\mathscr{R}}\leqslant\overline{\mathscr{R}_{\sigma}(\mathscr{R})}=\overline{\bigcup\limits_{a\in A}\mathscr{R}_a}\leqslant\overline{A}\cdot\aleph\leqslant\aleph\cdot\aleph=\aleph.
\end{align*}
根据\hyperref[Set Theory-theorem:Cantor-Bernstein定理]{Cantor-Bernstein定理}得到
\begin{align*}
\overline{\mathscr{R}_{\sigma}(\mathscr{R})}=\aleph.
\end{align*}

\end{proof}

\begin{theorem}\label{theorem:振幅函数的定义和性质}
设实函数$f$在$B_E(\boldsymbol{x}^0;\delta_0)=E\cap B(\boldsymbol{x}^0;\delta_0)\subseteq\mathbb{R}^n$上有定义,我们称$(0<\delta<\delta_0)$
\begin{align*}
\omega_E(\boldsymbol{x}^0)=\lim\limits_{\delta\to0^+}\omega_E(\boldsymbol{x}^0;\delta)
=\lim\limits_{\delta\to0^+}\sup\{|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|\mid \boldsymbol{x}',\boldsymbol{x}''\in B_E(\boldsymbol{x}^0;\delta)\}
\end{align*}
为$f$在点$\boldsymbol{x}^0\in E$处的\textbf{振幅}.特别当$E\subseteq\mathbb{R}^n$为开集时,简记为$\omega(\boldsymbol{x}^0)$.

如果$E\subseteq\mathbb{R}^n$为开集,$f$定义在$E$上,则对任意固定的$t\in\mathbb{R}$,点集
\begin{align*}
E_t=\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})<t\}\text{与}\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})>t\}
\end{align*}
为开集.而
\begin{align*}
\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})\geqslant t\}\text{与}\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})\leqslant t\}
\end{align*}
为闭集.
\end{theorem}
\begin{proof}
如果$E_t=\varnothing\in\mathscr{T}$,则$E_t$为开集.

如果$E_t\neq\varnothing$.对$\forall\,\boldsymbol{x}^0\in E_t$,因$\omega(\boldsymbol{x}^0)<t$,故$\exists\,\delta_1\in(0,\delta_0)$,s.t.$B(\boldsymbol{x}^0;\delta_1)\subset E$,且有
\begin{align*}
\sup\{|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|\mid \boldsymbol{x}',\boldsymbol{x}''\in B(\boldsymbol{x}^0;\delta_1)\}\leqslant t_1<t.
\end{align*}
现对$\forall\,\boldsymbol{x}\in B(\boldsymbol{x}^0;\delta_1)$,可选$\delta_2>0$,s.t.$B(\boldsymbol{x};\delta_2)\subset B(\boldsymbol{x}^0;\delta_1)$.显然,有
\begin{align*}
\sup\{|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|\mid \boldsymbol{x}',\boldsymbol{x}''\in B(\boldsymbol{x};\delta_2)\}\leqslant t_1<t.
\end{align*}
由此可知
\begin{align*}
\omega(\boldsymbol{x})\leqslant t_1<t,\quad B(\boldsymbol{x}^0;\delta_1)\subset E_t.
\end{align*}
这就证明了$E_t$为$(\mathbb{R}^n,\mathscr{T})$中的开集.

类似可证$\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})>t\}$为开集.而
\begin{align*}
\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})\geqslant t\}=\mathbb{R}^n\setminus\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})<t\},
\end{align*}
\begin{align*}
\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})\leqslant t\}=\mathbb{R}^n\setminus\{\boldsymbol{x}\in E\mid \omega(\boldsymbol{x})>t\}
\end{align*}
为闭集.

\end{proof}

\begin{lemma}\label{lemma:实变函数--引理1.4.8}
设$E\subseteq\mathbb{R}^n$,$\mathscr{T}$为$\mathbb{R}^n$上的通常拓扑,则$\mathscr{T}_E=\{H\subseteq E\mid H=E\cap G,G\in\mathscr{T}\}$为$E$上的一个拓扑.
\end{lemma}
\begin{proof}
因为
\begin{enumerate}[(1)]
\item $\varnothing_E=E\cap\varnothing^{\mathbb{R}^n}\in\mathscr{T}_E$,$E=E\cap\mathbb{R}^n\in\mathscr{T}_E$;
\item 若$H_1,H_2\in\mathscr{T}_E$,即$H_1=E\cap G_1,H_2=E\cap G_2,G_1\in\mathscr{T},G_2\in\mathscr{T}$,则
\begin{align*}
H_1\cap H_2=(E\cap G_1)\cap(E\cap G_2)=E\cap(G_1\cap G_2)\in\mathscr{T}_E;
\end{align*}
\item 若$H_{\alpha}\in\mathscr{T}_E,\alpha\in\Gamma$,即$H_{\alpha}=E\cap G_{\alpha},G_{\alpha}\in\mathscr{T},\alpha\in\Gamma$,则
\begin{align*}
\bigcup_{\alpha\in\Gamma}H_{\alpha}=\bigcup_{\alpha\in\Gamma}(E\cap G_{\alpha})=E\cap\left(\bigcup_{\alpha\in\Gamma}G_{\alpha}\right)\in\mathscr{T}_E,
\end{align*}
\end{enumerate}
所以,$\mathscr{T}_E$为$E$上的一个拓扑.

\end{proof}

\begin{definition}
设$E\subseteq\mathbb{R}^n$,$\mathscr{T}$为$\mathbb{R}^n$上的通常拓扑,则由\reflem{lemma:实变函数--引理1.4.8},有
\begin{align*}
\mathscr{T}_E&=\{H\subseteq E\mid H=E\cap G,G\in\mathscr{T}\}\\
&=\{H\subseteq E\mid \forall\,\boldsymbol{x}\in H,\exists\,\delta>0,\text{s.t.}E\cap B(\boldsymbol{x},\delta)\subseteq H\}\\
&=\{H\subseteq E\mid \forall\,\boldsymbol{x}\in H,\exists\,\delta>0,\text{s.t.}B_E(\boldsymbol{x},\delta)\subseteq H\}
\end{align*}
(其中$B_E(\boldsymbol{x};\delta)=E\cap B(\boldsymbol{x};\delta)=\{\boldsymbol{y}\in E\mid \rho(\boldsymbol{y},\boldsymbol{x})<\delta\}$)为$E$上的一个拓扑.于是,$(E,\mathscr{T}_E)$为一个\textbf{拓扑空间},称为$(\mathbb{R}^n,\mathscr{T})$的\textbf{子拓扑空间}.
\end{definition}

\begin{proposition}\label{proposition:连续函数的基本性质命题}
设$E\subseteq\mathbb{R}^n$,$f:E\to\mathbb{R}$为连续函数,则对$\forall\,t\in\mathbb{R}$,
\begin{align*}
E_t=\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})>t\}
\end{align*}
为$(E,\mathscr{T}_E)$中的开集(当$E\subseteq\mathbb{R}^n$为开集时,$E_t$也为$(\mathbb{R}^n,\mathscr{T})$中的开集).
\begin{align*}
F_t=\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})\geqslant t\}
\end{align*}
为$(E,\mathscr{T}_E)$中的闭集(当$E\subseteq\mathbb{R}^n$为闭集时,$F_t$也为$(\mathbb{R}^n,\mathscr{T})$中的闭集).

类似地,$$E_t'=\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})<t\}$$
为$(E,\mathscr{T}_E)$中的开集(当$E\subseteq\mathbb{R}^n$为开集时,$E_t'$也为$(\mathbb{R}^n,\mathscr{T})$中的开集).
$$F_t'=\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})\leqslant t\}$$
为$(E,\mathscr{T}_E)$中的闭集(当$E\subseteq\mathbb{R}^n$为开集时,$F_t'$也为$(\mathbb{R}^n,\mathscr{T})$中的闭集).
\end{proposition}
\begin{proof}
\begin{enumerate}[(1)]
\item $\forall\,\boldsymbol{x}\in E_t$,则$f(\boldsymbol{x})>t$.由$f$连续,$\exists\,\delta_{\boldsymbol{x}}>0$,s.t.
\begin{align*}
f(\boldsymbol{y})>t,\quad \forall\,\boldsymbol{y}\in B_E(\boldsymbol{x};\delta_{\boldsymbol{x}})=E\cap B(\boldsymbol{x};\delta_{\boldsymbol{x}}),
\end{align*}
所以,$\boldsymbol{x}\in B_E(\boldsymbol{x};\delta_{\boldsymbol{x}})\subseteq E_t$,从而$E_t$为$(E,\mathscr{T}_E)$中的开集.并且
\begin{align*}
E_t&=\bigcup_{\boldsymbol{x}\in E_t}B_E(\boldsymbol{x};\delta_{\boldsymbol{x}})=\bigcup_{\boldsymbol{x}\in E_t}(E\cap B(\boldsymbol{x};\delta_{\boldsymbol{x}}))\\
&=E\cap\left(\bigcup_{\boldsymbol{x}\in E_t}B(\boldsymbol{x};\delta_{\boldsymbol{x}})\right)=E\cap G_t,
\end{align*}
其中$G_t=\bigcup_{\boldsymbol{x}\in E_t}B(\boldsymbol{x};\delta_{\boldsymbol{x}})$为$(\mathbb{R}^n,\mathscr{T})$中的开集.这也说明了$E_t$为$(\mathbb{R}^n,\mathscr{T}_E)$中的开集.

类似可证$\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})<t\}$为$(E,\mathscr{T}_E)$中的开集.或者由上述结果知
\begin{align*}
\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})<t\}=\{\boldsymbol{x}\in E\mid -f(\boldsymbol{x})>-t\}
\end{align*}
为$(E,\mathscr{T}_E)$中的开集.

\item $F_t=\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})\geqslant t\}=E\setminus\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})<t\}$为$(E,\mathscr{T}_E)$中的闭集.或者由$\boldsymbol{x}^m\in F_t,\forall m\in \mathbb{N}$,$\,\,\boldsymbol{x}^0=\lim\limits_{m\to+\infty}\boldsymbol{x}^m$立即得到
\begin{align*}
f(\boldsymbol{x}^m)\geqslant t,\,\forall m\in \mathbb{N}.
\end{align*}
于是
\begin{align*}
f(\boldsymbol{x}^0)=\lim\limits_{m\to+\infty}f(\boldsymbol{x}^m)\geqslant t\Longrightarrow \boldsymbol{x}^0\in F_t.
\end{align*}
由\rrefthe{theorem:Rn中闭集的充要条件}{theorem:Rn中闭集的充要条件-4}知,$F_t$为$(E,\mathscr{T}_E)$中的闭集.

类似可证$\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})\leqslant t\}$为$(E,\mathscr{T}_E)$中的闭集.或者由上述结果知
\begin{align*}
\{\boldsymbol{x}\in E\mid f(\boldsymbol{x})\leqslant t\}=\{\boldsymbol{x}\in E\mid -f(\boldsymbol{x})\geqslant -t\}
\end{align*}
为$(E,\mathscr{T}_E)$中的闭集.
\end{enumerate}

\end{proof}

\begin{proposition}\label{proposition:实变函数连续点和导数点的结构}
\begin{enumerate}[(1)]
\item\label{proposition:实变函数连续点和导数点的结构-1} (\textbf{函数在点$\boldsymbol{x}^0$连续的充要条件}) 设$E\subseteq\mathbb{R}^n$,$f:E\to\mathbb{R}$,则
\begin{align*}
f\text{在}\boldsymbol{x}^0\text{连续}\iff \omega_E(\boldsymbol{x}^0)=0.
\end{align*}

\item\label{proposition:实变函数连续点和导数点的结构-2} (\textbf{函数连续点集的结构}) 设$G\subseteq\mathbb{R}^n$为开集,$f:G\to\mathbb{R}$为实函数,则$f$的连续点集为
\begin{align*}
\bigcap\limits_{k=1}^{\infty}\left\{\boldsymbol{x}\in G\mid \omega(\boldsymbol{x})<\frac{1}{k}\right\}
\end{align*}
是$G_{\delta}$集,因此,它也为Borel集.

\item\label{proposition:实变函数连续点和导数点的结构-3} (\textbf{连续函数可导点集的结构}) 设$f:\mathbb{R}^1\to\mathbb{R}$为连续函数,则$f$的可导点集为$F_{\sigma\delta}$集(可数个$F_{\sigma}$集的交集).
\end{enumerate}
\end{proposition}
\begin{proof}
\begin{enumerate}[(1)]
\item $(\Rightarrow)$$f$在$\boldsymbol{x}^0$连续,即$\forall\,\varepsilon>0,\exists\,\delta>0$,当$\boldsymbol{x}\in B_E(\boldsymbol{x}^0;\delta)$时,有
\begin{align*}
|f(\boldsymbol{x})-f(\boldsymbol{x}^0)|<\frac{\varepsilon}{4}.
\end{align*}
因此,当$\boldsymbol{x}',\boldsymbol{x}''\in B_E(\boldsymbol{x}^0;\delta)$时,有
\begin{align*}
|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|\leqslant|f(\boldsymbol{x}')-f(\boldsymbol{x}^0)|+|f(\boldsymbol{x}^0)-f(\boldsymbol{x}'')|<\frac{\varepsilon}{4}+\frac{\varepsilon}{4}=\frac{\varepsilon}{2},
\end{align*}
\begin{align*}
\sup\{|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|\mid \boldsymbol{x}',\boldsymbol{x}''\in B_E(\boldsymbol{x}^0;\delta)\}\leqslant\frac{\varepsilon}{2}<\varepsilon.
\end{align*}
于是
\begin{align*}
\omega_E(\boldsymbol{x}^0)=\lim\limits_{\delta\to0^+}\sup\{|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|\mid \boldsymbol{x}',\boldsymbol{x}''\in B_E(\boldsymbol{x}^0;\delta)\}=0.
\end{align*}

$(\Leftarrow)$设
\begin{align*}
\omega_E(\boldsymbol{x}^0)=\lim\limits_{\delta\to0^+}\sup\{|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|\mid \boldsymbol{x}',\boldsymbol{x}''\in B_E(\boldsymbol{x}^0;\delta)\}=0,
\end{align*}
即对$\forall\,\varepsilon>0,\exists\,\eta>0$,当$0<\delta<\eta$时,有
\begin{align*}
\sup\{|f(\boldsymbol{x}')-f(\boldsymbol{x}'')|\mid \boldsymbol{x}',\boldsymbol{x}''\in B_E(\boldsymbol{x}^0;\delta)\}<\varepsilon.
\end{align*}
特别当$\boldsymbol{x}'=\boldsymbol{x},\boldsymbol{x}''=\boldsymbol{x}^0$时,有
\begin{align*}
|f(\boldsymbol{x})-f(\boldsymbol{x}^0)|<\varepsilon,\quad \forall\,\boldsymbol{x}\in B_E(\boldsymbol{x}^0;\delta).
\end{align*}
这就证明了$f$在点$\boldsymbol{x}^0$处连续.

\item 由\rrefpro{proposition:实变函数连续点和导数点的结构}{proposition:实变函数连续点和导数点的结构-1}知
\begin{align*}
f\text{在}\boldsymbol{x}^0\in G\text{连续}\iff \omega(\boldsymbol{x}^0)=0.
\end{align*}
因此,$f$的连续点集可表示为
\begin{align*}
\bigcap\limits_{k=1}^{\infty}\left\{\boldsymbol{x}\in G\mid \omega(\boldsymbol{x})<\frac{1}{k}\right\}.
\end{align*}
根据\refthe{theorem:振幅函数的定义和性质},$\left\{\boldsymbol{x}\in G\mid \omega(\boldsymbol{x})<\frac{1}{k}\right\}$为$G$(也是$(\mathbb{R}^n,\mathscr{T})$)中的开集,所以$f$的连续点集为$G_{\delta}$集.

\item 根据De Morgan公式,有
\begin{align*}
\left(\bigcap\limits_{i=1}^{\infty}\bigcup\limits_{j=1}^{\infty}F_{ij}\right)^c=\bigcup\limits_{i=1}^{\infty}\left(\bigcup\limits_{j=1}^{\infty}F_{ij}\right)^c=\bigcup\limits_{i=1}^{\infty}\bigcap\limits_{j=1}^{\infty}F_{ij}^c.
\end{align*}
因此
\begin{align*}
f\text{的可导点集为}F_{\sigma\delta}\text{集}\iff f\text{的不可导点集为}G_{\delta\sigma}\text{集}(\text{可数个}G_{\delta}\text{集的并集}).
\end{align*}
应用上、下导数$\left(\varliminf\limits_{x\to a}\frac{f(x)-f(a)}{x-a},\varlimsup\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\right)$的概念,$f$的不可导点集为下面三个集合的并:
\begin{align*}
A&=\left\{a\mid \varliminf\limits_{x\to a}\frac{f(x)-f(a)}{x-a}<\varlimsup\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\right\}\\
&=\bigcup_{\substack{r,R\in\mathbb{Q}\\r<R}}\left\{a\mid \varliminf\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\leqslant r<R\leqslant\varlimsup\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\right\}\\
&=\bigcup_{\substack{r,R\in\mathbb{Q}\\r<R}}\left(\left\{a\mid \varliminf\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\leqslant r\right\}\cap\left\{a\mid \varlimsup\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\geqslant R\right\}\right),
\end{align*}
\begin{align*}
B&=\left\{a\mid \varlimsup\limits_{x\to a}\frac{f(x)-f(a)}{x-a}=+\infty\right\}=\bigcap\limits_{r\in\mathbb{Q}}\left\{a\mid \varlimsup\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\geqslant r\right\},
\end{align*}
\begin{align*}
C&=\left\{a\mid \varliminf\limits_{x\to a}\frac{f(x)-f(a)}{x-a}=-\infty\right\}=\bigcap\limits_{r\in\mathbb{Q}}\left\{a\mid \varliminf\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\leqslant r\right\}.
\end{align*}
另一方面,对$\forall t\in \mathbb{R}$以及任何固定的$n,k\in\mathbb{N}$,由$f$的连续性和\refpro{proposition:连续函数的基本性质命题}知
\begin{align*}
\left\{ a\mid \exists \,x,\mathrm{s}.\mathrm{t}. 0<|x-a|<\frac{1}{n},\text{且}\frac{f(x)-f(a)}{x-a}>t-\frac{1}{k} \right\} =\bigcup_{x\in \left( a-\frac{1}{n},a+\frac{1}{n} \right)}{\left\{ a\mid \frac{f(x)-f(a)}{x-a}>t-\frac{1}{k} \right\}}
\end{align*}
为开集.从而
\begin{align*}
\left\{a\mid \varlimsup\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\geqslant t\right\}
=\bigcap\limits_{k=1}^{\infty}\bigcap\limits_{n=1}^{\infty}\left\{a\mid \exists\,x,\text{s.t.}0<|x-a|<\frac{1}{n},\text{且}\frac{f(x)-f(a)}{x-a}>t-\frac{1}{k}\right\}
\end{align*}
为$G_{\delta}$集.

同理可证
\begin{align*}
\left\{a\mid \lim\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\leqslant t\right\}
\end{align*}
亦为$G_{\delta}$集.或者从前面结果知
\begin{align*}
\left\{a\mid \lim\limits_{x\to a}\frac{f(x)-f(a)}{x-a}\leqslant t\right\}=\left\{a\mid \varlimsup\limits_{x\to a}\frac{-f(x)-(-f(a))}{x-a}\geqslant -t\right\}
\end{align*}
为$G_{\delta}$集.

于是,$f$的不可导点集$A\cup B\cup C$为至多可数个$G_{\delta}$集的并集,即为$G_{\delta\sigma}$集.
\end{enumerate}

\end{proof}

\begin{example}
设$f: \mathbb{R}^1 \to \mathbb{R}$为单调增的函数. 证明: 点集
\[
E = \left\{ x \in \mathbb{R}^1 \mid \text{对 } \forall\ \delta > 0, f(x+\delta) - f(x-\delta) > 0 \right\}
\]
为$\mathbb{R}^1$中的闭集.
\end{example}
\begin{proof}
{\color{blue}证法一:}
因为
\begin{align*}
E^c &= \left\{ x \mid \exists\ \varepsilon > 0, \text{s.t. } f(x+\varepsilon) - f(x-\varepsilon) \leqslant 0 \right\} \\
&\xlongequal{f\text{单调增}}\left\{ x \mid \exists\ \varepsilon > 0, \text{s.t. } f(x+\varepsilon) - f(x-\varepsilon) = 0 \right\} \\
&= \left\{ x \mid \exists\ \varepsilon > 0, \text{s.t. } f|_{(x-\varepsilon,x+\varepsilon)} = f(x)\text{（常值）} \right\},
\end{align*}
所以, $x \in E^c \Rightarrow \exists\ \varepsilon > 0, \text{s.t. } (x-\varepsilon,x+\varepsilon) \subseteq E^c$, 从而$E^c$为开集, $E$为闭集.

{\color{blue}证法二:}
设$x \in E'$, 故$\exists\ x_k \to x, x_k \in E$. 于是, 对$\forall\ \delta > 0$, 有$x_k \in (x-\delta,x+\delta)$, 故$\exists\ \delta_1 > 0, \text{s.t. } (x_k-\delta_1,x_k+\delta_1) \subseteq (x-\delta,x+\delta)$. 则由$f$单调递增可得
\begin{align*}
f(x+\delta) - f(x-\delta) \geqslant f(x_k+\delta_1) - f(x_k-\delta_1) > 0.
\end{align*}
因此, $x \in E, E' \subseteq E$. 这就证明了$E$为闭集.

\end{proof}
