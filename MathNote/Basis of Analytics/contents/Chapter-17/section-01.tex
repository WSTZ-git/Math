\section{Fourier级数及基本性质}

\begin{definition}
设 $f$ 是 $[-\pi,\pi]$ 上的Riemann可积函数，定义其$\mathbf{Fourier}$\textbf{系数}为
\[
a_n = \frac{1}{\pi}\int_{-\pi}^{\pi} f(x)\cos(nx)\,\mathrm{d}x,\quad n=0,1,2,\cdots,
\]
\[
b_n = \frac{1}{\pi}\int_{-\pi}^{\pi} f(x)\sin(nx)\,\mathrm{d}x,\quad n=1,2,3,\cdots,
\]
称
\[
\frac{a_0}{2}+\sum_{n=1}^{\infty}\bigl(a_n\cos(nx)+b_n\sin(nx)\bigr).
\]
为$f$的$\mathbf{Fourier}$\textbf{级数}.也常记为
\[
f(x)\sim \frac{a_0}{2}+\sum_{n=1}^{\infty}\bigl(a_n\cos(nx)+b_n\sin(nx)\bigr).
\]

\end{definition}

\begin{definition}[分段连续]
设函数 $f$ 在闭区间 $[a,b]$ 上有定义。若存在 $a = x_0 < x_1 < \cdots < x_m = b$，使得 $f$ 在每个开区间 $(x_{i-1}, x_i)$ 上连续，且
\begin{gather*}
f(x_0^+),f(x_m^-)\text{存在且有限},
\\
f(x_i^-),f(x_i^+)\text{存在且有限},i=1,2,\cdots,m-1.
\end{gather*}
则称 $f$ 在 $[a,b]$ 上\textbf{分段连续}.
\end{definition}

\begin{definition}[分段光滑]
设函数 $f$ 在闭区间 $[a,b]$ 上分段连续。若 $f$ 的导函数 $f'$ 在 $[a,b]$ 上分段连续，则称 $f$ 在 $[a,b]$ 上\textbf{分段光滑}.
\end{definition}

\begin{theorem}[Dirichlet收敛定理]\label{theorem:Dirichlet收敛定理}
设 $f$ 是以 $2\pi$ 为周期的周期函数，且在 $[-\pi,\pi]$ 上分段光滑。记 $f$ 的 Fourier 系数为
\[
a_n = \frac{1}{\pi}\int_{-\pi}^{\pi} f(x)\cos(nx)\,\mathrm{d}x,\quad n=0,1,2,\cdots,
\]
\[
b_n = \frac{1}{\pi}\int_{-\pi}^{\pi} f(x)\sin(nx)\,\mathrm{d}x,\quad n=1,2,3,\cdots,
\]
则 $f$ 的 Fourier 级数
\[
\frac{a_0}{2} + \sum_{n=1}^{\infty} \bigl( a_n\cos(nx) + b_n\sin(nx) \bigr)
\]
在 $\mathbb{R}$ 上处处收敛，且其和函数 $S(x)$ 为
\[
S(x) = 
\begin{cases}
f(x), & \text{若 $f$ 在 $x$ 处连续},\\[6pt]
\dfrac{f(x^-) + f(x^+)}{2}, & \text{若 $x$ 是 $f$ 的间断点},
\end{cases}
\]
其中 $f(x^-)$、$f(x^+)$ 分别表示 $f$ 在 $x$ 处的左、右极限。特别地，若 $f$ 在 $[-\pi,\pi]$ 上连续且 $f(-\pi)=f(\pi)$，则 Fourier 级数在 $\mathbb{R}$ 上一致收敛到 $f$。
\end{theorem}

\begin{theorem}[Fourier级数的逐项积分定理]\label{theorem:Fourier级数的逐项积分定理}
设 $f$ 在 $[-\pi,\pi]$ 上Riemann可积（或绝对可积），其 Fourier 级数为
\[
f(x)\sim\frac{a_0}{2}+\sum_{n=1}^{\infty}(a_n\cos nx + b_n\sin nx).
\]
则对任意 $c,x\in[-\pi,\pi]$，有
\[
\int_{c}^{x}f(t)\,\mathrm{d}t
= \int_{c}^{x}\frac{a_0}{2}\,\mathrm{d}t
+ \sum_{n=1}^{\infty}\int_{c}^{x}\bigl(a_n\cos nt + b_n\sin nt\bigr)\,\mathrm{d}t,
\]
其中右边的级数在 $[-\pi,\pi]$ 上一致收敛。
\end{theorem}

\begin{theorem}[Fourier级数的逐项微分定理]\label{theorem:Fourier级数的逐项微分定理}
设 $f$ 在 $[-\pi,\pi]$ 上连续，且 $f(-\pi)=f(\pi)$。若 $f$ 在 $[-\pi,\pi]$ 上除有限个点外可导，且导函数 $f'$ 在 $[-\pi,\pi]$ 上Riemann可积。设 $f$ 的 Fourier级数为
\[
f(x)\sim\frac{a_0}{2}+\sum_{n=1}^{\infty}(a_n\cos nx + b_n\sin nx),
\]
则 $f'$ 的 Fourier 级数为
\[
f'(x)\sim \frac{\mathrm{d}}{\mathrm{d}x}\left(\frac{a_0}{2}\right)
+\sum_{n=1}^{\infty}\frac{\mathrm{d}}{\mathrm{d}x}\bigl(a_n\cos nx + b_n\sin nx\bigr)
= \sum_{n=1}^{\infty}\bigl(-a_n n\sin nx + b_n n\cos nx\bigr).
\]
\end{theorem}
\begin{remark}
当 $f$ 在 $[-\pi,\pi]$ 上除有限个点外可导时,$f'$在无定义的点可任意补充定义，不影响其可积性.
\end{remark}

\begin{corollary}
若三角级数
$$
\frac{a_0}{2}+\sum_{n=1}^{\infty}(a_n\cos nx + b_n\sin nx)
$$
是某个在 $[-\pi,\pi]$ 上Riemann可积函数的 Fourier 级数，则级数 $\sum_{n=1}^{\infty}\frac{b_n}{n}$ 收敛。
\end{corollary}

\begin{theorem}[Bessel不等式]\label{theorem:Bessel不等式}
设 $f$ 在 $[-\pi,\pi]$ 上平方可积,即 $f^2$是Riemann可积的，则$f$的Fourier 系数$\{a_n\}_{n=0}^{\infty},\{b_n\}_{n=0}^{\infty}$满足
\[
\frac{a_0^2}{2}+\sum_{n=1}^{\infty}(a_n^2 + b_n^2)\leqslant \frac{1}{\pi}\int_{-\pi}^{\pi}f^2(x)\,\mathrm{d}x.
\]
\end{theorem}
\begin{remark}
该不等式表明 Fourier 系数的平方和收敛。
\end{remark}

\begin{theorem}[Parseval恒等式]\label{theorem:Parseval恒等式}
设 $f$ 在 $[-\pi,\pi]$ 上平方可积,即 $f^2$是Riemann可积的，则$f$的Fourier 系数$\{a_n\}_{n=0}^{\infty},\{b_n\}_{n=0}^{\infty}$满足
\[
\frac{a_0^2}{2}+\sum_{n=1}^{\infty}(a_n^2 + b_n^2)=\frac{1}{\pi}\int_{-\pi}^{\pi}f^2(x)\,\mathrm{d}x.
\]
\end{theorem}

\begin{lemma}\label{lemma:Fourier级数与其导函数的系数关系}
设 $f$ 在 $[-\pi,\pi]$ 上连续可微，且 $f(-\pi)=f(\pi)$。记 $\{a_n\}_{n=0}^{\infty},\{b_n\}_{n=0}^{\infty}$ 为 $f$ 的 Fourier 系数，$\{a_n'\}_{n=0}^{\infty},\{b_n'\}_{n=0}^{\infty}$ 为 $f'$ 的 Fourier 系数。则
\[
a_0' = 0,\qquad a_n' = n b_n,\qquad b_n' = -n a_n\quad (n=1,2,\cdots).
\]
\end{lemma}
\begin{remark}
分部积分的条件，需要$f$的导函数$f'$在积分区域上连续．
\end{remark}
\begin{proof}

由于$f$为$[-\pi,\pi]$上的连续可微函数，因此$f'\in C([-\pi,\pi])$．又$f(\pi)=f(-\pi)$，故
\begin{align*}
a_{0}'&=\frac{1}{\pi}\int_{-\pi}^{\pi}f'(x)\mathrm{d}x=\frac{1}{\pi}f(x)\Big|_{-\pi}^{\pi}=0,\\
a_{n}'&=\frac{1}{\pi}\int_{-\pi}^{\pi}f'(x)\cos nx\mathrm{d}x=\frac{1}{\pi}f(x)\cos nx\Big|_{-\pi}^{\pi}+\frac{n}{\pi}\int_{-\pi}^{\pi}f(x)\sin nx\mathrm{d}x=nb_n(n=1,2,\cdots),\\
b_{n}'&=\frac{1}{\pi}\int_{-\pi}^{\pi}f'(x)\sin nx\mathrm{d}x=\frac{1}{\pi}f(x)\sin nx\Big|_{-\pi}^{\pi}-\frac{n}{\pi}\int_{-\pi}^{\pi}f(x)\cos nx\mathrm{d}x=-na_n(n=1,2,\cdots),
\end{align*}
因此结论得证．

\end{proof}

\begin{example}
设$f$以$2\pi$为周期且具有二阶连续的导函数,证明$f$的Fourier级数在$(-\infty,+\infty)$上一致收敛于$f$.
\end{example}
\begin{proof}

因为$f(x)$是以$2\pi$为周期的具有二阶连续导数的函数，故$f(x)$,$f'(x)$可展开成傅里叶级数，不妨设
\begin{align*}
f(x) &=\frac{a_0}{2}+\sum_{n=1}^{\infty}(a_n\cos nx+b_n\sin nx),\quad f'(x)=\frac{a_{0}'}{2}+\sum_{n=1}^{\infty}(a_{n}'\cos nx+b_{n}'\sin nx).
\end{align*}
先证$\frac{|a_0|}{2}+\sum_{n=1}^{\infty}(|a_n|+|b_n|)$收敛．由\reflem{lemma:Fourier级数与其导函数的系数关系}可知
$$a_{0}'=0,a_{n}'=nb_n,b_{n}'=-nb_n(n=1,2,\cdots),$$
从而
\begin{align}
|a_n|+|b_n|=\frac{|b_{n}'|}{n}+\frac{|a_{n}'|}{n}\leqslant \frac{1}{2}\left[(b_{n}')^2+\frac{1}{n^2}\right]+\frac{1}{2}\left[(a_{n}')^2+\frac{1}{n^2}\right]
=\frac{1}{n^2}+\frac{1}{2}\left[(a_{n}')^2+(b_{n}')^2\right].\label{100.103}
\end{align}
又由\hyperref[theorem:Bessel不等式]{Bessel不等式}可知
$$\frac{a_{0}'}{2}+\sum_{n=1}^{\infty}\left[(a_{n}')^2+(b_{n}')^2\right]\leqslant \frac{1}{\pi}\int_{-\pi}^{\pi}[f'(x)]^2\mathrm{d}x<+\infty.$$
故$\sum_{n=1}^{\infty}\left[(a_{n}')^2+(b_{n}')^2\right]$收敛．再结合$\sum_{n=1}^{\infty}\frac{1}{n^2}$收敛及\eqref{100.103}式可知$\sum_{n=1}^{\infty}(|a_n|+|b_n|)$收敛，进而$\frac{|a_0|}{2}+\sum_{n=1}^{\infty}(|a_n|+|b_n|)$收敛．注意到
$$\left|\frac{a_0}{2}+\sum_{n=1}^{\infty}(a_n\cos nx+b_n\sin nx)\right|\leqslant \frac{|a_0|}{2}+\sum_{n=1}^{\infty}(|a_n|+|b_n|),\forall x\in(-\infty,+\infty).$$
因此由Weierstrass判别法可知，$\frac{a_0}{2}+\sum_{n=1}^{\infty}(a_n\cos nx+b_n\sin nx)$在$(-\infty,+\infty)$上一致收敛，即$f$的Fourier级数在$(-\infty,+\infty)$上一致收敛于$f$．

\end{proof}
