\section{Euler积分}

\begin{definition}
我们将
\begin{align}
\Gamma(s) &= \int_{0}^{+\infty} x^{s-1} \mathrm{e}^{-x} \mathrm{d}x,\ s > 0, \label{eq::020290rj30jr0jwef-1} \\
\mathrm{B}(p,q) &= \int_{0}^{1} x^{p-1} (1-x)^{q-1} \mathrm{d}x,\ p > 0,\ q > 0 .\label{eq::020290rj30jr0jwef-2}
\end{align}
统称为$\mathbf{Euler}$(\textbf{欧拉})\textbf{积分},其中前者又称为$\mathbf{Gamma}$(\textbf{伽马})\textbf{函数}(或写作$\mathbf{Gamma}$\textbf{函数}),后者称为$\mathrm{Beta}$(\textbf{贝塔})\textbf{函数}(或写作$\mathrm{B}$\textbf{函数}).
\end{definition}

\begin{theorem}[$\Gamma$函数和$\mathrm{B}$函数的其他形式]\label{theorem:Euler积分的其他形式}
\begin{enumerate}[(1)]
\item \begin{align*}
\Gamma \left( s \right) &=\int_0^{+\infty}{x^{s-1}\mathrm{e}^{-x}\mathrm{d}x=2\int_0^{+\infty}{y^{2s-1}\mathrm{e}^{-y^2}\mathrm{d}y\quad \left( s>0 \right) }}.
\\
\Gamma \left( s \right) &=\int_0^{+\infty}{x^{s-1}\mathrm{e}^{-x}\mathrm{d}x}\xlongequal{x=py}p^s\int_0^{+\infty}{y^{s-1}\mathrm{e}^{-py}\mathrm{d}y}\quad \left( s>0,p>0 \right) .
\end{align*}

\item \begin{align}
\mathrm{B}\left( p,q \right) &=2\int_0^{\frac{\pi}{2}}{\left( \sin ^{2q-1}\varphi \right) \left( \cos ^{2p-1}\varphi \right) \mathrm{d}\varphi }.\label{eq::32r38u23u4239hjr2hjj3jr892jr29jr2hjr9923jr-1}
\\
\mathrm{B}\left( p,q \right) &=\int_0^{+\infty}{\frac{y^{p-1}}{\left( 1+y \right) ^{p+q}}\mathrm{d}y}.\label{eq::32r38u23u4239hjr2hjj3jr892jr29jr2hjr9923jr-2}
\\
\mathrm{B}\left( p,q \right) &=\int_0^1{\frac{y^{p-1}+y^{q-1}}{\left( 1+y \right) ^{p+q}}\mathrm{d}y}.\label{eq::32r38u23u4239hjr2hjj3jr892jr29jr2hjr9923jr-3}
\end{align}
\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item 在\eqref{eq::020290rj30jr0jwef-1}式中,分别令$x=y^2$,$\,x=py$即得.

\item 在\eqref{eq::020290rj30jr0jwef-2}式中,分别令$x=\cos^2\varphi$,$x=\frac{y}{1+y},1-x=\frac{1}{1+y},\mathrm{d}x = \frac{\mathrm{d}y}{(1+y)^2}$即得\eqref{eq::32r38u23u4239hjr2hjj3jr892jr29jr2hjr9923jr-1}\eqref{eq::32r38u23u4239hjr2hjj3jr892jr29jr2hjr9923jr-2}式.再在\eqref{eq::32r38u23u4239hjr2hjj3jr892jr29jr2hjr9923jr-2}式中,令$y=\frac{1}{t}$即得\eqref{eq::32r38u23u4239hjr2hjj3jr892jr29jr2hjr9923jr-3}式.
\end{enumerate}
\end{proof}


\subsection{$\Gamma$函数}

\begin{theorem}[$\Gamma$函数的性质]\label{theorem:Gamma函数的性质}
\begin{enumerate}[(1)]
\item  $\Gamma(s)$ 在定义在$s>0$上,且在定义域内连续,并且存在任意阶导数.对$\forall n\in \mathbb{N}$,有
\begin{align*}
\Gamma^{(n)}(s) = \int_{0}^{+\infty} x^{s-1} \mathrm{e}^{-x} (\ln x)^n \mathrm{d}x,\ s > 0.
\end{align*}

\item $\Gamma$函数的递推公式:
\begin{align}
\Gamma(s+1) = s\Gamma(s),\,\,\forall s\in \mathbb{R}.\label{eq::290-jriowuthj3f9ejfgs}
\end{align}
进而,若 $s$ 为正整数 $n+1$, 则
\begin{align}\label{eq:902uu9099999999}
\Gamma(n+1) = n(n-1)\cdots2\cdot1\cdot\Gamma(1) = n! \int_{0}^{+\infty} \mathrm{e}^{-x} \mathrm{d}x = n!. 
\end{align}

\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item $\Gamma$ 函数可写成如下两个积分之和:
\begin{align*}
\Gamma(s) = \int_{0}^{1} x^{s-1} \mathrm{e}^{-x} \mathrm{d}x + \int_{1}^{+\infty} x^{s-1} \mathrm{e}^{-x} \mathrm{d}x = \Phi(s) + \Psi(s),
\end{align*}
其中 $\Phi(s)$ 当 $s\geqslant 1$ 时是正常积分, 当 $0<s<1$ 时是收敛的无界函数反常积分(可用柯西判别法推得); $\Psi(s)$ 当 $s>0$ 时是收敛的无穷限反常积分(也可用柯西判别法推得). 所以$\Gamma$ 函数在 $s>0$ 时收敛, 即 $\Gamma$ 函数的定义域为 $s>0$.

在任何闭区间 $[a,b]$ ($a>0$) 上, 对于函数 $\Phi(s)$, 当 $0<x\leqslant 1$ 时有 $x^{s-1} \mathrm{e}^{-x} \leqslant x^{a-1} \mathrm{e}^{-x}$, 由于 $\int_{0}^{1} x^{a-1} \mathrm{e}^{-x} \mathrm{d}x$ 收敛, 从而 $\Phi(s)$ 在 $[a,b]$ 上一致收敛; 对于 $\Psi(s)$, 当 $1\leqslant x<+\infty$ 时, 有 $x^{s-1} \mathrm{e}^{-x} \leqslant x^{b-1} \mathrm{e}^{-x}$, 由于 $\int_{1}^{+\infty} x^{b-1} \mathrm{e}^{-x} \mathrm{d}x$ 收敛, 从而 $\Psi(s)$ 在 $[a,b]$ 上也一致收敛. 于是 $\Gamma(s)$ 在 $s>0$ 上连续.

用上述相同的方法考察积分
\begin{align*}
\int_{0}^{+\infty} \frac{\partial}{\partial s}\left(x^{s-1} \mathrm{e}^{-x}\right) \mathrm{d}x = \int_{0}^{+\infty} x^{s-1} \mathrm{e}^{-x} \ln x \mathrm{d}x.
\end{align*}
它在任何闭区间 $[a,b]$ ($a>0$) 上一致收敛. 于是由\refthe{theorem:含参反常积分的可微性}得到 $\Gamma(s)$ 在 $[a,b]$ 上可导, 由 $a,b$ 的任意性, $\Gamma(s)$ 在 $s>0$ 上可导, 且
\begin{align*}
\Gamma'(s) = \int_{0}^{+\infty} x^{s-1} \mathrm{e}^{-x} \ln x \mathrm{d}x,\ s > 0.
\end{align*}
仿照上面的办法, 还可推得 $\Gamma(s)$ 在 $s>0$ 上存在任意阶导数:
\begin{align*}
\Gamma^{(n)}(s) = \int_{0}^{+\infty} x^{s-1} \mathrm{e}^{-x} (\ln x)^n \mathrm{d}x,\ s > 0.
\end{align*}

\item 对下述积分应用分部积分法,有
\begin{align*}
\int_{0}^{A} x^s \mathrm{e}^{-x} \mathrm{d}x &= -x^s \mathrm{e}^{-x} \bigg|_{0}^{A} + s\int_{0}^{A} x^{s-1} \mathrm{e}^{-x} \mathrm{d}x \\
&= -A^s \mathrm{e}^{-A} + s\int_{0}^{A} x^{s-1} \mathrm{e}^{-x} \mathrm{d}x.
\end{align*}
让 $A\to+\infty$ 就得到 $\Gamma$ 函数的递推公式:
\begin{align}
\Gamma(s+1) = s\Gamma(s). \label{eq:gamma-recurrence}
\end{align}
设 $n < s \leqslant n+1$, 即 $0 < s-n \leqslant 1$, 应用递推公式 \eqref{eq:gamma-recurrence} $n$ 次可得到
\begin{align}
\begin{split}
\Gamma(s+1) &= s\Gamma(s) = s(s-1)\Gamma(s-1) = \dots \\
&= s(s-1)\cdots(s-n)\Gamma(s-n). \label{eq:gamma-recurrence-n}
\end{split}
\end{align}
公式 \eqref{eq:gamma-recurrence} 还指出, 如果已知 $\Gamma(s)$ 在 $0<s\leqslant 1$ 上的值, 那么在其他范围内的函数值可由它计算出来.

若 $s$ 为正整数 $n+1$, 则 \eqref{eq:gamma-recurrence-n} 式可写成
\begin{align*}
\Gamma(n+1) = n(n-1)\cdots2\cdot1\cdot\Gamma(1) = n! \int_{0}^{+\infty} \mathrm{e}^{-x} \mathrm{d}x = n!. 
\end{align*}
\end{enumerate}

\end{proof}

\begin{theorem}[$\mathbf{\Gamma }$函数的图像和延拓]\label{theorem:Gamma函数的图像和延拓}
改写递推公式\eqref{eq::290-jriowuthj3f9ejfgs}为
\begin{align}
\Gamma(s) = \frac{\Gamma(s+1)}{s}. \label{eq:gamma-extension}
\end{align}
当 $-1<s<0$ 时, \eqref{eq:gamma-extension} 式右端有意义, 于是可应用 \eqref{eq:gamma-extension} 式来定义左端函数 $\Gamma(s)$ 在 $(-1,0)$ 上的值, 并且可推得这时 $\Gamma(s)<0$.

用同样的方法, 利用 $\Gamma(s)$ 已在 $(-1,0)$ 上有定义这一事实, 由 \eqref{eq:gamma-extension} 式又可定义 $\Gamma(s)$ 在 $(-2,-1)$ 上的值, 而且这时 $\Gamma(s)>0$. 依此下去可把 $\Gamma(s)$ 延拓到整个数轴(除了 $s=0,-1,-2,\dots$ 以外), 其图像如\reffig{figure:Gamma函数的图像}所示.
\begin{figure}[H]
\centering
\includegraphics[scale=0.3]{Gamma函数的图像.png}
\caption{Gamma函数的图像}
\label{figure:Gamma函数的图像}
\end{figure}
\end{theorem}
\begin{remark}
对一切 $s>0$, $\Gamma(s)$ 和 $\Gamma''(s)$ 恒大于 $0$, 因此 $\Gamma(s)$ 的图形位于 $s$ 轴上方, 且是严格凸的. 因为 $\Gamma(1)=\Gamma(2)=1$, 所以 $\Gamma(s)$ 在 $s>0$ 上存在惟一的极小点 $x_0$ 且 $x_0\in(1,2)$. 又 $\Gamma(s)$ 在 $(0,x_0)$ 上严格减, 在 $(x_0,+\infty)$ 上严格增.

由于 $\Gamma(s)=\frac{s\Gamma(s)}{s}=\frac{\Gamma(s+1)}{s}\ (s>0)$ 及 $\lim\limits_{s\to0^+}\Gamma(s+1)=\Gamma(1)=1$, 故有
\begin{align*}
\lim\limits_{s\to0^+}\Gamma(s) = \lim\limits_{s\to0^+}\frac{\Gamma(s+1)}{s} = +\infty.
\end{align*}
由\eqref{eq:902uu9099999999}式及 $\Gamma(s)$ 在 $(x_0,+\infty)$ 上严格增可推得
\begin{align*}
\lim\limits_{s\to+\infty}\Gamma(s) = +\infty.
\end{align*}
综上所述, $\Gamma$ 函数的图像如\reffig{figure:Gamma函数的图像}中 $s>0$ 部分所示.
\end{remark}

\begin{proposition}\label{proposition:分式化积分-Gamma函数相关性质}
对$\forall s>0$,都有
\begin{align}\label{euqation:将分式化成积分形式}
\frac{1}{t^s}=\frac{1}{\Gamma \left( s \right)}\int_0^{+\infty}{y^{s-1}e^{-ty}\mathrm{d}y},\forall t\in \mathbb{R}.
\end{align}
\end{proposition}
\begin{proof}
已知$\Gamma $函数:\[\Gamma \left( s \right) =\int_0^{+\infty}{x^{s-1}e^{-x}\mathrm{d}x},\quad s>0.\]
令$x=ty$，其中$t\in \mathbb{R}$，得
\begin{align*}
\Gamma \left( s \right) &= t^s\int_0^{+\infty}{y^{s-1}e^{-ty}\mathrm{d}y} .
\end{align*}
故
\begin{align*}
\frac{1}{t^s}=\frac{1}{\Gamma \left( s \right)}\int_0^{+\infty}{y^{s-1}e^{-ty}\mathrm{d}y}.
\end{align*}

\end{proof}


\subsection{$\mathrm{B}$函数}

\begin{theorem}[$\mathrm{B}$函数的性质]\label{theorem:Beta函数的性质}
\begin{enumerate}[(1)]
\item  $\mathrm{B}(p,q)$ 在定义在$p>0,q>0$上,且在定义域内连续.

\item 对称性:$\mathrm{B}(p,q) = \mathrm{B}(q,p)$.

\item $\mathrm{B}$函数的递推公式:
\begin{align}
\mathrm{B}(p,q) &= \frac{q-1}{p+q-1}\mathrm{B}(p,q-1)\quad (p > 0,\ q > 1), \label{eq:B-recurrence-1} \\
\mathrm{B}(p,q) &= \frac{p-1}{p+q-1}\mathrm{B}(p-1,q)\quad (p > 1,\ q > 0), \label{eq:B-recurrence-2} \\
\mathrm{B}(p,q) &= \frac{(p-1)(q-1)}{(p+q-1)(p+q-2)}\mathrm{B}(p-1,q-1)\quad (p > 1,\ q > 1). \label{eq:B-recurrence-3}
\end{align}
\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item $\mathrm{B}$函数当 $p<1$ 时,是以 $x=0$ 为瑕点的无界函数反常积分;当 $q<1$ 时,是以 $x=1$ 为瑕点的无界函数反常积分.应用柯西判别法可证得当 $p>0,q>0$ 时这两个无界函数反常积分都收敛,所以函数 $\mathrm{B}(p,q)$ 的定义域为 $p>0,q>0$.

由于对任何 $p_0>0,q_0>0$ 成立不等式
$x^{p-1}(1-x)^{q-1} \leqslant x^{p_0-1}(1-x)^{q_0-1},\ p \geqslant p_0,\ q \geqslant q_0,$
而积分 $\int_{0}^{1} x^{p_0-1}(1-x)^{q_0-1}\mathrm{d}x$ 收敛,故由魏尔斯特拉斯 $M$ 判别法知 $\mathrm{B}(p,q)$ 在 $p_0\leqslant p<+\infty,\ q_0\leqslant q<+\infty$ 上一致收敛.因而推得 $\mathrm{B}(p,q)$ 在 $p>0,q>0$ 上连续.

\item 作变换 $x=1-y$,得
\begin{align*}
\mathrm{B}(p,q) &= \int_{0}^{1} x^{p-1}(1-x)^{q-1}\mathrm{d}x = \int_{0}^{1} (1-y)^{p-1}y^{q-1}\mathrm{d}y = \mathrm{B}(q,p).
\end{align*}

\item 下面只证公式\eqref{eq:B-recurrence-1},公式\eqref{eq:B-recurrence-2}可由对称性及公式\eqref{eq:B-recurrence-1}推得,而最后一个公式则可由公式\eqref{eq:B-recurrence-1},\eqref{eq:B-recurrence-2}推得.

当 $p>0,q>1$ 时,有
\begin{align*}
\mathrm{B}(p,q) &= \int_{0}^{1} x^{p-1}(1-x)^{q-1}\mathrm{d}x \\
&= \left.\frac{x^p(1-x)^{q-1}}{p}\right|_{0}^{1} + \frac{q-1}{p}\int_{0}^{1} x^p(1-x)^{q-2}\mathrm{d}x \\
&= \frac{q-1}{p}\int_{0}^{1} \left[x^{p-1} - x^{p-1}(1-x)\right](1-x)^{q-2}\mathrm{d}x \\
&= \frac{q-1}{p}\int_{0}^{1} x^{p-1}(1-x)^{q-2}\mathrm{d}x - \frac{q-1}{p}\int_{0}^{1} x^{p-1}(1-x)^{q-1}\mathrm{d}x \\
&= \frac{q-1}{p}\mathrm{B}(p,q-1) - \frac{q-1}{p}\mathrm{B}(p,q).
\end{align*}
移项并整理就得\eqref{eq:B-recurrence-1}.
\end{enumerate}

\end{proof}






\subsection{$\Gamma$函数和$\mathrm{B}$函数的联系}

\begin{theorem}[$\Gamma$函数和$\mathrm{B}$函数的联系]\label{theorem:Euler积分之间的联系}
\begin{enumerate}[(1)]
\item 对$\forall n,m\in \mathbb{N}$,有
\begin{align*}
\mathrm{B}(m,n) = \frac{\Gamma(n)\Gamma(m)}{\Gamma(n+m)}. 
\end{align*}

\item 对$\forall p,q>0$,也有相同的关系:
\begin{align}
\mathrm{B}(p,q) = \frac{\Gamma(p)\Gamma(q)}{\Gamma(p+q)}. \label{eq:B-Gamma}
\end{align}
\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item 当 $m,n$ 为正整数时,反复应用 B 函数的递推公式\eqref{eq:B-recurrence-1}可得
\begin{align*}
\mathrm{B}(m,n) = \frac{n-1}{m+n-1}\mathrm{B}(m,n-1) = \frac{n-1}{m+n-1}\frac{n-2}{m+n-2}\cdots\frac{1}{m+1}\mathrm{B}(m,1).
\end{align*}
又由于 $\mathrm{B}(m,1) = \int_{0}^{1} x^{m-1}\mathrm{d}x = \frac{1}{m}$,所以
\begin{align*}
\mathrm{B}(m,n) = \frac{n-1}{m+n-1}\frac{n-2}{m+n-2}\cdots\frac{1}{m+1}\frac{1}{m}\cdot\frac{(m-1)!}{(m-1)!} 
= \frac{(n-1)!\,(m-1)!}{(m+n-1)!},
\end{align*}
即
\begin{align}
\mathrm{B}(m,n) = \frac{\Gamma(n)\Gamma(m)}{\Gamma(n+m)}. \label{eq:B-Gamma-int}
\end{align}

\item 对于 $\Gamma$ 函数,令 $x=u^2$,则 $\mathrm{d}x=2u\mathrm{d}u$,于是
\begin{align*}
\Gamma(p) = \int_{0}^{+\infty} x^{p-1}\mathrm{e}^{-x}\mathrm{d}x = 2\int_{0}^{+\infty} u^{2p-1}\mathrm{e}^{-u^2}\mathrm{d}u.
\end{align*}
从而
\begin{align*}
\Gamma(p)\Gamma(q) = 4\int_{0}^{+\infty} x^{2p-1}\mathrm{e}^{-x^2}\mathrm{d}x \cdot \int_{0}^{+\infty} y^{2q-1}\mathrm{e}^{-y^2}\mathrm{d}y 
= \lim_{R\to+\infty} 4\int_{0}^{R} x^{2p-1}\mathrm{e}^{-x^2}\mathrm{d}x \cdot \int_{0}^{R} y^{2q-1}\mathrm{e}^{-y^2}\mathrm{d}y.
\end{align*}
令 $D_R=[0,R]\times[0,R]$,由二重积分化为累次积分计算公式有
\begin{align*}
\iint_{D_R} x^{2p-1}y^{2q-1}\mathrm{e}^{-(x^2+y^2)}\mathrm{d}\sigma
= \int_{0}^{R} x^{2p-1}\mathrm{e}^{-x^2}\mathrm{d}x \cdot \int_{0}^{R} y^{2q-1}\mathrm{e}^{-y^2}\mathrm{d}y.
\end{align*}
所以
\begin{align}
\begin{split}
\Gamma(p)\Gamma(q) = \lim_{R\to+\infty} 4\iint_{D_R} x^{2p-1}y^{2q-1}\mathrm{e}^{-(x^2+y^2)}\mathrm{d}\sigma 
= 4\iint_{D} x^{2p-1}y^{2q-1}\mathrm{e}^{-(x^2+y^2)}\mathrm{d}\sigma,
\end{split}
\label{eq:Gamma-product-double}
\end{align}
这里 $D$ 为平面上第一象限部分.下面讨论\eqref{eq:Gamma-product-double}式右边的反常二重积分. 记
$D_r = \{(x,y)\mid x^2+y^2 \leqslant r^2, x \geqslant 0, y \geqslant 0\}$.
于是有
\begin{align*}
\Gamma(p)\Gamma(q) = 4\iint_{D} x^{2p-1}y^{2q-1}\mathrm{e}^{-(x^2+y^2)}\mathrm{d}\sigma 
= \lim_{r\to+\infty} 4\iint_{D_r} x^{2p-1}y^{2q-1}\mathrm{e}^{-(x^2+y^2)}\mathrm{d}\sigma.
\end{align*}
对上式右边积分应用极坐标变换,则可得
\begin{align*}
\Gamma(p)\Gamma(q) &= \lim_{r\to+\infty} 4\int_{0}^{\frac{\pi}{2}} \mathrm{d}\theta \int_{0}^{r} r^{2(p+q)-2}(\cos\theta)^{2p-1}(\sin\theta)^{2q-1}\mathrm{e}^{-r^2}r\mathrm{d}r \\
&= \lim_{r\to+\infty} 2\int_{0}^{\frac{\pi}{2}} (\cos\theta)^{2p-1}(\sin\theta)^{2q-1}\mathrm{d}\theta \cdot 2\int_{0}^{r} r^{2(p+q)-1}\mathrm{e}^{-r^2}\mathrm{d}r \\
&= 2\int_{0}^{\frac{\pi}{2}} (\cos\theta)^{2p-1}(\sin\theta)^{2q-1}\mathrm{d}\theta \cdot \Gamma(p+q).
\end{align*}
再由\eqref{eq::32r38u23u4239hjr2hjj3jr892jr29jr2hjr9923jr-1}式就得到
\begin{align*}
\Gamma(p)\Gamma(q) = \mathrm{B}(p,q)\Gamma(p+q).
\end{align*}
\end{enumerate}

\end{proof}
