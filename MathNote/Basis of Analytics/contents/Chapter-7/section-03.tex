\section{基本组合学公式}

\begin{theorem}[二项式定理的推广]\label{theorem:二项式定理的推广}
$\left( a_1+b_1 \right)\cdots \left( a_n+b_n \right) =\sum_{I\subset \left\{ 1,2,\cdots ,n \right\}}{\left( \prod_{i\in I}{a_i}\prod_{j\in \left\{ 1,2,\cdots ,n \right\} -I}{b_j} \right)}.$
\end{theorem}
\begin{proof}
用数学归纳法证明即可.
\end{proof}
