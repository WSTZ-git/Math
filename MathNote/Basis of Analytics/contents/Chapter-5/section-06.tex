\section{变分法}

\begin{example}
设$f\in C[a,b]$,且对$\forall g\in C^1[a,b]$,若$g(a)=g(b)=g'(a)=g'(b)=0$,则
\begin{align*}
\int_a^b f(x)g'(x)\mathrm{d}x=0.
\end{align*}
对$\forall \psi\in C[a,b]$,若$\int_a^b \psi(x)\mathrm{d}x=0$,则
\begin{align*}
\int_a^b f(x)\psi(x)\mathrm{d}x=0.
\end{align*}
\end{example}
\begin{proof}
令
\begin{align*}
g(x)=\int_a^x (x-a)(b-x)\left(f(x)-C\right)\mathrm{d}x,
\end{align*}
其中
\begin{align*}
C=\frac{\int_a^b (x-a)(b-x)f(x)\mathrm{d}x}{\int_a^b (x-a)(b-x)\mathrm{d}x}.
\end{align*}
容易验证$g(a)=g(b)=g'(a)=g'(b)=0$.从而由题设知
\begin{align*}
0&=\int_a^b f(x)g'(x)\mathrm{d}x=\int_a^b (x-a)(b-x)\left(f(x)-C\right)f(x)\mathrm{d}x\\
&=\int_a^b (x-a)(b-x)\left(f(x)-C\right)^2\mathrm{d}x+C\int_a^b (x-a)(b-x)\left(f(x)-C\right)\mathrm{d}x\\
&=\int_a^b (x-a)(b-x)\left(f(x)-C\right)^2\mathrm{d}x.
\end{align*}
因为$(x-a)(b-x)\left(f(x)-C\right)^2\in C[a,b]$且$(x-a)(b-x)\left(f(x)-C\right)^2\geqslant 0,\forall x\in [a,b]$.所以
\begin{align*}
(x-a)(b-x)\left(f(x)-C\right)^2=0,\,\,\forall x\in [a,b] \Longrightarrow f(x)=C,\,\,\forall x\in (a,b).
\end{align*}
再结合$f$的连续性知$f(x)=C,\forall x\in [a,b]$.故对$\forall \psi\in C[a,b]$,若$\int_a^b \psi(x)\mathrm{d}x=0$,则
\begin{align*}
\int_a^b f(x)\psi(x)\mathrm{d}x=C\int_a^b \psi(x)\mathrm{d}x=0.
\end{align*}

\end{proof}
