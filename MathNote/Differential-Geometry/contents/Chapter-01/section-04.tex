\section{曲线的挠率和Frenet公式}

\begin{definition}
设$\beta(s)$和$\gamma(s)$分别是正则曲线$C$的主法向量和次法向量，其中$s$是曲线的弧长参数，则$\tau(s) = -\gamma'(s) \cdot \beta(s)$称为曲线$C$的\textbf{挠率}.
\end{definition}
\begin{remark}
\begin{enumerate}
\item 根据\refthe{theorem:微分几何-定理3.1}，单位切向量$\alpha$关于弧长参数$s$的导数的长度$|\alpha'(s)|$反映了曲线的切线方向转动的快慢；

\item 由\refpro{proposition:挠率的基本性质},次法向量$\gamma$关于弧长参数$s$的导数的长度$|\gamma'(s)|$反映了曲线的密切平面方向转动的快慢，因而它刻画了曲线偏离平面曲线的程度，反映了曲线扭曲的程度，即曲线的“挠率”。
\end{enumerate}

\end{remark}

\begin{proposition}\label{proposition:挠率的基本性质}
设$\bm{\alpha}(s)$、$\beta(s)$和$\gamma(s)$分别是正则曲线$C$的单位切向量场、主法向量场和次法向量场，$\tau(s)$为曲线$C$的挠率,其中$s$是曲线的弧长参数.则
\begin{align*}
\boldsymbol{\gamma}'\left( s \right) \text{与}\boldsymbol{\beta }\left( s \right) \text{共线},\text{且}|\tau(s)| = |\bm{\gamma}'(s)|. 
\end{align*}
进而
\begin{align*}
\bm{\gamma}'(s)=-\tau(s)\cdot \bm{\beta}(s),\quad \bm{\beta}(s)=-\tau(s)\cdot \bm{\gamma}'(s).
\end{align*}
\end{proposition}
\begin{proof}

因为$\gamma(s)$是单位向量场，所以$\gamma^2(s)=1$,再求导可得$\gamma(s)\gamma'(s)=0$,故$\gamma'(s) \perp \gamma(s)$。此外，
\begin{align*}
\gamma(s) = \alpha(s) \times \beta(s), \quad \alpha'(s) \varparallel  \beta(s),
\end{align*}
所以
\begin{align*}
\gamma'(s) = \alpha'(s) \times \beta(s) + \alpha(s) \times \beta'(s) = \alpha(s) \times \beta'(s).
\end{align*}
这说明$\gamma'(s) \perp \alpha(s)$。于是$\gamma'(s)$必定是与$\beta(s)$共线的，不妨设
\begin{align*}
\gamma'(s) = -\tau \beta(s),
\end{align*}
因此
\begin{align*}
\tau(s) = -\gamma'(s) \cdot \beta(s), 
\end{align*}
并且
\begin{align*}
|\tau(s)| = |\gamma'(s)|. 
\end{align*}

\end{proof}

\begin{theorem}\label{theorem:挠率为0等价于是平面曲线}
设正则曲线$C$不是直线，即曲率不是常数,则它是平面曲线当且仅当它的挠率为零。
\end{theorem}
\begin{remark}
实际上,若曲线$C$是直线,则必然也是平面曲线.
\end{remark}
\begin{proof}

{\heiti 必要性:}若曲线$C$为平面曲线，则曲线整体落在固定平面上，该平面即为曲线的密切平面，对应的次法向量$\bm{\gamma}(s)$为常向量。
而常向量关于弧长参数的导数满足$\bm{\gamma}'(s) \equiv \bm{0}$，代入挠率定义式$\tau(s) = -\bm{\gamma}'(s) \cdot \bm{\beta}(s)$，直接可得$\tau(s) \equiv 0$。

{\heiti 充分性:}设曲线$C$的参数方程是$\bm{r} = \bm{r}(s)$，$s$是弧长参数，并且$\kappa(s) \neq 0, \tau(s) \equiv 0$。此时，曲线有确定的Frenet标架$\{\bm{r}(s); \alpha(s), \beta(s), \gamma(s)\}$，并且
\begin{align*}
\gamma'(s) = -\tau(s) \cdot \beta(s) \equiv 0,
\end{align*}
因此$\gamma(s) = \gamma_0 $是常向量。但是
\begin{align*}
0 = \bm{r}'(s) \cdot \gamma(s) = \bm{r}'(s) \cdot \gamma_0 = \frac{\mathrm{d}}{\mathrm{d}s} (\bm{r}(s) \cdot \gamma_0),
\end{align*}
所以$\bm{r}(s) \cdot \gamma_0 = \bm{r}(s_0) \cdot \gamma_0,$进而
\begin{align*}
(\bm{r}(s) - \bm{r}(s_0)) \cdot \gamma_0 = 0.
\end{align*}
这说明曲线$\bm{r} = \bm{r}(s)$落在经过点$\bm{r}(s_0)$、以常向量$\gamma_0$为法向量的平面内.

\end{proof}

\begin{theorem}[Frenet 公式]\label{theorem:微分几何-Frenet公式}
设正则曲线 $C$的参数方程为$\bm{r}(s)$，其中$s$是弧长参数.其Frenet标架为 $\{\bm{r}(s); \bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s)\}$，其中 $\bm{\alpha}(s)=\bm{r}'(s)$ 是单位切向量，$\bm{\beta}(s)$ 是主法向量，$\bm{\gamma}(s)$ 是副法向量。$\kappa(s)$ 和 $\tau(s)$ 分别为曲线$C$的曲率和挠率.则曲线$C$的Frenet标架满足如下微分方程组：
\begin{align}\label{eq::Frenet公式}
\begin{cases}
\boldsymbol{r}' (s)=\boldsymbol{\alpha }(s),\\
\boldsymbol{\alpha }' (s)=\kappa (s)\boldsymbol{\beta }(s),\\
\boldsymbol{\beta }' (s)=-\kappa (s)\boldsymbol{\alpha }(s)+\tau (s)\boldsymbol{\gamma }(s),\\
\boldsymbol{\gamma }' (s)=-\tau (s)\boldsymbol{\beta }(s).\\
\end{cases}
\end{align}
上述公式称为$\mathbf{Frenet公式}$,也可写作矩阵形式
\begin{align*}
\begin{pmatrix}
\bm{\alpha}'(s) \\
\bm{\beta}'(s) \\
\bm{\gamma}'(s)
\end{pmatrix}
=
\begin{pmatrix}
0 & \kappa(s) & 0 \\
-\kappa(s) & 0 & \tau(s) \\
0 & -\tau(s) & 0
\end{pmatrix}
\cdot
\begin{pmatrix}
\bm{\alpha}(s) \\
\bm{\beta}(s) \\
\bm{\gamma}(s)
\end{pmatrix}. 
\end{align*}
\end{theorem}
\begin{remark}
沿曲线定义的任意一个单位正交标架场的导数公式的系数矩阵都是反对称矩阵。Frenet 公式是曲线论中最重要、最基本的公式，由法国数学家 Serret 和 Frenet 分别在 1851 年和 1852 年发表（他们给出的是切线、主法线、次法线的方向余弦的导数公式）。
\end{remark}
\begin{proof}

由已知条件，$\{\bm{r}(s); \bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s)\}$ 是空间 $E^3$ 的一个单位正交标架，因此 $\bm{\beta}'(s)$ 可表示为 $\bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s)$ 的线性组合。设
\begin{align}
\bm{\beta}'(s) = a \bm{\alpha}(s) + b \bm{\beta}(s) + c \bm{\gamma}(s). \label{eq:::0-2fjw89jf903wdeghr5hfrt4.632dsgagds23141342}
\end{align}
分别与 $\bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s)$ 作点乘，再结合$\bm{\beta}(s)\cdot\bm{\alpha}(s)=\bm{\beta}(s)\cdot \bm{\gamma}(s)=0$求导后仍为0,以及单位正交性可得
\begin{align*}
a &= \bm{\beta}'(s)\cdot\bm{\alpha}(s) = -\bm{\beta}(s)\cdot\bm{\alpha}'(s) = -\kappa(s), \\
b &= \bm{\beta}'(s)\cdot\bm{\beta}(s) = 0, \\
c &= \bm{\beta}'(s)\cdot\bm{\gamma}(s) = -\bm{\beta}(s)\cdot\bm{\gamma}'(s) = \tau(s).
\end{align*}
代入 \eqref{eq:::0-2fjw89jf903wdeghr5hfrt4.632dsgagds23141342} 即得\eqref{eq::Frenet公式}式中的第三个公式。又由\refdef{definition:Frenet标架的定义}和\refpro{proposition:挠率的基本性质}可知,$\bm{r}'(s)=\bm{\alpha}(s)$、$\bm{\alpha}'(s)=\kappa(s)\bm{\beta}(s)$和$\bm{\gamma}'(s)=-\tau(s)\bm{\beta}(s)$,这样就得到完整的Frenet公式.
\end{proof}

\begin{theorem}\label{theorem:曲线落在球面上的性质}
\begin{enumerate}[(1)]
\item 设正则曲线$\bm{r} = \bm{r}(s)$的曲率$\kappa(s)$和挠率$\tau(s)$都不为零，$s$是弧长参数。如果该曲线落在一个球面上，则它的曲率和挠率必满足关系式
\begin{align*}
\left( \frac{1}{\kappa(s)} \right)^2 + \left( \frac{1}{\tau(s)} \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\kappa(s)} \right) \right)^2 = C,
\end{align*}
其中$C$为一个常数.并且
\begin{align*}
\frac{\tau(s)}{\kappa(s)} + \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\tau(s)} \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\kappa(s)} \right) \right) = 0.
\end{align*}

\item 设正则曲线$\bm{r} = \bm{r}(s)$的曲率$\kappa(s)$和挠率$\tau(s)$都不为零，$s$是弧长参数.并且$\kappa(s)$不是常数,则该曲线落在一个球面上的充要条件是它的曲率和挠率满足关系式
\begin{align*}
\left( \frac{1}{\kappa(s)} \right)^2 + \left( \frac{1}{\tau(s)} \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\kappa(s)} \right) \right)^2 = C,
\end{align*}
其中$C$为一个常数.
\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}[(1)]
\item 设曲线的Frenet标架为 $\{\bm{r}(s); \bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s)\}$，其中 $\bm{\alpha}(s)=\bm{r}'(s)$ 是单位切向量，$\bm{\beta}(s)$ 是主法向量，$\bm{\gamma}(s)$ 是副法向量。假定曲线$\bm{r} = \bm{r}(s)$落在一个球面上，该球面的球心是$\bm{r}_0$，半径是$a$，则有关系式
\begin{align}
(\bm{r}(s) - \bm{r}_0)^2 = a^2. \label{eq:::0-2fjw89jf903wdeghr5hfrt4.11}
\end{align}
将上式两边对于$s$求导，得到
\begin{align*}
\alpha(s) \cdot (\bm{r}(s) - \bm{r}_0) = 0,
\end{align*}
故$(\bm{r}(s) - \bm{r}_0)$落在$\bm{\alpha}(s)$的二维正交补空间$\mathrm{span}\left\{ \boldsymbol{\beta }\left( s \right) ,\boldsymbol{\gamma }\left( s \right) \right\}$里。不妨设
\begin{align}
\bm{r}(s) - \bm{r}_0 = \lambda(s) \beta(s) + \mu(s) \gamma(s). \label{eq:::0-2fjw89jf903wdeghr5hfrt4.12}
\end{align}
将上式对于$s$求导并且利用Frenet公式得到
\begin{align*}
\alpha(s) = -\lambda(s) \kappa(s) \alpha(s) + (\lambda'(s) - \mu(s) \tau(s)) \beta(s) + (\lambda(s) \tau(s) + \mu'(s)) \gamma(s).
\end{align*}
因此比较等式两边的系数得到
\begin{align}\label{eq::209rj29j8wur3i0r14}
\lambda(s) \kappa(s) = -1,\quad \lambda'(s) = \mu(s) \tau(s),\quad \mu'(s) = -\lambda(s) \tau(s),
\end{align}
于是
\begin{align}
\lambda(s) = -\frac{1}{\kappa(s)},\quad \mu(s) = \frac{\lambda'(s)}{\tau(s)} = -\frac{1}{\tau(s)} \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\kappa(s)} \right).
\label{eq:::0-2fjw89jf903wdeghr5hfrt4.14}
\end{align}
将\eqref{eq:::0-2fjw89jf903wdeghr5hfrt4.14}式代入\eqref{eq:::0-2fjw89jf903wdeghr5hfrt4.12}式得到
\begin{align*}
\bm{r}(s) - \bm{r}_0 = -\frac{1}{\kappa(s)} \beta(s) - \frac{1}{\tau(s)} \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\kappa(s)} \right) \gamma(s).
\end{align*}
因此根据关系式\eqref{eq:::0-2fjw89jf903wdeghr5hfrt4.11}和单位正交性得到
\begin{align*}
\left( \frac{1}{\kappa(s)} \right)^2 + \left( \frac{1}{\tau(s)} \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\kappa(s)} \right) \right)^2 = a^2.
\end{align*}
将\eqref{eq:::0-2fjw89jf903wdeghr5hfrt4.14}式代入\eqref{eq::209rj29j8wur3i0r14}的第三式便得到
\begin{align*}
\frac{\tau(s)}{\kappa(s)} + \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\tau(s)} \frac{\mathrm{d}}{\mathrm{d}s} \left( \frac{1}{\kappa(s)} \right) \right) = 0.
\end{align*}

\item 必要性由(1)立得.下面证明充分性.
设曲线的Frenet标架为 $\{\bm{r}(s); \bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s)\}$，其中 $\bm{\alpha}(s)=\bm{r}'(s)$ 是单位切向量，$\bm{\beta}(s)$ 是主法向量，$\bm{\gamma}(s)$ 是副法向量.记$f(s)=\bm{r}(s)+\frac{1}{\kappa(s)}\bm{\beta}(s)+\frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)\bm{\gamma}(s)$，则由条件和\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}可知
\begin{align*}
&\frac{\mathrm{d}}{\mathrm{d}s}\left( \left( \frac{1}{\kappa(s)} \right)^2+\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right)^2 \right)=0\\
\Longleftrightarrow&\frac{2}{\kappa(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)+\frac{2}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)\cdot \frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right)=0\\
\Longleftrightarrow&\frac{1}{\kappa(s)}+\frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right)=0\\
\Longleftrightarrow&\frac{\tau(s)}{\kappa(s)}+\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right)=0.
\end{align*}
于是再利用\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}可得
\begin{align*}
f'(s)&=\bm{r}'(s)+\bm{\beta}(s)\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)+\frac{1}{\kappa(s)}\bm{\beta}'(s)+\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right)\bm{\gamma}(s)+\frac{\bm{\gamma}'(s)}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)\\
&=\bm{\alpha}(s)+\bm{\beta}(s)\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)+\frac{1}{\kappa(s)}\left[ -\kappa(s)\bm{\alpha}(s)+\tau(s)\bm{\gamma}(s) \right]+\bm{\gamma}(s)\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right)-\bm{\beta}(s)\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)\\
&=\bm{\gamma}(s)\left[ \frac{\tau(s)}{\kappa(s)}+\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right) \right]=0.
\end{align*}
故存在$\bm{r}_0\in\mathbb{R}^3$，使得
\begin{align*}
\bm{r}_0=f(s)=\bm{r}(s)+\frac{1}{\kappa(s)}\bm{\beta}(s)+\frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)\bm{\gamma}(s).
\end{align*}
再利用$\bm{\beta}(s),\bm{\gamma}(s)$的单位正交性可得
\begin{align*}
|\bm{r}(s)-\bm{r}_0|^2=\left| \frac{1}{\kappa(s)}\bm{\beta}(s)+\frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right)\bm{\gamma}(s) \right|^2=\left( \frac{1}{\kappa(s)} \right)^2+\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right)^2=C.
\end{align*}
因此该曲线落在一个球面上.
\end{enumerate}

\end{proof}

\begin{theorem}[曲线挠率的计算公式]\label{theorem:微分几何-曲线挠率的计算公式}
\begin{enumerate}
\item 设曲线 $\bm{r}=\bm{r}(s)$ 是正则参数曲线，其中$s$为弧长参数.其曲率 $\kappa(s)\neq 0$，则其挠率为
\begin{align}
\tau (s)=\frac{(\boldsymbol{r}\prime (s),\boldsymbol{r}'' (s),\boldsymbol{r}''' (s))}{|\boldsymbol{r}'' (s)|^2}=\frac{(\boldsymbol{r}\prime (s),\boldsymbol{r}'' (s),\boldsymbol{r}''' (s))}{\kappa ^2\left( s \right)}. \label{eq:tor_arc}
\end{align}
其中 $(\bm{r}'(s), \bm{r}''(s), \bm{r}'''(s))$ 表示三个向量的混合积。

\item 设曲线 $\bm{r}=\bm{r}(t)$ 是正则参数曲线，其中$t$为任意参数.其曲率 $\kappa(t)\neq 0$，则其挠率为
\begin{align}
\tau(t) = \frac{(\bm{r}'(t), \bm{r}''(t), \bm{r}'''(t))}{|\bm{r}'(t) \times \bm{r}''(t)|^2}. \label{eq:tor_gen}
\end{align}
其中 $(\bm{r}'(t), \bm{r}''(t), \bm{r}'''(t))$ 表示三个向量的混合积。
\end{enumerate}
\end{theorem}
\begin{proof}
\begin{enumerate}
\item 设曲线的 Frenet 标架为 $\{\bm{r}(s); \bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s)\}$，其中 $\bm{\alpha}(s)=\bm{r}'(s)$ 是单位切向量，$\bm{\beta}(s)$ 是主法向量，$\bm{\gamma}(s)$ 是副法向量. 由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}知
\[
\kappa(s)=|\bm{r}''(s)|\neq 0,\quad \bm{\beta}(s)=\frac{\bm{r}''(s)}{|\bm{r}''(s)|},\quad \bm{\gamma}(s)=\bm{\alpha}(s)\times\bm{\beta}(s)=\frac{\bm{r}'(s)\times\bm{r}''(s)}{|\bm{r}''(s)|}.
\]
对 $\bm{\gamma}(s)$ 求导得
\begin{align*}
\bm{\gamma}'(s)=\frac{\mathrm{d}}{\mathrm{d}s}\left(\frac{\bm{r}'(s)\times\bm{r}''(s)}{|\bm{r}''(s)|}\right)=\frac{\bm{r}'(s)\times\bm{r}'''(s)}{|\bm{r}''(s)|}+(\bm{r}'(s)\times\bm{r}''(s))\frac{\mathrm{d}}{\mathrm{d}s}\left(\frac{1}{|\bm{r}''(s)|}\right).
\end{align*}
两边与 $\bm{\beta}(s)=\dfrac{\bm{r}''(s)}{|\bm{r}''(s)|}$ 作点乘.又由挠率定义知$\bm{\gamma}'(s)\cdot\bm{\beta}(s)=-\tau(s)$,因此
\[
-\tau \left( s \right) =\frac{(\boldsymbol{r}' (s)\times \boldsymbol{r}''' (s))\cdot \boldsymbol{r}'' (s)}{|\boldsymbol{r}'' (s)|^2}=-\frac{(\boldsymbol{r}' (s),\boldsymbol{r}'' (s),\boldsymbol{r}''' (s))}{|\boldsymbol{r}'' (s)|^2}.
\]
上式两边同乘$-1$即得\eqref{eq:tor_arc}式.

\item 设曲线的Frenet标架为 $\{\bm{r}(t); \bm{\alpha}(t), \bm{\beta}(t), \bm{\gamma}(t)\}$，其中 $\bm{\alpha}(t)=\bm{r}'(t)$ 是单位切向量，$\bm{\beta}(t)$ 是主法向量，$\bm{\gamma}(t)$ 是副法向量.由弧长微分 $\frac{\mathrm{d}s}{\mathrm{d}t}=|\bm{r}'(t)|$ 及\hyperref[theorem:曲线论基本公式]{曲线论基本公式}可得
\begin{gather}
\bm{\alpha}(t) = \frac{\bm{r}'(t)}{|\bm{r}'(t)|},  \quad 
\bm{\gamma}(t) = \frac{\bm{r}'(t) \times \bm{r}''(t)}{|\bm{r}'(t) \times \bm{r}''(t)|}, \nonumber \\
\bm{\beta}(t) =\bm{\gamma}(t)\times \bm{\alpha}(t)= \frac{|\bm{r}'(t)|}{|\bm{r}'(t) \times \bm{r}''(t)|}\,\bm{r}''(t) - \frac{\bm{r}'(t)\cdot\bm{r}''(t)}{|\bm{r}'(t)|\cdot|\bm{r}'(t) \times \bm{r}''(t)|}\,\bm{r}'(t). \label{eq::28rj29jr29020---2--190i-1}
\end{gather}
将 $\bm{\gamma}(t)$ 对 $t$ 求导得
\begin{align*}
\frac{\mathrm{d}\bm{\gamma}}{\mathrm{d}t} = \frac{\mathrm{d}}{\mathrm{d}t}\left( \frac{\bm{r}'(t) \times \bm{r}''(t)}{|\bm{r}'(t) \times \bm{r}''(t)|} \right)= \frac{\bm{r}'(t) \times \bm{r}'''(t)}{|\bm{r}'(t) \times \bm{r}''(t)|} + \frac{\mathrm{d}}{\mathrm{d}t}\left( \frac{1}{|\bm{r}'(t) \times \bm{r}''(t)|} \right) \bigl(\bm{r}'(t) \times \bm{r}''(t)\bigr).
\end{align*}
利用\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}中的$\frac{\mathrm{d}\bm{\gamma}}{\mathrm{d}s} = -\tau(s)\bm{\beta}(s)$ 以及链式法则可知
\begin{align*}

\begin{proposition}\label{proposition:曲线的曲率和挠率在刚体运动下不变}
设$E^3$中的一条正则曲线$C$的参数方程为$\bm{r}=\bm{r}(s)$，其中$s$是弧长参数。再设$\sigma$是$E^3$的一个刚体运动，曲线$C$点$\bm{r}(s)$处的曲率为$\kappa(s)$，挠率为$\tau(s)$。记曲线$\sigma(\bm{r})=\sigma(\bm{r}(s))$的曲率为$\kappa_{\sigma}(s)$，挠率为$\tau_{\sigma}(s)$，则
\[
\kappa_{\sigma}(s)=\kappa(s),\quad \tau_{\sigma}(s)=\tau(s),\qquad \forall s.
\]
\end{proposition}
\begin{proof}

由\refthe{theorem:刚体运动等距变换的充要条件}知存在$\bm{a}\in E^3$，$A\in SO(3)$，使得
\[
\sigma(\bm{x})=\bm{a}+A\bm{x},\quad \forall \bm{x}\in E^3.
\]
于是
\[
\sigma(\bm{r}(s))=\bm{a}+A\bm{r}(s),\quad \forall s.
\]
从而
\begin{align*}
\frac{\mathrm{d}\sigma(\bm{r}(s))}{\mathrm{d}s}=A\bm{r}'(s),\qquad 
\frac{\mathrm{d}^2\sigma(\bm{r}(s))}{\mathrm{d}s^2}=A\bm{r}''(s),\qquad 
\frac{\mathrm{d}^3\sigma(\bm{r}(s))}{\mathrm{d}s^3}=A\bm{r}'''(s).
\end{align*}
因此
\begin{align*}
\bigl(A\bm{r}'(s),A\bm{r}''(s),A\bm{r}'''(s)\bigr)
&=\det(A)\,\bigl(\bm{r}'(s),\bm{r}''(s),\bm{r}'''(s)\bigr) \\
&=\bigl(\bm{r}'(s),\bm{r}''(s),\bm{r}'''(s)\bigr) \qquad (\det(A)=1).
\end{align*}
由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}
和\hyperref[theorem:微分几何-曲线挠率的计算公式]{曲线挠率的计算公式}的计算公式可得
\begin{gather*}
\kappa_{\sigma}(s)=\left|\frac{\mathrm{d}^2\sigma(\bm{r}(s))}{\mathrm{d}s^2}\right|
=|A\bm{r}''(s)|=|\bm{r}''(s)|=\kappa(s),\\
\tau_{\sigma}(s)=\frac{\bigl(\frac{\mathrm{d}\sigma(\bm{r}(s))}{\mathrm{d}s},\frac{\mathrm{d}^2\sigma(\bm{r}(s))}{\mathrm{d}s^2},\frac{\mathrm{d}^3\sigma(\bm{r}(s))}{\mathrm{d}s^3}\bigr)}
{\left|\frac{\mathrm{d}^2\sigma(\bm{r}(s))}{\mathrm{d}s^2}\right|^2}
=\frac{\bigl(\bm{r}'(s),\bm{r}''(s),\bm{r}'''(s)\bigr)}{|\bm{r}''(s)|^2}
=\tau(s).
\end{gather*}

\end{proof}

&\bm{e}_{2}^{\prime}\left( t \right) =\frac{\mathrm{d}\bm{e}_2\left( t \right)}{\mathrm{d}s}\cdot \frac{\mathrm{d}s}{\mathrm{d}t}=-\left| \bm{r}^\prime \left( t \right) \right|\kappa \left( t \right) \bm{e}_1\left( t \right) +\left| \bm{r}^\prime \left( t \right) \right|\tau \left( t \right) \bm{e}_3\left( t \right) =\sum_{j=1}^3{\lambda _{2j}\left( t \right) \bm{e}_j\left( t \right)},
\\
&\bm{e}_{3}^{\prime}\left( t \right) =\frac{\mathrm{d}\bm{e}_3\left( t \right)}{\mathrm{d}s}\cdot \frac{\mathrm{d}s}{\mathrm{d}t}=-\left| \bm{r}^\prime \left( t \right) \right|\tau \left( t \right) \bm{e}_2\left( t \right) =\sum_{j=1}^3{\lambda _{3j}\left( t \right) \bm{e}_j\left( t \right)}.
\end{align*}
由$\bm{e}_1\left( t \right) ,\bm{e}_2\left( t \right) ,\bm{e}_3\left( t \right)$的单位正交性可得
\begin{align*}
\lambda_{11}=\lambda_{22}=\lambda_{33}=0,
\end{align*}
\begin{align*}
\lambda _{12}\left( t \right) =\left| \bm{r}^\prime \left( t \right) \right|\kappa \left( t \right) ,\quad \lambda _{13}\left( t \right) =0,
\end{align*}
\begin{align*}
\lambda _{21}\left( t \right) =-\left| \bm{r}^\prime \left( t \right) \right|\kappa \left( t \right) ,\quad \lambda _{23}\left( t \right) =\left| \bm{r}^\prime \left( t \right) \right|\tau \left( t \right) ,
\end{align*}
\begin{align*}
\lambda _{31}\left( t \right) =0,\quad \lambda _{32}\left( t \right) =-\left| \bm{r}^\prime \left( t \right) \right|\tau \left( t \right) .
\end{align*}
故
\begin{align*}
\lambda _{ij}+\lambda _{ji}=0,\quad i,j=1,2,3.
\end{align*}

\end{proof}
