\section{曲线参数方程在一点的标准展开}

曲线的Frenet标架$\{\alpha(t),\beta(t),\gamma(t)\}$是$\{r'(t),r''(t),r'''(t)\}$经过Schmidt正交化得到的．$r'(t)$决定切线，曲率是切线方向关于弧长的变化率；$r''(t),r'''(t)$决定密切平面．若密切平面方向不变则曲线为平面曲线；否则挠率反映曲线的扭曲程度．曲线方程的更高阶导数可表示成Frenet标架的线性组合，系数是曲率、挠率及其各阶导数．

\begin{theorem}
设$r=r(s)$是以弧长$s$为参数的正则曲线，在$s=s_0$处取Frenet标架$\{r(s_0);\alpha(s_0),\beta(s_0),\gamma(s_0)\}$作为$E^3$的笛卡儿直角坐标系的标架，则曲线在$s=s_0$附近的参数方程为
\begin{align}\label{eq::ho239032jfjijefjs}
\begin{cases}
x=s-s_0-\frac{\kappa (s_0)^2}{6}\left( s-s_0 \right) ^3+o(\Delta s^3),\\
y=\frac{\kappa (s_0)}{2}\left( s-s_0 \right) ^2+\frac{\kappa '(s_0)}{6}\left( s-s_0 \right) ^3+o(\left( s-s_0 \right) ^3),\\
z=\frac{\kappa (s_0)\tau (s_0)}{6}\left( s-s_0 \right) ^3+o(\left( s-s_0 \right) ^3),\\
\end{cases}
\end{align}
其中$\kappa(s_0),\ \kappa'(s_0),\ \tau(s_0)$分别为曲线在$s_0$处的曲率、曲率导数和挠率．上式称为曲线在$s=s_0$处的\textbf{标准展开式}．

当$\kappa(s_0)\tau(s_0) \neq 0$时,称曲线
\begin{align*}
\tilde{r}(s)=\left( s-s_0, \frac{\kappa (s_0)}{2}\left( s-s_0 \right) ^2, \frac{\kappa (s_0)\tau (s_0)}{6}\left( s-s_0 \right) ^3 \right) 
\end{align*}
为曲线$\bm{r}=\bm{r}(s)$在$s=s_0$处的\textbf{近似曲线}．并且曲线$r=r(s)$在$s=s_0$处的近似曲线$\tilde{r}(s)$在$s=s_0$处与原曲线$r=r(s)$具有相同的曲率$\kappa(s_0)$、挠率$\tau(s_0)$以及Frenet标架．
\end{theorem}
\begin{remark}
曲线$\bm{r}=\bm{r}(s)$的弧长参数$s$不一定是曲线$\tilde{r}(s)$的弧长参数.
\end{remark}
\begin{proof}

令$\Delta s = s-s_0$，由Taylor展开式，曲线$r(s)$在$s=s_0$处有
\begin{align*}
r(s)=r(s_0)+\frac{\Delta s}{1!}r'(s_0)+\frac{\Delta s^2}{2!}r''(s_0)+\frac{\Delta s^3}{3!}r'''(s_0)+o(\Delta s^3),
\end{align*}
其中$\Delta s=s-s_0$，余项$o(\Delta s^3)$满足$\lim\limits_{\Delta s\to0}\frac{|o(\Delta s^3)|}{\Delta s^3}=0$．根据Frenet公式，有
\begin{align*}
r'(s_0)=\alpha(s_0),\qquad 
r''(s_0)=\kappa(s_0)\beta(s_0),\qquad 
r'''(s_0)=-\kappa^2(s_0)\alpha(s_0)+\kappa'(s_0)\beta(s_0)+\kappa(s_0)\tau(s_0)\gamma(s_0).
\end{align*}
代入Taylor展开式，得
\begin{align*}
r(s)&=r(s_0)+\Delta s\,\alpha(s_0)+\frac{\Delta s^2}{2}\kappa(s_0)\beta(s_0)
+\frac{\Delta s^3}{6}\left(-\kappa^2(s_0)\alpha(s_0)+\kappa'(s_0)\beta(s_0)+\kappa(s_0)\tau(s_0)\gamma(s_0)\right)+o(\Delta s^3)\\
&=r(s_0)+\left(\Delta s-\frac{\kappa(s_0)^2}{6}\Delta s^3\right)\alpha(s_0)+\left(\frac{\kappa(s_0)}{2}\Delta s^2+\frac{\kappa'(s_0)}{6}\Delta s^3\right)\beta(s_0)
+\frac{\kappa(s_0)\tau(s_0)}{6}\Delta s^3\gamma(s_0)+o(\Delta s^3).
\end{align*}
现将Frenet标架$\{r(s_0);\alpha(s_0),\beta(s_0),\gamma(s_0)\}$取为$E^3$的笛卡儿直角坐标系的标架，即$r(s_0)$为坐标原点，$\alpha(s_0),\beta(s_0),\gamma(s_0)$分别为$x,y,z$轴正向的单位向量．于是点$r(s)$的坐标$(x,y,z)$满足
\begin{align*}
(x,y,z)=r(s)-r(s_0),
\end{align*}
从而由上式直接得到
\begin{align*}
x=\Delta s-\frac{\kappa(s_0)^2}{6}\Delta s^3+o(\Delta s^3),\quad 
y=\frac{\kappa(s_0)}{2}\Delta s^2+\frac{\kappa'(s_0)}{6}\Delta s^3+o(\Delta s^3),\quad 
z=\frac{\kappa(s_0)\tau(s_0)}{6}\Delta s^3+o(\Delta s^3).
\end{align*}

计算$\tilde{r}(s)$在$s=s_0$处的各阶导数得
\begin{align*}
\tilde{r}'(s_0)=(1,0,0),\qquad 
\tilde{r}''(s_0)=(0,\kappa_0,0),\qquad 
\tilde{r}'''(s_0)=(0,0,\kappa_0\tau_0).
\end{align*}
从而由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}可得,曲线$\tilde{r}(s)$在$s=s_0$处的曲率、主法向量、次法向量分别为
\begin{gather*}
\tilde{\kappa}(s_0)=\frac{|\tilde{r}'(s_0)\times\tilde{r}''(s_0)|}{|\tilde{r}'(s_0)|^3}=\kappa_0,
\quad
\tilde{\beta}(s_0)=\frac{\tilde{r}' (s_0)\times \tilde{r}'' (s_0)}{|\tilde{r}' (s_0)\times \tilde{r}'' (s_0)|}=\beta (s_0),
\\
\tilde{\gamma}(s_0)=\tilde{\alpha}(s_0)\times\tilde{\beta}(s_0)=\alpha(s_0)\times\beta(s_0)=\gamma(s_0).
\end{gather*}
由\hyperref[theorem:微分几何-曲线挠率的计算公式]{曲线挠率的计算公式}可知，曲线$\tilde{r}(s)$在$s=s_0$处的挠率为
\begin{align*}
\tilde{\tau}(s_0)=\frac{(\tilde{r}'(s_0)\times\tilde{r}''(s_0))\cdot\tilde{r}'''(s_0)}{|\tilde{r}'(s_0)\times\tilde{r}''(s_0)|^2}=\frac{\kappa_0^2\tau_0}{\kappa_0^2}=\tau_0.
\end{align*}
因此，近似曲线$\tilde{r}(s)$在$s=s_0$处与原曲线$r(s)$具有相同的曲率$\kappa_0$、挠率$\tau_0$以及Frenet标架$\{r(s_0);\alpha(s_0),\beta(s_0),\gamma(s_0)\}$.

\end{proof}

\begin{remark}
设正则曲线$r=r(s)$在$s=0$处的近似曲线为$\tilde{r}(s)=\left(s,\ \frac{\kappa_0}{2}s^2,\ \frac{\kappa_0\tau_0}{6}s^3\right)$．它在三个坐标平面上的投影分别为：
\begin{enumerate}[(1)]
\item 在密切平面（$z=0$）上的投影：$x=s,\ y=\dfrac{\kappa_0}{2}s^2$；
\item 在从切平面（$y=0$）上的投影：$x=s,\ z=\dfrac{\kappa_0\tau_0}{6}s^3$；
\item 在法平面（$x=0$）上的投影：$y=\dfrac{\kappa_0}{2}s^2,\ z=\dfrac{\kappa_0\tau_0}{6}s^3$．
\end{enumerate}

\begin{definition}[切触]
设曲线$C_1$和$C_2$相交于点$p_0$，在$C_1$和$C_2$上各取一点$p_1$和$p_2$，使得曲线$C_1$在点$p_0$和$p_1$之间的弧长是$\Delta s$，$C_2$在点$p_0$和$p_2$之间的弧长也是$\Delta s$．若有正整数$n$使得
\begin{align*}
\lim_{\Delta s \to 0} \frac{|p_1p_2|}{\Delta s^n} = 0,\qquad 
\lim_{\Delta s \to 0} \frac{|p_1p_2|}{(\Delta s)^{n+1}} \neq 0,
\end{align*}
则称曲线$C_1$和$C_2$在交点$p_0$处有$\bm{n}$\textbf{阶切触}．
\end{definition}

\begin{theorem}\label{theorem:微分几何-定理6.1}
设曲线$r_1(s)$和$r_2(s)$都以$s$为弧长参数，且$r_1(s_0)=r_2(s_0)$，则它们在$s=s_0$处有$n$阶切触的充分必要条件是
\begin{align}
r_1^{(i)}(s_0) = r_2^{(i)}(s_0),\quad \forall i \leqslant n;\qquad 
r_1^{(n+1)}(s_0) \neq r_2^{(n+1)}(s_0). \label{thm:contact_cond_s0}
\end{align}
\end{theorem}
\begin{proof}

在条件\eqref{thm:contact_cond_s0}下，由Taylor展开式得到
\begin{align*}
r_1(s)-r_2(s)=\frac{(s-s_0)^{n+1}}{(n+1)!}\left(r_1^{(n+1)}(s_0)-r_2^{(n+1)}(s_0)\right)+o\left((s-s_0)^{n+1}\right),
\end{align*}
因此
\begin{align*}
\lim_{s \to s_0} \frac{|r_1(s)-r_2(s)|}{(s-s_0)^n}=0,\qquad 
\lim_{s \to s_0} \frac{|r_1(s)-r_2(s)|}{(s-s_0)^{n+1}}=\frac{1}{(n+1)!}\left|r_1^{(n+1)}(s_0)-r_2^{(n+1)}(s_0)\right| \neq 0.
\end{align*}
反之亦然．

\end{proof}

\begin{corollary}
两条相交的正则曲线在交点处有$2$阶以上切触的充分必要条件是它们在交点处相切、有相同的有向密切平面和相同的曲率．
\end{corollary}
\begin{proof}

由\refthe{theorem:微分几何-定理6.1},两条相交的正则曲线在交点 $s=0$ 处有2阶以上的切触的充分必要条件是
\begin{align*}
\bm{r}_1(0) = \bm{r}_2(0),\quad \bm{r}_1^{(1)}(0) = \bm{r}_2^{(1)}(0),\quad \bm{r}_1^{(2)}(0) = \bm{r}_2^{(2)}(0),
\end{align*}
上式前两式说明这两条曲线相切, 第三式意味着
\begin{align*}
\kappa_1(0)\bm{\beta}_1(0) = \kappa_2(0)\bm{\beta}_2(0),
\end{align*}
即
\begin{align*}
\kappa_1(0) = \kappa_2(0),\quad \bm{\beta}_1(0) = \bm{\beta}_2(0).
\end{align*}
此即这两条曲线在交点$s=0$处有相同的有向密切平面和相同的曲率．

\end{proof}

\begin{definition}
对于曲线$r=r(s)$上$\kappa(s_0)>0$的点$s=s_0$，在密切平面上以$r(s_0)+\dfrac{\beta(s_0)}{\kappa(s_0)}$为中心、以$\dfrac{1}{\kappa(s_0)}$为半径的圆周称为曲线在$s=s_0$处的\textbf{曲率圆}，其圆心称为\textbf{曲率中心}，半径$\dfrac{1}{\kappa(s_0)}$称为\textbf{曲率半径}．
\end{definition}

\begin{theorem}\label{theorem:曲线的密切球面和曲率轴}
设$C: r=r(s)$是曲率和挠率都不为零的正则参数曲线，$s$是弧长参数，则在$s$处与曲线$C$有$3$阶以上切触的球面$\Sigma$的球心为
\begin{align*}
r(s)+\frac{1}{\kappa (s)}\beta (s)+\frac{1}{\tau (s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa (s)} \right) \gamma (s), 
\end{align*}
半径为
\begin{align*}
\sqrt{\left( \frac{1}{\kappa (s)} \right) ^2+\left( \frac{1}{\tau (s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa (s)} \right) \right) ^2}.
\end{align*}
该球面称为曲线$C$在$s$处的\textbf{密切球面}，其球心所在的直线
\begin{align*}
r(\lambda) = r(s)+\frac{1}{\kappa(s)}\beta(s)+\lambda\gamma(s)
\end{align*}
是通过曲率中心、垂直于密切平面的直线，称为曲线$C$在$s$处的\textbf{曲率轴}．
\end{theorem}
\begin{proof}

由弧长参数平移不变性，不失一般性只证$s=0$的情形.记$\kappa_0=\kappa(0),\ \kappa'_0=\kappa'(0),\ \tau_0=\tau(0)$，且$\kappa_0\tau_0\neq0$．曲线在$s=0$处的Frenet标架为$\{r(0);\alpha_0,\beta_0,\gamma_0\}$．设球面$\Sigma$与曲线交于$p_0=r(0)$，则球心$A$满足
\begin{align*}
\overrightarrow{p_0A}=a\alpha_0+b\beta_0+c\gamma_0,\qquad a^2+b^2+c^2=r^2,
\end{align*}
其中$r$为球面半径．利用标准展开式\eqref{eq::ho239032jfjijefjs}知,曲线上在$r(0)$附近的点$p=r(s)$满足
\begin{align*}
|pA|^2-r^2=&\left(s-\frac{\kappa_0^2}{6}s^3+o(s^3)-a\right)^2+\left(\frac{\kappa_0}{2}s^2+\frac{\kappa'_0}{6}s^3+o(s^3)-b\right)^2\\
&+\left(\frac{\kappa_0\tau_0}{6}s^3+o(s^3)-c\right)^2-(a^2+b^2+c^2)\\
=&-2a\left(s-\frac{\kappa_0^2}{6}s^3+o(s^3)\right)+\left(s-\frac{\kappa_0^2}{6}s^3+o(s^3)\right)^2\\
&-2b\left(\frac{\kappa_0}{2}s^2+\frac{\kappa'_0}{6}s^3+o(s^3)\right)+\left(\frac{\kappa_0\tau_0}{6}s^3+o(s^3)\right)^2\\
&-2c\left(\frac{\kappa_0\tau_0}{6}s^3+o(s^3)\right)+o(s^3).
\end{align*}
由\refthe{theorem:微分几何-定理6.1}知,曲线与球面在$s=0$处有$1$阶以上切触要求$s$的一次项系数为零，即$a=0$．在此条件下，上式成为
\begin{align*}
|pA|^2-r^2=\left( 1-b\kappa _0 \right) s^2-\left( \frac{\kappa _{0}^{\prime}}{3}+c\frac{\kappa _0\tau _0}{3} \right) s^3+o\left( s^3 \right) .
\end{align*}
再由\refthe{theorem:微分几何-定理6.1}知,曲线与球面在$s=0$处有$2$阶以上切触要求$s^2$项系数为零，即$1-b\kappa_0=0$，故$b=\dfrac{1}{\kappa_0}$,$a=0$.在此条件下,上式成为
\begin{align*}
|pA|^2-r^2=-\left( \frac{\kappa _{0}^{\prime}}{3}+c\frac{\kappa _0\tau _0}{3} \right) s^3+o\left( s^3 \right) .
\end{align*}
由\refthe{theorem:微分几何-定理6.1}知,曲线与球面在$s=0$处有$3$阶以上切触要求$s^3$项系数为零，即
\begin{align*}
c=-\frac{\kappa'_0}{\kappa_0^2\tau_0}=\frac{1}{\tau \left( 0 \right)}\left.\left( \frac{1}{\kappa \left( s \right)} \right) ' \right|_{s=0}.
\end{align*}
故
\begin{align*}
a=0,\quad b=\frac{1}{\kappa_0},\quad c=-\frac{\kappa'_0}{\kappa_0^2\tau_0}=\frac{1}{\tau \left( 0 \right)}\left.\left( \frac{1}{\kappa \left( s \right)} \right) ' \right|_{s=0}.
\end{align*}
即
\begin{align*}
\overrightarrow{p_0A}=\frac{1}{\kappa _0}\beta _0+\frac{1}{\tau \left( 0 \right)}\left. \left( \frac{1}{\kappa \left( s \right)} \right) ' \right|_{s=0}\gamma _0.
\end{align*}
进而球心$A$的坐标为
\begin{align*}
\overrightarrow{OA}=\overrightarrow{Op_0}+\frac{1}{\kappa _0}\beta _0+\frac{1}{\tau \left( 0 \right)}\left. \left( \frac{1}{\kappa \left( s \right)} \right) ' \right|_{s=0}\gamma _0=r\left( 0 \right) + \frac{1}{\kappa _0}\beta _0+\frac{1}{\tau \left( 0 \right)}\left. \left( \frac{1}{\kappa \left( s \right)} \right) ' \right|_{s=0}\gamma _0.
\end{align*}
半径$r$满足
\begin{align*}
r^2=|\overrightarrow{OA}-r(0)|^2=\frac{1}{\kappa _{0}^{2}}+\frac{1}{\tau \left( 0 \right)}\left. \left( \frac{1}{\kappa \left( s \right)} \right) \prime \right|_{s=0}.
\end{align*}
球心所在的直线$r(\lambda)=r(0)+\frac{1}{\kappa_0}\beta_0+\lambda\gamma_0$显然通过曲率中心$r(0)+\frac{1}{\kappa_0}\beta_0$且垂直于密切平面.

\end{proof}

\begin{example}
作正则参数曲线$C$关于一张平面的对称曲线$C^*$, 证明: 曲线$C$和$C^*$在对应点的曲率相同, 挠率的绝对值相同而符号相反.
\end{example}
\begin{proof}

设曲线$C$的参数方程为
\begin{align*}
\bm{r}(s) = \left( x(s), y(s), z(s) \right),
\end{align*}
其中$s$是弧长参数，其Frenet标架为$\left\{ \bm{r}(s); \bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s) \right\}$。由\hyperref[theorem:曲线论基本公式]{曲线论基本公式}和\hyperref[theorem:微分几何-曲线挠率的计算公式]{曲线挠率的计算公式}可得，曲线$C$的曲率和挠率分别为
\begin{gather*}
\kappa(s) = \left| \bm{r}''(s) \right| = \sqrt{x''^2(s) + y''^2(s) + z''^2(s)}, \\
\tau(s) = \frac{\left( \bm{r}'(s), \bm{r}''(s), \bm{r}'''(s) \right)}{\left| \bm{r}''(s) \right|^2} = \frac{x''' \left( y'x'' - y''z' \right) + y''' \left( x''z' - x'z'' \right) + z''' \left( x'y'' - x''y' \right)}{x''^2 + y''^2 + z''^2}.
\end{gather*}
曲线$C$关于$xOy$面对称的曲线$C^*$的参数方程为
\begin{align*}
\bm{r}^*(s) = \left( x(s), y(s), -z(s) \right).
\end{align*}
注意到
\begin{align*}
\left| \bm{r}^{*'}(s) \right| = \sqrt{x'^2(s) + y'^2(s) + z'^2(s)} = \left| \bm{r}'(s) \right| = 1,
\end{align*}
故由\refthe{theorem:微分几何-定理2.1}知$s$也是曲线$C^*$的弧长参数。再利用\hyperref[theorem:曲线论基本公式]{曲线论基本公式}和\hyperref[theorem:微分几何-曲线挠率的计算公式]{曲线挠率的计算公式}可得，曲线$C^*$的曲率和挠率分别为
\begin{gather*}
\kappa^*(s) = \left| \bm{r}^{*''}(s) \right| = \sqrt{x''^2(s) + y''^2(s) + z''^2(s)} = \kappa(s), \\
\tau^*(s) = \frac{\left( \bm{r}^{*'}(s), \bm{r}^{*''}(s), \bm{r}^{*'''}(s) \right)}{\left| \bm{r}^{*''}(s) \right|^2} = -\frac{x''' \left( y'x'' - y''z' \right) + y''' \left( x''z' - x'z'' \right) + z''' \left( x'y'' - x''y' \right)}{x''^2 + y''^2 + z''^2} = -\tau(s).
\end{gather*}

\end{proof}

\begin{example}
如果正则参数曲线的向径$\bm{r}(s)$关于弧长参数$s$的$n$阶导数是
\begin{align*}
\bm{r}^{(n)}(s) = a_n(s)\bm{\alpha}(s) + b_n(s)\bm{\beta}(s) + c_n(s)\bm{\gamma}(s),
\end{align*}
其中 $\bm{\alpha}(s)$ 是单位切向量，$\bm{\beta}(s)$ 是主法向量，$\bm{\gamma}(s)$ 是副法向量.求它的$n+1$阶导数.
\end{example}
\begin{solution}

由条件和\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}可得
\begin{align*}
r^{\left( n+1 \right)}\left( s \right) &=a_{n}^{\prime}\left( s \right) \alpha \left( s \right) +a_n\left( s \right) \alpha \prime \left( s \right) +b_{n}^{\prime}\left( s \right) \beta \left( s \right) +b_n\left( s \right) \beta \prime \left( s \right) +c_{n}^{\prime}\left( s \right) \gamma \left( s \right) +c_n\left( s \right) \gamma \prime \left( s \right) .
\\
&=\left[ a_{n}^{\prime}\left( s \right) -b_n\left( s \right) \kappa \left( s \right) \right] \alpha \left( s \right) +\left[ a_n\left( s \right) \kappa \left( s \right) +b_{n}^{\prime}\left( s \right) -c_n\left( s \right) \tau \left( s \right) \right] \beta \left( s \right) +\left[ b_n\left( s \right) \tau \left( s \right) +c_{n}^{\prime}\left( s \right) \right] \gamma \left( s \right) .
\end{align*}

\end{solution}

\begin{example}
假设正则曲线$\bm{r} = \bm{r}(s)$的曲率和挠率都不为零. 证明: 如果它在各点的密切球面的球心是一个固定点, 则它必定是一条球面曲线.
\end{example}
\begin{proof}

设正则曲线的参数方程为$\bm{r}=\bm{r}(s)$，其中$s$为弧长参数。其Frenet标架为$\left\{ \bm{r}(s); \bm{\alpha}(s), \bm{\beta}(s), \bm{\gamma}(s) \right\}$。由条件和\refthe{theorem:曲线的密切球面和曲率轴}可设，曲线在各点的密切球面球心为
\begin{align}
\bm{r}(s) + \frac{1}{\kappa(s)}\bm{\beta}(s) + \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \bm{\gamma}(s) \equiv \bm{n}\in E^3,\quad \forall s.
\label{eq:fwf::IJIOJLLIJDwdf31.1}
\end{align}
对上式关于$s$求导，再结合\hyperref[theorem:微分几何-Frenet公式]{Frenet公式}得，对$\forall s$，都有
\begin{align*}
0&=\bm{\alpha}(s) +\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \bm{\beta}(s) +\frac{1}{\kappa(s)}\bm{\beta}'(s) +\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right) \bm{\gamma}(s) +\frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \bm{\gamma}'(s) \\
&=\left[ \frac{\tau(s)}{\kappa(s)}+\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right) \right] \bm{\gamma}(s).
\end{align*}
又$\bm{\gamma}(s) \ne \bm{0}$，故
\begin{align}
\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right) =-\frac{\tau(s)}{\kappa(s)},\quad \forall s.
\label{eq:fwf::IJIOJLLIJDwdf31.2}
\end{align}
令$f(s)=\left| \bm{r}(s)-\bm{n} \right|^2$，则由\eqref{eq:fwf::IJIOJLLIJDwdf31.1}式知
\begin{align*}
f(s)=\left( \frac{1}{\kappa(s)}\bm{\beta}(s) +\frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \bm{\gamma}(s) \right) ^2=\left( \frac{1}{\kappa(s)} \right) ^2+\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right) ^2.
\end{align*}
再由\eqref{eq:fwf::IJIOJLLIJDwdf31.1}式可得
\begin{align*}
f'(s)&=2\frac{1}{\kappa(s)}\cdot \frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) +2\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right) \cdot \frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right) \\
&=2\frac{1}{\kappa(s)}\cdot \frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) -2\left( \frac{1}{\tau(s)}\frac{\mathrm{d}}{\mathrm{d}s}\left( \frac{1}{\kappa(s)} \right) \right) \cdot \frac{\tau(s)}{\kappa(s)}=0.
\end{align*}
故$f(s)=\left| \bm{r}(s)-\bm{n} \right|^2$是常数，因此$\bm{r}(s)$是球面曲线。

\end{proof}
