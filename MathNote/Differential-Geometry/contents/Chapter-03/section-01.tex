\section{第二基本形式}

\begin{definition}
设正则参数曲面$S:\boldsymbol{r}=\boldsymbol{r}(u,v)$，在任意点$(u_0,v_0)$处的切平面单位法向量为
\begin{align}
\boldsymbol{n}=\left.\frac{\boldsymbol{r}_u\times\boldsymbol{r}_v}{|\boldsymbol{r}_u\times\boldsymbol{r}_v|}\right|_{(u_0,v_0)}.\label{eq:unit_normal}
\end{align}
对$(u_0,v_0)$的邻近点$(u_0+\Delta u,v_0+\Delta v)$，其到切平面的\textbf{有向距离}为
\begin{align}
\delta(\Delta u,\Delta v)=\big(\boldsymbol{r}(u_0+\Delta u,v_0+\Delta v)-\boldsymbol{r}(u_0,v_0)\big)\cdot\boldsymbol{n}.\label{eq:oriented_distance}
\end{align}
将$\boldsymbol{r}(u_0+\Delta u,v_0+\Delta v)-\boldsymbol{r}(u_0,v_0)$作Taylor展开：
\begin{align*}
&\boldsymbol{r}(u_0+\Delta u,v_0+\Delta v)-\boldsymbol{r}(u_0,v_0)\\
=&\boldsymbol{r}_u\Delta u+\boldsymbol{r}_v\Delta v+\frac{1}{2}\big(\boldsymbol{r}_{uu}(\Delta u)^2+2\boldsymbol{r}_{uv}\Delta u\Delta v+\boldsymbol{r}_{vv}(\Delta v)^2\big)+o\big((\Delta u)^2+(\Delta v)^2\big),
\end{align*}
代入\eqref{eq:oriented_distance}得
\begin{align}
\delta(\Delta u,\Delta v)=\frac{1}{2}\big[L(\Delta u)^2+2M\Delta u\Delta v+N(\Delta v)^2\big]+o\big((\Delta u)^2+(\Delta v)^2\big),\label{eq:delta_taylor}
\end{align}
其中
\begin{align}
\begin{cases}
L=\boldsymbol{r}_{uu}(u_0,v_0)\cdot\boldsymbol{n}(u_0,v_0),\\
M=\boldsymbol{r}_{uv}(u_0,v_0)\cdot\boldsymbol{n}(u_0,v_0),\\
N=\boldsymbol{r}_{vv}(u_0,v_0)\cdot\boldsymbol{n}(u_0,v_0).
\end{cases}\label{eq:LMN_def}
\end{align}
利用$\boldsymbol{r}_u\cdot\boldsymbol{n}=\boldsymbol{r}_v\cdot\boldsymbol{n}=0$，可推得
\begin{align}
L=-\boldsymbol{r}_u\cdot\boldsymbol{n}_u,\quad M=-\boldsymbol{r}_u\cdot\boldsymbol{n}_v=-\boldsymbol{r}_v\cdot\boldsymbol{n}_u,\quad N=-\boldsymbol{r}_v\cdot\boldsymbol{n}_v.\label{eq:LMN_alt}
\end{align}
称二次微分式
\begin{align}
\mathrm{II}=\mathrm{d}^2\boldsymbol{r}\cdot\boldsymbol{n}=-\mathrm{d}\boldsymbol{r}\cdot\mathrm{d}\boldsymbol{n}=L(\mathrm{d}u)^2+2M\mathrm{d}u\mathrm{d}v+N(\mathrm{d}v)^2\label{eq:second_fundamental_form}
\end{align}
为\textbf{曲面的第二基本形式}，$L,M,N$称为\textbf{第二基本量}。
\end{definition}

\centering
\includegraphics[scale=0.3]{邻近点到切平面的有向距离.png}
\caption{邻近点到切平面的有向距离}
\label{figure:oriented_distance}
\end{figure}

保持定向的容许参数变换$u=u(\tilde{u},\tilde{v}),v=v(\tilde{u},\tilde{v})$满足$\frac{\partial(u,v)}{\partial(\tilde{u},\tilde{v})}>0$，此时$\tilde{\boldsymbol{n}}=\boldsymbol{n}$，第二基本形式在该变换下不变；若参数变换翻转定向，则$\tilde{\boldsymbol{n}}=-\boldsymbol{n}$，第二基本形式变号。
第二基本形式的几何意义为$\mathrm{II}\approx2\delta(\mathrm{d}u,\mathrm{d}v)$，即其为有向距离$\delta$作为无穷小的主部的2倍，刻画曲面的弯曲程度。

\begin{example}
求平面和圆柱面的第二基本形式。
\end{example}
\begin{solution}

设$S_1$为$\mathbb{E}^3$中的$Oxy$平面，参数方程为$\boldsymbol{r}=(u,v,0)$，单位法向量$\boldsymbol{n}=(0,0,1)$，则
\begin{align*}
\mathrm{I}=\mathrm{d}\boldsymbol{r}\cdot\mathrm{d}\boldsymbol{r}=(\mathrm{d}u)^2+(\mathrm{d}v)^2,\quad \mathrm{II}=-\mathrm{d}\boldsymbol{r}\cdot\mathrm{d}\boldsymbol{n}=0.
\end{align*}

设圆柱面$S_2$的参数方程为$\boldsymbol{r}=\big(a\cos\frac{u}{a},a\sin\frac{u}{a},v\big)$，计算得
\begin{align*}
\boldsymbol{r}_u=\big(-\sin\frac{u}{a},\cos\frac{u}{a},0\big),\quad \boldsymbol{r}_v=(0,0,1),\\
\boldsymbol{r}_u\times\boldsymbol{r}_v=\big(\cos\frac{u}{a},\sin\frac{u}{a},0\big)=\boldsymbol{n},\\
\boldsymbol{r}_{uu}=\big(-\frac{1}{a}\cos\frac{u}{a},-\frac{1}{a}\sin\frac{u}{a},0\big),\quad \boldsymbol{r}_{uv}=\boldsymbol{r}_{vv}=\boldsymbol{0}.
\end{align*}
因此第一基本量与第二基本量为
\begin{align*}
E=\boldsymbol{r}_u\cdot\boldsymbol{r}_u=1,\quad F=\boldsymbol{r}_u\cdot\boldsymbol{r}_v=0,\quad G=\boldsymbol{r}_v\cdot\boldsymbol{r}_v=1,\\
L=\boldsymbol{r}_{uu}\cdot\boldsymbol{n}=-\frac{1}{a},\quad M=\boldsymbol{r}_{uv}\cdot\boldsymbol{n}=0,\quad N=\boldsymbol{r}_{vv}\cdot\boldsymbol{n}=0.
\end{align*}
故
\begin{align*}
\mathrm{I}=(\mathrm{d}u)^2+(\mathrm{d}v)^2,\quad \mathrm{II}=-\frac{1}{a}(\mathrm{d}u)^2.
\end{align*}
平面与圆柱面第一基本形式相同，但第二基本形式不同，反映二者外形的差异。

\end{solution}

\begin{theorem}
一块正则曲面是平面的一部分，当且仅当其第二基本形式恒等于零。
\end{theorem}
\begin{proof}

必要性：平面的第二基本形式恒为零，已由前述例题验证。

充分性：设曲面$S:\boldsymbol{r}=\boldsymbol{r}(u,v)$的第二基本形式恒为零，则
\begin{align*}
L=-\boldsymbol{r}_u\cdot\boldsymbol{n}_u=0,\quad M=-\boldsymbol{r}_u\cdot\boldsymbol{n}_v=-\boldsymbol{r}_v\cdot\boldsymbol{n}_u=0,\quad N=-\boldsymbol{r}_v\cdot\boldsymbol{n}_v=0.
\end{align*}
由$\boldsymbol{n}$为单位向量场，得$\boldsymbol{n}_u\cdot\boldsymbol{n}=\boldsymbol{n}_v\cdot\boldsymbol{n}=0$。
$\{\boldsymbol{r};\boldsymbol{r}_u,\boldsymbol{r}_v,\boldsymbol{n}\}$为$\mathbb{E}^3$的标架，故$\boldsymbol{n}_u,\boldsymbol{n}_v$在标架各向量上投影均为零，即$\boldsymbol{n}_u=\boldsymbol{n}_v=\boldsymbol{0}$，$\boldsymbol{n}$为常向量场。

由$\mathrm{d}\boldsymbol{r}\cdot\boldsymbol{n}=0$，微分得
\begin{align*}
\mathrm{d}(\boldsymbol{r}\cdot\boldsymbol{n})=\mathrm{d}\boldsymbol{r}\cdot\boldsymbol{n}+\boldsymbol{r}\cdot\mathrm{d}\boldsymbol{n}=0,
\end{align*}
故$\boldsymbol{r}\cdot\boldsymbol{n}=$常数，即$(\boldsymbol{r}(u,v)-\boldsymbol{r}(u_0,v_0))\cdot\boldsymbol{n}=0$，曲面落在以$\boldsymbol{n}$为法向量的平面内。

\end{proof}

\begin{theorem}
一块正则曲面是球面的一部分，当且仅当曲面上每一点的第二基本形式是第一基本形式的非零倍数。
\end{theorem}
\begin{remark}
这个定理的几何意义为：曲面上每一点沿各个方向的弯曲程度均相同；若曲面上每个定点沿各方向弯曲程度相同，则曲面在所有点、所有方向的弯曲程度均相同。
\end{remark}
\begin{proof}

必要性：设曲面$S$落在以$\boldsymbol{r}_0$为中心、$R$为半径的球面上，则$(\boldsymbol{r}(u,v)-\boldsymbol{r}_0)^2=R^2$。
微分得$\mathrm{d}\boldsymbol{r}\cdot(\boldsymbol{r}-\boldsymbol{r}_0)=0$，故$\boldsymbol{r}-\boldsymbol{r}_0$为法向量，即$\boldsymbol{n}=\frac{1}{R}(\boldsymbol{r}-\boldsymbol{r}_0)$。
因此
\begin{align*}
\mathrm{II}=-\mathrm{d}\boldsymbol{r}\cdot\mathrm{d}\boldsymbol{n}=-\frac{1}{R}\mathrm{d}\boldsymbol{r}\cdot\mathrm{d}\boldsymbol{r}=-\frac{1}{R}\mathrm{I}.
\end{align*}

充分性：设存在处处非零的函数$c(u,v)$，使得$\mathrm{II}=c(u,v)\mathrm{I}$，即
\begin{align*}
(L-cE)(\mathrm{d}u)^2+2(M-cF)\mathrm{d}u\mathrm{d}v+(N-cG)(\mathrm{d}v)^2=0,
\end{align*}
对任意$(\mathrm{d}u,\mathrm{d}v)$成立，故
\begin{align*}
L=cE,\quad M=cF,\quad N=cG.
\end{align*}
由第二基本量的等价定义，上式等价于
\begin{align*}
(\boldsymbol{n}_u+c\boldsymbol{r}_u)\cdot\boldsymbol{r}_u=0,\quad (\boldsymbol{n}_u+c\boldsymbol{r}_u)\cdot\boldsymbol{r}_v=(\boldsymbol{n}_v+c\boldsymbol{r}_v)\cdot\boldsymbol{r}_u=0,\quad (\boldsymbol{n}_v+c\boldsymbol{r}_v)\cdot\boldsymbol{r}_v=0.
\end{align*}
又$\boldsymbol{n}$为单位向量场，故$(\boldsymbol{n}_u+c\boldsymbol{r}_u)\cdot\boldsymbol{n}=(\boldsymbol{n}_v+c\boldsymbol{r}_v)\cdot\boldsymbol{n}=0$，因此$\boldsymbol{n}_u+c\boldsymbol{r}_u=\boldsymbol{0},\boldsymbol{n}_v+c\boldsymbol{r}_v=\boldsymbol{0}$。

将两式分别对$v,u$求偏导得
\begin{align*}
\boldsymbol{n}_{uv}+c_v\boldsymbol{r}_u+c\boldsymbol{r}_{uv}=\boldsymbol{0},\quad \boldsymbol{n}_{vu}+c_u\boldsymbol{r}_v+c\boldsymbol{r}_{vu}=\boldsymbol{0},
\end{align*}
两式相减得$c_v\boldsymbol{r}_u=c_u\boldsymbol{r}_v$。因$\boldsymbol{r}_u,\boldsymbol{r}_v$线性无关，故$c_u=c_v=0$，即$c$为常数。

由$\mathrm{d}(\boldsymbol{n}+c\boldsymbol{r})=(\boldsymbol{n}_u+c\boldsymbol{r}_u)\mathrm{d}u+(\boldsymbol{n}_v+c\boldsymbol{r}_v)\mathrm{d}v=\boldsymbol{0}$，得$\boldsymbol{n}+c\boldsymbol{r}=\boldsymbol{r}_0$（常向量），因此
\begin{align*}
\boldsymbol{r}(u,v)-\boldsymbol{r}_0=-\frac{1}{c}\boldsymbol{n}(u,v),\quad (\boldsymbol{r}(u,v)-\boldsymbol{r}_0)^2=\frac{1}{c^2},
\end{align*}
即曲面落在以$\boldsymbol{r}_0$为中心、$\frac{1}{|c|}$为半径的球面上。

\end{proof}



\begin{example}
求下列曲面的第二基本形式:
\begin{enumerate}[(1)]
\item $\bm{r} = (a\cos\varphi\cos\theta,a\cos\varphi\sin\theta,b\sin\varphi)$(椭球面);
\item $\bm{r} = \left(u,v,\frac{1}{2}(u^2+v^2)\right)$(旋转椭圆抛物面);
\item $\bm{r} = (a(u+v),a(u-v),2uv)$(双曲抛物面);
\item $\bm{r} = (f(u),g(u),v)$(一般柱面);
\item $\bm{r} = (u\cos v,u\sin v,f(v))$(劈锥曲面).
\end{enumerate}
\end{example}

\begin{example}
求下列曲面的第二基本形式:
\begin{enumerate}[(1)]
\item $z(x^2+y^2)=2xy$;
\item $x^2+y^2=(\tan^2\alpha)z^2$,$\alpha$是常数;
\item $xyz=k^3$,$k\neq0$是常数.
\end{enumerate}
\end{example}
