\section{某些特殊曲面}

\begin{definition}
平均曲率$H=0$的曲面称为\textbf{极小曲面}。
\end{definition}

\begin{proposition}
考虑旋转曲面$S:\boldsymbol{r}=(u\cos v,u\sin v,f(u))$，其中$u>0,\ 0\leqslant v<2\pi$，$f(u)$为光滑函数。
这个旋转曲面的第一、第二基本形式与Gauss曲率分别为
\begin{align}
\mathrm{I}&=\big(1+f'^2(u)\big)(\mathrm{d}u)^2+u^2(\mathrm{d}v)^2,\label{eq:rot_surf_I}\\
\mathrm{II}&=\frac{f''(u)}{\sqrt{1+f'^2(u)}}(\mathrm{d}u)^2+\frac{uf'(u)}{\sqrt{1+f'^2(u)}}(\mathrm{d}v)^2,\label{eq:rot_surf_II}\\
K&=\frac{f'(u)f''(u)}{u\big(1+f'^2(u)\big)^2}.\label{eq:rot_surf_K}
\end{align}
\end{proposition}
\begin{proof}

直接计算偏导$\boldsymbol{r}_u=(\cos v,\sin v,f'(u))$，$\boldsymbol{r}_v=(-u\sin v,u\cos v,0)$，求得第一基本量$E=1+f'^2(u),\ F=0,\ G=u^2$；
法向量$\boldsymbol{n}=\frac{(-f'(u)\cos v,-f'(u)\sin v,1)}{\sqrt{1+f'^2(u)}}$，求得第二基本量$L=\frac{f''(u)}{\sqrt{1+f'^2(u)}},\ M=0,\ N=\frac{uf'(u)}{\sqrt{1+f'^2(u)}}$。
代入Gauss曲率公式$K=\frac{LN-M^2}{EG-F^2}$即得\eqref{eq:rot_surf_K}.

\end{proof}

若$S$具有常数Gauss曲率$K$，则$f(u)$满足微分方程
\begin{align}
f'(u)f''(u)=Ku\big(1+f'^2(u)\big)^2.\label{eq:rot_const_K_ode}
\end{align}
对\eqref{eq:rot_const_K_ode}积分一次，得
\begin{align}
\frac{1}{1+f'^2(u)}=c-Ku^2,\label{eq:rot_const_K_int1}
\end{align}
其中$c$为积分常数。再次积分，得
\begin{align}
f(u)=\pm\int\sqrt{\frac{1-c+Ku^2}{c-Ku^2}}\mathrm{d}u.\label{eq:rot_const_K_int2}
\end{align}

\begin{example}
分三类情形讨论常数Gauss曲率的旋转曲面：
\begin{enumerate}[(1)]
\item $K=0$
此时$\frac{1}{1+f'^2(u)}=c$，解得$f(u)=au+b$，其中$a=\sqrt{\frac{1-c}{c}},\ 0<c<1$。
对应曲面为平面（$a=0$）或圆锥面（$a\neq0$）；圆柱面$\boldsymbol{r}=(a\cos v,a\sin v,u)$也是$K=0$的旋转曲面。

\item $K>0$，取$K=\frac{1}{a^2}\ (a>0)$
此时$c>0$，令$c=b^2\ (b>0)$，则
\begin{align*}
f(u)=\pm\int\sqrt{\frac{a^2(1-b^2)+u^2}{a^2b^2-u^2}}\mathrm{d}u.
\end{align*}
当$b^2=1$时，$f(u)=\mp\sqrt{a^2-u^2}+c_0$，对应半径为$a$的球面；$b^2\neq1$时得到非球面的常正Gauss曲率旋转曲面。

\item $K<0$，取$K=-\frac{1}{a^2}\ (a>0)$
此时$c<1$，令$c=1-b^2\ (b>0)$，则
\begin{align*}
f(u)=\pm\int\sqrt{\frac{a^2b^2-u^2}{a^2(1-b^2)+u^2}}\mathrm{d}u.
\end{align*}
当$b^2=1$时，作变量替换$u=a\cos\varphi\ \big(0\leqslant\varphi<\frac{\pi}{2}\big)$，得
\begin{align*}
f=\pm a\big(\ln(\sec\varphi+\tan\varphi)-\sin\varphi\big),
\end{align*}
对应\textbf{伪球面}，参数方程为
\begin{align*}
\boldsymbol{r}=\big(a\cos\varphi\cos\theta,a\cos\varphi\sin\theta,a(\ln(\sec\varphi+\tan\varphi)-\sin\varphi)\big),\quad 0\leqslant\varphi<\frac{\pi}{2},\ 0\leqslant\theta<2\pi.
\end{align*}
\end{enumerate}
\end{example}

对旋转曲面$\boldsymbol{r}=(u\cos v,u\sin v,f(u))$，由\eqref{eq:rot_surf_I},\eqref{eq:rot_surf_II}，其平均曲率为
\begin{align}
H&=\frac{1}{2}\left(\frac{f''(u)}{\big(\sqrt{1+f'^2(u)}\big)^3}+\frac{f'(u)}{u\sqrt{1+f'^2(u)}}\right)\nonumber\\
&=\frac{uf''(u)+f'(u)\big(1+f'^2(u)\big)}{2u\big(\sqrt{1+f'^2(u)}\big)^3}.\label{eq:rot_surf_H}
\end{align}

\begin{proposition}
旋转曲面为极小曲面的充要条件是
\begin{align}
uf''(u)+f'(u)\big(1+f'^2(u)\big)=0.\label{eq:rot_minimal_ode}
\end{align}
也等价于
\begin{align*}
\frac{f'^2(u)}{1+f'^2(u)}=\frac{c}{u^2},
\end{align*}
其中常数$c\geqslant0$。
\end{proposition}
\begin{proof}

极小曲面满足$H=0$，由\eqref{eq:rot_surf_H}，分母恒正，故分子为零即得\eqref{eq:rot_minimal_ode}.
将\eqref{eq:rot_minimal_ode}变形为$\frac{\mathrm{d}}{\mathrm{d}u}\left(\frac{uf'(u)}{\sqrt{1+f'^2(u)}}\right)=0$，积分得
\begin{align*}
\frac{f'^2(u)}{1+f'^2(u)}=\frac{c}{u^2},
\end{align*}
其中常数$c\geqslant0$。
\end{proof}



\begin{example}
分两类情形求解旋转极小曲面：
\begin{enumerate}[(1)]
\item $c=0$：$f'(u)=0$，$f(u)$为常数，对应平面。
\item $c=a^2\ (a>0)$：解得
\begin{align*}
f'(u)=\pm\frac{a}{\sqrt{u^2-a^2}},\quad f(u)=\pm a\ln\big(u+\sqrt{u^2-a^2}\big),
\end{align*}
对应\textbf{悬链面}，参数方程为
\begin{align*}
\boldsymbol{r}=\big(u\cos v,u\sin v,\pm a\ln\big(u+\sqrt{u^2-a^2}\big)\big).
\end{align*}
作参数替换$u=a\cosh\frac{r}{a},\ v=\theta$，悬链面方程化为
\begin{align*}
\boldsymbol{r}=\left(a\cosh\frac{r}{a}\cos\theta,a\cosh\frac{r}{a}\sin\theta,r\right).
\end{align*}
\end{enumerate}
\end{example}

\begin{theorem}
极小曲面满足以下性质：
\begin{enumerate}[(1)]
\item Gauss曲率$K\leqslant0$，曲面上无椭圆点，仅含平点或双曲点；
\item 双曲点处的两个渐近方向彼此正交。
\end{enumerate}
\end{theorem}
\begin{remark}
平均曲率为非零常数的曲面包括球面、圆柱面.
\end{remark}
\begin{proof}
\begin{enumerate}[(1)]
\item 由平均曲率定义$2H=\kappa_1+\kappa_2$，极小曲面$H=0$，故$\kappa_2=-\kappa_1$，因此$K=\kappa_1\kappa_2=-\kappa_1^2\leqslant0$。

\item 设渐近方向与主方向$\kappa_1$的夹角为$\theta$，由Euler公式，法曲率
\begin{align*}
0=\kappa_n(\theta)=\kappa_1\cos^2\theta+\kappa_2\sin^2\theta=\kappa_1(\cos^2\theta-\sin^2\theta)=\kappa_1\cos2\theta,
\end{align*}
得$2\theta=\frac{\pi}{2}$，即两渐近方向夹角为$\frac{\pi}{2}$，彼此正交.
\end{enumerate}
\end{proof}

\begin{example}
$z = c \cdot \arctan \frac{y}{x}$是极小曲面，并求它的主曲率。
\end{example}

\begin{example}
\begin{enumerate}[(1)]
\item \begin{align*}
z = \frac{1}{a}\ln\frac{\cos ay}{\cos ax}
\end{align*}
是极小曲面，其中$a$是非零常数。该曲面称为Scherk曲面。

\item 形如$z = f(x) + g(y)$的极小曲面必定是Scherk曲面。
\end{enumerate}
\end{example}

\begin{example}
\begin{enumerate}[(1)]
\item \begin{align*}
\bm{r} = \big(3u(1+v^{2})-u^{3},3v(1+u^{2})-v^{3},3(u^{2}-v^{2})\big)
\end{align*}
是极小曲面，该曲面称为Enneper曲面。

\item Enneper曲面的曲率线必定是平面曲线，求出曲率线所在的平面。
\end{enumerate}
\end{example}

\begin{example}
\begin{enumerate}[(1)]
\item 正螺面$\bm{r}=(u\cos v,u\sin v,bv)$是极小曲面。

\item 形如$z = f\left(\frac{x}{y}\right)$的极小曲面必定是正螺面。
\end{enumerate}
\end{example}

\begin{example}
如果从曲面$S$到单位球面$\Sigma$的Gauss映射是保角对应，则该曲面$S$或者是球面，或者是极小曲面。
\end{example}

\begin{example}
推导极小曲面$z = f(x,y)$所满足的微分方程
\begin{align*}
(1 + f_{y}^{2})f_{xx} - 2f_{x}f_{y}f_{xy} + (1 + f_{x}^{2})f_{yy} = 0.
\end{align*}
\end{example}

\begin{example}
将一个旋转曲面$S$沿它的旋转轴平移得到一个单参数曲面族。
\begin{enumerate}[(1)]
\item 求一个共轴的旋转曲面$S^*$，使得它与已知单参数曲面族中的每一个曲面都垂直相交；
\item 假设旋转曲面$S$和$S^*$的Gauss曲率分别是$K$和$K^*$，则$K = -K^*$.
\end{enumerate}
\end{example}
