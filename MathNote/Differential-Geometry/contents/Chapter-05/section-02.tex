\section{测地线}

\begin{definition}
在曲面$S$上测地曲率恒等于零的曲线称为曲面$S$上的\textbf{测地线}。
\end{definition}

\begin{remark}
平面曲线的测地曲率为其相对曲率，因此平面上的测地线即为直线，测地线是平面上直线概念在曲面上的推广。
\end{remark}

\begin{theorem}
曲面$S$上的一条曲线$C$是测地线，当且仅当它或者是一条直线，或者它的主法向量处处是曲面$S$的法向量。
\end{theorem}

\begin{proof}

曲面$S$上曲线$C$的测地曲率满足
\begin{align}
\kappa_g = \kappa \cos\tilde{\theta},
\end{align}
其中$\tilde{\theta}$是曲线$C$的次法向量和曲面$S$的法向量的夹角。
若$C$为测地线，则$\kappa_g \equiv 0$，故对$C$上每一点，必有$\kappa=0$或$\cos\tilde{\theta}=0$。
若$\kappa \equiv 0$，则$C$为直线；若存在点$p$使得$\kappa(p)\neq0$，由曲率函数$\kappa$的连续性，在$p$的邻域内$\kappa\neq0$，故$\cos\tilde{\theta}\equiv0$，即$\tilde{\theta}=\frac{\pi}{2}$，此时曲线$C$的主法向量是曲面$S$的法向量。
反之显然成立。

\end{proof}

\begin{example}
试证：旋转面$S$上的经线是该曲面$S$的测地线。
\end{example}

\begin{proof}

旋转面$S$上的经线为过旋转轴的平面与曲面的交线，是旋转面的母线。经线作为平面曲线，其法线即为主法线，且该法线过旋转轴，与旋转面的平行圆垂直，因此经线的主法线同时为曲面$S$的法线，故经线是测地线。特别地，球面上的大圆周均为测地线。

\end{proof}

\begin{example}
证明：若在曲面$S$上运动的质点$p$仅受曲面约束作用，无其他外力，则点$p$的轨迹$C$是曲面$S$上的测地线。
\end{example}

\begin{proof}

设质点运动方程为$\boldsymbol{r}=\boldsymbol{r}(t)$，约束力$\boldsymbol{F}$沿曲面法向。由牛顿第二定律
\begin{align}
\boldsymbol{F}=m\frac{\mathrm{d}^2\boldsymbol{r}(t)}{\mathrm{d}t^2},
\end{align}
其中$m$为质点质量。因此
\begin{align*}
\frac{\mathrm{d}^2\boldsymbol{r}(t)}{\mathrm{d}t^2}\cdot\frac{\mathrm{d}\boldsymbol{r}(t)}{\mathrm{d}t}=\frac{1}{m}\boldsymbol{F}\cdot\frac{\mathrm{d}\boldsymbol{r}(t)}{\mathrm{d}t}=0,
\end{align*}
进而
\begin{align*}
\frac{1}{2}\frac{\mathrm{d}}{\mathrm{d}t}\left(\frac{\mathrm{d}\boldsymbol{r}(t)}{\mathrm{d}t}\cdot\frac{\mathrm{d}\boldsymbol{r}(t)}{\mathrm{d}t}\right)=\frac{\mathrm{d}^2\boldsymbol{r}(t)}{\mathrm{d}t^2}\cdot\frac{\mathrm{d}\boldsymbol{r}(t)}{\mathrm{d}t}=0,
\end{align*}
故
\begin{align}
\left|\frac{\mathrm{d}\boldsymbol{r}(t)}{\mathrm{d}t}\right|=\text{常数},
\end{align}
即质点做匀速运动，时间参数$t$与弧长参数$s$成比例。因此$\frac{\mathrm{d}^2\boldsymbol{r}(t)}{\mathrm{d}t^2}$指向轨迹主法线方向，结合式\eqref{2.1}，曲线$C$的主法向量是曲面$S$的法向量，故$C$为测地线。

\end{proof}

由测地曲率表达式
\begin{align}
\kappa_g \boldsymbol{e}_2=\left(\frac{\mathrm{d}^2u^\gamma}{\mathrm{d}s^2}+\Gamma^\gamma_{\alpha\beta}\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\right)\boldsymbol{r}_\gamma,
\end{align}
结合$\boldsymbol{r}_1,\boldsymbol{r}_2$线性无关，得$\kappa_g\equiv0$的充要条件为
\begin{align}
\frac{\mathrm{d}^2u^\gamma}{\mathrm{d}s^2}+\Gamma^\gamma_{\alpha\beta}\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}=0,\quad \gamma=1,2.
\end{align}
该方程仅涉及曲面第一基本形式，属于内蕴几何范畴。

引入未知函数$v^\gamma$，将二阶方程组降阶为一阶拟线性常微分方程组
\begin{align}
\begin{cases}
\frac{\mathrm{d}u^\gamma}{\mathrm{d}s}=v^\gamma,\\
\frac{\mathrm{d}v^\gamma}{\mathrm{d}s}=-\Gamma^\gamma_{\alpha\beta}v^\alpha v^\beta,
\end{cases}
\end{align}
由常微分方程理论，对任意初始值$(u_0^1,u_0^2;v_0^1,v_0^2)$，存在$\varepsilon>0$，使得方程组在$(-\varepsilon,\varepsilon)$上存在唯一解$u^\gamma=u^\gamma(s),v^\gamma=v^\gamma(s)$，满足
\begin{align}
u^\gamma(0)=u_0^\gamma,\quad v^\gamma(0)=v_0^\gamma.
\end{align}

若初始值满足
\begin{align}
g_{\alpha\beta}(u_0^1,u_0^2)v_0^\alpha v_0^\beta=1,
\end{align}
定义
\begin{align}
f(s)=g_{\alpha\beta}(u^1(s),u^2(s))\frac{\mathrm{d}u^\alpha(s)}{\mathrm{d}s}\frac{\mathrm{d}u^\beta(s)}{\mathrm{d}s}-1,
\end{align}
求导可得
\begin{align*}
\frac{\mathrm{d}f(s)}{\mathrm{d}s}&=\frac{\partial g_{\alpha\beta}}{\partial u^\gamma}\frac{\mathrm{d}u^\gamma}{\mathrm{d}s}\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}+2g_{\alpha\beta}\frac{\mathrm{d}^2u^\alpha}{\mathrm{d}s^2}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\\
&=(\Gamma_{\alpha\beta\gamma}+\Gamma_{\beta\alpha\gamma})\frac{\mathrm{d}u^\gamma}{\mathrm{d}s}\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}-2g_{\alpha\beta}\Gamma^\alpha_{\gamma\delta}\frac{\mathrm{d}u^\gamma}{\mathrm{d}s}\frac{\mathrm{d}u^\delta}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}=0,
\end{align*}
结合$f(0)=0$，得$f(s)\equiv0$，即$s$为曲线$C$的弧长参数。

\begin{theorem}
对于曲面$S$上的任意一点$p$和曲面$S$在点$p$的任意一个单位切向量$\boldsymbol{v}$，在曲面$S$上必存在唯一的一条以弧长为参数的测地线$C$通过点$p$，并且在点$p$以$\boldsymbol{v}$为它的切向量。
\end{theorem}

\begin{remark}
平面上的直线满足该定理所述性质。
\end{remark}

同时有
\begin{align}
\frac{\mathrm{d}}{\mathrm{d}s}\left(g_{\alpha\beta}(u^1(s),u^2(s))\frac{\mathrm{d}u^\alpha(s)}{\mathrm{d}s}\frac{\mathrm{d}u^\beta(s)}{\mathrm{d}s}\right)=0,
\end{align}
故当初始切向量非零时，曲线$C$为正则曲线，参数$s$与弧长成比例；当切向量为单位向量时，$s$为弧长参数。

取正交参数系$(u,v)$，由测地曲率的Liouville公式，测地线的微分方程组为
\begin{align}
\begin{cases}
\frac{\mathrm{d}u}{\mathrm{d}s}=\frac{1}{\sqrt{E}}\cos\theta,\\
\frac{\mathrm{d}v}{\mathrm{d}s}=\frac{1}{\sqrt{G}}\sin\theta,\\
\frac{\mathrm{d}\theta}{\mathrm{d}s}=\frac{1}{2\sqrt{G}}\frac{\partial\ln E}{\partial v}\cos\theta-\frac{1}{2\sqrt{E}}\frac{\partial\ln G}{\partial u}\sin\theta.
\end{cases}
\end{align}

\begin{example}
求旋转面$S:\boldsymbol{r}=(u\cos v,u\sin v,f(u))$上的测地线。
\end{example}

\begin{solution}

曲面$S$的第一基本形式为
\begin{align}
\mathrm{I}=(1+(f'(u))^2)(\mathrm{d}u)^2+u^2(\mathrm{d}v)^2,
\end{align}
故$(u,v)$为正交参数系。代入测地线的微分方程组得
\begin{align}
\begin{cases}
\frac{\mathrm{d}u}{\mathrm{d}s}=\frac{1}{\sqrt{1+(f'(u))^2}}\cos\theta,\\
\frac{\mathrm{d}v}{\mathrm{d}s}=\frac{1}{u}\sin\theta,\\
\frac{\mathrm{d}\theta}{\mathrm{d}s}=-\frac{1}{u\sqrt{1+(f'(u))^2}}\sin\theta.
\end{cases}
\end{align}
消去参数$s$，得
\begin{align}
\begin{cases}
\frac{\mathrm{d}v}{\mathrm{d}u}=\frac{\sqrt{1+(f'(u))^2}}{u}\tan\theta,\\
\frac{\mathrm{d}\theta}{\mathrm{d}u}=-\frac{1}{u}\tan\theta.
\end{cases}
\end{align}
对第二式积分得
\begin{align}
u\sin\theta=c,
\end{align}
因此
\begin{align*}
\cos\theta=\sqrt{1-\frac{c^2}{u^2}},\quad \tan\theta=\frac{c}{\sqrt{u^2-c^2}},\quad \frac{\mathrm{d}v}{\mathrm{d}u}=\frac{c\sqrt{1+(f'(u))^2}}{u\sqrt{u^2-c^2}},
\end{align*}
积分得
\begin{align}
v=v_0+\int_{u_0}^{u}\frac{c\sqrt{1+(f'(u))^2}}{u\sqrt{u^2-c^2}}\mathrm{d}u,
\end{align}
此即为旋转面的测地线。当$c=0$时，经线为测地线。

\end{solution}

\begin{definition}
设$C:u^\alpha=u^\alpha(s),\alpha=1,2$是曲面$S:\boldsymbol{r}=\boldsymbol{r}(u^1,u^2)$上的曲线，$a\leqslant s\leqslant b$，$s$为弧长参数。若存在定义在区域$[a,b]\times(-\varepsilon,\varepsilon)$上的可微函数
\begin{align}\label{eq::HUHWIIO2823}
u^\alpha=u^\alpha(s,t),\quad \alpha=1,2,
\end{align}
满足
\begin{align}
u^\alpha(s,0)=u^\alpha(s),
\end{align}
则称式\eqref{eq::HUHWIIO2823}是曲线$C$的一个\textbf{变分}。若进一步满足
\begin{align}
u^\alpha(a,t)=u^\alpha(a),\quad u^\alpha(b,t)=u^\alpha(b),\quad -\varepsilon<t<\varepsilon,
\end{align}
则称该变分有\textbf{固定的端点}。
\end{definition}

\begin{remark}
变分可将曲线$C$嵌入曲线族$C_t:u^\alpha=u^\alpha_t(s)=u^\alpha(s,t)$，$C=C_0$；$s$未必是$C_t$的弧长参数。
\end{remark}

定义变分向量场：
\begin{align}
v^\alpha(s)=\left.\frac{\partial}{\partial t}\right|_{t=0}u^\alpha(s,t),\quad \boldsymbol{v}(s)=v^\alpha(s)\boldsymbol{r}_\alpha(u^1(s),u^2(s)),
\end{align}
$\boldsymbol{v}(s)$为曲面$S$上沿$C$的切向量场，称为变分的\textbf{变分向量场}。固定端点的变分满足$v(a)=0,v(b)=0$；反之，给定切向量场$\boldsymbol{v}(s)$，可取$u^\alpha(s,t)=u^\alpha(s)+tv^\alpha(s)$构造对应变分。

曲线$C$的长度为
\begin{align}
L(C)=\int_{a}^{b}\sqrt{g_{\alpha\beta}(u^1(s),u^2(s))\frac{\mathrm{d}u^\alpha(s)}{\mathrm{d}s}\frac{\mathrm{d}u^\beta(s)}{\mathrm{d}s}}\mathrm{d}s.
\end{align}
若曲线$C$是连接端点的最短线，则对任意固定端点的变分曲线族$C_t$，有$L(C)\leqslant L(C_t)$，即
\begin{align}
\left.\frac{\mathrm{d}}{\mathrm{d}t}\right|_{t=0}L(C_t)=0.
\end{align}
此时称曲线$C$的弧长在变分曲线族中达到\textbf{临界值}，该性质仅与第一基本形式有关，属于内蕴几何。

设曲面第一基本形式为
\begin{align}
\mathrm{I}=g_{\alpha\beta}(u^1,u^2)\mathrm{d}u^\alpha\mathrm{d}u^\beta,
\end{align}
则
\begin{align}
L(C_t)=\int_{a}^{b}\sqrt{g_{\alpha\beta}(u^1(s,t),u^2(s,t))\frac{\partial u^\alpha(s,t)}{\partial s}\frac{\partial u^\beta(s,t)}{\partial s}}\mathrm{d}s,
\end{align}
求导得
\begin{align*}
\frac{\mathrm{d}}{\mathrm{d}t}L(C_t)=\frac{1}{2}\int_{a}^{b}\left(\frac{\partial g_{\alpha\beta}}{\partial u^\gamma}\frac{\partial u^\gamma}{\partial t}\frac{\partial u^\alpha}{\partial s}\frac{\partial u^\beta}{\partial s}+2g_{\alpha\beta}\frac{\partial^2u^\alpha}{\partial s\partial t}\frac{\partial u^\beta}{\partial s}\right)\cdot\left(g_{\alpha\beta}\frac{\partial u^\alpha}{\partial s}\frac{\partial u^\beta}{\partial s}\right)^{-\frac{1}{2}}\mathrm{d}s.
\end{align*}
因$s$为$C=C_0$的弧长参数，故$g_{\alpha\beta}\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\big|_{t=0}=1$，因此
\begin{align*}
\left.\frac{\mathrm{d}}{\mathrm{d}t}\right|_{t=0}L(C_t)=\int_{a}^{b}\frac{1}{2}\left(\frac{\partial g_{\alpha\beta}}{\partial u^\gamma}v^\gamma(s)\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}+2g_{\alpha\beta}\frac{\mathrm{d}v^\alpha}{\mathrm{d}s}\frac{\mathrm{d}u^\beta}{\mathrm{d}s}\right)\mathrm{d}s.
\end{align*}
利用分部积分与克里斯托费尔符号化简，结合固定端点条件$v^\alpha(a)=v^\alpha(b)=0$，得
\begin{align}
\left.\frac{\mathrm{d}}{\mathrm{d}t}\right|_{t=0}L(C_t)=-\int_{a}^{b}g_{\alpha\beta}v^\alpha\left(\frac{\mathrm{d}^2u^\beta}{\mathrm{d}s^2}+\Gamma^\beta_{\gamma\delta}\frac{\mathrm{d}u^\gamma}{\mathrm{d}s}\frac{\mathrm{d}u^\delta}{\mathrm{d}s}\right)\mathrm{d}s.
\end{align}

\begin{theorem}
曲面$S$上的曲线$C$是测地线，当且仅当曲线$C$的弧长在任意固定端点的变分下达到临界值。
\end{theorem}

\begin{proof}

曲线$C$弧长取临界值即$\left.\frac{\mathrm{d}}{\mathrm{d}t}\right|_{t=0}L(C_t)=0$对任意固定端点的变分成立。
取变分向量场$v^\alpha(s)=\sin\frac{(s-a)\pi}{b-a}\cdot\delta^\alpha_\beta$，代入得
\begin{align*}
\left.\frac{\mathrm{d}}{\mathrm{d}t}\right|_{t=0}L(C_t)=\int_{a}^{b}\sin\frac{(s-a)\pi}{b-a}(\kappa_g)^2\mathrm{d}s=0.
\end{align*}
因被积函数非负，故$(\kappa_g)^2\sin\frac{(s-a)\pi}{b-a}=0,\ a\leqslant s\leqslant b$，即$\kappa_g\equiv0$，故$C$为测地线。

\end{proof}

\begin{corollary}
设$p,q$是曲面$S$上的任意两点，若曲线$C$是曲面$S$上连接$p,q$两点的最短线，则$C$必是曲面$S$上的测地线。
\end{corollary}
\begin{remark}
该定理证明仅涉及曲面$S$的第一基本形式，与曲面外在特征无关。
\end{remark}

\begin{example}
证明:一般柱面上的测地线与直母线的夹角为常数.特别地,圆柱面上的测地线或者是直母线,或者是正螺旋线.
\end{example}

\begin{example}
设曲线$C$是旋转曲面
\begin{align*}
\boldsymbol{r}(u,v)=(f(u)\cos v,f(u)\sin v,g(u))
\end{align*}
上的一条测地线,用$\theta$表示曲线$C$与曲面的经线之间的夹角,证明:沿测地线$C$下式为恒等式:
\begin{align*}
f(u)\sin\theta \equiv \text{常数}.
\end{align*}
\end{example}

\begin{example}
设在旋转曲面$S$上有一条测地线$C$与经线的夹角$\theta$是常数,并且$\theta \neq 0^\circ,90^\circ$.证明:该旋转曲面$S$必定是圆柱面.
\end{example}

\begin{example}
设曲面$S$由方程$z=f(x,y)$给出.
\begin{enumerate}[(1)]
\item 证明:曲面$S$上的测地线微分方程可以写成
\begin{align*}
\frac{\mathrm{d}^2y}{\mathrm{d}x^2} = \Gamma_{22}^{1}\left(\frac{\mathrm{d}y}{\mathrm{d}x}\right)^{3} - \left(\Gamma_{22}^{2}-2\Gamma_{12}^{1}\right)\left(\frac{\mathrm{d}y}{\mathrm{d}x}\right)^{2} + \left(\Gamma_{11}^{1}-2\Gamma_{12}^{2}\right)\frac{\mathrm{d}y}{\mathrm{d}x} - \Gamma_{11}^{2},
\end{align*}
其中记号$\Gamma_{\alpha\beta}^{\gamma}$意指把$x,y$作为曲面的参数$u^1,u^2$;
\item 证明:曲面$S$上的测地线微分方程是
\begin{align*}
\frac{\mathrm{d}^2y}{\mathrm{d}x^2}(1+p^2+q^2) = \left(r+2s\frac{\mathrm{d}y}{\mathrm{d}x}+t\left(\frac{\mathrm{d}y}{\mathrm{d}x}\right)^{2}\right)\left(p\frac{\mathrm{d}y}{\mathrm{d}x}-q\right),
\end{align*}
其中
\begin{align*}
p=\frac{\partial f}{\partial x},\quad q=\frac{\partial f}{\partial y},\quad r=\frac{\partial^2 f}{\partial x\partial x},\quad s=\frac{\partial^2 f}{\partial x\partial y},\quad t=\frac{\partial^2 f}{\partial y\partial y},
\end{align*}
其中曲面$S$以$x,y$为参数,并且在计算$\Gamma_{\alpha\beta}^{\gamma}$时把$x,y$当成$u^1,u^2$.
\end{enumerate}
\end{example}
