\section{曲面上切向量的平行移动}

\begin{definition}
设$S$为欧氏空间$E^3$中的曲面，其参数方程为$\boldsymbol{r}=\boldsymbol{r}(u^1,u^2)$。对定义在$S$上的切向量场$\boldsymbol{X}(u^1,u^2)=x^\alpha(u^1,u^2)\boldsymbol{r}_\alpha(u^1,u^2)$，若分量$x^\alpha(u^1,u^2)$为可微函数，则称$\boldsymbol{X}(u^1,u^2)$为\textbf{可微切向量场}。
\end{definition}

\begin{definition}
对可微切向量场$\boldsymbol{X}(u^1,u^2)$，将其微分$\mathrm{d}\boldsymbol{X}(u^1,u^2)$向曲面$S$在$(u^1,u^2)$处的切空间作正交投影，所得向量称为$\boldsymbol{X}(u^1,u^2)$的\textbf{协变微分}，记作$\mathrm{D}\boldsymbol{X}(u^1,u^2)$，即
\begin{align}
\mathrm{D}\boldsymbol{X}(u^1,u^2)=(\mathrm{d}\boldsymbol{X}(u^1,u^2))^\top=(\mathrm{d}x^\alpha+x^\beta\Gamma_{\beta\gamma}^\alpha\mathrm{d}u^\gamma)\boldsymbol{r}_\alpha,
\label{eq:cov_diff_vec}
\end{align}
其中$(\cdot)^\top$表示正交投影，$\Gamma_{\beta\gamma}^\alpha$为Christoffel记号。记
\begin{align}
\mathrm{D}x^\alpha=\mathrm{d}x^\alpha+x^\beta\Gamma_{\beta\gamma}^\alpha\mathrm{d}u^\gamma,
\label{eq:cov_diff_comp}
\end{align}
则$\mathrm{D}\boldsymbol{X}=\mathrm{D}x^\alpha \boldsymbol{r}_\alpha$，称$\mathrm{D}x^\alpha$为分量$x^\alpha(u^1,u^2)$的协变微分。
\end{definition}

\begin{theorem}
设$\sigma:S\to\tilde{S}$为保长对应，则对曲面$S$上任意可微切向量场$\boldsymbol{X}$，有
\begin{align}
\sigma_*(\mathrm{D}\boldsymbol{X})=\mathrm{D}(\sigma_*\boldsymbol{X}).
\label{eq:cov_diff_isometry}
\end{align}
\end{theorem}

\begin{proof}

在曲面$S$与$\tilde{S}$上取适配参数系，使保长对应$\sigma$为相同参数值点的对应，则切映射$\sigma_*$将$S$的自然基底$\boldsymbol{r}_\alpha$映为$\tilde{S}$的自然基底$\tilde{\boldsymbol{r}}_\alpha$，故
\[
\boldsymbol{X}=x^\alpha\boldsymbol{r}_\alpha,\quad \sigma_*\boldsymbol{X}=x^\alpha\tilde{\boldsymbol{r}}_\alpha.
\]
由$\sigma$保长，二者第一基本量一致，故Christoffel记号$\Gamma_{\beta\gamma}^\alpha$相同。代入协变微分定义得
\[
\mathrm{D}\boldsymbol{X}=(\mathrm{d}x^\alpha+x^\beta\Gamma_{\beta\gamma}^\alpha\mathrm{d}u^\gamma)\boldsymbol{r}_\alpha,\quad \mathrm{D}(\sigma_*\boldsymbol{X})=(\mathrm{d}x^\alpha+x^\beta\Gamma_{\beta\gamma}^\alpha\mathrm{d}u^\gamma)\tilde{\boldsymbol{r}}_\alpha=\sigma_*(\mathrm{D}\boldsymbol{X}).
\]

\end{proof}

\begin{remark}
协变微分仅依赖曲面第一基本形式与Christoffel记号，属于内蕴几何概念，与第二基本形式无关，可在仅给定第一基本形式的抽象曲面上定义。
\end{remark}

\begin{theorem}
曲面$S$上可微切向量场的协变微分满足如下运算性质：
\begin{enumerate}[(1)]
\item $\mathrm{D}(\boldsymbol{X}+\boldsymbol{Y})=\mathrm{D}\boldsymbol{X}+\mathrm{D}\boldsymbol{Y}$；
\item $\mathrm{D}(f\cdot\boldsymbol{X})=\mathrm{d}f\cdot\boldsymbol{X}+f\cdot\mathrm{D}\boldsymbol{X}$；
\item $\mathrm{d}(\boldsymbol{X}\cdot\boldsymbol{Y})=\mathrm{D}\boldsymbol{X}\cdot\boldsymbol{Y}+\boldsymbol{X}\cdot\mathrm{D}\boldsymbol{Y}$，
\end{enumerate}
其中$\boldsymbol{X},\boldsymbol{Y}$为可微切向量场，$f$为$S$上的可微函数。
\end{theorem}

\begin{remark}
协变微分算子$\mathrm{D}$具备与普通微分算子$\mathrm{d}$一致的运算规则。
\end{remark}

\begin{definition}
设$C:u^\alpha=u^\alpha(t)$为曲面$S$上的曲线，$\boldsymbol{X}(t)=x^\alpha(t)\boldsymbol{r}_\alpha(u^1(t),u^2(t))$为沿$C$定义的切向量场。将$\frac{\mathrm{d}\boldsymbol{X}(t)}{\mathrm{d}t}$正交投影至$S$在对应点的切空间，所得向量称为$\boldsymbol{X}(t)$沿$C$的\textbf{协变导数}，记作$\frac{\mathrm{D}\boldsymbol{X}(t)}{\mathrm{d}t}$，即
\begin{align}
\frac{\mathrm{D}\boldsymbol{X}(t)}{\mathrm{d}t}=\left(\frac{\mathrm{d}\boldsymbol{X}(t)}{\mathrm{d}t}\right)^\top.
\label{eq:cov_der_vec}
\end{align}
记
\begin{align}
\frac{\mathrm{D}x^\alpha(t)}{\mathrm{d}t}=\frac{\mathrm{d}x^\alpha(t)}{\mathrm{d}t}+\Gamma_{\beta\gamma}^\alpha x^\beta(t)\frac{\mathrm{d}u^\gamma(t)}{\mathrm{d}t},
\label{eq:cov_der_comp}
\end{align}
则$\frac{\mathrm{D}\boldsymbol{X}(t)}{\mathrm{d}t}=\frac{\mathrm{D}x^\alpha(t)}{\mathrm{d}t}\boldsymbol{r}_\alpha$，称$\frac{\mathrm{D}x^\alpha(t)}{\mathrm{d}t}$为分量$x^\alpha(t)$沿曲线$C$的协变导数。
\end{definition}

\begin{remark}
协变导数为内蕴几何概念，在曲面保长变换下保持不变，算子$\frac{\mathrm{D}}{\mathrm{d}t}$满足与$\mathrm{D}$一致的运算规则。
\end{remark}

对曲面$S$上可微切向量场的分量$x^\alpha(u^1,u^2)$，定义其沿参数曲线的协变导数
\begin{align*}
x^\alpha_{,\gamma}=\frac{\partial x^\alpha(u^1,u^2)}{\partial u^\gamma}+\Gamma_{\beta\gamma}^\alpha x^\beta(u^1,u^2),
\end{align*}
则分量的协变微分可表示为
\begin{align*}
\mathrm{D}x^\alpha=x^\alpha_{,\gamma}\mathrm{d}u^\gamma.
\end{align*}

\begin{definition}
设$\boldsymbol{X}(t)$为曲面$S$上沿曲线$C:u^\gamma=u^\gamma(t)$定义的可微切向量场，若
\begin{align}
\frac{\mathrm{D}\boldsymbol{X}(t)}{\mathrm{d}t}=0,
\label{eq:parallel_def}
\end{align}
则称$\boldsymbol{X}(t)$沿曲线$C$\textbf{{平行}}。
\end{definition}

\begin{remark}
切向量场$\boldsymbol{X}(t)$沿$C$平行的充要条件为其分量满足一阶线性齐次常微分方程组
\begin{align}
\frac{\mathrm{d}x^\alpha(t)}{\mathrm{d}t}+\Gamma_{\beta\gamma}^\alpha x^\beta(t)\frac{\mathrm{d}u^\gamma(t)}{\mathrm{d}t}=0,\quad \alpha=1,2.
\label{eq:parallel_ode}
\end{align}
该方程组对给定初值$x_0^\alpha$存在唯一解，对应切向量沿曲线$C$的平行移动。
\end{remark}

\begin{theorem}
设$C:u^\gamma=u^\gamma(t),a\leqslant t\leqslant b$为曲面$S$上连接点$A,B$的可微曲线，记$\mathrm{P}_a^b$为切向量沿$C$从$A$到$B$的平行移动，则映射
\[
\mathrm{P}_a^b:T_A S\to T_B S
\]
为切空间之间的\textbf{等距同构}。
\end{theorem}

\begin{proof}

平行移动对应的微分方程组解空间与曲面在定点的切空间线性同构，故$\mathrm{P}_a^b$为线性同构。
对沿$C$平行的切向量场$\boldsymbol{X}(t),\boldsymbol{Y}(t)$，由协变微分的运算性质，
\[
\frac{\mathrm{d}}{\mathrm{d}t}(\boldsymbol{X}(t)\cdot\boldsymbol{Y}(t))=\frac{\mathrm{D}\boldsymbol{X}(t)}{\mathrm{d}t}\cdot\boldsymbol{Y}(t)+\boldsymbol{X}(t)\cdot\frac{\mathrm{D}\boldsymbol{Y}(t)}{\mathrm{d}t}=0,
\]
故切向量内积沿平行移动保持不变，因此$\mathrm{P}_a^b$为等距同构。

\end{proof}

\begin{theorem}
设曲面$S_1,S_2$沿曲线$C$相切，$\frac{\mathrm{D}^{(1)}}{\mathrm{d}t},\frac{\mathrm{D}^{(2)}}{\mathrm{d}t}$分别为二者沿$C$的协变导数算子，$\boldsymbol{X}(t)$为沿$C$的切向量场，则
\[
\frac{\mathrm{D}^{(1)}\boldsymbol{X}(t)}{\mathrm{d}t}=\frac{\mathrm{D}^{(2)}\boldsymbol{X}(t)}{\mathrm{d}t}.
\]
特别地，若$\boldsymbol{X}(t)$在$S_1$上沿$C$平行，则其在$S_2$上也沿$C$平行。
\end{theorem}

\begin{example}
设$S$为欧氏空间$E^3$中的单位球面，$\tilde{S}$为与$S$沿圆周$C$相切的锥面，求球面$S$上的切向量沿圆周$C$平行移动一周后，与原切向量的夹角。
\begin{figure}[H]
\centering
\includegraphics[scale=0.3]{球面上切向量的平行移动.png}
\caption{球面上切向量的平行移动}
\label{figure:sphere_parallel}
\end{figure}
\begin{solution}

切向量沿$C$在$S$与$\tilde{S}$上的平行移动等价。设圆周$C$的半径为$r_0$，锥面$\tilde{S}$保长展开为平面$\Pi$时，$C$对应平面内半径为$\frac{r_0}{\sqrt{1-r_0^2}}$的圆弧$\tilde{C}$，其圆心角为$\alpha=2\pi\sqrt{1-r_0^2}$。
设初始切向量与圆周$C$的夹角为$\theta$，展开后该向量与圆弧$\tilde{C}$的夹角仍为$\theta$。平面内平行移动一周后，向量与$\tilde{C}$的夹角为$2\pi+\theta-\alpha$，故球面上移动后的切向量与原向量的夹角为$2\pi-\alpha=2\pi-2\pi\sqrt{1-r_0^2}$。

\end{solution}
\end{example}

\begin{remark}
弯曲曲面上切向量沿闭曲线平行移动一周后，一般不与原切向量重合，这是弯曲曲面几何与平面欧氏几何的本质差异。
\end{remark}

设曲线$C$的参数为弧长$s$，则其测地曲率可表示为
\begin{align}
\kappa_g=\frac{\mathrm{d}^2\boldsymbol{r}}{\mathrm{d}s^2}\cdot\boldsymbol{e}_2=\frac{\mathrm{D}\boldsymbol{e}_1(s)}{\mathrm{d}s}\cdot\boldsymbol{e}_2(s)=\sqrt{g_{11}g_{22}-(g_{12})^2}\begin{vmatrix}
\frac{\mathrm{d}u^1}{\mathrm{d}s}&\frac{\mathrm{d}u^2}{\mathrm{d}s}\\
\frac{\mathrm{D}}{\mathrm{d}s}\left(\frac{\mathrm{d}u^1}{\mathrm{d}s}\right)&\frac{\mathrm{D}}{\mathrm{d}s}\left(\frac{\mathrm{d}u^2}{\mathrm{d}s}\right)
\end{vmatrix}.
\label{eq:geod_curv}
\end{align}

\begin{remark}
平面$\mathbb{R}^2$在直角坐标系下，协变导数等价于普通导数，此时式\eqref{eq:geod_curv}即为平面曲线的相对曲率公式。
\end{remark}

\begin{definition}
曲面$S$上以弧长$s$为参数的曲线$C:\boldsymbol{r}=\boldsymbol{r}(s)$，若满足
\begin{align}
\frac{\mathrm{D}}{\mathrm{d}s}\left(\frac{\mathrm{d}\boldsymbol{r}(s)}{\mathrm{d}s}\right)=0,
\label{eq:geod_line_vec}
\end{align}
或分量形式
\begin{align}
\frac{\mathrm{D}}{\mathrm{d}s}\left(\frac{\mathrm{d}u^\alpha}{\mathrm{d}s}\right)=0,\quad \alpha=1,2,
\label{eq:geod_line_comp}
\end{align}
则称$C$为\textbf{测地线}。
\end{definition}

\begin{remark}
测地线是其单位切向量沿自身平行的曲线，可视为平面上直线概念在曲面上的推广。
\end{remark}

\begin{example}
设$x^\alpha = x^\alpha(u^1,u^2)$是偏微分方程组
\begin{align*}
\frac{\partial x^\alpha}{\partial u^\beta} = -\Gamma^\alpha_{\beta\gamma}x^\gamma.
\end{align*}
的非零解，其中$\Gamma^\alpha_{\beta\gamma}$是曲面$S$关于它的第一基本形式$\mathrm{I}=g_{\alpha\beta}\mathrm{d}u^\alpha\mathrm{d}u^\beta$的Christoffel记号。证明:
\begin{enumerate}[(1)]
\item $f = g_{\alpha\beta}x^\alpha(u^1,u^2)x^\beta(u^1,u^2)$是非零常数;
\item $\boldsymbol{X}(u^1,u^2) = x^\alpha(u^1,u^2)\boldsymbol{r}_\alpha(u^1,u^2)$是曲面$S$上沿任意一条曲线平行的切向量场.
\end{enumerate}
\end{example}

\begin{example}
证明: 在曲面$S$上存在一个非零的、与路径无关的平行切向量场，当且仅当曲面$S$的Gauss曲率恒等于零.
\end{example}

\begin{example}
证明: 曲面$S$上的$u^\alpha$-曲线($\alpha$固定)的单位切向量沿曲线$C:u^\gamma = u^\gamma(t)$是平行的充分必要条件是, 对于$\beta \neq \alpha$, 沿曲线$C$下式成立:
\begin{align*}
\Gamma^\beta_{\alpha\gamma}(u^1(t),u^2(t))\frac{\mathrm{d}u^\gamma(t)}{\mathrm{d}t} = 0.
\end{align*}
\end{example}
