\section{切平面与法线}

\begin{definition}
设$S$为正则参数曲面，$p\in S$，$S$上过点$p$的任意一条连续可微曲线在$p$处的切向量，称为曲面$S$在点$p$的\textbf{切向量}。
\end{definition}

\begin{proposition}
设正则参数曲面$S:\bm r=\bm r(u,v)$，$S$上曲线的参数表示为$u=u(t),v=v(t)$，点$p$对应$t=0$，则该曲线在$p$处的切向量为
\begin{align}
\left.\frac{\mathrm{d}\bm r(u(t),v(t))}{\mathrm{d}t}\right|_{t=0}=\left.\bm r_u\frac{\mathrm{d}u(t)}{\mathrm{d}t}\right|_{t=0}+\left.\bm r_v\frac{\mathrm{d}v(t)}{\mathrm{d}t}\right|_{t=0}.
\label{eq:tangentvector1}
\end{align}
反之，$\bm r_u,\bm r_v$的任意线性组合$a\bm r_u+b\bm r_v\;(a,b\in\mathbb{R})$均为$S$在$p$处的切向量。
\end{proposition}

\begin{proof}

对$\bm r=\bm r(u(t),v(t))$关于$t$求导，由链式法则可得
\begin{align*}
\frac{\mathrm{d}\bm r(u(t),v(t))}{\mathrm{d}t}=\frac{\partial\bm r}{\partial u}\frac{\mathrm{d}u}{\mathrm{d}t}+\frac{\partial\bm r}{\partial v}\frac{\mathrm{d}v}{\mathrm{d}t}=\bm r_u\frac{\mathrm{d}u}{\mathrm{d}t}+\bm r_v\frac{\mathrm{d}v}{\mathrm{d}t}.
\end{align*}
代入$t=0$即得式\eqref{eq:tangentvector1}。

任取$a,b\in\mathbb{R}$，取$S$上曲线$u(t)=u_0+at,\;v(t)=v_0+bt$，其中$(u_0,v_0)$为点$p$对应的参数，则
\begin{align*}
\left.\frac{\mathrm{d}\bm r(u(t),v(t))}{\mathrm{d}t}\right|_{t=0}=a\bm r_u+b\bm r_v,
\end{align*}
故$a\bm r_u+b\bm r_v$为$S$在$p$处的切向量.

\end{proof}

\begin{definition}
正则参数曲面$S$在点$p$处的全体切向量构成二维实线性空间，称为$S$在$p$处的\textbf{切空间}，记作$T_pS$，$\{\bm r_u,\bm r_v\}$为$T_pS$的一组基底。
\end{definition}

\begin{definition}
$\mathbb{E}^3$中过点$p$、由切向量$\bm r_u,\bm r_v$张成的二维平面，称为曲面$S$在点$p$的\textbf{切平面}，其参数方程为
\begin{align}
\bm X(\lambda,\mu)=\bm r(u,v)+\lambda\bm r_u(u,v)+\mu\bm r_v(u,v),
\label{eq:tangentplane}
\end{align}
其中$\lambda,\mu\in\mathbb{R}$为参数；切平面的单位法向量为
\begin{align}
\bm n(u,v)=\frac{\bm r_u(u,v)\times\bm r_v(u,v)}{|\bm r_u(u,v)\times\bm r_v(u,v)|}.
\label{eq:unitnormal}
\end{align}
\end{definition}

\begin{definition}
$\mathbb{E}^3$中过点$p$、以单位法向量$\bm n(u,v)$为方向向量的直线，称为曲面$S$在点$p$的\textbf{法线}，其参数方程为
\begin{align}
\bm X(t)=\bm r(u,v)+t\bm n(u,v),
\label{eq:normalline}
\end{align}
其中$t\in\mathbb{R}$为参数。
\end{definition}

\begin{theorem}
正则参数曲面的切空间、切平面、法线在容许参数变换下保持不变；可定向正则曲面上存在连续可微的单位法向量场，且仅有两个互为反向的单位法向量场，选取其一即确定曲面的定向。
\end{theorem}

\begin{proof}

由曲面参数变换的微分性质，切向量的线性结构在容许参数变换下不变，故切空间、切平面、法线与参数选取无关。
容许参数变换保持定向时，$\bm r_u\times\bm r_v$的方向不变，故单位法向量$\bm n$指向不变；翻转定向时，$\bm r_u\times\bm r_v$反向，$\bm n$随之反向。因此可定向正则曲面上恰有两个反向的连续可微单位法向量场，选定其一即确定曲面定向.

\end{proof}

\begin{definition}
正则参数曲面$S:\bm r=\bm r(u,v)$上每一点处的标架$\{\bm r(u,v);\bm r_u(u,v),\bm r_v(u,v),\bm n(u,v)\}$，称为该曲面的\textbf{自然标架}。
\end{definition}

\begin{proposition}
设$f(x,y,z)$为$\mathbb{E}^3$区域上的连续可微函数，若$\left(\frac{\partial f}{\partial x},\frac{\partial f}{\partial y},\frac{\partial f}{\partial z}\right)\neq\bm 0$，则等值面$f(x,y,z)=c$为$\mathbb{E}^3$中的正则曲面，且$\nabla f=\left(\frac{\partial f}{\partial x},\frac{\partial f}{\partial y},\frac{\partial f}{\partial z}\right)$为该等值面的法向量。
\end{proposition}
\begin{proof}

任取等值面上一点$p(x_0,y_0,z_0)$，不妨设$\left.\frac{\partial f}{\partial z}\right|_p\neq0$，由隐函数定理，存在$Oxy$平面上$(x_0,y_0)$的邻域，使得$z=g(x,y)$满足$f(x,y,g(x,y))=c$，故等值面的参数表示为$\bm r(x,y)=(x,y,g(x,y))$，满足正则曲面的条件。

设$x=x(t),y=y(t),z=z(t)$为等值面上任意曲线，则$f(x(t),y(t),z(t))=c$，对$t$求导得
\begin{align}
\frac{\partial f}{\partial x}\frac{\mathrm{d}x}{\mathrm{d}t}+\frac{\partial f}{\partial y}\frac{\mathrm{d}y}{\mathrm{d}t}+\frac{\partial f}{\partial z}\frac{\mathrm{d}z}{\mathrm{d}t}=0.
\label{eq:gradienttangentortho}
\end{align}
式\eqref{eq:gradienttangentortho}表明$\nabla f$与曲线切向量正交，故$\nabla f$为等值面的法向量.

\end{proof}

\begin{proposition}
正则参数曲面$S:\bm r=\bm r(u,v)$的微分为
\begin{align}
\mathrm{d}\bm r(u,v)=\bm r_u(u,v)\mathrm{d}u+\bm r_v(u,v)\mathrm{d}v,
\label{eq:surfdiff}
\end{align}
$\mathrm{d}u,\mathrm{d}v$为切空间$T_pS$上的线性函数，对任意切向量$\bm X=X^1\bm r_u+X^2\bm r_v$，有$\mathrm{d}u(\bm X)=X^1,\;\mathrm{d}v(\bm X)=X^2$。
\end{proposition}
\begin{remark}
\begin{enumerate}[(1)]
\item 切平面的参数方程可写为$\bm X=\bm r(u,v)+\bm r_u\mathrm{d}u+\bm r_v\mathrm{d}v$，$\mathrm{d}u,\mathrm{d}v$为切平面上动点的参数；
\item 自然基底$\{\bm r_u,\bm r_v\}$一般非单位正交基，故切向量分量$(\mathrm{d}u,\mathrm{d}v)$非笛卡尔直角坐标系下的分量；
\item 比值$\mathrm{d}u:\mathrm{d}v$可表示曲面$S$上的一个方向。
\end{enumerate}
\end{remark}
\begin{proof}

由多元函数微分的定义，$\bm r(u,v)$的增量满足
\begin{align*}
\lim\limits_{\rho\to0}\frac{|\bm r(u+\Delta u,v+\Delta v)-\bm r(u,v)-\mathrm{d}\bm r(u,v)|}{\rho}=0,
\end{align*}
其中$\rho=\sqrt{(\Delta u)^2+(\Delta v)^2}$，故$\mathrm{d}\bm r=\bm r_u\mathrm{d}u+\bm r_v\mathrm{d}v$。

设$V$为$n$维线性空间，取基底$\{\bm e_1,\dots,\bm e_n\}$，对任意$\bm w=\sum\limits_{i=1}^nw^i\bm e_i$，定义$f^i(\bm w)=w^i$，则
\begin{align*}
f^i(\bm w+\bm z)=w^i+z^i=f^i(\bm w)+f^i(\bm z),\quad f^i(\lambda\bm w)=\lambda w^i=\lambda f^i(\bm w),
\end{align*}
故$f^i$为$V$上的线性函数。
切空间$T_pS$为二维线性空间，基底为$\{\bm r_u,\bm r_v\}$，因此$\mathrm{d}u,\mathrm{d}v$为$T_pS$上的线性函数，对应切向量的分量.

\end{proof}

\begin{example}
证明: 一个正则参数曲面是球面的充分必要条件是, 它的所有法线都经过一个固定点.
\end{example}

\begin{example}
证明: 一个正则参数曲面是锥面的充分必要条件是, 它的所有切平面都经过一个固定点.
\end{example}

\begin{example}
证明: 一个正则参数曲面是旋转面的充分必要条件是, 它的所有法线都与一条固定的直线相交.
\end{example}

\begin{example}
假定在方程
\begin{align*}
\frac{x^2}{a-\lambda}+\frac{y^2}{b-\lambda}+\frac{z^2}{c-\lambda}=1.
\end{align*}
中,$a,b,c$是常数,并且$a>b>c$,$\lambda$是参数. 当$\lambda\in(-\infty,c)$时,方程给出一族椭球面; 当$\lambda\in(c,b)$时,方程给出一族单叶双曲面; 当$\lambda\in(b,a)$时,方程给出一族双叶双曲面. 证明: 经过空间中不在各坐标面上的任意一点有且只有分别属于这三族曲面的三个二次曲面,并且它们沿交线是彼此正交的.
\end{example}

\begin{example}
设$S$是圆锥面$\bm{r}=(v\cos u,v\sin u,v)$,$C$是$S$上的一条曲线,其方程是$u=\sqrt{2}t$,$v=\mathrm{e}^t$.
\begin{enumerate}[(1)]
\item 将曲线$C$的切向量用$\bm{r}_u,\bm{r}_v$的线性组合表示出来;
\item 证明: $C$的切向量平分了$\bm{r}_u$和$\bm{r}_v$的夹角.
\end{enumerate}
\end{example}
